70 inline bool is_host_memory(const void* ptr)
71 {
72  CUpointer_attribute attrs[1] = {
73  CU_POINTER_ATTRIBUTE_MEMORY_TYPE,
74  };
75  CUmemorytype memtype{};
76  void* data[1] = {&memtype};
77  CUDA_DRIVER_TRY(
78  cudaAPI::instance().PointerGetAttributes(1, attrs, data, convert_void2deviceptr(ptr)));
79  // Notice, queying `CU_POINTER_ATTRIBUTE_MEMORY_TYPE` returns zero when the memory
80  // is unregistered host memory. This is undocumented but how the Runtime CUDA API
81  // does it to support `cudaMemoryTypeUnregistered`.
82  return memtype == 0 || memtype == CU_MEMORYTYPE_HOST;
83 }

115 // Find the base and offset of the memory allocation `devPtr` is in
116 inline std::tuple<void*, std::size_t, std::size_t> get_alloc_info(const void* devPtr,
117  CUcontext* ctx = nullptr)
118 {
119  auto dev = convert_void2deviceptr(devPtr);
120  CUdeviceptr base_ptr{};
121  std::size_t base_size{};
122  CUcontext _ctx{};
123  if (ctx != nullptr) {
124  _ctx = *ctx;
125  } else {
126  _ctx = get_context_from_device_pointer(devPtr);
127  }
128  PushAndPopContext context(_ctx);
129  CUDA_DRIVER_TRY(cudaAPI::instance().MemGetAddressRange(&base_ptr, &base_size, dev));
130  std::size_t offset = dev - base_ptr;
131  // NOLINTNEXTLINE(performance-no-int-to-ptr, cppcoreguidelines-pro-type-reinterpret-cast)
132  return std::make_tuple(reinterpret_cast<void*>(base_ptr), base_size, offset);
133 }
