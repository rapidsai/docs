62  [[nodiscard]] Alloc get()
63  {
64  const std::lock_guard lock(_mutex);
65  // Check if we have an allocation available
66  if (!_free_allocs.empty()) {
67  void* ret = _free_allocs.top();
68  _free_allocs.pop();
69  return Alloc(this, ret);
70  }
71 
72  // If no available allocation, allocate and register a new one
73  void* alloc{};
74  // Allocate page-locked host memory
75  CUDA_DRIVER_TRY(
76  cudaAPI::instance().MemHostAlloc(&alloc, chunk_size, CU_MEMHOSTREGISTER_PORTABLE));
77  return Alloc(this, alloc);
78  }

112 inline void pwrite_all(int fd, const void* buf, size_t count, off_t offset)
113 {
114  off_t cur_offset = offset;
115  size_t byte_remaining = count;
116  const char* buffer = static_cast<const char*>(buf);
117  while (byte_remaining > 0) {
118  ssize_t nbytes_written = ::pwrite(fd, buffer, byte_remaining, cur_offset);
119  if (nbytes_written == -1) {
120  if (errno == EBADF) {
121  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
122  KVIKIO_STRINGIFY(__LINE__) + ": unsupported file open flags"};
123  }
124  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
125  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(errno)};
126  }
127  if (nbytes_written == 0) {
128  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
129  KVIKIO_STRINGIFY(__LINE__) + ": EOF"};
130  }
131 
132  buffer += nbytes_written; // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
133  cur_offset += nbytes_written;
134  byte_remaining -= nbytes_written;
135  }
136 }

147 inline ssize_t pread_some(int fd, void* buf, size_t count, off_t offset)
148 {
149  ssize_t ret = ::pread(fd, buf, count, offset);
150  if (ret == -1) {
151  if (errno == EBADF) {
152  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
153  KVIKIO_STRINGIFY(__LINE__) + ": unsupported file open flags"};
154  }
155  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
156  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(errno)};
157  }
158  if (ret == 0) {
159  throw CUfileException{std::string{"POSIX error on pread at: "} + __FILE__ + ":" +
160  KVIKIO_STRINGIFY(__LINE__) + ": EOF"};
161  }
162  return ret;
163 }

176 template <bool IsReadOperation>
177 inline std::size_t posix_io(int fd,
178  const void* devPtr_base,
179  std::size_t size,
180  std::size_t file_offset,
181  std::size_t devPtr_offset)
182 {
183  auto alloc = manager.get();
184  CUdeviceptr devPtr = convert_void2deviceptr(devPtr_base) + devPtr_offset;
185  off_t cur_file_offset = convert_size2off(file_offset);
186  off_t byte_remaining = convert_size2off(size);
187  const off_t chunk_size2 = convert_size2off(chunk_size);
188 
189  while (byte_remaining > 0) {
190  const off_t nbytes_requested = std::min(chunk_size2, byte_remaining);
191  ssize_t nbytes_got = nbytes_requested;
192  if constexpr (IsReadOperation) {
193  nbytes_got = pread_some(fd, alloc.get(), nbytes_requested, cur_file_offset);
194  CUDA_DRIVER_TRY(cudaAPI::instance().MemcpyHtoD(devPtr, alloc.get(), nbytes_got));
195  } else { // Is a write operation
196  CUDA_DRIVER_TRY(cudaAPI::instance().MemcpyDtoH(alloc.get(), devPtr, nbytes_requested));
197  pwrite_all(fd, alloc.get(), nbytes_requested, cur_file_offset);
198  }
199  cur_file_offset += nbytes_got;
200  devPtr += nbytes_got;
201  byte_remaining -= nbytes_got;
202  }
203  return size;
204 }
