86 // Find the base and offset of the memory allocation `devPtr` is in
87 inline std::tuple<void*, std::size_t, std::size_t> get_alloc_info(const void* devPtr,
88  CUcontext* ctx = nullptr)
89 {
90  auto dev = convert_void2deviceptr(devPtr);
91  CUdeviceptr base_ptr{};
92  std::size_t base_size{};
93  CUcontext _ctx{};
94  if (ctx != nullptr) {
95  _ctx = *ctx;
96  } else {
97  _ctx = get_context_from_device_pointer(devPtr);
98  }
99  PushAndPopContext context(_ctx);
100  CUDA_DRIVER_TRY(cuMemGetAddressRange(&base_ptr, &base_size, dev));
101  std::size_t offset = dev - base_ptr;
102  // NOLINTNEXTLINE(performance-no-int-to-ptr, cppcoreguidelines-pro-type-reinterpret-cast)
103  return std::make_tuple(reinterpret_cast<void*>(base_ptr), base_size, offset);
104 }
