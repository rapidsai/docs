28 [[nodiscard]] inline bool get_driver_flag(unsigned int prop, unsigned int flag) noexcept
29 {
30  return (prop & (1U << flag)) != 0;
31 }
32 
33 inline void set_driver_flag(unsigned int& prop, unsigned int flag, bool val) noexcept
34 {
35  if (val) {
36  prop |= (1U << flag);
37  } else {
38  prop &= ~(1U << flag);
39  }
40 }
