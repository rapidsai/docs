70 inline bool is_host_memory(const void* ptr)
71 {
72  CUpointer_attribute attrs[1] = {
73  CU_POINTER_ATTRIBUTE_MEMORY_TYPE,
74  };
75  CUmemorytype memtype{};
76  void* data[1] = {&memtype};
77  CUresult result =
78  cudaAPI::instance().PointerGetAttributes(1, attrs, data, convert_void2deviceptr(ptr));
79 
80  // We assume that `ptr` is host memory when CUDA_ERROR_NOT_INITIALIZED
81  if (result == CUDA_ERROR_NOT_INITIALIZED) { return true; }
82  CUDA_DRIVER_TRY(result);
83 
84  // Notice, queying `CU_POINTER_ATTRIBUTE_MEMORY_TYPE` returns zero when the memory
85  // is unregistered host memory. This is undocumented but how the Runtime CUDA API
86  // does it to support `cudaMemoryTypeUnregistered`.
87  return memtype == 0 || memtype == CU_MEMORYTYPE_HOST;
88 }

120 // Find the base and offset of the memory allocation `devPtr` is in
121 inline std::tuple<void*, std::size_t, std::size_t> get_alloc_info(const void* devPtr,
122  CUcontext* ctx = nullptr)
123 {
124  auto dev = convert_void2deviceptr(devPtr);
125  CUdeviceptr base_ptr{};
126  std::size_t base_size{};
127  CUcontext _ctx{};
128  if (ctx != nullptr) {
129  _ctx = *ctx;
130  } else {
131  _ctx = get_context_from_device_pointer(devPtr);
132  }
133  PushAndPopContext context(_ctx);
134  CUDA_DRIVER_TRY(cudaAPI::instance().MemGetAddressRange(&base_ptr, &base_size, dev));
135  std::size_t offset = dev - base_ptr;
136  // NOLINTNEXTLINE(performance-no-int-to-ptr, cppcoreguidelines-pro-type-reinterpret-cast)
137  return std::make_tuple(reinterpret_cast<void*>(base_ptr), base_size, offset);
138 }
