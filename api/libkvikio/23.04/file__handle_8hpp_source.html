48 inline int open_fd_parse_flags(const std::string& flags, bool o_direct)
49 {
50  int file_flags = -1;
51  if (flags.empty()) { throw std::invalid_argument("Unknown file open flag"); }
52  switch (flags[0]) {
53  case 'r':
54  file_flags = O_RDONLY;
55  if (flags[1] == '+') { file_flags = O_RDWR; }
56  break;
57  case 'w':
58  file_flags = O_WRONLY;
59  if (flags[1] == '+') { file_flags = O_RDWR; }
60  file_flags |= O_CREAT | O_TRUNC;
61  break;
62  case 'a': throw std::invalid_argument("Open flag 'a' isn't supported");
63  default: throw std::invalid_argument("Unknown file open flag");
64  }
65  file_flags |= O_CLOEXEC;
66  if (o_direct) { file_flags |= O_DIRECT; }
67  return file_flags;
68 }

377  std::future<std::size_t> pread(void* buf,
378  std::size_t size,
379  std::size_t file_offset = 0,
380  std::size_t task_size = defaults::task_size())
381  {
382  if (is_host_memory(buf)) {
383  auto op = [this](void* hostPtr_base,
384  std::size_t size,
385  std::size_t file_offset,
386  std::size_t hostPtr_offset) -> std::size_t {
387  char* buf = static_cast<char*>(hostPtr_base) + hostPtr_offset;
388  return posix_host_read(_fd_direct_off, buf, size, file_offset, false);
389  };
390 
391  return parallel_io(op, buf, size, file_offset, task_size, 0);
392  }
393 
394  CUcontext ctx = get_context_from_device_pointer(buf);
395  auto task = [this, ctx](void* devPtr_base,
396  std::size_t size,
397  std::size_t file_offset,
398  std::size_t devPtr_offset) -> std::size_t {
399  PushAndPopContext c(ctx);
400  return read(devPtr_base, size, file_offset, devPtr_offset);
401  };
402 
403  auto [devPtr_base, base_size, devPtr_offset] = get_alloc_info(buf, &ctx);
404  return parallel_io(task, devPtr_base, size, file_offset, task_size, devPtr_offset);
