67  // cuFile is supposed to open and close the driver automatically but because of a bug in
68  // CUDA 11.8, it sometimes segfault. See <https://github.com/rapidsai/kvikio/issues/159>.
69  CUfileError_t const error = DriverOpen();
70  if (error.err != CU_FILE_SUCCESS) {
71  throw std::runtime_error(std::string{"cuFile error at: "} + __FILE__ + ":" +
72  KVIKIO_STRINGIFY(__LINE__) + ": " +
73  cufileop_status_error(error.err));
74  }
75  }
76  ~cuFileAPI()
77  {
78  CUfileError_t const error = DriverClose();
79  if (error.err != CU_FILE_SUCCESS) {
80  std::cerr << "Unable to close GDS file driver: " << cufileop_status_error(error.err)
81  << std::endl;
82  }
83  }
