64  {
65  const std::lock_guard lock(_mutex);
66  // Check if we have an allocation available
67  if (!_free_allocs.empty()) {

69  _free_allocs.pop();

116 template <bool IsReadOperation>
117 ssize_t posix_host_io(int fd, const void* buf, size_t count, off_t offset, bool partial)
118 {
119  off_t cur_offset = offset;
120  size_t byte_remaining = count;
121  char* buffer = const_cast<char*>(static_cast<const char*>(buf));
122  while (byte_remaining > 0) {
123  ssize_t nbytes = 0;
124  if constexpr (IsReadOperation) {
125  nbytes = ::pread(fd, buffer, byte_remaining, cur_offset);
126  } else {
127  nbytes = ::pwrite(fd, buffer, byte_remaining, cur_offset);
128  }
129  if (nbytes == -1) {
130  if (errno == EBADF) {
131  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
132  KVIKIO_STRINGIFY(__LINE__) + ": unsupported file open flags"};
133  }
134  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
135  KVIKIO_STRINGIFY(__LINE__) + ": " + strerror(errno)};
136  }
137  if (nbytes == 0) {
138  throw CUfileException{std::string{"POSIX error on pwrite at: "} + __FILE__ + ":" +
139  KVIKIO_STRINGIFY(__LINE__) + ": EOF"};
140  }
141  if (partial) { return nbytes; }
142  buffer += nbytes; // NOLINT(cppcoreguidelines-pro-bounds-pointer-arithmetic)
143  cur_offset += nbytes;
144  byte_remaining -= nbytes;
145  }
146  return convert_size2ssize(count);
147 }

160 template <bool IsReadOperation>
161 std::size_t posix_device_io(int fd,
162  const void* devPtr_base,
163  std::size_t size,
164  std::size_t file_offset,
165  std::size_t devPtr_offset)
166 {
167  auto alloc = manager.get();
168  CUdeviceptr devPtr = convert_void2deviceptr(devPtr_base) + devPtr_offset;
169  off_t cur_file_offset = convert_size2off(file_offset);
170  off_t byte_remaining = convert_size2off(size);
171  const off_t chunk_size2 = convert_size2off(posix_bounce_buffer_size);
172 
173  while (byte_remaining > 0) {
174  const off_t nbytes_requested = std::min(chunk_size2, byte_remaining);
175  ssize_t nbytes_got = nbytes_requested;
176  if constexpr (IsReadOperation) {
177  nbytes_got = posix_host_io<true>(fd, alloc.get(), nbytes_requested, cur_file_offset, true);
178  CUDA_DRIVER_TRY(cudaAPI::instance().MemcpyHtoD(devPtr, alloc.get(), nbytes_got));
179  } else { // Is a write operation
180  CUDA_DRIVER_TRY(cudaAPI::instance().MemcpyDtoH(alloc.get(), devPtr, nbytes_requested));
181  posix_host_io<false>(fd, alloc.get(), nbytes_requested, cur_file_offset, false);
182  }
183  cur_file_offset += nbytes_got;
184  devPtr += nbytes_got;
185  byte_remaining -= nbytes_got;
186  }
187  return size;
188 }
