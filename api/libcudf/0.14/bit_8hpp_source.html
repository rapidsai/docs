45 template < typename T>
46 constexpr CUDA_HOST_DEVICE_CALLABLE std::size_t size_in_bits() {
47 static_assert(CHAR_BIT == 8, "Size of a byte must be 8 bits." );
48 return sizeof (T) * CHAR_BIT;

121 constexpr CUDA_HOST_DEVICE_CALLABLE bitmask_type
122 set_least_significant_bits (size_type n) {
123 constexpr_assert(0 <= n && n < static_cast<size_type>(
124 detail::size_in_bits<bitmask_type>()));
125 return ((bitmask_type{1} << n) - 1);

136 constexpr CUDA_HOST_DEVICE_CALLABLE bitmask_type
137 set_most_significant_bits (size_type n) {
138 constexpr size_type word_size{detail::size_in_bits<bitmask_type>()};
139 constexpr_assert(0 <= n && n < word_size);
140 return ~((bitmask_type{1} << (word_size - n)) - 1);
141 }
