70 template <typename T, gdf_dtype type_id>
71 struct wrapper {
72  static constexpr gdf_dtype corresponding_column_type{type_id};
73  using value_type = T;

77  CUDA_HOST_DEVICE_CALLABLE constexpr explicit wrapper(value_type v) : value{v}
78  {
79  }

85  {

88  template <gdf_dtype dtype = type_id, std::enable_if_t<(dtype != GDF_BOOL8)> * = nullptr>
89  CUDA_HOST_DEVICE_CALLABLE explicit operator value_type() const
90  {
91  return this->value;
92  }
93 
94  template <gdf_dtype dtype = type_id, std::enable_if_t<(dtype == GDF_BOOL8)> * = nullptr>
95  CUDA_HOST_DEVICE_CALLABLE explicit operator value_type() const
96  {
97  return static_cast<value_type>(static_cast<bool>(this->value));
98  }
99 
100  // enable conversion to arithmetic types *only* for the cudf::bool8 wrapper
101  // (defined later in this file as wrapper<int8_t, GDF_BOOL8>)
102  template <typename T_out,
103  gdf_dtype dtype = type_id,
104  typename std::enable_if<(dtype == GDF_BOOL8) && std::is_arithmetic<T_out>::value,
105  int>::type * = nullptr>
106  CUDA_HOST_DEVICE_CALLABLE explicit operator T_out() const
107  {
108  // Casting a cudf::bool8 to arithmetic type should always be the same as
109  // casting a bool to arithmetic type, and not the same as casting the
110  // underlying type to arithmetic type. Therefore we cast the value to bool
111  // first, then the output type
112  return static_cast<T_out>(static_cast<bool>(this->value));
113  }

115  wrapper(wrapper const &w) = default;

117  wrapper() = default;
