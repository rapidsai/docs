29 struct CompressedStreamInfo {
30 CompressedStreamInfo () = default ;
31 explicit constexpr CompressedStreamInfo ( const uint8_t *compressed_data_,
32 size_t compressed_size_)
33 : compressed_data(compressed_data_),
34 compressed_data_size(compressed_size_),
35 uncompressed_data( nullptr ),
36 decctl( nullptr ),
37 decstatus( nullptr ),
38 copyctl( nullptr ),
39 num_compressed_blocks(0),
40 num_uncompressed_blocks(0),
41 max_uncompressed_size(0) {}
42 const uint8_t *compressed_data; // [in] base ptr to compressed stream data
43 uint8_t *uncompressed_data; // [in] base ptr to uncompressed stream data or NULL if not known yet
44 size_t compressed_data_size; // [in] compressed data size for this stream
45 gpu_inflate_input_s *decctl; // [in] base ptr to decompression structure to be filled
46 gpu_inflate_status_s *decstatus; // [in] results of decompression
47 gpu_inflate_input_s *copyctl; // [in] base ptr to copy structure to be filled for uncompressed blocks
48 uint32_t num_compressed_blocks; // [in,out] number of entries in decctl(in), number of compressed blocks(out)
49 uint32_t num_uncompressed_blocks; // [in,out] number of entries in copyctl(in), number of uncompressed blocks(out)
50 uint64_t max_uncompressed_size; // [out] maximum uncompressed data size
51 };

89 struct ColumnDesc
90 {
91 const uint8_t *streams[CI_NUM_STREAMS]; // ptr to data stream index
92 uint32_t strm_id[CI_NUM_STREAMS]; // stream ids
93 uint32_t strm_len[CI_NUM_STREAMS]; // stream length
94 uint32_t *valid_map_base; // base pointer of valid bit map for this column

96 uint32_t start_row; // starting row of the stripe

98 uint32_t dictionary_start; // start position in global dictionary
99 uint32_t dict_len; // length of local dictionary
100 uint32_t null_count; // number of null values in this stripe's column
101 uint32_t skip_count; // number of non-null values to skip
102 uint32_t rowgroup_id; // row group position
103 uint8_t encoding_kind; // column encoding kind (orc::ColumnEncodingKind)
104 uint8_t type_kind; // column data type (orc::TypeKind)
105 uint8_t dtype_len; // data type length (for types that can be mapped to different sizes)
106 uint8_t decimal_scale; // number of fractional decimal digits for decimal type (bit 7 set if converting to float64)
107 int32_t ts_clock_rate; // output timestamp clock frequency (0=default, 1000=ms, 1000000000=ns)
108 };
