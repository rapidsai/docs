60 template <JoinType join_type,
61  typename multimap_type>
62 gdf_error estimate_join_output_size(device_table const & build_table,
63  device_table const & probe_table,
64  multimap_type hash_table,
65  gdf_size_type * join_output_size_estimate)
66 {
67  const gdf_size_type build_table_num_rows{build_table.num_rows()};
68  const gdf_size_type probe_table_num_rows{probe_table.num_rows()};
69 
70  // If the probe table is significantly larger (5x) than the build table,
71  // then we attempt to only use a subset of the probe table rows to compute an
72  // estimate of the join output size.
73  gdf_size_type probe_to_build_ratio{0};
74  if(build_table_num_rows > 0) {
75  probe_to_build_ratio = static_cast<gdf_size_type>(std::ceil(static_cast<float>(probe_table_num_rows)/build_table_num_rows));
76  }
77  else {
78  // If the build table is empty, we know exactly how large the output
79  // will be for the different types of joins and can return immediately
80  switch(join_type)
81  {
82  case JoinType::INNER_JOIN:
83  {
84  // Inner join with an empty table will have no output
85  *join_output_size_estimate = 0;
86  break;
87  }
88  case JoinType::LEFT_JOIN:
89  {
90  // Left join with an empty table will have an output of NULL rows
91  // equal to the number of rows in the probe table
92  *join_output_size_estimate = probe_table_num_rows;
93  break;
94  }
95  default:
96  return GDF_UNSUPPORTED_JOIN_TYPE;
97  }
98  return GDF_SUCCESS;
99  }
100 
101  gdf_size_type sample_probe_num_rows{probe_table_num_rows};
102  constexpr gdf_size_type MAX_RATIO{5};
103  if(probe_to_build_ratio > MAX_RATIO)
104  {
105  sample_probe_num_rows = build_table_num_rows;
106  }
107 
108  // Allocate storage for the counter used to get the size of the join output
109  gdf_size_type * d_size_estimate{nullptr};
110  gdf_size_type h_size_estimate{0};
111 
112  CUDA_TRY(cudaMallocHost(&d_size_estimate, sizeof(size_t)));
113  *d_size_estimate = 0;
114 
115  CUDA_TRY( cudaGetLastError() );
116 
117  constexpr int block_size {DEFAULT_CUDA_BLOCK_SIZE};
118  int numBlocks {-1};
119 
120  CUDA_TRY(cudaOccupancyMaxActiveBlocksPerMultiprocessor(
121  &numBlocks, compute_join_output_size<join_type, multimap_type, block_size, DEFAULT_CUDA_CACHE_SIZE>,
122  block_size, 0
123  ));
124 
125  int dev_id {-1};
126  CUDA_TRY(cudaGetDevice(&dev_id));
127 
128  int num_sms {-1};
129  CUDA_TRY(cudaDeviceGetAttribute(&num_sms, cudaDevAttrMultiProcessorCount, dev_id));
130 
131  // Continue probing with a subset of the probe table until either:
132  // a non-zero output size estimate is found OR
133  // all of the rows in the probe table have been sampled
134  do{
135 
136  sample_probe_num_rows = std::min(sample_probe_num_rows, probe_table_num_rows);
137 
138  *d_size_estimate = 0;
139 
140  // Probe the hash table without actually building the output to simply
141  // find what the size of the output will be.
142  compute_join_output_size<join_type,
143  multimap_type,
144  block_size,
145  DEFAULT_CUDA_CACHE_SIZE>
146  <<<numBlocks * num_sms, block_size>>>(hash_table,
147  build_table,
148  probe_table,
149  sample_probe_num_rows,
150  d_size_estimate);
151 
152  // Device sync is required to ensure d_size_estimate is updated
153  CUDA_TRY( cudaDeviceSynchronize() );
154 
155  // Only in case subset of probe table is chosen,
156  // increase the estimated output size by a factor of the ratio between the
157  // probe and build tables
158  if(sample_probe_num_rows < probe_table_num_rows) {
159  h_size_estimate = *d_size_estimate * probe_to_build_ratio;
160  } else {
161  h_size_estimate = *d_size_estimate;
162  }
163 
164  // If the size estimate is non-zero, then we have a valid estimate and can break
165  // If sample_probe_num_rows >= probe_table_num_rows, then we've sampled the entire
166  // probe table, in which case the estimate is exact and we can break
167  if((h_size_estimate > 0)
168  || (sample_probe_num_rows >= probe_table_num_rows))
169  {
170  break;
171  }
172 
173  // If the size estimate is zero, then double the number of sampled rows in the probe
174  // table. Reduce the ratio of the number of probe rows sampled to the
175  // number of rows in the build table by the same factor
176  if(0 == h_size_estimate)
177  {
178  constexpr gdf_size_type GROW_RATIO{2};
179  sample_probe_num_rows *= GROW_RATIO;
180  probe_to_build_ratio = static_cast<gdf_size_type>(std::ceil(static_cast<float>(probe_to_build_ratio)/GROW_RATIO));
181  }
182 
183  } while(true);
184 
185  CUDA_TRY( cudaFreeHost(d_size_estimate) );
186 
187  *join_output_size_estimate = h_size_estimate;
188 
189  return GDF_SUCCESS;
190 }

217  bool flip_results = false)
218 {
219  gdf_error gdf_error_code{GDF_SUCCESS};
220 
221  gdf_column_view(output_l, nullptr, nullptr, 0, N_GDF_TYPES);
222  gdf_column_view(output_r, nullptr, nullptr, 0, N_GDF_TYPES);
223 
224  // The LEGACY allocator allocates the hash table array with normal cudaMalloc,
225  // the non-legacy allocator uses managed memory
226 #ifdef HT_LEGACY_ALLOCATOR
227  using multimap_type = concurrent_unordered_multimap<hash_value_type,
228  output_index_type,
229  size_t,
230  std::numeric_limits<hash_value_type>::max(),
231  std::numeric_limits<output_index_type>::max(),
232  default_hash<hash_value_type>,
233  equal_to<hash_value_type>,
234  legacy_allocator< thrust::pair<hash_value_type, output_index_type> > >;
235 #else
236  using multimap_type = concurrent_unordered_multimap<hash_value_type,
237  output_index_type,
238  size_t,
239  std::numeric_limits<hash_value_type>::max(),
240  std::numeric_limits<output_index_type>::max()>;
241 #endif
242  //If FULL_JOIN is selected then we process as LEFT_JOIN till we need to take care of unmatched indices
243  constexpr JoinType base_join_type = (join_type == JoinType::FULL_JOIN)? JoinType::LEFT_JOIN : join_type;
244 
245  // Hash table will be built on the right table
246  auto build_table = device_table::create(right_table);
247  const gdf_size_type build_table_num_rows{build_table->num_rows()};
248 
249  // Probe with the left table
250  auto probe_table = device_table::create(left_table);
251  const gdf_size_type probe_table_num_rows{probe_table->num_rows()};
252 
253  // Hash table size must be at least 1 in order to have a valid allocation.
254  // Even if the hash table will be empty, it still must be allocated for the
255  // probing phase in the event of an outer join
256  size_t const hash_table_size =
257  std::max(compute_hash_table_size(build_table_num_rows), size_t{1});
258 
259  auto hash_table = multimap_type::create(hash_table_size);
260 
261  // FIXME: use GPU device id from the context?
262  // (although should be possible once we move to Arrow)
263  hash_table->prefetch(0);
264 
265  CUDA_TRY( cudaDeviceSynchronize() );
266 
267  // Allocate a gdf_error for the device to hold error code returned from
268  // the build kernel and intialize with GDF_SUCCESS
269  // Use Page Locked memory to avoid overhead of memcpys
270  gdf_error * d_gdf_error_code{nullptr};
271  CUDA_TRY( cudaMallocHost(&d_gdf_error_code, sizeof(gdf_error)) );
272  *d_gdf_error_code = GDF_SUCCESS;
273 
274  constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
275 
276  // build the hash table
277  if(build_table_num_rows > 0)
278  {
279  const gdf_size_type build_grid_size{(build_table_num_rows + block_size - 1)/block_size};
280  build_hash_table<<<build_grid_size, block_size>>>(*hash_table,
281  *build_table,
282  build_table_num_rows,
283  d_gdf_error_code);
284 
285  // Device synch is required to ensure d_gdf_error_code
286  // has been written
287  CUDA_TRY( cudaDeviceSynchronize() );
288  }
289 
290  // Check error code from the kernel
291  gdf_error_code = *d_gdf_error_code;
292  if(GDF_SUCCESS != gdf_error_code){
293  return gdf_error_code;
294  }
295 
296 
297  gdf_size_type estimated_join_output_size{0};
298  gdf_error_code = estimate_join_output_size<base_join_type, multimap_type>(
299  *build_table, *probe_table, *hash_table, &estimated_join_output_size);
300 
301  if(GDF_SUCCESS != gdf_error_code){
302  return gdf_error_code;
303  }
304 
305  // If the estimated output size is zero, return immediately
306  if(0 == estimated_join_output_size){
307  return GDF_SUCCESS;
308  }
309 
310  // Because we are approximating the number of joined elements, our approximation
311  // might be incorrect and we might have underestimated the number of joined elements.
312  // As such we will need to de-allocate memory and re-allocate memory to ensure
313  // that the final output is correct.
314  gdf_size_type h_actual_found{0};
315  output_index_type *output_l_ptr{nullptr};
316  output_index_type *output_r_ptr{nullptr};
317  bool cont = true;
318 
319  // Allocate device global counter used by threads to determine output write location
320  gdf_size_type *d_global_write_index{nullptr};
321  RMM_TRY( RMM_ALLOC((void**)&d_global_write_index, sizeof(gdf_size_type), 0) ); // TODO non-default stream?
322 
323  // Because we only have an estimate of the output size, we may need to probe the
324  // hash table multiple times until we've found an output buffer size that is large enough
325  // to hold the output
326  while(cont)
327  {
328  output_l_ptr = nullptr;
329  output_r_ptr = nullptr;
330 
331  // Allocate temporary device buffer for join output
332  RMM_TRY( RMM_ALLOC((void**)&output_l_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
333  RMM_TRY( RMM_ALLOC((void**)&output_r_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
334  CUDA_TRY( cudaMemsetAsync(d_global_write_index, 0, sizeof(gdf_size_type), 0) );
335 
336  const gdf_size_type probe_grid_size{(probe_table_num_rows + block_size -1)/block_size};
337 
338  // Do the probe of the hash table with the probe table and generate the output for the join
339  probe_hash_table<base_join_type,
340  multimap_type,
341  hash_value_type,
342  output_index_type,
343  block_size,
344  DEFAULT_CUDA_CACHE_SIZE>
345  <<<probe_grid_size, block_size>>> (*hash_table,
346  *build_table,
347  *probe_table,
348  probe_table->num_rows(),
349  output_l_ptr,
350  output_r_ptr,
351  d_global_write_index,
352  estimated_join_output_size,
353  flip_results);
354 
355  CUDA_TRY( cudaGetLastError() );
356 
357  CUDA_TRY( cudaMemcpy(&h_actual_found, d_global_write_index, sizeof(gdf_size_type), cudaMemcpyDeviceToHost));
358 
359  // The estimate was too small. Double the estimate and try again
360  if(estimated_join_output_size < h_actual_found){
361  cont = true;
362  estimated_join_output_size *= 2;
363  // Free the old buffers to prevent a memory leak on the new allocation
364  RMM_TRY( RMM_FREE(output_l_ptr, 0) );
365  RMM_TRY( RMM_FREE(output_r_ptr, 0) );
366  }
367  else
368  {
369  cont = false;
370  }
371  }
372 
373  // free memory used for the counters
374  RMM_TRY( RMM_FREE(d_global_write_index, 0) );
375 
376  cudaStream_t stream = 0;
377  if (join_type == JoinType::FULL_JOIN) {
378  append_full_join_indices(
379  &output_l_ptr, &output_r_ptr,
380  estimated_join_output_size,
381  h_actual_found, build_table_num_rows,
382  stream);
383  }
384 
385  // If the estimated join output size was larger than the actual output size,
386  // then the buffers are larger than necessary. Allocate buffers of the actual
387  // output size and copy the results to the buffers of the correct size
388  // FIXME Is this really necessary? It's probably okay to have the buffers be oversized
389  // and avoid the extra allocation/memcopy
390  if (estimated_join_output_size > h_actual_found) {
391  output_index_type *copy_output_l_ptr{nullptr};
392  output_index_type *copy_output_r_ptr{nullptr};
393  RMM_TRY( RMM_ALLOC((void**)&copy_output_l_ptr, h_actual_found*sizeof(output_index_type), 0) ); // TODO non-default stream?
394  RMM_TRY( RMM_ALLOC((void**)&copy_output_r_ptr, h_actual_found*sizeof(output_index_type), 0) );
395  CUDA_TRY( cudaMemcpy(copy_output_l_ptr, output_l_ptr, h_actual_found*sizeof(output_index_type), cudaMemcpyDeviceToDevice) );
396  CUDA_TRY( cudaMemcpy(copy_output_r_ptr, output_r_ptr, h_actual_found*sizeof(output_index_type), cudaMemcpyDeviceToDevice) );
397  RMM_TRY( RMM_FREE(output_l_ptr, 0) );
398  RMM_TRY( RMM_FREE(output_r_ptr, 0) );
399  output_l_ptr = copy_output_l_ptr;
400  output_r_ptr = copy_output_r_ptr;
401  }
402 
403  // Free the device error code
404  CUDA_TRY( cudaFreeHost(d_gdf_error_code) );
405 
406  // Deduce the type of the output gdf_columns
407  gdf_dtype dtype;
408  switch(sizeof(output_index_type))
409  {
410  case 1 : dtype = GDF_INT8; break;
411  case 2 : dtype = GDF_INT16; break;
412  case 4 : dtype = GDF_INT32; break;
413  case 8 : dtype = GDF_INT64; break;
414  }
415  gdf_column_view(output_l, output_l_ptr, nullptr, h_actual_found, dtype);
416  gdf_column_view(output_r, output_r_ptr, nullptr, h_actual_found, dtype);
417 
418  return gdf_error_code;
