189 // This pragma disables a compiler warning that complains about the valid usage
190 // of calling a __host__ functor from this function which is __host__ __device__
191 #pragma nv_exec_check_disable
192 template < template <cudf::type_ id > typename IdTypeMap = id_to_type_impl,
193 typename Functor, typename ... Ts>
194 CUDA_HOST_DEVICE_CALLABLE constexpr decltype( auto ) type_dispatcher(
195 cudf :: data_type dtype, Functor f, Ts&&... args) {
196 switch (dtype.id()) {
197 case INT8 :
198 return f.template operator()< typename IdTypeMap<INT8>::type>(
199 std::forward<Ts>(args)...);
200 case INT16 :
201 return f.template operator()< typename IdTypeMap<INT16>::type>(
202 std::forward<Ts>(args)...);
203 case INT32 :
204 return f.template operator()< typename IdTypeMap<INT32>::type>(
205 std::forward<Ts>(args)...);
206 case INT64 :
207 return f.template operator()< typename IdTypeMap<INT64>::type>(
208 std::forward<Ts>(args)...);
209 case FLOAT32 :
210 return f.template operator()< typename IdTypeMap<FLOAT32>::type>(
211 std::forward<Ts>(args)...);
212 case FLOAT64 :
213 return f.template operator()< typename IdTypeMap<FLOAT64>::type>(
214 std::forward<Ts>(args)...);
215 case STRING :
216 return f.template operator()< typename IdTypeMap<STRING>::type>(
217 std::forward<Ts>(args)...);
218 default : {
219 #ifndef __CUDA_ARCH__
220 CUDF_FAIL( "Unsupported type_id." );
221 #else
222 release_assert( false && "Unsuported type_id." );

229 // return
230 using return_type =

232 return return_type();
233 #endif
