83 inline std::vector<gdf_dtype> target_dtypes(
84 std::vector<gdf_dtype> const & source_dtypes,
85 std::vector<operators> const & ops) {
86 std::vector<gdf_dtype> output_dtypes(source_dtypes.size());
87
88 std::transform(
89 source_dtypes.begin(), source_dtypes.end(), ops.begin(),
90 output_dtypes.begin(), [](gdf_dtype source_dtype, operators op) {
91 gdf_dtype t =
92 cudf::type_dispatcher(source_dtype, target_type_mapper{}, op);
93 CUDF_EXPECTS(
94 t != GDF_invalid,
95 "Invalid combination of input type and aggregation operation." );
96 return t;
97 });
98
99 return output_dtypes;
100 }

106 struct identity_initializer {
107 template < typename T>
108 T get_identity(operators op) {
109 switch (op) {
110 case SUM:
111 return corresponding_functor_t<SUM>::identity<T>();
112 case MIN:
113 return corresponding_functor_t<MIN>::identity<T>();
114 case MAX:
115 return corresponding_functor_t<MAX>::identity<T>();
116 case COUNT:
117 return corresponding_functor_t<COUNT>::identity<T>();
118 default :
119 CUDF_FAIL( "Invalid aggregation operation." );
120 }
121 }
122
123 template < typename T>
124 void operator()( gdf_column const & col, operators op,
125 cudaStream_t stream = 0) {
126 T* typed_data = static_cast<T*>(col. data );
127 thrust::fill(rmm::exec_policy(stream)->on(stream), typed_data,
128 typed_data + col. size , get_identity<T>(op));
129
130 // For COUNT operator, initialize column's bitmask to be all valid
131 if (( nullptr != col. valid ) and (COUNT == op)) {
132 CUDA_TRY(cudaMemsetAsync(
133 col. valid , 0xff,
134 sizeof (gdf_valid_type) * gdf_valid_allocation_size(col. size ),
135 stream));
136 }
137 }
138 };

179 static void update_nvcategories(table const & input_keys, table& output_keys,
180 table const & input_values, table& output_values) {
181 gdf_error update_err = nvcategory_gather_table(input_keys, output_keys);
182 CUDF_EXPECTS(update_err == GDF_SUCCESS, "nvcategory_gather_table error for keys" );
183
184 // Filter out columns from (input/output)_values where the output datatype is
185 // not GDF_STRING_CATEGORY. This is possible in aggregations like `count`.
186 std::vector<gdf_column *> string_input_value_cols;
187 std::vector<gdf_column *> string_output_value_cols;
188
189 for (gdf_size_type i = 0; i < input_values.num_columns(); i++) {
190 if (output_values.get_column(i)->dtype == GDF_STRING_CATEGORY) {
191 string_input_value_cols.push_back(
192 const_cast<gdf_column*>(input_values.get_column(i)));
193 string_output_value_cols.push_back(
194 const_cast<gdf_column*>(output_values.get_column(i)));
195 }
196 }
197
198 cudf::table string_input_values(string_input_value_cols);
199 cudf::table string_output_values(string_output_value_cols);
200
201 update_err = nvcategory_gather_table(string_input_values, string_output_values);
202 CUDF_EXPECTS(update_err == GDF_SUCCESS, "nvcategory_gather_table error for values" );
203 }
