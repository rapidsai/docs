30 struct CompressedStreamInfo {
31 CompressedStreamInfo () = default ;
32 explicit constexpr CompressedStreamInfo (uint8_t *compressed_data_,
33 size_t compressed_size_)
34 : compressed_data(compressed_data_),
35 compressed_data_size(compressed_size_),
36 uncompressed_data( nullptr ),
37 decctl( nullptr ),
38 decstatus( nullptr ),
39 max_compressed_blocks(0),
40 num_compressed_blocks(0),
41 max_uncompressed_size(0) {}
42 uint8_t *compressed_data; // [in] base ptr to compressed stream data
43 uint8_t *uncompressed_data; // [in] base ptr to uncompressed stream data or NULL if not known yet
44 size_t compressed_data_size; // [in] compressed data size for this stream
45 gpu_inflate_input_s *decctl; // [in] base ptr to decompression structure to be filled
46 gpu_inflate_status_s *decstatus; // [in] results of decompression
47 uint32_t max_compressed_blocks; // [in] number of entries in decctl
48 uint32_t num_compressed_blocks; // [out] total number of compressed blocks in this stream
49 uint64_t max_uncompressed_size; // [out] maximum uncompressed data size
50 };
