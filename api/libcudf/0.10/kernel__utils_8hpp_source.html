140 template < bool source_has_nulls>
141 struct elementwise_aggregator {
142 template < typename SourceType>
143 __device__ inline void operator()( gdf_column const & target,
144 gdf_size_type target_index,
145 gdf_column const & source,
146 gdf_size_type source_index, operators op) {
147 switch (op) {
148 case MIN: {
149 update_target_element<SourceType, MIN, source_has_nulls> {}(
150 target, target_index, source, source_index);
151 break ;
152 }
153 case MAX: {
154 update_target_element<SourceType, MAX, source_has_nulls> {}(
155 target, target_index, source, source_index);
156 break ;
157 }
158 case SUM: {
159 update_target_element<SourceType, SUM, source_has_nulls> {}(
160 target, target_index, source, source_index);
161 break ;
162 }
163 case COUNT: {
164 update_target_element<SourceType, COUNT, source_has_nulls> {}(
165 target, target_index, source, source_index);
166 }
167 default :
168 return ;
169 }
170 }

201 template < bool values_have_nulls = true>
202 __device__ inline void aggregate_row( device_table const & target,
203 gdf_size_type target_index,
204 device_table const & source,
205 gdf_size_type source_index,
206 operators* ops) {
207 using namespace bit_mask;
208 for (gdf_size_type i = 0; i < target.num_columns(); ++i) {
209 bit_mask_t const * const __restrict__ source_mask{
210 reinterpret_cast<bit_mask_t const*>(source.get_column(i)-> valid )};
211
212 if (values_have_nulls and nullptr != source_mask and
213 not is_valid(source_mask, source_index)) {
214 continue ;
215 }
216
217 cudf::type_dispatcher(source.get_column(i)-> dtype ,
218 elementwise_aggregator<values_have_nulls>{},
219 *target.get_column(i), target_index,
220 *source.get_column(i), source_index, ops[i]);
221 }
222 }
