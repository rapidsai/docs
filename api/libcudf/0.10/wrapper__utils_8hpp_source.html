66  void allocate() {
67  const auto num_rows = std::max(col->size, 1);
68  const auto valid_size = gdf_valid_allocation_size(num_rows);
69 
70  // For strings, just store the <ptr, length>. Eventually, the column's data
71  // ptr is replaced with an `NvString` instance created from these pairs.
72  // NvStrings does not use the valid mask so it expects invalid rows to be
73  // <nullptr, 0> initialized.
74  if (col->dtype == GDF_STRING) {
75  RMM_TRY(RMM_ALLOC(&col->data, num_rows * sizeof(str_pair), 0));
76  CUDA_TRY(cudaMemsetAsync(col->data, 0, num_rows * sizeof(str_pair)));
77  } else {
78  RMM_TRY(RMM_ALLOC(&col->data, num_rows * cudf::byte_width(*col), 0));
79  }
80  RMM_TRY(RMM_ALLOC(&col->valid, valid_size, 0));
81  CUDA_TRY(cudaMemsetAsync(col->valid, 0, valid_size));
82  }
83 
84  void finalize() {
85  // Create and initialize an `NvStrings` instance from <ptr, length> data.
86  // The container copies the string data to its internal memory so the source
87  // memory must not be released prior to calling this method.
88  if (col->dtype == GDF_STRING) {
89  auto str_list = static_cast<str_pair *>(col->data);
90  str_ptr str_data(NVStrings::create_from_index(str_list, col->size),
91  &NVStrings::destroy);
92  CUDF_EXPECTS(str_data != nullptr, "Cannot create `NvStrings` instance");
93  RMM_FREE(std::exchange(col->data, str_data.release()), 0);
94  }
95  }

97  gdf_column *operator->() const noexcept { return col; }
98  gdf_column *get() const noexcept { return col; }
99  gdf_column *release() noexcept {
100  auto temp = col;
101  col = nullptr;
102  return temp;
103  }
104 
105  private:
106  gdf_column *col = nullptr;
107 };

139  bool insert(const T &data) {
140  if (num_elements < max_elements) {
141  h_data[num_elements] = data;
142  num_elements++;
143  return true;
144  }
145  return false;
146  }
147 
148  size_t max_size() const noexcept { return max_elements; }
149  size_t size() const noexcept { return num_elements; }
150  size_t memory_size() const noexcept { return sizeof(T) * num_elements; }
151 
152  T &operator[](size_t i) const { return h_data[i]; }
153  T *host_ptr(size_t offset = 0) const { return h_data + offset; }

155

171 class device_buffer {
172  T* d_data_ = nullptr;
173  size_t count_ = 0;
174  cudaStream_t stream_ = 0;
175 
176 public:
177  device_buffer() noexcept = default;
178  device_buffer(size_t cnt, cudaStream_t stream = 0):
179  stream_(stream){
180  resize(cnt);
181  }
182 
183  T* data() const noexcept {return d_data_;}
184  size_t size() const noexcept {return count_;}
185  bool empty() const noexcept {return count_ == 0;}
186 
187  void resize(size_t cnt) {
188  if (cnt == count_) {
189  return;
190  }
191  // new size is zero, free the buffer if not null
192  if(cnt == 0 && d_data_ != nullptr) {
193  RMM_FREE(d_data_, stream_);
194  d_data_ = nullptr;
195  count_ = cnt;
196  return;
197  }
198 
199  T* new_ptr = nullptr;
200  const auto error = RMM_ALLOC(&new_ptr, cnt*sizeof(T), stream_);
201  if(error != RMM_SUCCESS) {
202  cudf::detail::throw_cuda_error(cudaErrorMemoryAllocation, __FILE__, __LINE__);
203  }
204  // Copy to the new buffer, if some memory was already allocated
205  if (count_ != 0) {
206  const size_t copy_bytes = std::min(cnt, count_)*sizeof(T);
207  CUDA_TRY(cudaMemcpyAsync(new_ptr, d_data_, copy_bytes, cudaMemcpyDefault, stream_));
208  RMM_FREE(d_data_, stream_);
209  }
210 
211  d_data_ = new_ptr;
212  count_ = cnt;
213  }
214 
215  device_buffer(device_buffer& ) = delete;
216  device_buffer(device_buffer&& rh) noexcept {
217  d_data_ = rh.d_data_;
218  count_ = rh.count_;
219  stream_ = rh.stream_;
220 
221  rh.d_data_ = nullptr;
222  rh.count_ = 0;
223  }
224 
225  device_buffer& operator=(device_buffer& ) = delete;
226  device_buffer& operator=(device_buffer&& rh) noexcept {
227  RMM_FREE(d_data_, stream_);
228 
229  d_data_ = rh.d_data_;
230  count_ = rh.count_;
231  stream_ = rh.stream_;
232 
233  rh.d_data_ = nullptr;
234  rh.count_ = 0;

237  }
