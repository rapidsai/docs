32 template <typename S>
33 inline S round_up_safe(S number_to_round, S modulus) {
34  auto remainder = number_to_round % modulus;
35  if (remainder == 0) { return number_to_round; }
36  auto rounded_up = number_to_round - remainder + modulus;
37  if (rounded_up < number_to_round) {
38  throw std::invalid_argument("Attempt to round up beyond the type's maximum value");
39  }
40  return rounded_up;
41 }
