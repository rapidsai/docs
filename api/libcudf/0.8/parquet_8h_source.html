180 static inline int CountLeadingZeros32(uint32_t value) {
181 #if defined(__clang__) || defined(__GNUC__)
182 if (value == 0) return 32;
183 return static_cast< int > (__builtin_clz(value));
184 #elif defined(_MSC_VER)
185 unsigned long index;
186 return (_BitScanReverse(&index, static_cast<unsigned long>(value)))
187 ? 31 - static_cast<int>(index)
188 : 32;
189 #else
190 int bitpos = 0;
191 while (value != 0) {
192 value >>= 1;
193 ++bitpos;
194 }
195 return 32 - bitpos;
196 #endif
197 }
