136 template <typename aggregation_type, typename aggregation_operation>
137 gdf_error GroupbyHash(cudf::table const& input_keys,
138  const aggregation_type* const in_aggregation_column,
139  cudf::table& output_keys,
140  aggregation_type* out_aggregation_column,
141  gdf_size_type* out_size,
142  aggregation_operation aggregation_op,
143  bool sort_result = false) {
144  const gdf_size_type input_num_rows = input_keys.num_rows();
145 
146  // The map will store (row index, aggregation value)
147  // Where row index is the row number of the first row to be successfully inserted
148  // for a given unique row
149  using map_type = concurrent_unordered_map<gdf_size_type, aggregation_type>;
150 
151  // The hash table occupancy and the input size determines the size of the hash table
152  const size_t hash_table_size{ compute_hash_table_size(input_num_rows) };
153 
154  // Initialize the hash table with the aggregation operation functor's identity value
155  std::unique_ptr<map_type> the_map(
156  new map_type(hash_table_size, aggregation_operation::IDENTITY));
157 
158  const dim3 build_grid_size ((input_num_rows + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
159  const dim3 block_size (THREAD_BLOCK_SIZE, 1, 1);
160 
161  CUDA_TRY(cudaGetLastError());
162 
163 
164  auto d_input_keys = device_table::create(input_keys);
165 
166  // Inserts (i, aggregation_column[i]) as a key-value pair into the
167  // hash table. When a given key already exists in the table, the aggregation operation
168  // is computed between the new and existing value, and the result is stored back.
169  build_aggregation_table<<<build_grid_size, block_size>>>(the_map.get(),
170  *d_input_keys,
171  in_aggregation_column,
172  input_num_rows,
173  aggregation_op,
174  row_comparator<map_type>(*the_map, *d_input_keys, *d_input_keys));
175  CUDA_TRY(cudaGetLastError());
176 
177 
178  auto d_output_keys = device_table::create(output_keys);
179 
180  // Used by threads to coordinate where to write their results
181  gdf_size_type * global_write_index{nullptr};
182  RMM_TRY(RMM_ALLOC((void**)&global_write_index, sizeof(gdf_size_type), 0)); // TODO: non-default stream?
183  CUDA_TRY(cudaMemset(global_write_index, 0, sizeof(gdf_size_type)));
184  const dim3 extract_grid_size ((hash_table_size + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
185 
186  // Extracts every non-empty key and value into separate contiguous arrays,
187  // which provides the result of the groupby operation
188  extract_groupby_result<<<extract_grid_size, block_size>>>(the_map.get(),
189  hash_table_size,
190  *d_output_keys,
191  *d_input_keys,
192  out_aggregation_column,
193  global_write_index);
194 
195  CUDA_TRY(cudaGetLastError());
196 
197  // At the end of the extraction kernel, the global write index will be equal to
198  // the size of the output. Update the output size.
199  CUDA_TRY( cudaMemcpy(out_size, global_write_index, sizeof(gdf_size_type), cudaMemcpyDeviceToHost) );
200  RMM_TRY( RMM_FREE(global_write_index, 0) );
201 
202  // Update the size of the output key columns
203  std::for_each(output_keys.begin(), output_keys.end(),
204  [out_size](gdf_column* col) { col->size = *out_size; });
205 
206 
207  // Optionally sort the groupby/aggregation result columns
208  if(true == sort_result) {
209 
210  rmm::device_vector<gdf_index_type> sorted_indices(*out_size);
211  thrust::sequence(rmm::exec_policy()->on(0), sorted_indices.begin(), sorted_indices.end());
212 
213  gdf_column sorted_indices_col{};
214  gdf_error status = gdf_column_view(&sorted_indices_col, sorted_indices.data().get(),
215  nullptr, *out_size, GDF_INT32);
216  if (status != GDF_SUCCESS)
217  return status;
218 
219  gdf_context ctxt;
220  ctxt.flag_null_sort_behavior = GDF_NULL_AS_LARGEST;
221  status = gdf_order_by(output_keys.begin(), nullptr,
222  output_keys.num_columns(), &sorted_indices_col, &ctxt);
223 
224  if (status != GDF_SUCCESS)
225  return status;
226 
227  // Reorder table according to indices from order_by
228  cudf::detail::gather(&output_keys, sorted_indices.data().get(),
229  &output_keys);
230 
231  rmm::device_vector<aggregation_type> temporary_aggregation_buffer(*out_size);
232  thrust::gather(rmm::exec_policy()->on(0),
233  sorted_indices.begin(), sorted_indices.end(),
234  out_aggregation_column,
235  temporary_aggregation_buffer.begin());
236  thrust::copy(rmm::exec_policy()->on(0),
237  temporary_aggregation_buffer.begin(),
238  temporary_aggregation_buffer.end(), out_aggregation_column);
239  }
240 
241  return GDF_SUCCESS;
242 }
