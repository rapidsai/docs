403  explicit arrow_io_source(std::string_view arrow_uri)
404  {
405  const std::string uri_start_delimiter = "//";
406  const std::string uri_end_delimiter = "?";
407 
408  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
409  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
410  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
411  filesystem = result.ValueOrDie();
412 
413  // Parse the path from the URI
414  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
415  ? 0
416  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
417  size_t end = arrow_uri.find(uri_end_delimiter) - start;
418  std::string_view path = arrow_uri.substr(start, end);
419 
420  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
421  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
422  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
423  arrow_file = in_stream.ValueOrDie();
424  }
