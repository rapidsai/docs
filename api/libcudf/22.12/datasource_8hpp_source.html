396  explicit arrow_io_source(std::string_view arrow_uri)
397  {
398  const std::string uri_start_delimiter = "//";
399  const std::string uri_end_delimiter = "?";
400 
401  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
402  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
403  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
404  filesystem = result.ValueOrDie();
405 
406  // Parse the path from the URI
407  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
408  ? 0
409  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
410  size_t end = arrow_uri.find(uri_end_delimiter) - start;
411  std::string_view path = arrow_uri.substr(start, end);
412 
413  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
414  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
415  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
416  arrow_file = in_stream.ValueOrDie();
417  }
