137 inline void throw_cuda_error(cudaError_t error, const char* file, unsigned int line)
138 {
139  // Calls cudaGetLastError to clear the error status. It is nearly certain that a fatal error
140  // occurred if it still returns the same error after a cleanup.
141  cudaGetLastError();
142  auto const last = cudaFree(0);
143  auto const msg = std::string{"CUDA error encountered at: " + std::string{file} + ":" +
144  std::to_string(line) + ": " + std::to_string(error) + " " +
145  cudaGetErrorName(error) + " " + cudaGetErrorString(error)};
146  // Call cudaDeviceSynchronize to ensure `last` did not result from an asynchronous error.
147  // between two calls.
148  if (error == last && last == cudaDeviceSynchronize()) {
149  throw fatal_cuda_error{"Fatal " + msg, error};
150  } else {
151  throw cuda_error{msg, error};
152  }
153 }
154 // @endcond
