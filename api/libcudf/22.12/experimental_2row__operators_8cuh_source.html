216 // The has_nested_columns template parameter of the device_row_comparator is
217 // necessary to help the compiler optimize our code. Without it, the list and
218 // struct view specializations are present in the code paths used for primitive
219 // types, and the compiler fails to inline this nearly as well resulting in a
220 // significant performance drop. As a result, there is some minor tension in
221 // the current design between the presence of this parameter and the way that
222 // the Dremel data is passed around, first as a
223 // std::optional<device_span<dremel_device_view>> in the
224 // preprocessed_table/device_row_comparator (which is always valid when
225 // has_nested_columns and is otherwise invalid) that is then unpacked to a
226 // thrust::optional<dremel_device_view> at the element_comparator level (which
227 // is always valid for a list column and otherwise invalid). We cannot use an
228 // additional template parameter for the element_comparator on a per-column
229 // basis because we cannot conditionally define dremel_device_view member
230 // variables without jumping through extra hoops with inheritance, so the
231 // thrust::optional<dremel_device_view> member must be an optional rather than
232 // a raw dremel_device_view.

346  __device__ cuda::std::pair<weak_ordering, int> operator()(
347  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept

372  template <typename Element,
373  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::struct_view>)>
374  __device__ cuda::std::pair<weak_ordering, int> operator()(
375  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
376  {
377  column_device_view lcol = _lhs;
378  column_device_view rcol = _rhs;
379  int depth = _depth;
380  while (lcol.type().id() == type_id::STRUCT) {
381  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
382  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
383 
384  if (lhs_is_null or rhs_is_null) { // at least one is null
385  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
386  return cuda::std::pair(state, depth);
387  }
388 
389  if (lcol.num_child_columns() == 0) {
390  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
391  }
392 
393  // Non-empty structs have been modified to only have 1 child when using this.
394  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
395  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
396  ++depth;
397  }
398 
399  return cudf::type_dispatcher<dispatch_void_if_nested>(
400  lcol.type(),
401  element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth, _comparator},
402  lhs_element_index,
403  rhs_element_index);
404  }

406  template <typename Element,
407  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::list_view>)>
408  __device__ cuda::std::pair<weak_ordering, int> operator()(size_type lhs_element_index,
409  size_type rhs_element_index)
410  {
411  // These are all the values from the Dremel encoding.
412  auto const l_max_def_level = _l_dremel_device_view->max_def_level;
413  auto const l_def_levels = _l_dremel_device_view->def_levels;
414  auto const r_def_levels = _r_dremel_device_view->def_levels;
415  auto const l_rep_levels = _l_dremel_device_view->rep_levels;
416  auto const r_rep_levels = _r_dremel_device_view->rep_levels;
417 
418  // Traverse the nested list hierarchy to get a column device view
419  // pointing to the underlying child data.
420  column_device_view lcol = _lhs.slice(lhs_element_index, 1);
421  column_device_view rcol = _rhs.slice(rhs_element_index, 1);
422  while (lcol.type().id() == type_id::LIST) {
423  lcol = detail::lists_column_device_view(lcol).get_sliced_child();
424  rcol = detail::lists_column_device_view(rcol).get_sliced_child();
425  }
426 
427  // These start and end values indicate the start and end points of all
428  // the elements of the lists in the current list element
429  // (`[lhs|rhs]_element_index`) that we are comparing.
430  auto const l_offsets = _l_dremel_device_view->offsets;
431  auto const r_offsets = _r_dremel_device_view->offsets;
432  auto l_start = l_offsets[lhs_element_index];
433  auto l_end = l_offsets[lhs_element_index + 1];
434  auto r_start = r_offsets[rhs_element_index];
435  auto r_end = r_offsets[rhs_element_index + 1];
436 
437  // This comparator will be used to compare leaf (non-nested) data types.
438  auto comparator =
439  element_comparator{_check_nulls, lcol, rcol, _null_precedence, _depth, _comparator};
440 
441  // Loop over each element in the encoding. Note that this includes nulls
442  // and empty lists, so not every index corresponds to an actual element
443  // in the child column. The element_index is used to keep track of the current
444  // child element that we're actually comparing.
445  weak_ordering state{weak_ordering::EQUIVALENT};
446  for (int l_dremel_index = l_start, r_dremel_index = r_start, element_index = 0;
447  l_dremel_index < l_end and r_dremel_index < r_end;
448  ++l_dremel_index, ++r_dremel_index) {
449  // First early exit: the definition levels do not match.
450  if (l_def_levels[l_dremel_index] != r_def_levels[r_dremel_index]) {
451  state = (l_def_levels[l_dremel_index] < r_def_levels[r_dremel_index])
452  ? weak_ordering::LESS
453  : weak_ordering::GREATER;
454  return cuda::std::pair(state, _depth);
455  }
456 
457  // Second early exit: the repetition levels do not match.
458  if (l_rep_levels[l_dremel_index] != r_rep_levels[r_dremel_index]) {
459  state = (l_rep_levels[l_dremel_index] < r_rep_levels[r_dremel_index])
460  ? weak_ordering::LESS
461  : weak_ordering::GREATER;
462  return cuda::std::pair(state, _depth);
463  }
464 
465  // Third early exit: This case has two branches.
466  // 1) If we are at the maximum definition level, then we actually have
467  // an underlying element to compare, not just an empty list or a
468  // null. Therefore, we access the element_index element of each list
469  // and compare the values.
470  // 2) If we are one level below the maximum definition level and the
471  // column is nullable, the current element must be a null in the
472  // leaf data. In this case we ignore the null and skip to the next
473  // element.
474  if (l_def_levels[l_dremel_index] == l_max_def_level) {
475  int last_null_depth = _depth;
476  cuda::std::tie(state, last_null_depth) = cudf::type_dispatcher<dispatch_void_if_nested>(
477  lcol.type(), comparator, element_index, element_index);
478  if (state != weak_ordering::EQUIVALENT) { return cuda::std::pair(state, _depth); }
479  ++element_index;
480  } else if (lcol.nullable() and l_def_levels[l_dremel_index] == l_max_def_level - 1) {
481  ++element_index;
482  }
483  }
484 
485  // If we have reached this stage, we know that definition levels,
486  // repetition levels, and actual elements are identical in both list
487  // columns up to the `min(l_end - l_start, r_end - r_start)` element of
488  // the Dremel encoding. However, two lists can only compare equivalent if
489  // they are of the same length. Otherwise, the shorter of the two is less
490  // than the longer. This final check determines the appropriate resulting
491  // ordering by checking how many total elements each list is composed of.
492  return cuda::std::pair(detail::compare_elements(l_end - l_start, r_end - r_start), _depth);
493  }

516  __device__ constexpr weak_ordering operator()(size_type const lhs_index,
517  size_type const rhs_index) const noexcept
518  {
519  int last_null_depth = std::numeric_limits<int>::max();
520  size_type list_column_index{0};
521  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
522  int const depth = _depth.has_value() ? (*_depth)[i] : 0;
523  if (depth > last_null_depth) { continue; }
524 
525  bool const ascending =
526  _column_order.has_value() ? (*_column_order)[i] == order::ASCENDING : true;

529  _null_precedence.has_value() ? (*_null_precedence)[i] : null_order::BEFORE;

533  // here, otherwise the current code would be failing.
534  auto [l_dremel_i, r_dremel_i] = [&]() {
535  if (_lhs.column(i).type().id() == type_id::LIST) {
536  auto idx = list_column_index++;
537  return std::make_tuple(optional_dremel_view(_l_dremel[idx]),
538  optional_dremel_view(_r_dremel[idx]));
539  } else {
540  return std::make_tuple(optional_dremel_view{}, optional_dremel_view{});
541  }
542  }();
543  auto element_comp = element_comparator{_check_nulls,
544  _lhs.column(i),
545  _rhs.column(i),
546  null_precedence,
547  depth,
548  _comparator,
