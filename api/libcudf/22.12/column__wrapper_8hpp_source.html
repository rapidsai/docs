291 template <typename StringsIterator, typename ValidityIterator>
292 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
293 {
294  std::vector<char> chars{};
295  std::vector<cudf::size_type> offsets(1, 0);
296  for (auto str = begin; str < end; ++str) {
297  std::string tmp = (*v++) ? std::string(*str) : std::string{};
298  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
299  offsets.push_back(offsets.back() + tmp.length());
300  }
301  return std::pair(std::move(chars), std::move(offsets));
302 };

1530  std::vector<bool> const& v)
1531  {
1532  auto valids = cudf::detail::make_counting_transform_iterator(
1533  0, [&v](auto i) { return v.empty() ? true : v[i]; });
1534 
1535  // compute the expected hierarchy and depth
1536  auto const hierarchy_and_depth =
1537  std::accumulate(elements.begin(),
1538  elements.end(),
1539  std::pair<column_view, int32_t>{{}, -1},
1540  [](auto acc, lists_column_wrapper const& lcw) {
1541  return lcw.depth > acc.second ? std::pair(lcw.get_view(), lcw.depth) : acc;
1542  });
1543  column_view expected_hierarchy = hierarchy_and_depth.first;
1544  int32_t const expected_depth = hierarchy_and_depth.second;
1545 
1546  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
1547  auto [cols, stubs] = preprocess_columns(elements, expected_hierarchy, expected_depth);
1548 
1549  // generate offsets
1550  size_type count = 0;
1551  std::vector<size_type> offsetv;
1552  std::transform(cols.cbegin(),
1553  cols.cend(),
1554  valids,
1555  std::back_inserter(offsetv),
1556  [&](cudf::column_view const& col, bool valid) {
1557  // nulls are represented as a repeated offset
1558  size_type ret = count;
1559  if (valid) { count += col.size(); }
1560  return ret;
1561  });
1562  // add the final offset
1563  offsetv.push_back(count);
1564  auto offsets =
1565  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();
1566 
1567  // concatenate them together, skipping children that are null.
1568  std::vector<column_view> children;
1569  thrust::copy_if(
1570  std::cbegin(cols), std::cend(cols), valids, std::back_inserter(children), thrust::identity{});
1571 
1572  auto data = children.empty() ? cudf::empty_like(expected_hierarchy) : concatenate(children);
1573 
1574  // increment depth
1575  depth = expected_depth + 1;
1576 
1577  // construct the list column
1578  wrapped =
1579  make_lists_column(cols.size(),
1580  std::move(offsets),
1581  std::move(data),
1582  v.size() <= 0 ? 0 : cudf::UNKNOWN_NULL_COUNT,
1583  v.size() <= 0 ? rmm::device_buffer{}
1584  : cudf::test::detail::make_null_mask(v.begin(), v.end()));

1670  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1671  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1672  column_view& expected_hierarchy,
1673  int expected_depth)
1674  {
1675  std::vector<std::unique_ptr<column>> stubs;
1676  std::vector<column_view> cols;
1677 
1678  // preprocess the incoming lists.
1679  // - unwrap any "root" lists
1680  // - handle incomplete hierarchies
1681  std::transform(elements.begin(),
1682  elements.end(),
1683  std::back_inserter(cols),
1684  [&](lists_column_wrapper const& l) -> column_view {
1685  // depth mismatch. attempt to normalize the short column.
1686  // this function will also catch if this is a legitimately broken
1687  // set of input
1688  if (l.depth < expected_depth) {
1689  if (l.root) {
1690  // this exception distinguishes between the following two cases:
1691  //
1692  // { {{{1, 2, 3}}}, {} }
1693  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1694  // just a List<> which is an apparent mismatch. However, because row 1
1695  // is empty we will allow that to semantically mean
1696  // "a List<List<List<int>>> that's empty at the top level"
1697  //
1698  // { {{{1, 2, 3}}}, {4, 5, 6} }
1699  // In this case, row 1 is a a concrete List<int> with actual values.
1700  // There is no way to rectify the differences so we will treat it as a
1701  // true column mismatch.
1702  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1703  stubs.push_back(empty_like(expected_hierarchy));
1704  } else {
1705  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1706  }
1707  return *(stubs.back());
1708  }
1709  // the empty hierarchy case
1710  return l.get_view();
1711  });
1712 
1713  return {std::move(cols), std::move(stubs)};
1714  }
1715 
1716  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1717 
1718  int depth = 0;
1719  bool root = false;
1720 };
