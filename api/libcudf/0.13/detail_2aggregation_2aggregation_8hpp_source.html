303 #pragma nv_exec_check_disable
304 template <typename F, typename... Ts>
305 CUDA_HOST_DEVICE_CALLABLE decltype(auto) aggregation_dispatcher(
306  aggregation::Kind k, F&& f, Ts&&... args) {
307  switch (k) {
308  case aggregation::SUM:
309  return f.template operator()<aggregation::SUM>(std::forward<Ts>(args)...);
310  case aggregation::PRODUCT:
311  return f.template operator()<aggregation::PRODUCT>(std::forward<Ts>(args)...);
312  case aggregation::MIN:
313  return f.template operator()<aggregation::MIN>(std::forward<Ts>(args)...);
314  case aggregation::MAX:
315  return f.template operator()<aggregation::MAX>(std::forward<Ts>(args)...);
316  case aggregation::COUNT_VALID:
317  return f.template operator()<aggregation::COUNT_VALID>(
318  std::forward<Ts>(args)...);
319  case aggregation::COUNT_ALL:
320  return f.template operator()<aggregation::COUNT_ALL>(
321  std::forward<Ts>(args)...);
322  case aggregation::ANY:
323  return f.template operator()<aggregation::ANY>(std::forward<Ts>(args)...);
324  case aggregation::ALL:
325  return f.template operator()<aggregation::ALL>(std::forward<Ts>(args)...);
326  case aggregation::SUM_OF_SQUARES:
327  return f.template operator()<aggregation::SUM_OF_SQUARES>(std::forward<Ts>(args)...);
328  case aggregation::MEAN:
329  return f.template operator()<aggregation::MEAN>(
330  std::forward<Ts>(args)...);
331  case aggregation::VARIANCE:
332  return f.template operator()<aggregation::VARIANCE>(
333  std::forward<Ts>(args)...);
334  case aggregation::STD:
335  return f.template operator()<aggregation::STD>(
336  std::forward<Ts>(args)...);
337  case aggregation::MEDIAN:
338  return f.template operator()<aggregation::MEDIAN>(
339  std::forward<Ts>(args)...);
340  case aggregation::QUANTILE:
341  return f.template operator()<aggregation::QUANTILE>(
342  std::forward<Ts>(args)...);
343  case aggregation::ARGMAX:
344  return f.template operator()<aggregation::ARGMAX>(
345  std::forward<Ts>(args)...);
346  case aggregation::ARGMIN:
347  return f.template operator()<aggregation::ARGMIN>(
348  std::forward<Ts>(args)...);
349  case aggregation::NUNIQUE:
350  return f.template operator()<aggregation::NUNIQUE>(
351  std::forward<Ts>(args)...);
352  case aggregation::NTH_ELEMENT:
353  return f.template operator()<aggregation::NTH_ELEMENT>(
354  std::forward<Ts>(args)...);
355  default: {
356 #ifndef __CUDA_ARCH__
357  CUDF_FAIL("Unsupported aggregation.");
358 #else
359  release_assert(false && "Unsupported aggregation.");
360 
361  // The following code will never be reached, but the compiler generates a
362  // warning if there isn't a return value.
363 
364  // Need to find out what the return type is in order to have a default
365  // return value and solve the compiler warning for lack of a default
366  // return
367  using return_type =
368  decltype(f.template operator()<aggregation::SUM>(
369  std::forward<Ts>(args)...));
370  return return_type();
371 #endif
372  }
373  }
374 }
