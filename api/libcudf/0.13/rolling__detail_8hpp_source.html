30 // return true if ColumnType is arithmetic type or
31 // AggOp is min_op/max_op/count_op for wrapper (non-arithmetic) types
32 template < typename ColumnType, class AggOp, cudf::experimental::aggregation::Kind op, bool is_mean>
33 static constexpr bool is_supported()
34 {
35 constexpr bool is_comparable_countable_op =
36 std::is_same<AggOp, DeviceMin>::value ||
37 std::is_same<AggOp, DeviceMax>::value ||
38 std::is_same<AggOp, DeviceCount>::value;
39
40 constexpr bool is_operation_supported = (op == experimental::aggregation::SUM ) or
41 (op == experimental::aggregation::MIN ) or
42 (op == experimental::aggregation::MAX ) or
43 (op == experimental::aggregation::COUNT_VALID ) or
44 (op == experimental::aggregation::COUNT_ALL ) or
45 (op == experimental::aggregation::MEAN );
46
47 constexpr bool is_valid_timestamp_agg = cudf::is_timestamp<ColumnType>() and
48 (op == experimental::aggregation::MIN or
49 op == experimental::aggregation::MAX or
50 op == experimental::aggregation::COUNT_VALID or
51 op == experimental::aggregation::COUNT_ALL or
52 op == experimental::aggregation::MEAN);
53
54
55 constexpr bool is_valid_numeric_agg = ( cudf :: is_numeric <ColumnType>() or
56 is_comparable_countable_op) and
57 is_operation_supported;
58
59 constexpr bool is_valid_rolling_agg = !std::is_same<ColumnType, cudf ::string_view>::value and
60 (is_valid_timestamp_agg or is_valid_numeric_agg);
61
62 return is_valid_rolling_agg and
63 cudf ::experimental::detail::is_valid_aggregation<ColumnType, op>();
64 }

66 template <typename ColumnType, class AggOp, cudf ::experimental::aggregation::Kind Op>
67 static constexpr bool is_string_supported()
68 {
69 return std::is_same<ColumnType, cudf::string_view>::value and
70 (( cudf::experimental::aggregation::MIN == Op and std::is_same<AggOp, DeviceMin>::value) or
71 ( cudf::experimental::aggregation::MAX == Op and std::is_same<AggOp, DeviceMax>::value) or
72 ( cudf::experimental::aggregation::COUNT_VALID == Op and std::is_same<AggOp, DeviceCount>::value) or
73 ( cudf::experimental::aggregation::COUNT_ALL == Op and std::is_same<AggOp, DeviceCount>::value));
74 }

76 // store functor
77 template < typename T, bool is_mean>
78 struct store_output_functor
79 {
80 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
81 {
82 out = val;
83 }
84 };
85
86 // Specialization for MEAN
87 template < typename _T>
88 struct store_output_functor <_T, true>
89 {
90 // SFINAE for non-bool types
91 template < typename T = _T,
92 std::enable_if_t<!(cudf::is_boolean<T>() || cudf::is_timestamp<T>())>* = nullptr >
93 CUDA_HOST_DEVICE_CALLABLE void operator ()(T &out, T &val, size_type count)
94 {
95 out = val / count;
96 }
97
98 // SFINAE for bool type
99 template <typename T = _T, std::enable_if_t<cudf::is_boolean<T>()>* = nullptr >
100 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
101 {
102 out = static_cast<int32_t>(val) / count;
103 }
104
105 // SFINAE for timestamp types
106 template <typename T = _T, std::enable_if_t<cudf::is_timestamp<T>()>* = nullptr >
107 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
108 {
109 out = val.time_since_epoch() / count;
110 }
111 };
