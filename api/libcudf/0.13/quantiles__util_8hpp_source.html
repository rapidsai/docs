41 template < typename Result, typename T>
42 CUDA_HOST_DEVICE_CALLABLE
43 Result linear(T lhs, T rhs, double frac)
44 {
45 // TODO: safe operation to avoid overflow/underflow
46 // double can fully represent int8-32 value range.
47 // Since the fraction part of double is 52 bits,
48 // double cannot fully represent int64.
49 // Underflow may occur when converting int64 to double
50 // detail: https://github.com/rapidsai/cudf/issues/1417
51
52 double dlhs = static_cast<double>(lhs);
53 double drhs = static_cast<double>(rhs);
54 double one_minus_frac = 1.0 - frac;
55 return static_cast<Result>(one_minus_frac * dlhs + frac * drhs);
56 }

58 template < typename Result, typename T>
59 CUDA_HOST_DEVICE_CALLABLE
60 Result midpoint(T lhs, T rhs)
61 {
62 // TODO: try std::midpoint (C++20) if available
63 double dlhs = static_cast<double>(lhs);
64 double drhs = static_cast<double>(rhs);
65 return static_cast<Result>(dlhs / 2 + drhs / 2);
66 }
67
68 template < typename Result>
69 CUDA_HOST_DEVICE_CALLABLE
70 Result midpoint(int64_t lhs, int64_t rhs)
71 {
72 // caring to avoid integer overflow and underflow between int64_t and Result( double )
73 int64_t half = lhs / 2 + rhs / 2;
74 int64_t rest = lhs % 2 + rhs % 2;
75 return static_cast<Result>(static_cast<Result>(half) +
76 static_cast<Result>(rest) * 0.5);
77 }
78
79 template <>
80 CUDA_HOST_DEVICE_CALLABLE
81 int64_t midpoint(int64_t lhs, int64_t rhs)
82 {
83 // caring to avoid integer overflow
84 int64_t half = lhs / 2 + rhs / 2;
85 int64_t rest = lhs % 2 + rhs % 2;
86 int64_t result = half;
87
88 // rounding toward zero
89 result += ( half >= 0 && rest != 0 ) ? rest/2 : 0;
90 result += ( half < 0 && rest != 0 ) ? 1 : 0;
91
92 return result;
93 }

97 struct quantile_index
98 {
99 size_type lower;
100 size_type higher;
101 size_type nearest;
102 double fraction;
103
104 CUDA_HOST_DEVICE_CALLABLE
105 quantile_index (size_type count, double quantile )
106 {
107 quantile = std::min(std::max( quantile , 0.0), 1.0);
108
109 double val = quantile * (count - 1);
110 lower = std::floor(val);
111 higher = static_cast<size_type>(std::ceil(val));
112 nearest = static_cast<size_type>(std::nearbyint(val));
113 fraction = val - lower;
114 }
115 };

118 /* @brief computes a quantile value.
119 *
120 * Computes a value for a quantile by interpolating between two values on either
121 * side of the desired quantile.
122 *
123 * `get_value` must have signature: `T <T>(size_type)` where T can be
124 * `static_cast` to `Result`.
125 *
126 * @param[in] get_value Gets the value at a given index in range [0, size].
127 * @param[in] size Number of values indexed by `get_value`.
128 * @param[in] q Desired quantile in range [0, 1].
129 * @param[in] interp Strategy used to interpolate between the two values
130 * on either side of the desired quantile.
131 *
132 * @returns Value of the desired quantile.
133 */
134 template < typename Result, typename ValueAccessor>
135 CUDA_HOST_DEVICE_CALLABLE
136 Result
137 select_quantile(ValueAccessor get_value,
138 size_type size,
139 double q,
140 interpolation interp)
141 {
142 if (size < 2) {
143 return get_value(0);
144 }
145
146 quantile_index idx(size, q);
147
148 switch (interp) {
149 case interpolation::LINEAR :
150 return interpolate::linear<Result>(get_value(idx.lower),
151 get_value(idx.higher),
152 idx.fraction);
153
154 case interpolation::MIDPOINT :
155 return interpolate::midpoint<Result>(get_value(idx.lower),
156 get_value(idx.higher));
157
158 case interpolation::LOWER :
159 return static_cast<Result>(get_value(idx.lower));
160
161 case interpolation::HIGHER :
162 return static_cast<Result>(get_value(idx.higher));
163
164 case interpolation::NEAREST :
165 return static_cast<Result>(get_value(idx.nearest));
166
167 default :
168 #if defined(__CUDA_ARCH__)
169 release_assert( false && "Invalid interpolation operation for quantiles" );
170 return Result();
171 #else
172 CUDF_FAIL( "Invalid interpolation operation for quantiles." );
173 #endif
174 }
175 }
