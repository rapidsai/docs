184 static inline int CountLeadingZeros32(uint32_t value) {
185 #if defined(__clang__) || defined(__GNUC__)
186 if (value == 0) return 32;
187 return static_cast<int>(__builtin_clz(value));
188 #elif defined(_MSC_VER)
189 unsigned long index;
190 return (_BitScanReverse(&index, static_cast<unsigned long>(value)))
191 ? 31 - static_cast<int>(index)
192 : 32;
193 #else
194 int bitpos = 0;
195 while (value != 0) {
196 value >>= 1;
197 ++bitpos;
198 }
199 return 32 - bitpos;
200 #endif
201 }
