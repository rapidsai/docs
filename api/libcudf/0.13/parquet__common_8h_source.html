42 enum ConvertedType {
43 UNKNOWN = -1, // No type information present
44 UTF8 = 0, // a BYTE_ARRAY actually contains UTF8 encoded chars
45 MAP = 1, // a map is converted as an optional field containing a repeated key/value pair
46 MAP_KEY_VALUE = 2, // a key/value pair is converted into a group of two fields
47 LIST = 3, // a list is converted into an optional field containing a repeated field for its values
48 ENUM = 4, // an enum is converted into a binary field
49 DECIMAL = 5, // A decimal value. 10^(-scale) encoded as 2's complement big endian (precision=number of digits, scale=location of decimal point)
50 DATE = 6, // A Date, stored as days since Unix epoch, encoded as the INT32 physical type.
51 TIME_MILLIS = 7, // A time. The total number of milliseconds since midnight.The value is stored as an INT32 physical type.
52 TIME_MICROS = 8, // A time. The total number of microseconds since midnight. The value is stored as an INT64 physical type.
53 TIMESTAMP_MILLIS = 9, // A date/time combination, recorded as milliseconds since the Unix epoch using physical type of INT64.
54 TIMESTAMP_MICROS = 10, // A date/time combination, microseconds since the Unix epoch as INT64
55 UINT_8 = 11, // An unsigned integer 8-bit value as INT32
56 UINT_16 = 12, // An unsigned integer 16-bit value as INT32
57 UINT_32 = 13, // An unsigned integer 32-bit value as INT32
58 UINT_64 = 14, // An unsigned integer 64-bit value as INT64
59 INT_8 = 15, // A signed integer 8-bit value as INT32
60 INT_16 = 16, // A signed integer 16-bit value as INT32
61 INT_32 = 17, // A signed integer 32-bit value as INT32
62 INT_64 = 18, // A signed integer 8-bit value as INT64
63 JSON = 19, // A JSON document embedded within a single UTF8 column.
64 BSON = 20, // A BSON document embedded within a single BINARY column.
65 INTERVAL = 21, // This type annotates a time interval stored as a FIXED_LEN_BYTE_ARRAY of length 12 for 3 integers {months,days,milliseconds}
66 };
