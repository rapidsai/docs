278 // of calling a __host__ functor from this function which is __host__ __device__
279 #pragma nv_exec_check_disable
280 template <template <cudf::type_id> typename IdTypeMap = id_to_type_impl,
281  typename Functor, typename... Ts>
282 CUDA_HOST_DEVICE_CALLABLE constexpr decltype(auto) type_dispatcher(
283  cudf::data_type dtype, Functor f, Ts&&... args) {
284  switch (dtype.id()) {
285  case BOOL8:
286  return f.template operator()<typename IdTypeMap<BOOL8>::type>(
287  std::forward<Ts>(args)...);
288  case INT8:
289  return f.template operator()<typename IdTypeMap<INT8>::type>(
290  std::forward<Ts>(args)...);
291  case INT16:
292  return f.template operator()<typename IdTypeMap<INT16>::type>(
293  std::forward<Ts>(args)...);
294  case INT32:
295  return f.template operator()<typename IdTypeMap<INT32>::type>(
296  std::forward<Ts>(args)...);
297  case INT64:
298  return f.template operator()<typename IdTypeMap<INT64>::type>(
299  std::forward<Ts>(args)...);
300  case FLOAT32:
301  return f.template operator()<typename IdTypeMap<FLOAT32>::type>(
302  std::forward<Ts>(args)...);
303  case FLOAT64:
304  return f.template operator()<typename IdTypeMap<FLOAT64>::type>(
305  std::forward<Ts>(args)...);
306  case STRING:
307  return f.template operator()<typename IdTypeMap<STRING>::type>(
308  std::forward<Ts>(args)...);
309  case TIMESTAMP_DAYS:
310  return f.template operator()<typename IdTypeMap<TIMESTAMP_DAYS>::type>(
311  std::forward<Ts>(args)...);
312  case TIMESTAMP_SECONDS:
313  return f.template operator()<typename IdTypeMap<TIMESTAMP_SECONDS>::type>(
314  std::forward<Ts>(args)...);
315  case TIMESTAMP_MILLISECONDS:
316  return f.template operator()<typename IdTypeMap<TIMESTAMP_MILLISECONDS>::type>(
317  std::forward<Ts>(args)...);
318  case TIMESTAMP_MICROSECONDS:
319  return f.template operator()<typename IdTypeMap<TIMESTAMP_MICROSECONDS>::type>(
320  std::forward<Ts>(args)...);
321  case TIMESTAMP_NANOSECONDS:
322  return f.template operator()<typename IdTypeMap<TIMESTAMP_NANOSECONDS>::type>(
323  std::forward<Ts>(args)...);
324  case DICTIONARY32:
325  return f.template operator()<typename IdTypeMap<DICTIONARY32>::type>(
326  std::forward<Ts>(args)...);
327  default: {
328 #ifndef __CUDA_ARCH__
329  CUDF_FAIL("Unsupported type_id.");
330 #else
331  release_assert(false && "Unsuported type_id.");
332 
333  // The following code will never be reached, but the compiler generates a
334  // warning if there isn't a return value.
335 
336  // Need to find out what the return type is in order to have a default
337  // return value and solve the compiler warning for lack of a default
338  // return
339  using return_type =
340  decltype(f.template operator()<int8_t>(std::forward<Ts>(args)...));
341  return return_type();
342 #endif
343  }
344  }
345 }
