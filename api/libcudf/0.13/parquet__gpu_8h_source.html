48 struct PageInfo {
49 uint8_t *page_data; // Compressed page data before decompression, or uncompressed data after decompression
50 int32_t compressed_page_size; // compressed data size in bytes
51 int32_t uncompressed_page_size; // uncompressed data size in bytes
52 int32_t num_values; // Number of values in this data page or dictionary
53 int32_t chunk_row; // starting row of this page relative to the start of the chunk
54 int32_t num_rows; // number of rows in this page
55 int32_t chunk_idx; // column chunk this page belongs to
56 uint8_t flags; // PAGEINFO_FLAGS_XXX
57 uint8_t encoding; // Encoding for data or dictionary page
58 uint8_t definition_level_encoding; // Encoding used for definition levels (data page)
59 uint8_t repetition_level_encoding; // Encoding used for repetition levels (data page)
60 int32_t valid_count; // Count of valid (non-null) values in this page (negative values indicate data error)
61 };

66 struct ColumnChunkDesc {
67 ColumnChunkDesc () = default ;
68 explicit constexpr ColumnChunkDesc (
69 size_t compressed_size_, uint8_t *compressed_data_, size_t num_values_,
70 uint16_t datatype_, uint16_t datatype_length_, uint32_t start_row_,
71 uint32_t num_rows_, int16_t max_definition_level_,
72 int16_t max_repetition_level_, uint8_t def_level_bits_,
73 uint8_t rep_level_bits_, int8_t codec_, int8_t converted_type_,

75 : compressed_data(compressed_data_),
76 compressed_size(compressed_size_),
77 num_values(num_values_),
78 start_row(start_row_),
79 num_rows(num_rows_),
80 max_def_level(max_definition_level_),
81 max_rep_level(max_repetition_level_),
82 def_level_bits(def_level_bits_),
83 rep_level_bits(rep_level_bits_),
84 data_type (datatype_ | (datatype_length_ << 3)),
85 num_data_pages(0),
86 num_dict_pages(0),
87 max_num_pages(0),
88 page_info( nullptr ),
89 str_dict_index( nullptr ),
90 valid_map_base( nullptr ),
91 column_data_base( nullptr ),
92 codec(codec_),
93 converted_type(converted_type_),
94 decimal_scale(decimal_scale_),
95 ts_clock_rate(ts_clock_rate_) {}

97 uint8_t *compressed_data; // pointer to compressed column chunk data
98 size_t compressed_size; // total compressed data size for this chunk
99 size_t num_values; // total number of values in this column
100 size_t start_row; // starting row of this chunk
101 uint32_t num_rows; // number of rows in this chunk
102 int16_t max_def_level; // max definition level
103 int16_t max_rep_level; // max repetition level
104 uint16_t data_type ; // basic column data type, ((type_length << 3) |
105 // parquet::Type)
106 uint8_t def_level_bits; // bits to encode max definition level
107 uint8_t rep_level_bits; // bits to encode max repetition level
108 int32_t num_data_pages; // number of data pages
109 int32_t num_dict_pages; // number of dictionary pages
110 int32_t max_num_pages; // size of page_info array
111 PageInfo *page_info; // output page info for up to num_dict_pages +
112 // num_data_pages (dictionary pages first)
113 nvstrdesc_s *str_dict_index; // index for string dictionary
114 uint32_t *valid_map_base; // base pointer of valid bit map for this column
115 void *column_data_base; // base pointer of column data
116 int8_t codec; // compressed codec enum
117 int8_t converted_type; // converted type enum
118 int8_t decimal_scale; // decimal scale pow(10, -decimal_scale)
119 int32_t ts_clock_rate; // output timestamp clock frequency (0=default, 1000=ms, 1000000000=ns)
