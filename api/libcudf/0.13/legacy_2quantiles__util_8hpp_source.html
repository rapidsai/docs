29 template <typename T_out, typename T_in>
30 CUDA_HOST_DEVICE_CALLABLE
31 void linear(T_out& result, T_in lhs, T_in rhs, double frac)
32 {
33  // TODO: safe operation to avoid overflow/underflow
34  // double can fully represent int8-32 value range.
35  // Since the fractoin part of double is 52 bits,
36  // double cannot fully represent int64.
37  // The underflow will be occurs at converting int64 to double
38  // detail: https://github.com/rapidsai/cudf/issues/1417
39 
40  double dlhs = static_cast<double>(lhs);
41  double drhs = static_cast<double>(rhs);
42  double one_minus_frac = 1.0 - frac;
43 
44 // result = static_cast<T_out>(static_cast<T_out>(lhs) + frac*static_cast<T_out>(rhs-lhs));
45  result = static_cast<T_out>(one_minus_frac * dlhs + frac * drhs);
46 }

63 template <typename T_out>
64 CUDA_HOST_DEVICE_CALLABLE
65 void midpoint(T_out& result, int64_t lhs, int64_t rhs)
66 {
67  // caring to avoid integer overflow and underflow between int64_t and T_out( double )
68  int64_t half = lhs/2 + rhs/2;
69  int64_t rest = (lhs%2 + rhs%2);
70 
71  result = static_cast<T_out>(static_cast<T_out>(half) + static_cast<T_out>(rest)*0.5);
72 }

75 template <>
76 CUDA_HOST_DEVICE_CALLABLE
77 void midpoint(int64_t& result, int64_t lhs, int64_t rhs)
78 {
79  // caring to avoid integer overflow
80  int64_t half = lhs/2 + rhs/2;
81  int64_t rest = (lhs%2 + rhs%2);
82  result = half;
83 
84  // rounding toward zero
85  result += ( half >= 0 && rest != 0 )? rest/2 : 0;
86  result += ( half < 0 && rest != 0 )? 1 : 0;
87 }

104 struct quantile_index {
105  cudf::size_type lower_bound;
106  cudf::size_type upper_bound;
107  cudf::size_type nearest;
108  double fraction;
109 
110  CUDA_HOST_DEVICE_CALLABLE
111  quantile_index(cudf::size_type length, double quant)
112  {
113  // clamp quant value.
114  // Todo: use std::clamp if c++17 is supported.
115  quant = std::min(std::max(quant, 0.0), 1.0);
116 
117  // since cudf::size_type is int32_t, there is no underflow/overflow
118  double val = quant*(length -1);
119  lower_bound = std::floor(val);
120  upper_bound = static_cast<size_t>(std::ceil(val));
121  nearest = static_cast<size_t>(std::nearbyint(val));
122  fraction = val - lower_bound;
123  }
124 };

148 template <typename T,
149  typename RetT = double>
150 CUDA_HOST_DEVICE_CALLABLE
151 RetT select_quantile(T const* devarr, cudf::size_type size, double quantile,
152  interpolation interpolation)
153 {
154  T temp[2];
155  RetT result;
156 
157  if( size < 2 )
158  {
159  temp[0] = get_array_value(devarr, 0);
160  result = static_cast<RetT>( temp[0] );
161  return result;
162  }
163 
164  quantile_index qi(size, quantile);
165 
166  switch( interpolation )
167  {
168  case LINEAR:
169  temp[0] = get_array_value(devarr, qi.lower_bound);
170  temp[1] = get_array_value(devarr, qi.upper_bound);
171  cudf::interpolate::linear(result, temp[0], temp[1], qi.fraction);
172  break;
173  case MIDPOINT:
174  temp[0] = get_array_value(devarr, qi.lower_bound);
175  temp[1] = get_array_value(devarr, qi.upper_bound);
176  cudf::interpolate::midpoint(result, temp[0], temp[1]);
177  break;
178  case LOWER:
179  temp[0] = get_array_value(devarr, qi.lower_bound);
180  result = static_cast<RetT>( temp[0] );
181  break;
182  case HIGHER:
183  temp[0] = get_array_value(devarr, qi.upper_bound);
184  result = static_cast<RetT>( temp[0] );
185  break;
186  case NEAREST:
187  temp[0] = get_array_value(devarr, qi.nearest);
188  result = static_cast<RetT>( temp[0] );
189  break;
190  default:
191  #if defined(__CUDA_ARCH__)
192  release_assert(false && "Invalid interpolation operation for quantiles");
193  #else
194  CUDF_FAIL("Invalid interpolation operation for quantiles");
195  #endif
196  }
197 
198  return result;
199 }
