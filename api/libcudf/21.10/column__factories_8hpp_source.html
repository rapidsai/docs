304 template <typename B>
305 std::unique_ptr<column> make_fixed_width_column(
306  data_type type,
307  size_type size,
308  B&& null_mask,
309  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
310  rmm::cuda_stream_view stream = rmm::cuda_stream_default,
311  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
312 {
313  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
314  if (is_timestamp(type)) {
315  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
316  } else if (is_duration(type)) {
317  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
318  } else if (is_fixed_point(type)) {
319  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
320  }
321  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
322 }
