204 inline void throw_cuda_error(cudaError_t error, const char* file, unsigned int line)
205 {
206  // Calls cudaGetLastError to clear the error status. It is nearly certain that a fatal error
207  // occurred if it still returns the same error after a cleanup.
208  cudaGetLastError();
209  auto const last = cudaFree(0);
210  auto const msg = std::string{"CUDA error encountered at: " + std::string{file} + ":" +
211  std::to_string(line) + ": " + std::to_string(error) + " " +
212  cudaGetErrorName(error) + " " + cudaGetErrorString(error)};
213  // Call cudaDeviceSynchronize to ensure `last` did not result from an asynchronous error.
214  // between two calls.
215  if (error == last && last == cudaDeviceSynchronize()) {
216  throw fatal_cuda_error{"Fatal " + msg, error};
217  } else {
218  throw cuda_error{msg, error};
219  }
220 }
221 // @endcond
222 } // namespace detail
