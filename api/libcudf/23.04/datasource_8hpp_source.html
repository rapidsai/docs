411  public:
417  explicit arrow_io_source(std::string_view arrow_uri)
418  {
419  const std::string uri_start_delimiter = "//";
420  const std::string uri_end_delimiter = "?";
421 
422  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
423  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
424  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
425  filesystem = result.ValueOrDie();
426 
427  // Parse the path from the URI
428  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
429  ? 0
430  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
431  size_t end = arrow_uri.find(uri_end_delimiter) - start;
432  std::string_view path = arrow_uri.substr(start, end);
433 
434  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
435  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
436  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
437  arrow_file = in_stream.ValueOrDie();
438  }

454  std::unique_ptr<buffer> host_read(size_t offset, size_t size) override
455  {
