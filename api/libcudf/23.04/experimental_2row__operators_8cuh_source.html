216 // The has_nested_columns template parameter of the device_row_comparator is
217 // necessary to help the compiler optimize our code. Without it, the list and
218 // struct view specializations are present in the code paths used for primitive
219 // types, and the compiler fails to inline this nearly as well resulting in a
220 // significant performance drop. As a result, there is some minor tension in
221 // the current design between the presence of this parameter and the way that
222 // the Dremel data is passed around, first as a
223 // std::optional<device_span<dremel_device_view>> in the
224 // preprocessed_table/device_row_comparator (which is always valid when
225 // has_nested_columns and is otherwise invalid) that is then unpacked to a
226 // thrust::optional<dremel_device_view> at the element_comparator level (which
227 // is always valid for a list column and otherwise invalid). We cannot use an
228 // additional template parameter for the element_comparator on a per-column
229 // basis because we cannot conditionally define dremel_device_view member
230 // variables without jumping through extra hoops with inheritance, so the
231 // thrust::optional<dremel_device_view> member must be an optional rather than
232 // a raw dremel_device_view.

356  __device__ cuda::std::pair<weak_ordering, int> operator()(
357  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept

382  template <typename Element,
383  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::struct_view>)>
384  __device__ cuda::std::pair<weak_ordering, int> operator()(
385  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
386  {
387  column_device_view lcol = _lhs;
388  column_device_view rcol = _rhs;
389  int depth = _depth;
390  while (lcol.type().id() == type_id::STRUCT) {
391  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
392  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
393 
394  if (lhs_is_null or rhs_is_null) { // at least one is null
395  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
396  return cuda::std::pair(state, depth);
397  }
398 
399  if (lcol.num_child_columns() == 0) {
400  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
401  }
402 
403  // Non-empty structs have been modified to only have 1 child when using this.
404  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
405  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
406  ++depth;
407  }
408 
409  return cudf::type_dispatcher<dispatch_void_if_nested>(
410  lcol.type(),
411  element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth, _comparator},
412  lhs_element_index,
413  rhs_element_index);
414  }

416  template <typename Element,
417  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::list_view>)>
418  __device__ cuda::std::pair<weak_ordering, int> operator()(size_type lhs_element_index,
419  size_type rhs_element_index)
420  {
421  // only order top-NULLs according to null_order
422  auto const is_l_row_null = _lhs.is_null(lhs_element_index);
423  auto const is_r_row_null = _rhs.is_null(rhs_element_index);
424  if (is_l_row_null || is_r_row_null) {
425  return cuda::std::pair(null_compare(is_l_row_null, is_r_row_null, _null_precedence),
426  _depth);
427  }
428 
429  // These are all the values from the Dremel encoding.
430  auto const l_max_def_level = _l_dremel_device_view->max_def_level;
431  auto const r_max_def_level = _r_dremel_device_view->max_def_level;
432  auto const l_def_levels = _l_dremel_device_view->def_levels;
433  auto const r_def_levels = _r_dremel_device_view->def_levels;
434  auto const l_rep_levels = _l_dremel_device_view->rep_levels;
435  auto const r_rep_levels = _r_dremel_device_view->rep_levels;
436 
437  // Traverse the nested list hierarchy to get a column device view
438  // pointing to the underlying child data.
439  column_device_view lcol = _lhs.slice(lhs_element_index, 1);
440  column_device_view rcol = _rhs.slice(rhs_element_index, 1);
441 
442  while (lcol.type().id() == type_id::LIST) {
443  lcol = detail::lists_column_device_view(lcol).get_sliced_child();
444  rcol = detail::lists_column_device_view(rcol).get_sliced_child();
445  }
446 
447  // These start and end values indicate the start and end points of all
448  // the elements of the lists in the current list element
449  // (`[lhs|rhs]_element_index`) that we are comparing.
450  auto const l_offsets = _l_dremel_device_view->offsets;
451  auto const r_offsets = _r_dremel_device_view->offsets;
452  auto l_start = l_offsets[lhs_element_index];
453  auto l_end = l_offsets[lhs_element_index + 1];
454  auto r_start = r_offsets[rhs_element_index];
455  auto r_end = r_offsets[rhs_element_index + 1];
456 
457  // This comparator will be used to compare leaf (non-nested) data types.
458  auto comparator =
459  element_comparator{_check_nulls, lcol, rcol, _null_precedence, _depth, _comparator};
460 
461  // Loop over each element in the encoding. Note that this includes nulls
462  // and empty lists, so not every index corresponds to an actual element
463  // in the child column. The element_index is used to keep track of the current
464  // child element that we're actually comparing.
465  for (int l_dremel_index = l_start, r_dremel_index = r_start, element_index = 0;
466  l_dremel_index < l_end and r_dremel_index < r_end;
467  ++l_dremel_index, ++r_dremel_index) {
468  auto const l_rep_level = l_rep_levels[l_dremel_index];
469  auto const r_rep_level = r_rep_levels[r_dremel_index];
470 
471  // early exit for smaller sub-list
472  if (l_rep_level != r_rep_level) {
473  // the lower repetition level is a smaller sub-list
474  return l_rep_level < r_rep_level ? cuda::std::pair(weak_ordering::LESS, _depth)
475  : cuda::std::pair(weak_ordering::GREATER, _depth);
476  }
477 
478  // only compare if left and right are at same nesting level
479  auto const l_def_level = l_def_levels[l_dremel_index];
480  auto const r_def_level = r_def_levels[r_dremel_index];
481 
482  // either left or right are empty or NULLs of arbitrary nesting
483  if (l_def_level < l_max_def_level || r_def_level < r_max_def_level) {
484  // in the fully unraveled version of the list column, only the
485  // most nested NULLs and leafs are present
486  // In this rare condition that we get to the most nested NULL, we increment
487  // element_index because either both rows have a deeply nested NULL at the
488  // same position, and we'll "continue" in our iteration, or we will early
489  // exit if only one of the rows has a deeply nested NULL
490  if ((lcol.nullable() and l_def_levels[l_dremel_index] == l_max_def_level - 1) or
491  (rcol.nullable() and r_def_levels[r_dremel_index] == r_max_def_level - 1)) {
492  ++element_index;
493  }
494  if (l_def_level == r_def_level) { continue; }
495  // We require [] < [NULL] < [leaf] for nested nulls.
496  // The null_precedence only affects top level nulls.
497  return l_def_level < r_def_level ? cuda::std::pair(weak_ordering::LESS, _depth)
498  : cuda::std::pair(weak_ordering::GREATER, _depth);
499  }
500 
501  // finally, compare leaf to leaf
502  weak_ordering state{weak_ordering::EQUIVALENT};
503  int last_null_depth = _depth;
504  cuda::std::tie(state, last_null_depth) = cudf::type_dispatcher<dispatch_void_if_nested>(
505  lcol.type(), comparator, element_index, element_index);
506  if (state != weak_ordering::EQUIVALENT) { return cuda::std::pair(state, _depth); }
507  ++element_index;
508  }
509 
510  // If we have reached this stage, we know that definition levels,
511  // repetition levels, and actual elements are identical in both list
512  // columns up to the `min(l_end - l_start, r_end - r_start)` element of
513  // the Dremel encoding. However, two lists can only compare equivalent if
514  // they are of the same length. Otherwise, the shorter of the two is less
515  // than the longer. This final check determines the appropriate resulting
516  // ordering by checking how many total elements each list is composed of.
517  return cuda::std::pair(detail::compare_elements(l_end - l_start, r_end - r_start), _depth);
518  }
