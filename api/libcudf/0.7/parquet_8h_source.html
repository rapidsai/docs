178 static inline int CountLeadingZeros32(uint32_t value) {
179 #if defined(__clang__) || defined(__GNUC__)
180 if (value == 0) return 32;
181 return static_cast< int > (__builtin_clz(value));
182 #elif defined(_MSC_VER)
183 unsigned long index;
184 return (_BitScanReverse(&index, static_cast<unsigned long>(value)))
185 ? 31 - static_cast<int>(index)
186 : 32;
187 #else
188 int bitpos = 0;
189 while (value != 0) {
190 value >>= 1;
191 ++bitpos;
192 }
193 return 32 - bitpos;
194 #endif
195 }
