86 // enable conversion to arithmetic types *only* for the cudf::bool8 wrapper
87 // (defined later in this file as wrapper<gdf_bool8, GDF_BOOL8>)
88 template <gdf_dtype the_type = type_id,
89 typename T_out,
90 typename std::enable_if<(the_type == GDF_BOOL8) &&
91 std::is_arithmetic<T_out>::value,
92 int >::type* = nullptr >
93 CUDA_HOST_DEVICE_CALLABLE
94 explicit operator T_out() const {
95 // Casting a cudf::bool8 to arithmetic type should always be the same as
96 // casting a bool to arithmetic type, and not the same as casting the
97 // underlying type to arithmetic type. Therefore we cast the value to bool
98 // first, then the output type
99 return static_cast< T_out > ( static_cast< bool > (this-> value ));
100 }
101
102 wrapper(wrapper const & w) = default ;
103
104 wrapper() = default ;
105 };
