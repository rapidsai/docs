23 /* --------------------------------------------------------------------------*
24  * @brief Computes the rolling window function of the values in a column.
25  *
26  * This function aggregates values in a window around each element i of the input
27  * column, and invalidates the bit mask for element i if there are not enough observations. The
28  * window size and the number of observations can be static or dynamic (varying for
29  * each element). This matches Pandas' API for DataFrame.rolling with a few notable
30  * differences:
31  * - instead of the center flag it uses the forward window size to allow for
32  * more flexible windows. The total window size = window + forward_window.
33  * Element i uses elements [i-window+1, i+forward_window] to do the window
34  * computation.
35  * - instead of storing NA/NaN for output rows that do not meet the minimum
36  * number of observations this function updates the valid bitmask of the column
37  * to indicate which elements are valid.
38  * - support for dynamic rolling windows, i.e. window size or number of
39  * observations can be specified for each element using an additional array.
40  *
41  * @param[in] input_col The input column
42  * @param[in] window The static rolling window size. If window_col = NULL,
43  * output_col[i] accumulates values from input_col[i-window+1] to
44  * input_col[i] inclusive
45  * @param[in] min_periods Minimum number of observations in window required to
46  * have a value, otherwise 0 is stored in the valid bit mask for
47  * element i. If min_periods_col != NULL, then minimum number of
48  * observations for element i is obtained from min_periods_col[i]
49  * @param[in] forward_window The static window size in the forward direction. If
50  * forward_window_col = NULL, output_col[i] accumulates values from
51  * input_col[i] to input_col[i+forward_window] inclusive
52  * @param[in] agg_type The rolling window aggregtion type (sum, max, min, etc.)
53  * @param[in] window_col The window size values, window_col[i] specifies window
54  * size for element i. If window_col = NULL, then window is used as
55  * the static window size for all elements
56  * @param[in] min_periods_col The minimum number of observation values,
57  * min_periods_col[i] specifies minimum number of observations for
58  * element i. If min_periods_col = NULL, then min_periods is used as
59  * the static value for all elements
60  * @param[in] forward_window_col The forward window size values,
61  * forward_window_col[i] specifies forward window size for element i.
62  * If forward_window_col = NULL, then forward_window is used as the
63  * static forward window size for aill elements
64  * @param[in] stream Optional CUDA stream on which to execute kernels
65  *
66  * @returns gdf_column The output column
67  *
68  * --------------------------------------------------------------------------*/
69 gdf_column* rolling_window(const gdf_column &input_col,
70  gdf_size_type window,
71  gdf_size_type min_periods,
72  gdf_size_type forward_window,

75  const gdf_size_type *min_periods_col,
76  const gdf_size_type *forward_window_col,
77  cudaStream_t stream = 0);
