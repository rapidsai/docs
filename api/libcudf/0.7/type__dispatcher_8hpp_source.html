113 // This pragma disables a compiler warning that complains about the valid usage
114 // of calling a __host__ functor from this function which is __host__ __device__
115 #pragma hd_warning_disable
116 #pragma nv_exec_check_disable
117 template <class functor_t, typename... Ts>
118 CUDA_HOST_DEVICE_CALLABLE decltype(auto) type_dispatcher(gdf_dtype dtype,
119  functor_t f,
120  Ts&&... args) {
121  switch(dtype)
122  {
123  // The .template is known as a "template disambiguator"
124  // See here for more information:
125  // https://stackoverflow.com/questions/3786360/confusing-template-error
126  case GDF_INT8: { return f.template operator()< int8_t >(std::forward<Ts>(args)...); }
127  case GDF_INT16: { return f.template operator()< int16_t >(std::forward<Ts>(args)...); }
128  case GDF_INT32: { return f.template operator()< int32_t >(std::forward<Ts>(args)...); }
129  case GDF_INT64: { return f.template operator()< int64_t >(std::forward<Ts>(args)...); }
130  case GDF_FLOAT32: { return f.template operator()< float >(std::forward<Ts>(args)...); }
131  case GDF_FLOAT64: { return f.template operator()< double >(std::forward<Ts>(args)...); }
132  case GDF_BOOL8: { return f.template operator()< bool8 >(std::forward<Ts>(args)...); }
133  case GDF_DATE32: { return f.template operator()< date32 >(std::forward<Ts>(args)...); }
134  case GDF_DATE64: { return f.template operator()< date64 >(std::forward<Ts>(args)...); }
135  case GDF_TIMESTAMP: { return f.template operator()< timestamp >(std::forward<Ts>(args)...); }
136  case GDF_CATEGORY: { return f.template operator()< category >(std::forward<Ts>(args)...); }
137  case GDF_STRING_CATEGORY:
138  { return f.template operator()< nvstring_category >(std::forward<Ts>(args)...); }
139  default: {
140 #ifdef __CUDA_ARCH__
141 
142  // This will cause the calling kernel to crash as well as invalidate
143  // the GPU context
144  release_assert(false && "Invalid gdf_dtype in type_dispatcher");
145 
146  // The following code will never be reached, but the compiler generates a
147  // warning if there isn't a return value.
148 
149  // Need to find out what the return type is in order to have a default
150  // return value and solve the compiler warning for lack of a default
151  // return
152  using return_type =
153  decltype(f.template operator()<int8_t>(std::forward<Ts>(args)...));
154  return return_type();
155 #else
156  // In host-code, the compiler is smart enough to know we don't need a
157  // default return type since we're throwing an exception.
158  throw std::runtime_error("Invalid gdf_dtype in type_dispatcher");
159 #endif
160  }
161  }
162 }
