60 template <JoinType join_type,
61 typename multimap_type>
62 gdf_error estimate_join_output_size( device_table const & build_table,
63 device_table const & probe_table,
64 multimap_type const & hash_table,
65 gdf_size_type * join_output_size_estimate)
66 {
67 const gdf_size_type build_table_num_rows{build_table.num_rows()};
68 const gdf_size_type probe_table_num_rows{probe_table.num_rows()};
69
70 // If the probe table is significantly larger (5x) than the build table,
71 // then we attempt to only use a subset of the probe table rows to compute an
72 // estimate of the join output size.
73 gdf_size_type probe_to_build_ratio{0};
74 if (build_table_num_rows > 0) {
75 probe_to_build_ratio = static_cast< gdf_size_type > (std::ceil(static_cast<float>(probe_table_num_rows)/build_table_num_rows));
76 }
77 else {
78 // If the build table is empty, we know exactly how large the output
79 // will be for the different types of joins and can return immediately
80 switch (join_type)
81 {
82 case JoinType::INNER_JOIN:
83 {
84 // Inner join with an empty table will have no output
85 *join_output_size_estimate = 0;
86 break ;
87 }
88 case JoinType::LEFT_JOIN:
89 {
90 // Left join with an empty table will have an output of NULL rows
91 // equal to the number of rows in the probe table
92 *join_output_size_estimate = probe_table_num_rows;
93 break ;
94 }
95 default :
96 return GDF_UNSUPPORTED_JOIN_TYPE;
97 }
98 return GDF_SUCCESS;
99 }
100
101 gdf_size_type sample_probe_num_rows{probe_table_num_rows};
102 constexpr gdf_size_type MAX_RATIO{5};
103 if (probe_to_build_ratio > MAX_RATIO)
104 {
105 sample_probe_num_rows = build_table_num_rows;
106 }
107
108 // Allocate storage for the counter used to get the size of the join output
109 gdf_size_type * d_size_estimate{ nullptr };
110 gdf_size_type h_size_estimate{0};
111
112 CUDA_TRY(cudaMallocHost(&d_size_estimate, sizeof ( size_t )));
113 *d_size_estimate = 0;
114
115 CUDA_TRY( cudaGetLastError() );
116
117
118 // Continue probing with a subset of the probe table until either:
119 // a non-zero output size estimate is found OR
120 // all of the rows in the probe table have been sampled
121 do {
122
123 sample_probe_num_rows = std::min(sample_probe_num_rows, probe_table_num_rows);
124
125 *d_size_estimate = 0;
126
127 constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
128 const gdf_size_type probe_grid_size{(sample_probe_num_rows + block_size -1)/block_size};
129
130 // Probe the hash table without actually building the output to simply
131 // find what the size of the output will be.
132 compute_join_output_size<join_type,
133 multimap_type,
134 block_size,
135 DEFAULT_CUDA_CACHE_SIZE>
136 <<<probe_grid_size, block_size>>>(&hash_table,
137 build_table,
138 probe_table,
139 sample_probe_num_rows,
140 d_size_estimate);
141
142 // Device sync is required to ensure d_size_estimate is updated
143 CUDA_TRY( cudaDeviceSynchronize() );
144
145 // Only in case subset of probe table is chosen,
146 // increase the estimated output size by a factor of the ratio between the
147 // probe and build tables
148 if (sample_probe_num_rows < probe_table_num_rows) {
149 h_size_estimate = *d_size_estimate * probe_to_build_ratio;
150 } else {
151 h_size_estimate = *d_size_estimate;
152 }
153
154 // If the size estimate is non-zero, then we have a valid estimate and can break
155 // If sample_probe_num_rows >= probe_table_num_rows, then we've sampled the entire
156 // probe table, in which case the estimate is exact and we can break
157 if ((h_size_estimate > 0)
158 || (sample_probe_num_rows >= probe_table_num_rows))
159 {
160 break ;
161 }
162
163 // If the size estimate is zero, then double the number of sampled rows in the probe
164 // table. Reduce the ratio of the number of probe rows sampled to the
165 // number of rows in the build table by the same factor
166 if (0 == h_size_estimate)
167 {
168 constexpr gdf_size_type GROW_RATIO{2};
169 sample_probe_num_rows *= GROW_RATIO;
170 probe_to_build_ratio = static_cast< gdf_size_type > (std::ceil(static_cast<float>(probe_to_build_ratio)/GROW_RATIO));
171 }
172
173 } while ( true );
174
175 CUDA_TRY( cudaFreeHost(d_size_estimate) );
176
177 *join_output_size_estimate = h_size_estimate;
178
179 return GDF_SUCCESS;
180 }

200 template <JoinType join_type,
201 typename output_index_type>
202 gdf_error compute_hash_join(
203 gdf_column * const output_l,
204 gdf_column * const output_r,
205 cudf::table const & left_table,
206 cudf::table const & right_table,
207 bool flip_results = false )
208 {
209 gdf_error gdf_error_code{GDF_SUCCESS};
210
211 gdf_column_view (output_l, nullptr , nullptr , 0, N_GDF_TYPES);
212 gdf_column_view (output_r, nullptr , nullptr , 0, N_GDF_TYPES);
213
214 // The LEGACY allocator allocates the hash table array with normal cudaMalloc,
215 // the non-legacy allocator uses managed memory
216 #ifdef HT_LEGACY_ALLOCATOR
217 using multimap_type = concurrent_unordered_multimap <hash_value_type,
218 output_index_type,
219 size_t,
220 std::numeric_limits<hash_value_type>::max(),
221 std::numeric_limits<output_index_type>::max(),
222 default_hash<hash_value_type> ,
223 equal_to<hash_value_type> ,
224 legacy_allocator< thrust::pair<hash_value_type, output_index_type> > >;
225 #else
226 using multimap_type = concurrent_unordered_multimap <hash_value_type,
227 output_index_type,
228 size_t,
229 std::numeric_limits<hash_value_type>::max(),
230 std::numeric_limits<output_index_type>::max()>;
231 #endif
232 //If FULL_JOIN is selected then we process as LEFT_JOIN till we need to take care of unmatched indices
233 constexpr JoinType base_join_type = (join_type == JoinType::FULL_JOIN)? JoinType::LEFT_JOIN : join_type;
234
235 // Hash table will be built on the right table
236 auto build_table = device_table::create (right_table);
237 const gdf_size_type build_table_num_rows{build_table->num_rows()};
238
239 // Probe with the left table
240 auto probe_table = device_table::create (left_table);
241 const gdf_size_type probe_table_num_rows{probe_table->num_rows()};
242
243 // Hash table size must be at least 1 in order to have a valid allocation.
244 // Even if the hash table will be empty, it still must be allocated for the
245 // probing phase in the event of an outer join
246 size_t const hash_table_size =
247 std::max(compute_hash_table_size(build_table_num_rows), size_t {1});
248
249 std::unique_ptr<multimap_type> hash_table( new multimap_type(hash_table_size));
250
251 // FIXME: use GPU device id from the context?
252 // (although should be possible once we move to Arrow)
253 hash_table->prefetch(0);
254
255 CUDA_TRY( cudaDeviceSynchronize() );
256
257 // Allocate a gdf_error for the device to hold error code returned from
258 // the build kernel and intialize with GDF_SUCCESS
259 // Use Page Locked memory to avoid overhead of memcpys
260 gdf_error * d_gdf_error_code{ nullptr };
261 CUDA_TRY( cudaMallocHost(&d_gdf_error_code, sizeof (gdf_error)) );
262 *d_gdf_error_code = GDF_SUCCESS;
263
264 constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
265
266 // build the hash table
267 if (build_table_num_rows > 0)
268 {
269 const gdf_size_type build_grid_size{(build_table_num_rows + block_size - 1)/block_size};
270 build_hash_table<<<build_grid_size, block_size>>>(hash_table.get(),
271 *build_table,
272 build_table_num_rows,
273 d_gdf_error_code);
274
275 // Device synch is required to ensure d_gdf_error_code
276 // has been written
277 CUDA_TRY( cudaDeviceSynchronize() );
278 }
279
280 // Check error code from the kernel
281 gdf_error_code = *d_gdf_error_code;
282 if (GDF_SUCCESS != gdf_error_code){
283 return gdf_error_code;
284 }
285
286
287 gdf_size_type estimated_join_output_size{0};
288 gdf_error_code = estimate_join_output_size<base_join_type, multimap_type>(
289 *build_table, *probe_table, *hash_table, &estimated_join_output_size);
290
291 if (GDF_SUCCESS != gdf_error_code){
292 return gdf_error_code;
293 }
294
295 // If the estimated output size is zero, return immediately
296 if (0 == estimated_join_output_size){
297 return GDF_SUCCESS;
298 }
299
300 // Because we are approximating the number of joined elements, our approximation
301 // might be incorrect and we might have underestimated the number of joined elements.
302 // As such we will need to de-allocate memory and re-allocate memory to ensure
303 // that the final output is correct.
304 gdf_size_type h_actual_found{0};
305 output_index_type *output_l_ptr{ nullptr };
306 output_index_type *output_r_ptr{ nullptr };
307 bool cont = true ;
308
309 // Allocate device global counter used by threads to determine output write location
310 gdf_size_type *d_global_write_index{ nullptr };
311 RMM_TRY( RMM_ALLOC(( void **)&d_global_write_index, sizeof (gdf_size_type), 0) ); // TODO non-default stream?
312
313 // Because we only have an estimate of the output size, we may need to probe the
314 // hash table multiple times until we've found an output buffer size that is large enough
315 // to hold the output
316 while (cont)
317 {
318 output_l_ptr = nullptr ;
319 output_r_ptr = nullptr ;
320
321 // Allocate temporary device buffer for join output
322 RMM_TRY( RMM_ALLOC(( void **)&output_l_ptr, estimated_join_output_size* sizeof (output_index_type), 0) );
323 RMM_TRY( RMM_ALLOC(( void **)&output_r_ptr, estimated_join_output_size* sizeof (output_index_type), 0) );
324 CUDA_TRY( cudaMemsetAsync(d_global_write_index, 0, sizeof (gdf_size_type), 0) );
325
326 const gdf_size_type probe_grid_size{(probe_table_num_rows + block_size -1)/block_size};
327
328 // Do the probe of the hash table with the probe table and generate the output for the join
329 probe_hash_table<base_join_type,
330 multimap_type,
331 hash_value_type,
332 output_index_type,
333 block_size,
334 DEFAULT_CUDA_CACHE_SIZE>
335 <<<probe_grid_size, block_size>>> (hash_table.get(),
336 *build_table,
337 *probe_table,
338 probe_table->num_rows(),
339 output_l_ptr,
340 output_r_ptr,
341 d_global_write_index,
342 estimated_join_output_size,
343 flip_results);
344
345 CUDA_TRY( cudaGetLastError() );
346
347 CUDA_TRY( cudaMemcpy(&h_actual_found, d_global_write_index, sizeof (gdf_size_type), cudaMemcpyDeviceToHost));
348
349 // The estimate was too small. Double the estimate and try again
350 if (estimated_join_output_size < h_actual_found){
351 cont = true ;
352 estimated_join_output_size *= 2;
353 // Free the old buffers to prevent a memory leak on the new allocation
354 RMM_TRY( RMM_FREE(output_l_ptr, 0) );
355 RMM_TRY( RMM_FREE(output_r_ptr, 0) );
356 }
357 else
358 {
359 cont = false ;
360 }
361 }
362
363 // free memory used for the counters
364 RMM_TRY( RMM_FREE(d_global_write_index, 0) );
365
366 cudaStream_t stream = 0;
367 if (join_type == JoinType::FULL_JOIN) {
368 append_full_join_indices (
369 &output_l_ptr, &output_r_ptr,
370 estimated_join_output_size,
371 h_actual_found, build_table_num_rows,
372 stream);
373 }
374
375 // If the estimated join output size was larger than the actual output size,
376 // then the buffers are larger than necessary. Allocate buffers of the actual
377 // output size and copy the results to the buffers of the correct size
378 // FIXME Is this really necessary? It's probably okay to have the buffers be oversized
379 // and avoid the extra allocation/memcopy
380 if (estimated_join_output_size > h_actual_found) {
381 output_index_type *copy_output_l_ptr{ nullptr };
382 output_index_type *copy_output_r_ptr{ nullptr };
383 RMM_TRY( RMM_ALLOC(( void **)&copy_output_l_ptr, h_actual_found* sizeof (output_index_type), 0) ); // TODO non-default stream?
384 RMM_TRY( RMM_ALLOC(( void **)&copy_output_r_ptr, h_actual_found* sizeof (output_index_type), 0) );
385 CUDA_TRY( cudaMemcpy(copy_output_l_ptr, output_l_ptr, h_actual_found* sizeof (output_index_type), cudaMemcpyDeviceToDevice) );
386 CUDA_TRY( cudaMemcpy(copy_output_r_ptr, output_r_ptr, h_actual_found* sizeof (output_index_type), cudaMemcpyDeviceToDevice) );
387 RMM_TRY( RMM_FREE(output_l_ptr, 0) );
388 RMM_TRY( RMM_FREE(output_r_ptr, 0) );
389 output_l_ptr = copy_output_l_ptr;
390 output_r_ptr = copy_output_r_ptr;
391 }
392
393 // Free the device error code
394 CUDA_TRY( cudaFreeHost(d_gdf_error_code) );
395
396 // Deduce the type of the output gdf_columns
397 gdf_dtype dtype;
398 switch ( sizeof (output_index_type))
399 {
400 case 1 : dtype = GDF_INT8; break ;
401 case 2 : dtype = GDF_INT16; break ;
402 case 4 : dtype = GDF_INT32; break ;
403 case 8 : dtype = GDF_INT64; break ;
404 }
405 gdf_column_view (output_l, output_l_ptr, nullptr , h_actual_found, dtype);
406 gdf_column_view (output_r, output_r_ptr, nullptr , h_actual_found, dtype);
407
408 return gdf_error_code;
409 }
