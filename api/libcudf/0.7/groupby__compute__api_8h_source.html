96 __device__ bool operator() (key_type const & left_index,
97 key_type const & right_index) const
98 {
99 // The unused key is not a valid row index in the device_tables.
100 // Therefore, if comparing against the unused key, use the map's default
101 // comparison function
102 if ((unused_key == left_index) || (unused_key == right_index))
103 return default_comparator(left_index, right_index);
104
105 // Check for equality between the two rows of the two tables
106 return rows_equal(left_table, left_index, right_table, right_index);
107 }

135 template < typename aggregation_type, typename aggregation_operation>
136 gdf_error GroupbyHash( cudf::table const & input_keys,
137 const aggregation_type* const in_aggregation_column,
138 cudf::table & output_keys,
139 aggregation_type* out_aggregation_column,
140 gdf_size_type* out_size,
141 aggregation_operation aggregation_op,
142 bool sort_result = false ) {
143 const gdf_size_type input_num_rows = input_keys. num_rows ();
144
145 // The map will store (row index, aggregation value)
146 // Where row index is the row number of the first row to be successfully inserted
147 // for a given unique row
148 using map_type = concurrent_unordered_map <gdf_size_type,
149 aggregation_type,
150 std::numeric_limits<gdf_size_type>::max(),
151 default_hash<gdf_size_type> ,
152 equal_to<gdf_size_type> ,
153 legacy_allocator<thrust::pair<gdf_size_type, aggregation_type> > >;
154
155 // The hash table occupancy and the input size determines the size of the hash table
156 const size_t hash_table_size{ compute_hash_table_size(input_num_rows) };
157
158 // Initialize the hash table with the aggregation operation functor's identity value
159 std::unique_ptr<map_type> the_map( new map_type(hash_table_size, aggregation_operation::IDENTITY));
160
161 const dim3 build_grid_size ((input_num_rows + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
162 const dim3 block_size (THREAD_BLOCK_SIZE, 1, 1);
163
164 CUDA_TRY(cudaGetLastError());
165
166
167 auto d_input_keys = device_table::create (input_keys);
168
169 // Inserts (i, aggregation_column[i]) as a key-value pair into the
170 // hash table. When a given key already exists in the table, the aggregation operation
171 // is computed between the new and existing value, and the result is stored back.
172 build_aggregation_table<<<build_grid_size, block_size>>>(the_map.get(),
173 *d_input_keys,
174 in_aggregation_column,
175 input_num_rows,
176 aggregation_op,
177 row_comparator<map_type> (*the_map, *d_input_keys, *d_input_keys));
178 CUDA_TRY(cudaGetLastError());
179
180
181 auto d_output_keys = device_table::create (output_keys);
182
183 // Used by threads to coordinate where to write their results
184 gdf_size_type * global_write_index{ nullptr };
185 RMM_TRY(RMM_ALLOC(( void **)&global_write_index, sizeof (gdf_size_type), 0)); // TODO: non-default stream?
186 CUDA_TRY(cudaMemset(global_write_index, 0, sizeof (gdf_size_type)));
187 const dim3 extract_grid_size ((hash_table_size + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
188
189 // Extracts every non-empty key and value into separate contiguous arrays,
190 // which provides the result of the groupby operation
191 extract_groupby_result<<<extract_grid_size, block_size>>>(the_map.get(),
192 hash_table_size,
193 *d_output_keys,
194 *d_input_keys,
195 out_aggregation_column,
196 global_write_index);
197
198 CUDA_TRY(cudaGetLastError());
199
200 // At the end of the extraction kernel, the global write index will be equal to
201 // the size of the output. Update the output size.
202 CUDA_TRY( cudaMemcpy(out_size, global_write_index, sizeof (gdf_size_type), cudaMemcpyDeviceToHost) );
203 RMM_TRY( RMM_FREE(global_write_index, 0) );
204
205 // Update the size of the output key columns
206 std::for_each(output_keys. begin (), output_keys. end (),
207 [out_size]( gdf_column * col) { col-> size = *out_size; });
208
209
210 // Optionally sort the groupby/aggregation result columns
211 if ( true == sort_result) {
212
213 rmm::device_vector<gdf_index_type> sorted_indices(*out_size);
214 thrust::sequence(rmm::exec_policy()->on(0), sorted_indices.begin(), sorted_indices.end());
215
216 gdf_column sorted_indices_col;
217 gdf_error status = gdf_column_view (&sorted_indices_col, sorted_indices.data().get(),
218 nullptr , *out_size, GDF_INT32);
219 if (status != GDF_SUCCESS)
220 return status;
221
222 gdf_context ctxt;
223 ctxt. flag_null_sort_behavior = GDF_NULL_AS_LARGEST;
224 status = gdf_order_by(output_keys. begin (), nullptr ,
225 output_keys. num_columns (), &sorted_indices_col, &ctxt);
226
227 if (status != GDF_SUCCESS)
228 return status;
229
230 // Reorder table according to indices from order_by
231 cudf::detail::gather(&output_keys, sorted_indices.data().get(),
232 &output_keys);
233
234 rmm::device_vector<aggregation_type> temporary_aggregation_buffer(*out_size);
235 thrust::gather(rmm::exec_policy()->on(0),
236 sorted_indices.begin(), sorted_indices.end(),
237 out_aggregation_column,
238 temporary_aggregation_buffer.begin());
239 thrust::copy(rmm::exec_policy()->on(0),
240 temporary_aggregation_buffer.begin(),
241 temporary_aggregation_buffer.end(), out_aggregation_column);
242 }
243
244 return GDF_SUCCESS;
245 }
