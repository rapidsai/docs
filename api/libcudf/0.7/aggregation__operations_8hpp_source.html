35 template < typename value_type>
36 struct max_op {
37 constexpr static value_type IDENTITY{std::numeric_limits<value_type>::lowest()};
38
39 CUDA_HOST_DEVICE_CALLABLE
40 value_type operator()(value_type new_value, value_type old_value)
41 {
42 return (new_value > old_value ? new_value : old_value);
43 }
44 };

51 CUDA_HOST_DEVICE_CALLABLE
