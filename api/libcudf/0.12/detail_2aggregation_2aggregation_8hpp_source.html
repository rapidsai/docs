257 #pragma nv_exec_check_disable
258 template < typename F, typename ... Ts>
259 CUDA_HOST_DEVICE_CALLABLE decltype( auto ) aggregation_dispatcher(
260 aggregation ::Kind k, F&& f, Ts&&... args) {
261 switch (k) {
262 case aggregation::SUM :
263 return f.template operator()< aggregation::SUM >(std::forward<Ts>(args)...);
264 case aggregation::MIN :
265 return f.template operator()< aggregation::MIN >(std::forward<Ts>(args)...);
266 case aggregation::MAX :
267 return f.template operator()< aggregation::MAX >(std::forward<Ts>(args)...);
268 case aggregation::COUNT :
269 return f.template operator()< aggregation::COUNT >(
270 std::forward<Ts>(args)...);
271 case aggregation::MEAN :
272 return f.template operator()< aggregation::MEAN >(
273 std::forward<Ts>(args)...);
274 case aggregation::VARIANCE :
275 return f.template operator()< aggregation::VARIANCE >(
276 std::forward<Ts>(args)...);
277 case aggregation::STD :
278 return f.template operator()< aggregation::STD >(
279 std::forward<Ts>(args)...);
280 case aggregation::MEDIAN :
281 return f.template operator()< aggregation::MEDIAN >(
282 std::forward<Ts>(args)...);
283 case aggregation::QUANTILE :
284 return f.template operator()< aggregation::QUANTILE >(
285 std::forward<Ts>(args)...);
286 case aggregation::ARGMAX :
287 return f.template operator()< aggregation::ARGMAX >(
288 std::forward<Ts>(args)...);
289 case aggregation::ARGMIN :
290 return f.template operator()< aggregation::ARGMIN >(
291 std::forward<Ts>(args)...);
292 default : {
293 #ifndef __CUDA_ARCH__
294 CUDF_FAIL( "Unsupported aggregation." );
295 #else
296 release_assert( false && "Unsupported aggregation." );
297
298 // The following code will never be reached, but the compiler generates a
299 // warning if there isn't a return value.
300
301 // Need to find out what the return type is in order to have a default
302 // return value and solve the compiler warning for lack of a default
303 // return
304 using return_type =
305 decltype(f.template operator ()< aggregation::SUM >(
306 std::forward<Ts>(args)...));
307 return return_type();
308 #endif
309 }
310 }
311 }

323 struct dispatch_source {

327 F&& f, Ts&&... args) const
