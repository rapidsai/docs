36 template <typename S>
37 inline S round_up_safe(S number_to_round, S modulus) {
38  auto remainder = number_to_round % modulus;
39  if (remainder == 0) { return number_to_round; }
40  auto rounded_up = number_to_round - remainder + modulus;
41  if (rounded_up < number_to_round) {
42  throw std::invalid_argument("Attempt to round up beyond the type's maximum value");
43  }
44  return rounded_up;
45 }

52 template <typename S>
53 inline S round_down_safe(S number_to_round, S modulus) {
54  auto remainder = number_to_round % modulus;
55  auto rounded_down = number_to_round - remainder;
56  return rounded_down;
57 }
