30  // return true if ColumnType is arithmetic type or
31  // AggOp is min_op/max_op/count_op for wrapper (non-arithmetic) types
32  template <typename ColumnType, class AggOp, cudf::experimental::aggregation::Kind op, bool is_mean>
33  static constexpr bool is_supported()
34  {
35  constexpr bool comparable_countable_op =
36  std::is_same<AggOp, DeviceMin>::value ||
37  std::is_same<AggOp, DeviceMax>::value ||
38  std::is_same<AggOp, DeviceCount>::value;
39 
40  constexpr bool timestamp_mean =
41  is_mean &&
42  std::is_same<AggOp, DeviceSum>::value &&
43  cudf::is_timestamp<ColumnType>();
44 
45  constexpr bool is_valid_rolling_agg = !std::is_same<ColumnType, cudf::string_view>::value and
46  ((op == experimental::aggregation::SUM) or
47  (op == experimental::aggregation::MIN) or
48  (op == experimental::aggregation::MAX) or
49  (op == experimental::aggregation::COUNT) or
50  (op == experimental::aggregation::MEAN)) and
51  (cudf::is_numeric<ColumnType>() or
52  comparable_countable_op or
53  timestamp_mean);
54 
55 
56  return is_valid_rolling_agg and
57  cudf::experimental::detail::is_valid_aggregation<ColumnType, op>();
58  //return is_valid_rolling_agg;
59  }

77  std::enable_if_t<!(cudf::is_boolean<T>() || cudf::is_timestamp<T>())>* = nullptr>
78  CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)

83  // SFINAE for bool type
84  template <typename T = _T, std::enable_if_t<cudf::is_boolean<T>()>* = nullptr>
85  CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
86  {
87  out = static_cast<int32_t>(val) / count;

89 
90  // SFINAE for timestamp types

92  CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
93  {
