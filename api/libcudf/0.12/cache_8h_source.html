167 template < typename T, typename FallbackFunc>
168 named_prog<T> getCached(
169 std::string const & name,
170 umap_str_shptr<T>& map,
171 FallbackFunc func) {
172
173 // Find memory cached T object
174 auto it = map.find(name);
175 if ( it != map.end()) {
176 return std::make_pair(name, it->second);
177 }
178 else { // Find file cached T object
179 bool successful_read = false ;
180 std::string serialized;
181 #if defined(JITIFY_USE_CACHE)
182 boost::filesystem::path file_name = getCacheDir() / name;
183 cacheFile file{file_name.string()};
184 serialized = file.read();
185 successful_read = file.is_read_successful();
186 #endif
187 if (not successful_read) {
188 // JIT compile and write to file if possible
189 serialized = func().serialize();
190 #if defined(JITIFY_USE_CACHE)
191 file.write(serialized);
192 #endif
193 }
194 // Add deserialized T to cache and return
195 auto program = std::make_shared<T>(T::deserialize(serialized));
196 map[name] = program;
197 return std::make_pair(name, program);
198 }
199 }
