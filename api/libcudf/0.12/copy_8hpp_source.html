43 template <typename ColumnView>
44 ColumnView slice(ColumnView const& input,
45  cudf::size_type begin,
46  cudf::size_type end) {
47  static_assert(std::is_same<ColumnView, cudf::column_view>::value or
48  std::is_same<ColumnView, cudf::mutable_column_view>::value,
49  "slice can be performed only on column_view and mutable_column_view");
50  CUDF_EXPECTS(begin >= 0, "Invalid beginning of range.");
51  CUDF_EXPECTS(end >= begin, "Invalid end of range.");
52  CUDF_EXPECTS(end <= input.size(), "Slice range out of bounds.");
53 
54  std::vector<ColumnView> children {};
55  children.reserve(input.num_children());
56  for (size_type index = 0; index < input.num_children(); index++) {
57  children.emplace_back(input.child(index));
58  }
59 
60  return ColumnView(input.type(), end - begin,
61  input.head(), input.null_mask(),
62  cudf::UNKNOWN_NULL_COUNT,
63  input.offset() + begin, children);
64 }
