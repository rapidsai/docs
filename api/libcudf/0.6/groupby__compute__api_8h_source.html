99 /* ----------------------------------------------------------------------------*/
100 __device__ bool operator() (key_type const & left_index,
101 key_type const & right_index) const
102 {
103 // The unused key is not a valid row index in the gdf_tables.
104 // Therefore, if comparing against the unused key, use the map's default
105 // comparison function
106 if ((unused_key == left_index) || (unused_key == right_index))
107 return default_comparator(left_index, right_index);
108
109 // Check for equality between the two rows of the two tables
110 return left_table. rows_equal (right_table, left_index, right_index);
111 }
112

114 const key_type unused_key;
115 map_type const & the_map;
116 gdf_table<size_type> const & left_table;
117 gdf_table<size_type> const & right_table;
118 };

139 template < typename aggregation_type,
140 typename size_type,
141 typename aggregation_operation>
142 gdf_error GroupbyHash( gdf_table<size_type> const & groupby_input_table,
143 const aggregation_type * const in_aggregation_column,
144 gdf_table<size_type> & groupby_output_table,
145 aggregation_type * out_aggregation_column,
146 size_type * out_size,
147 aggregation_operation aggregation_op,
148 bool sort_result = false )
149 {
150 const size_type input_num_rows = groupby_input_table.get_column_length();
151
152 // The map will store (row index, aggregation value)
153 // Where row index is the row number of the first row to be successfully inserted
154 // for a given unique row
155 using map_type = concurrent_unordered_map <size_type,
156 aggregation_type,
157 std::numeric_limits<size_type>::max(),
158 default_hash<size_type> ,
159 equal_to<size_type> ,
160 legacy_allocator<thrust::pair<size_type, aggregation_type> > >;
161
162 // The hash table occupancy and the input size determines the size of the hash table
163 // e.g., for a 50% occupancy, the size of the hash table is twice that of the input
164 const size_type hash_table_size = static_cast< size_type > (( static_cast< uint64_t > (input_num_rows) * 100 / DEFAULT_HASH_TABLE_OCCUPANCY));
165
166 // Initialize the hash table with the aggregation operation functor's identity value
167 std::unique_ptr<map_type> the_map( new map_type(hash_table_size, aggregation_operation::IDENTITY));
168
169 const dim3 build_grid_size ((input_num_rows + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
170 const dim3 block_size (THREAD_BLOCK_SIZE, 1, 1);
171
172 CUDA_TRY(cudaGetLastError());
173
174 // Inserts (i, aggregation_column[i]) as a key-value pair into the
175 // hash table. When a given key already exists in the table, the aggregation operation
176 // is computed between the new and existing value, and the result is stored back.
177 build_aggregation_table<<<build_grid_size, block_size>>>(the_map.get(),
178 groupby_input_table,
179 in_aggregation_column,
180 input_num_rows,
181 aggregation_op,
182 row_comparator<map_type, size_type> (*the_map, groupby_input_table, groupby_input_table));
183 CUDA_TRY(cudaGetLastError());
184
185 // Used by threads to coordinate where to write their results
186 size_type * global_write_index{ nullptr };
187 RMM_TRY(RMM_ALLOC(( void **)&global_write_index, sizeof (size_type), 0)); // TODO: non-default stream?
188 CUDA_TRY(cudaMemset(global_write_index, 0, sizeof (size_type)));
189
190 const dim3 extract_grid_size ((hash_table_size + THREAD_BLOCK_SIZE - 1) / THREAD_BLOCK_SIZE, 1, 1);
191
192 // Extracts every non-empty key and value into separate contiguous arrays,
193 // which provides the result of the groupby operation
194 extract_groupby_result<<<extract_grid_size, block_size>>>(the_map.get(),
195 hash_table_size,
196 groupby_output_table,
197 groupby_input_table,
198 out_aggregation_column,
199 global_write_index);
200
201 CUDA_TRY(cudaGetLastError());
202
203 // At the end of the extraction kernel, the global write index will be equal to
204 // the size of the output. Update the output size.
205 CUDA_TRY( cudaMemcpy(out_size, global_write_index, sizeof (size_type), cudaMemcpyDeviceToHost) );
206 RMM_TRY( RMM_FREE(global_write_index, 0) );
207 groupby_output_table.set_column_length(*out_size);
208
209 // Optionally sort the groupby/aggregation result columns
210 if ( true == sort_result) {
211
212 rmm::device_vector<gdf_index_type> sorted_indices(*out_size);
213 thrust::sequence(rmm::exec_policy()->on(0), sorted_indices.begin(), sorted_indices.end());
214
215 gdf_column sorted_indices_col;
216 gdf_error status = gdf_column_view (&sorted_indices_col, sorted_indices.data().get(),
217 nullptr , *out_size, GDF_INT32);
218 if (status != GDF_SUCCESS)
219 return status;
220
221 status = gdf_order_by(groupby_output_table.get_columns(), //input columns
222 nullptr ,
223 groupby_output_table.get_num_columns(), //number of columns in the first parameter (e.g. number of columsn to sort by)
224 &sorted_indices_col, //a gdf_column that is pre allocated for storing sorted indices
225 0); //flag to indicate if nulls are to be considered smaller than non-nulls or viceversa
226 if (status != GDF_SUCCESS)
227 return status;
228
229 // Reorder table according to indices from order_by
230 cudf::table result_table(groupby_output_table.get_columns(),
231 groupby_output_table.get_num_columns());
232 cudf::detail::gather(&result_table, sorted_indices.data().get(), &result_table);
233
234 rmm::device_vector<aggregation_type> temporary_aggregation_buffer(*out_size);
235 thrust::gather(rmm::exec_policy()->on(0),
236 sorted_indices.begin(), sorted_indices.end(),
237 out_aggregation_column,
238 temporary_aggregation_buffer.begin());
239 thrust::copy(rmm::exec_policy()->on(0),
240 temporary_aggregation_buffer.begin(),
241 temporary_aggregation_buffer.end(), out_aggregation_column);
242 }
243
244 return GDF_SUCCESS;
245 }
