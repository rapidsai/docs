56 template <typename index_type, typename size_type>
57 thrust::device_vector<index_type>
58 create_missing_indices(
59  index_type const * const index_ptr,
60  const size_type max_index_value,
61  const size_type index_size) {
62  //Assume all the indices in invalid_index_map are invalid
63  thrust::device_vector<index_type> invalid_index_map(max_index_value, 1);
64  //Vector allocated for unmatched result
65  thrust::device_vector<index_type> unmatched_indices(max_index_value);
66  //lambda to check for index validity since left joins can create invalid indices
67  auto bounds_checker =
68  [max_index_value] __device__(gdf_index_type index) {
69  return ((index >= 0) && (index < max_index_value));
70  };
71 
72  //invalid_index_map[index_ptr[i]] = 0 for i = 0 to max_index_value
73  //Thus specifying that those locations are valid
74  thrust::scatter_if(
75  thrust::device,
76  thrust::make_constant_iterator(0),
77  thrust::make_constant_iterator(0) + index_size,
78  index_ptr,//Index locations
79  index_ptr,//Stencil - Check if index location is valid
80  invalid_index_map.begin(),//Output indices
81  bounds_checker);//Stencil Predicate
82  CHECK_STREAM(0);
83  size_type begin_counter = static_cast<size_type>(0);
84  size_type end_counter = static_cast<size_type>(invalid_index_map.size());
85  //Create list of indices that have been marked as invalid
86  size_type compacted_size = thrust::copy_if(
87  thrust::device,
88  thrust::make_counting_iterator(begin_counter),
89  thrust::make_counting_iterator(end_counter),
90  invalid_index_map.begin(),
91  unmatched_indices.begin(),
92  thrust::identity<index_type>()) -
93  unmatched_indices.begin();
94  CHECK_STREAM(0);
95  unmatched_indices.resize(compacted_size);
96  return unmatched_indices;
97 }

114 template <typename data_type, typename size_type>
115 gdf_error expand_buffer(
116  data_type ** buffer,
117  size_type * const buffer_capacity,
118  const size_type buffer_size,
119  const size_type expand_size) {
120  size_type requested_size = buffer_size + expand_size;
121  //No need to proceed if the buffer can contain requested additional elements
122  if (*buffer_capacity >= requested_size) {
123  return GDF_SUCCESS;
124  }
125  data_type * new_buffer{nullptr};
126  data_type * old_buffer = *buffer;
127  RMM_TRY( RMM_ALLOC((void**)&new_buffer, requested_size*sizeof(data_type), 0) );
128  CUDA_TRY( cudaMemcpy(new_buffer, old_buffer, buffer_size*sizeof(data_type), cudaMemcpyDeviceToDevice) );
129  RMM_TRY( RMM_FREE(old_buffer, 0) );
130  *buffer = new_buffer;
131  *buffer_capacity = requested_size;
132 
133  return GDF_SUCCESS;
134 }

153 template <typename index_type, typename size_type>
154 gdf_error append_full_join_indices(
155  index_type ** l_index_ptr,
156  index_type ** r_index_ptr,
157  size_type * const index_capacity,
158  size_type * const index_size,
159  const size_type max_index_value) {
160  gdf_error err;
161  //Get array of indices that do not appear in r_index_ptr
162  thrust::device_vector<index_type> unmatched_indices =
163  create_missing_indices(
164  *r_index_ptr, max_index_value, *index_size);
165  CUDA_CHECK_LAST()
166 
167  //Expand l_index_ptr and r_index_ptr if necessary
168  size_type mismatch_index_size = unmatched_indices.size();
169  size_type l_index_capacity = *index_capacity;
170  size_type r_index_capacity = *index_capacity;
171  err = expand_buffer(l_index_ptr, &l_index_capacity, *index_size, mismatch_index_size);
172  if (GDF_SUCCESS != err) return err;
173  err = expand_buffer(r_index_ptr, &r_index_capacity, *index_size, mismatch_index_size);
174  if (GDF_SUCCESS != err) return err;
175 
176  //Copy JoinNoneValue to l_index_ptr to denote that a match does not exist on the left
177  thrust::fill(
178  thrust::device,
179  *l_index_ptr + *index_size,
180  *l_index_ptr + *index_size + mismatch_index_size,
181  JoinNoneValue);
182 
183  //Copy unmatched indices to the r_index_ptr
184  thrust::copy(thrust::device,
185  unmatched_indices.begin(),
186  unmatched_indices.begin() + mismatch_index_size,
187  *r_index_ptr + *index_size);
188  *index_capacity = l_index_capacity;
189  *index_size = *index_size + mismatch_index_size;
190 
191  CUDA_CHECK_LAST()
192  return GDF_SUCCESS;
193 }

211 template <JoinType join_type,
212  typename multimap_type,
213  typename size_type>
214 gdf_error estimate_join_output_size(gdf_table<size_type> const & build_table,
215  gdf_table<size_type> const & probe_table,
216  multimap_type const & hash_table,
217  size_type * join_output_size_estimate)
218 {
219  const size_type build_table_num_rows{build_table.get_column_length()};
220  const size_type probe_table_num_rows{probe_table.get_column_length()};
221 
222  // If the probe table is significantly larger (5x) than the build table,
223  // then we attempt to only use a subset of the probe table rows to compute an
224  // estimate of the join output size.
225  size_type probe_to_build_ratio{0};
226  if(build_table_num_rows > 0) {
227  probe_to_build_ratio = static_cast<size_type>(std::ceil(static_cast<float>(probe_table_num_rows)/build_table_num_rows));
228  }
229  else {
230  // If the build table is empty, we know exactly how large the output
231  // will be for the different types of joins and can return immediately
232  switch(join_type)
233  {
234  case JoinType::INNER_JOIN:
235  {
236  // Inner join with an empty table will have no output
237  *join_output_size_estimate = 0;
238  break;
239  }
240  case JoinType::LEFT_JOIN:
241  {
242  // Left join with an empty table will have an output of NULL rows
243  // equal to the number of rows in the probe table
244  *join_output_size_estimate = probe_table_num_rows;
245  break;
246  }
247  default:
248  return GDF_UNSUPPORTED_JOIN_TYPE;
249  }
250  return GDF_SUCCESS;
251  }
252 
253  size_type sample_probe_num_rows{probe_table_num_rows};
254  constexpr size_type MAX_RATIO{5};
255  if(probe_to_build_ratio > MAX_RATIO)
256  {
257  sample_probe_num_rows = build_table_num_rows;
258  }
259 
260  // Allocate storage for the counter used to get the size of the join output
261  size_type * d_size_estimate{nullptr};
262  size_type h_size_estimate{0};
263 
264  CUDA_TRY(cudaMallocHost(&d_size_estimate, sizeof(size_type)));
265  *d_size_estimate = 0;
266 
267  CUDA_TRY( cudaGetLastError() );
268 
269  // Continue probing with a subset of the probe table until either:
270  // a non-zero output size estimate is found OR
271  // all of the rows in the probe table have been sampled
272  do{
273 
274  sample_probe_num_rows = std::min(sample_probe_num_rows, probe_table_num_rows);
275 
276  *d_size_estimate = 0;
277 
278  constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
279  const size_type probe_grid_size{(sample_probe_num_rows + block_size -1)/block_size};
280 
281  // Probe the hash table without actually building the output to simply
282  // find what the size of the output will be.
283  compute_join_output_size<join_type,
284  multimap_type,
285  size_type,
286  block_size,
287  DEFAULT_CUDA_CACHE_SIZE>
288  <<<probe_grid_size, block_size>>>(&hash_table,
289  build_table,
290  probe_table,
291  sample_probe_num_rows,
292  d_size_estimate);
293 
294  // Device sync is required to ensure d_size_estimate is updated
295  CUDA_TRY( cudaDeviceSynchronize() );
296 
297 
298  // Increase the estimated output size by a factor of the ratio between the
299  // probe and build tables
300  h_size_estimate = *d_size_estimate * probe_to_build_ratio;
301 
302  // If the size estimate is non-zero, then we have a valid estimate and can break
303  // If sample_probe_num_rows >= probe_table_num_rows, then we've sampled the entire
304  // probe table, in which case the estimate is exact and we can break
305  if((h_size_estimate > 0)
306  || (sample_probe_num_rows >= probe_table_num_rows))
307  {
308  break;
309  }
310 
311  // If the size estimate is zero, then double the number of sampled rows in the probe
312  // table. Reduce the ratio of the number of probe rows sampled to the
313  // number of rows in the build table by the same factor
314  if(0 == h_size_estimate)
315  {
316  constexpr size_type GROW_RATIO{2};
317  sample_probe_num_rows *= GROW_RATIO;
318  probe_to_build_ratio = static_cast<size_type>(std::ceil(static_cast<float>(probe_to_build_ratio)/GROW_RATIO));
319  }
320 
321  } while(true);
322 
323  CUDA_TRY( cudaFreeHost(d_size_estimate) );
324 
325  *join_output_size_estimate = h_size_estimate;
326 
327  return GDF_SUCCESS;
328 }

341 inline size_t compute_hash_table_size(
342  gdf_size_type num_keys_to_insert,
343  uint32_t desired_occupancy = DEFAULT_HASH_TABLE_OCCUPANCY) {
344  assert(desired_occupancy != 0);
345  assert(desired_occupancy <= 100);
346  double const grow_factor{100.0 / desired_occupancy};
347 
348  // Calculate size of hash map based on the desired occupancy
349  size_t hash_table_size{
350  static_cast<size_t>(std::ceil(num_keys_to_insert * grow_factor))};
351 
352  return hash_table_size;

373 template<JoinType join_type,
374  typename output_index_type,
375  typename size_type>
376 gdf_error compute_hash_join(
377  gdf_column * const output_l,
378  gdf_column * const output_r,
379  gdf_table<size_type> const & left_table,
380  gdf_table<size_type> const & right_table,
381  bool flip_results = false)
382 {
383  gdf_error gdf_error_code{GDF_SUCCESS};
384 
385  gdf_column_view(output_l, nullptr, nullptr, 0, N_GDF_TYPES);
386  gdf_column_view(output_r, nullptr, nullptr, 0, N_GDF_TYPES);
387 
388  // The LEGACY allocator allocates the hash table array with normal cudaMalloc,
389  // the non-legacy allocator uses managed memory
390 #ifdef HT_LEGACY_ALLOCATOR
391  using multimap_type = concurrent_unordered_multimap<hash_value_type,
392  output_index_type,
393  size_type,
394  std::numeric_limits<hash_value_type>::max(),
395  std::numeric_limits<output_index_type>::max(),
396  default_hash<hash_value_type>,
397  equal_to<hash_value_type>,
398  legacy_allocator< thrust::pair<hash_value_type, output_index_type> > >;
399 #else
400  using multimap_type = concurrent_unordered_multimap<hash_value_type,
401  output_index_type,
402  size_type,
403  std::numeric_limits<hash_value_type>::max(),
404  std::numeric_limits<output_index_type>::max()>;
405 #endif
406  //If FULL_JOIN is selected then we process as LEFT_JOIN till we need to take care of unmatched indices
407  constexpr JoinType base_join_type = (join_type == JoinType::FULL_JOIN)? JoinType::LEFT_JOIN : join_type;
408 
409  // Hash table will be built on the right table
410  gdf_table<size_type> const & build_table{right_table};
411  const size_type build_table_num_rows{build_table.get_column_length()};
412 
413  // Probe with the left table
414  gdf_table<size_type> const & probe_table{left_table};
415  const size_type probe_table_num_rows{probe_table.get_column_length()};
416 
417  // Hash table size must be at least 1 in order to have a valid allocation.
418  // Even if the hash table will be empty, it still must be allocated for the
419  // probing phase in the event of an outer join
420  size_t const hash_table_size =
421  std::max(compute_hash_table_size(build_table_num_rows), size_t{1});
422 
423  std::unique_ptr<multimap_type> hash_table(new multimap_type(hash_table_size));
424 
425  // FIXME: use GPU device id from the context?
426  // but moderngpu only provides cudaDeviceProp
427  // (although should be possible once we move to Arrow)
428  hash_table->prefetch(0);
429 
430  CUDA_TRY( cudaDeviceSynchronize() );
431 
432  // Allocate a gdf_error for the device to hold error code returned from
433  // the build kernel and intialize with GDF_SUCCESS
434  // Use Page Locked memory to avoid overhead of memcpys
435  gdf_error * d_gdf_error_code{nullptr};
436  CUDA_TRY( cudaMallocHost(&d_gdf_error_code, sizeof(gdf_error)) );
437  *d_gdf_error_code = GDF_SUCCESS;
438 
439  constexpr int block_size{DEFAULT_CUDA_BLOCK_SIZE};
440 
441  // build the hash table
442  if(build_table_num_rows > 0)
443  {
444  const size_type build_grid_size{(build_table_num_rows + block_size - 1)/block_size};
445  build_hash_table<<<build_grid_size, block_size>>>(hash_table.get(),
446  build_table,
447  build_table_num_rows,
448  d_gdf_error_code);
449 
450  // Device synch is required to ensure d_gdf_error_code
451  // has been written
452  CUDA_TRY( cudaDeviceSynchronize() );
453  }
454 
455  // Check error code from the kernel
456  gdf_error_code = *d_gdf_error_code;
457  if(GDF_SUCCESS != gdf_error_code){
458  return gdf_error_code;
459  }
460 
461 
462  size_type estimated_join_output_size{0};
463  gdf_error_code = estimate_join_output_size<base_join_type, multimap_type>(build_table, probe_table, *hash_table, &estimated_join_output_size);
464 
465  if(GDF_SUCCESS != gdf_error_code){
466  return gdf_error_code;
467  }
468 
469  // If the estimated output size is zero, return immediately
470  if(0 == estimated_join_output_size){
471  return GDF_SUCCESS;
472  }
473 
474  // Because we are approximating the number of joined elements, our approximation
475  // might be incorrect and we might have underestimated the number of joined elements.
476  // As such we will need to de-allocate memory and re-allocate memory to ensure
477  // that the final output is correct.
478  size_type h_actual_found{0};
479  output_index_type *output_l_ptr{nullptr};
480  output_index_type *output_r_ptr{nullptr};
481  bool cont = true;
482 
483  // Allocate device global counter used by threads to determine output write location
484  size_type *d_global_write_index{nullptr};
485  RMM_TRY( RMM_ALLOC((void**)&d_global_write_index, sizeof(size_type), 0) ); // TODO non-default stream?
486 
487  // Because we only have an estimate of the output size, we may need to probe the
488  // hash table multiple times until we've found an output buffer size that is large enough
489  // to hold the output
490  while(cont)
491  {
492  output_l_ptr = nullptr;
493  output_r_ptr = nullptr;
494 
495  // Allocate temporary device buffer for join output
496  RMM_TRY( RMM_ALLOC((void**)&output_l_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
497  RMM_TRY( RMM_ALLOC((void**)&output_r_ptr, estimated_join_output_size*sizeof(output_index_type), 0) );
498  CUDA_TRY( cudaMemsetAsync(d_global_write_index, 0, sizeof(size_type), 0) );
499 
500  const size_type probe_grid_size{(probe_table_num_rows + block_size -1)/block_size};
501 
502  // Do the probe of the hash table with the probe table and generate the output for the join
503  probe_hash_table<base_join_type,
504  multimap_type,
505  hash_value_type,
506  size_type,
507  output_index_type,
508  block_size,
509  DEFAULT_CUDA_CACHE_SIZE>
510  <<<probe_grid_size, block_size>>> (hash_table.get(),
511  build_table,
512  probe_table,
513  probe_table.get_column_length(),
514  output_l_ptr,
515  output_r_ptr,
516  d_global_write_index,
517  estimated_join_output_size,
518  flip_results);
519 
520  CUDA_TRY( cudaGetLastError() );
521 
522  CUDA_TRY( cudaMemcpy(&h_actual_found, d_global_write_index, sizeof(size_type), cudaMemcpyDeviceToHost));
523 
524  // The estimate was too small. Double the estimate and try again
525  if(estimated_join_output_size < h_actual_found){
526  cont = true;
527  estimated_join_output_size *= 2;
528  // Free the old buffers to prevent a memory leak on the new allocation
529  RMM_TRY( RMM_FREE(output_l_ptr, 0) );
530  RMM_TRY( RMM_FREE(output_r_ptr, 0) );
531  }
532  else
533  {
534  cont = false;
535  }
536  }
537 
538  // free memory used for the counters
539  RMM_TRY( RMM_FREE(d_global_write_index, 0) );
540 
541  if (join_type == JoinType::FULL_JOIN) {
542  append_full_join_indices(
543  &output_l_ptr, &output_r_ptr,
544  &estimated_join_output_size,
545  &h_actual_found, build_table_num_rows);
546  }
547 
548  // If the estimated join output size was larger than the actual output size,
549  // then the buffers are larger than necessary. Allocate buffers of the actual
550  // output size and copy the results to the buffers of the correct size
551  // FIXME Is this really necessary? It's probably okay to have the buffers be oversized
552  // and avoid the extra allocation/memcopy
553  if (estimated_join_output_size > h_actual_found) {
554  output_index_type *copy_output_l_ptr{nullptr};
555  output_index_type *copy_output_r_ptr{nullptr};
556  RMM_TRY( RMM_ALLOC((void**)&copy_output_l_ptr, h_actual_found*sizeof(output_index_type), 0) ); // TODO non-default stream?
557  RMM_TRY( RMM_ALLOC((void**)&copy_output_r_ptr, h_actual_found*sizeof(output_index_type), 0) );
558  CUDA_TRY( cudaMemcpy(copy_output_l_ptr, output_l_ptr, h_actual_found*sizeof(output_index_type), cudaMemcpyDeviceToDevice) );
559  CUDA_TRY( cudaMemcpy(copy_output_r_ptr, output_r_ptr, h_actual_found*sizeof(output_index_type), cudaMemcpyDeviceToDevice) );
560  RMM_TRY( RMM_FREE(output_l_ptr, 0) );
561  RMM_TRY( RMM_FREE(output_r_ptr, 0) );
562  output_l_ptr = copy_output_l_ptr;
563  output_r_ptr = copy_output_r_ptr;
564  }
565 
566  // Free the device error code
567  CUDA_TRY( cudaFreeHost(d_gdf_error_code) );
568 
569  // Deduce the type of the output gdf_columns
570  gdf_dtype dtype;
571  switch(sizeof(output_index_type))
572  {
573  case 1 : dtype = GDF_INT8; break;
574  case 2 : dtype = GDF_INT16; break;
575  case 4 : dtype = GDF_INT32; break;
576  case 8 : dtype = GDF_INT64; break;
577  }
578  gdf_column_view(output_l, output_l_ptr, nullptr, h_actual_found, dtype);
579  gdf_column_view(output_r, output_r_ptr, nullptr, h_actual_found, dtype);
580 
581  return gdf_error_code;
582 }
