344  {
345  const std::string uri_start_delimiter = "//";
346  const std::string uri_end_delimiter = "?";
347 
348  arrow::Result<std::shared_ptr<arrow::fs::FileSystem>> result =
349  arrow::fs::FileSystemFromUri(static_cast<std::string>(arrow_uri));
350  CUDF_EXPECTS(result.ok(), "Failed to generate Arrow Filesystem instance from URI.");
351  filesystem = result.ValueOrDie();
352 
353  // Parse the path from the URI
354  size_t start = arrow_uri.find(uri_start_delimiter) == std::string::npos
355  ? 0
356  : arrow_uri.find(uri_start_delimiter) + uri_start_delimiter.size();
357  size_t end = arrow_uri.find(uri_end_delimiter) - start;
358  std::string_view path = arrow_uri.substr(start, end);
359 
360  arrow::Result<std::shared_ptr<arrow::io::RandomAccessFile>> in_stream =
361  filesystem->OpenInputFile(static_cast<std::string>(path).c_str());
362  CUDF_EXPECTS(in_stream.ok(), "Failed to open Arrow RandomAccessFile");
363  arrow_file = in_stream.ValueOrDie();
364  }
