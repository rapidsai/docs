44 template <typename T, bool nullable = false>
45 inline cudf::test::fixed_width_column_wrapper<T, int64_t> generate_timestamps(int32_t count,
46  time_point_ms start,
47  time_point_ms stop)
48 {
49  using Rep = typename T::rep;
50  using Period = typename T::period;
51  using ToDuration = cuda::std::chrono::duration<Rep, Period>;
52 
53  auto lhs = start.time_since_epoch().count();
54  auto rhs = stop.time_since_epoch().count();
55 
56  // When C++17, auto [min, max] = std::minmax(lhs, rhs)
57  auto min = std::min(lhs, rhs);
58  auto max = std::max(lhs, rhs);
59  auto range = max - min;
60  auto iter = cudf::test::make_counting_transform_iterator(0, [=](auto i) {
61  return cuda::std::chrono::floor<ToDuration>(
62  cuda::std::chrono::milliseconds(min + (range / count) * i))
63  .count();
64  });
65 
66  if (nullable) {
67  auto mask = cudf::test::make_counting_transform_iterator(0, [](auto i) { return i % 2 == 0; });
68  return cudf::test::fixed_width_column_wrapper<T, int64_t>(iter, iter + count, mask);
69  } else {
70  // This needs to be in an else to quash `statement_not_reachable` warnings
71  return cudf::test::fixed_width_column_wrapper<T, int64_t>(iter, iter + count);
72  }
73 }
