115 template <typename From, typename To>
116 struct fixed_width_type_converter {
117  // Are the types same - simply copy elements from [begin, end) to out
118  template <typename FromT = From,
119  typename ToT = To,
120  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
121  ToT operator()(FromT element) const
122  {
123  return element;
124  }
125 
126  // Are the types convertible or can target be constructed from source?
127  template <typename FromT = From,
128  typename ToT = To,
129  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
130  (cudf::is_convertible<FromT, ToT>::value ||
131  std::is_constructible<ToT, FromT>::value),
132  void>::type* = nullptr>
133  ToT operator()(FromT element) const
134  {
135  return static_cast<ToT>(element);
136  }
137 
138  // Convert integral values to timestamps
139  template <
140  typename FromT = From,
141  typename ToT = To,
142  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
143  void>::type* = nullptr>
144  ToT operator()(FromT element) const
145  {
146  return ToT{typename ToT::duration{element}};
147  }
148 };

288 template <typename StringsIterator, typename ValidityIterator>
289 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
290 {
291  std::vector<char> chars{};
292  std::vector<cudf::size_type> offsets(1, 0);
293  for (auto str = begin; str < end; ++str) {
294  std::string tmp = (*v++) ? std::string(*str) : std::string{};
295  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
296  offsets.push_back(offsets.back() + tmp.length());
297  }
298  return std::make_pair(std::move(chars), std::move(offsets));
299 };

1622  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1623  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1624  column_view& expected_hierarchy,
1625  int expected_depth)
1626  {
1627  std::vector<std::unique_ptr<column>> stubs;
1628  std::vector<column_view> cols;
1629 
1630  // preprocess the incoming lists.
1631  // - unwrap any "root" lists
1632  // - handle incomplete hierarchies
1633  std::transform(elements.begin(),
1634  elements.end(),
1635  std::back_inserter(cols),
1636  [&](lists_column_wrapper const& l) -> column_view {
1637  // depth mismatch. attempt to normalize the short column.
1638  // this function will also catch if this is a legitmately broken
1639  // set of input
1640  if (l.depth < expected_depth) {
1641  if (l.root) {
1642  // this exception distinguishes between the following two cases:
1643  //
1644  // { {{{1, 2, 3}}}, {} }
1645  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1646  // just a List<> which is an apparent mismatch. However, because row 1
1647  // is empty we will allow that to semantically mean
1648  // "a List<List<List<int>>> that's empty at the top level"
1649  //
1650  // { {{{1, 2, 3}}}, {4, 5, 6} }
1651  // In this case, row 1 is a a concrete List<int> with actual values.
1652  // There is no way to rectify the differences so we will treat it as a
1653  // true column mismatch.
1654  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1655  stubs.push_back(empty_like(expected_hierarchy));
1656  } else {
1657  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1658  }
1659  return *(stubs.back());
1660  }
1661  // the empty hierarchy case
1662  return l.get_view();
1663  });
1664 
1665  return {std::move(cols), std::move(stubs)};
1666  }
1667 
1668  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1669 
1670  int depth = 0;
1671  bool root = false;
