312 template <typename T>
313 void throw_or_mimic(const std::string& text)
314 {
315  static_assert(std::is_base_of<std::exception, T>::value,
316  "throw_or_mimic only works on std::exception and "
317  "deriving classes");
318 
319 #ifndef CXXOPTS_NO_EXCEPTIONS
320  // If CXXOPTS_NO_EXCEPTIONS is not defined, just throw
321  throw T{text};
322 #else
323  // Otherwise manually instantiate the exception, print what() to stderr,
324  // and abort
325  T exception{text};
326  std::cerr << exception.what() << std::endl;
327  std::cerr << "Aborting (exceptions disabled)..." << std::endl;
328  std::abort();
329 #endif
330 }

375 template <typename R, typename T>
376 R checked_negate(T&& t, const std::string&, std::true_type)
377 {
378  // if we got to here, then `t` is a positive number that fits into
379  // `R`. So to avoid MSVC C4146, we first cast it to `R`.
380  // See https://github.com/jarro2783/cxxopts/issues/62 for more details.
381  return static_cast<R>(-static_cast<R>(t - 1) - 1);

391 template <typename T>
392 void integer_parser(const std::string& text, T& value)
393 {
394  std::smatch match;
395  std::regex_match(text, match, integer_pattern);
396 
397  if (match.length() == 0) { throw_or_mimic<argument_incorrect_type>(text); }
398 
399  if (match.length(4) > 0) {
400  value = 0;
401  return;
402  }
403 
404  using US = typename std::make_unsigned<T>::type;
405 
406  constexpr bool is_signed = std::numeric_limits<T>::is_signed;
407  const bool negative = match.length(1) > 0;
408  const uint8_t base = match.length(2) > 0 ? 16 : 10;
409 
410  auto value_match = match[3];
411 
412  US result = 0;
413 
414  for (auto iter = value_match.first; iter != value_match.second; ++iter) {
415  US digit = 0;
416 
417  if (*iter >= '0' && *iter <= '9') {
418  digit = static_cast<US>(*iter - '0');
419  } else if (base == 16 && *iter >= 'a' && *iter <= 'f') {
420  digit = static_cast<US>(*iter - 'a' + 10);
421  } else if (base == 16 && *iter >= 'A' && *iter <= 'F') {
422  digit = static_cast<US>(*iter - 'A' + 10);
423  } else {
424  throw_or_mimic<argument_incorrect_type>(text);
425  }
426 
427  const US next = static_cast<US>(result * base + digit);
428  if (result > next) { throw_or_mimic<argument_incorrect_type>(text); }
429 
430  result = next;
431  }
432 
433  detail::check_signed_range<T>(negative, result, text);
434 
435  if (negative) {
436  value = checked_negate<T>(result, text, std::integral_constant<bool, is_signed>());
437  } else {
438  value = static_cast<T>(result);
439  }
440 }

973 String format_option(const HelpOptionDetails& o)
974 {
975  auto& s = o.s;
976  auto& l = o.l;
977 
978  String result = " ";
979 
980  if (s.size() > 0) {
981  result += "-" + toLocalString(s) + ",";
982  } else {
983  result += " ";
984  }
985 
986  if (l.size() > 0) { result += " --" + toLocalString(l); }
987 
988  auto arg = o.arg_help.size() > 0 ? toLocalString(o.arg_help) : "arg";
989 
990  if (!o.is_boolean) {
991  if (o.has_implicit) {

993  } else {

1001 String format_description(const HelpOptionDetails& o, size_t start, size_t width)
1002 {
1003  auto desc = o.desc;
1004 
1005  if (o.has_default && (!o.is_boolean || o.default_value != "false")) {
1006  if (o.default_value != "") {
1007  desc += toLocalString(" (default: " + o.default_value + ")");
1008  } else {
1009  desc += toLocalString(" (default: \"\")");
1010  }
1011  }
1012 
1013  String result;
1014 
1015  auto current = std::begin(desc);
1016  auto startLine = current;
1017  auto lastSpace = current;
1018 
1019  auto size = size_t{};
1020 
1021  while (current != std::end(desc)) {
1022  if (*current == ' ') { lastSpace = current; }
1023 
1024  if (*current == '\n') {
1025  startLine = current + 1;
1026  lastSpace = startLine;
1027  } else if (size > width) {
1028  if (lastSpace == startLine) {
1029  stringAppend(result, startLine, current + 1);
1030  stringAppend(result, "\n");
1031  stringAppend(result, start, ' ');
1032  startLine = current + 1;
1033  lastSpace = startLine;
1034  } else {
1035  stringAppend(result, startLine, lastSpace);
1036  stringAppend(result, "\n");
1037  stringAppend(result, start, ' ');
1038  startLine = lastSpace + 1;
1039  lastSpace = startLine;
1040  }
1041  size = 0;
1042  } else {
1043  ++size;
1044  }
1045 
1046  ++current;
1047  }
1048 
1049  // append whatever is left
1050  stringAppend(result, startLine, current);
1051 
1052  return result;
1053 }

1218 inline void ParseResult::parse(int& argc, char**& argv)
1219 {
1220  int current = 1;
1221 
1222  int nextKeep = 1;
1223 
1224  bool consume_remaining = false;
1225 
1226  while (current != argc) {
1227  if (strcmp(argv[current], "--") == 0) {
1228  consume_remaining = true;
1229  ++current;
1230  break;
1231  }
1232 
1233  std::match_results<const char*> result;
1234  std::regex_match(argv[current], result, option_matcher);
1235 
1236  if (result.empty()) {
1237  // not a flag
1238 
1239  // but if it starts with a `-`, then it's an error
1240  if (argv[current][0] == '-' && argv[current][1] != '\0') {
1241  if (!m_allow_unrecognised) { throw_or_mimic<option_syntax_exception>(argv[current]); }
1242  }
1243 
1244  // if true is returned here then it was consumed, otherwise it is
1245  // ignored
1246  if (consume_positional(argv[current])) {
1247  } else {
1248  argv[nextKeep] = argv[current];
1249  ++nextKeep;
1250  }
1251  // if we return from here then it was parsed successfully, so continue
1252  } else {
1253  // short or long option?
1254  if (result[4].length() != 0) {
1255  const std::string& s = result[4];
1256 
1257  for (std::size_t i = 0; i != s.size(); ++i) {
1258  std::string name(1, s[i]);
1259  auto iter = m_options->find(name);
1260 
1261  if (iter == m_options->end()) {
1262  if (m_allow_unrecognised) {
1263  continue;
1264  } else {
1265  // error
1266  throw_or_mimic<option_not_exists_exception>(name);
1267  }
1268  }
1269 
1270  auto value = iter->second;
1271 
1272  if (i + 1 == s.size()) {
1273  // it must be the last argument
1274  checked_parse_arg(argc, argv, current, value, name);
1275  } else if (value->value().has_implicit()) {
1276  parse_option(value, name, value->value().get_implicit_value());
1277  } else {
1278  // error
1279  throw_or_mimic<option_requires_argument_exception>(name);
1280  }
1281  }
1282  } else if (result[1].length() != 0) {
1283  const std::string& name = result[1];
1284 
1285  auto iter = m_options->find(name);
1286 
1287  if (iter == m_options->end()) {
1288  if (m_allow_unrecognised) {
1289  // keep unrecognised options in argument list, skip to next argument
1290  argv[nextKeep] = argv[current];
1291  ++nextKeep;
1292  ++current;
1293  continue;
1294  } else {
1295  // error
1296  throw_or_mimic<option_not_exists_exception>(name);
1297  }
1298  }
1299 
1300  auto opt = iter->second;
1301 
1302  // equals provided for long option?
1303  if (result[2].length() != 0) {
1304  // parse the option given
1305 
1306  parse_option(opt, name, result[3]);
1307  } else {
1308  // parse the next argument
1309  checked_parse_arg(argc, argv, current, opt, name);
1310  }
1311  }
1312  }
1313 
1314  ++current;
1315  }
1316 
1317  for (auto& opt : *m_options) {
1318  auto& detail = opt.second;
1319  auto& value = detail->value();
1320 
1321  auto& store = m_results[detail];
1322 
1323  if (value.has_default() && !store.count() && !store.has_default()) { parse_default(detail); }
1324  }
1325 
1326  if (consume_remaining) {
1327  while (current < argc) {
1328  if (!consume_positional(argv[current])) { break; }
1329  ++current;
1330  }
1331 
1332  // adjust argv for any that couldn't be swallowed
1333  while (current != argc) {
1334  argv[nextKeep] = argv[current];
1335  ++nextKeep;
1336  ++current;
1337  }
1338  }
1339 
1340  argc = nextKeep;
1341 }

1385 inline String Options::help_one_group(const std::string& g) const
1386 {
1387  typedef std::vector<std::pair<String, String>> OptionHelp;
1388 
1389  auto group = m_help.find(g);
1390  if (group == m_help.end()) { return ""; }
1391 
1392  OptionHelp format;
1393 
1394  size_t longest = 0;
1395 
1396  String result;
1397 
1398  if (!g.empty()) { result += toLocalString(" " + g + " options:\n"); }
1399 
1400  for (const auto& o : group->second.options) {
1401  if (m_positional_set.find(o.l) != m_positional_set.end() && !m_show_positional) { continue; }
1402 
1403  auto s = format_option(o);
1404  longest = (std::max)(longest, stringLength(s));
1405  format.push_back(std::make_pair(s, String()));
1406  }
1407 
1408  longest = (std::min)(longest, static_cast<size_t>(OPTION_LONGEST));
1409 
1410  // widest allowed description
1411  auto allowed = size_t{76} - longest - OPTION_DESC_GAP;
1412
