351 constexpr size_type bytes_in_char_utf8(char_utf8 character)
352 {
353  return 1 + static_cast<size_type>((character & unsigned{0x0000FF00}) > 0) +

368 constexpr size_type bytes_in_utf8_byte(uint8_t byte)
369 {
370  return 1 + static_cast<size_type>((byte & 0xF0) == 0xF0) // 4-byte character prefix
371  + static_cast<size_type>((byte & 0xE0) == 0xE0) // 3-byte character prefix
372  + static_cast<size_type>((byte & 0xC0) == 0xC0) // 2-byte character prefix
373  - static_cast<size_type>((byte & 0xC0) == 0x80); // intermediate byte
374 }

383 CUDA_HOST_DEVICE_CALLABLE size_type to_char_utf8(const char* str, char_utf8& character)
384 {
385  size_type const chr_width = bytes_in_utf8_byte(static_cast<uint8_t>(*str));
386 
387  character = static_cast<char_utf8>(*str++) & 0xFF;
388  if (chr_width > 1) {
389  character = character << 8;
390  character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 8;
391  if (chr_width > 2) {
392  character = character << 8;
393  character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 16;
394  if (chr_width > 3) {
395  character = character << 8;
396  character |= (static_cast<char_utf8>(*str++) & 0xFF); // << 24;
397  }
398  }
399  }
400  return chr_width;
401 }

410 CUDA_HOST_DEVICE_CALLABLE size_type from_char_utf8(char_utf8 character, char* str)
411 {
412  size_type const chr_width = bytes_in_char_utf8(character);
413  for (size_type idx = 0; idx < chr_width; ++idx) {
414  str[chr_width - idx - 1] = static_cast<char>(character) & 0xFF;
415  character = character >> 8;
416  }
417  return chr_width;
418 }
