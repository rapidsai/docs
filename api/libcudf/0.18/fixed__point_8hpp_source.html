86 template <typename Rep,
87  Radix Base,
88  typename T,
89  typename cuda::std::enable_if_t<(cuda::std::is_same<int32_t, T>::value &&
90  is_supported_representation_type<Rep>())>* = nullptr>
91 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
92 {
93  release_assert(exponent >= 0 && "integer exponentiation with negative exponent is not possible.");
94  if (exponent == 0) return static_cast<Rep>(1);
95  auto extra = static_cast<Rep>(1);
96  auto square = static_cast<Rep>(Base);
97  while (exponent > 1) {
98  if (exponent & 1 /* odd */) {
99  extra *= square;
100  exponent -= 1;
101  }
102  exponent /= 2;
103  square *= square;
104  }
105  return square * extra;
106 }

607 template <typename Rep, typename T>
608 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)
609 {
610  return lhs == cuda::std::numeric_limits<Rep>::min() && rhs == -1;
611 }

621 template <typename Rep, typename T>
622 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
623 {
624  auto const min = cuda::std::numeric_limits<Rep>::min();
625  auto const max = cuda::std::numeric_limits<Rep>::max();
626  if (rhs > 0)
627  return lhs > max / rhs || lhs < min / rhs;
628  else if (rhs < -1)
629  return lhs > min / rhs || lhs < max / rhs;
630  else
631  return rhs == -1 && lhs == min;
632 }

635 template <typename Rep1, Radix Rad1>
636 CUDA_HOST_DEVICE_CALLABLE fixed_point<Rep1, Rad1> operator+(fixed_point<Rep1, Rad1> const& lhs,
637  fixed_point<Rep1, Rad1> const& rhs)
638 {
639  auto const scale = std::min(lhs._scale, rhs._scale);
640  auto const sum = lhs.rescaled(scale)._value + rhs.rescaled(scale)._value;
641 
642 #if defined(__CUDACC_DEBUG__)

645  "fixed_point overflow of underlying representation type " + print_rep<Rep1>());
646 
647 #endif
648 
649  return fixed_point<Rep1, Rad1>{scaled_integer<Rep1>{sum, scale}};
