114 template <typename From, typename To>
115 struct fixed_width_type_converter {
116  // Are the types same - simply copy elements from [begin, end) to out
117  template <typename FromT = From,
118  typename ToT = To,
119  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
120  ToT operator()(FromT element) const
121  {
122  return element;
123  }
124 
125  // Are the types convertible or can target be constructed from source?
126  template <typename FromT = From,
127  typename ToT = To,
128  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
129  (cudf::is_convertible<FromT, ToT>::value ||
130  std::is_constructible<ToT, FromT>::value),
131  void>::type* = nullptr>
132  ToT operator()(FromT element) const
133  {
134  return static_cast<ToT>(element);
135  }
136 
137  // Convert integral values to timestamps
138  template <
139  typename FromT = From,
140  typename ToT = To,
141  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
142  void>::type* = nullptr>
143  ToT operator()(FromT element) const
144  {
145  return ToT{typename ToT::duration{element}};
146  }
147 };

286 template <typename StringsIterator, typename ValidityIterator>
287 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
288 {
289  std::vector<char> chars{};
290  std::vector<int32_t> offsets(1, 0);
291  for (auto str = begin; str < end; ++str) {
292  std::string tmp = (*v++) ? std::string(*str) : std::string{};
293  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
294  offsets.push_back(offsets.back() + tmp.length());
295  }
296  return std::make_pair(std::move(chars), std::move(offsets));

1341  private:
1357  void build_from_nested(std::initializer_list<lists_column_wrapper<T, SourceElementT>> elements,
1358  std::vector<bool> const& v)
1359  {
1360  auto valids = cudf::test::make_counting_transform_iterator(
1361  0, [&v](auto i) { return v.empty() ? true : v[i]; });
1362 
1363  // compute the expected hierarchy and depth
1364  auto const hierarchy_and_depth = std::accumulate(
1365  elements.begin(),
1366  elements.end(),
1367  std::pair<column_view, int32_t>{{}, -1},
1368  [](auto acc, lists_column_wrapper const& lcw) {
1369  return lcw.depth > acc.second ? std::make_pair(lcw.get_view(), lcw.depth) : acc;
1370  });
1371  column_view expected_hierarchy = hierarchy_and_depth.first;
1372  int32_t const expected_depth = hierarchy_and_depth.second;
1373 
1374  // preprocess columns so that every column_view in 'cols' is an equivalent hierarchy
1375  std::vector<std::unique_ptr<column>> stubs;
1376  std::vector<column_view> cols;
1377  std::tie(cols, stubs) = preprocess_columns(elements, expected_hierarchy, expected_depth);
1378 
1379  // generate offsets
1380  size_type count = 0;
1381  std::vector<size_type> offsetv;
1382  std::transform(cols.cbegin(),
1383  cols.cend(),
1384  valids,
1385  std::back_inserter(offsetv),
1386  [&](cudf::column_view const& col, bool valid) {
1387  // nulls are represented as a repeated offset
1388  size_type ret = count;
1389  if (valid) { count += col.size(); }
1390  return ret;
1391  });
1392  // add the final offset
1393  offsetv.push_back(count);
1394  auto offsets =
1395  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();

1512  // - unwrap any "root" lists

1515  elements.end(),
1516  std::back_inserter(cols),
1517  [&](lists_column_wrapper const& l) -> column_view {
1518  // depth mismatch. attempt to normalize the short column.
1519  // this function will also catch if this is a legitmately broken
1520  // set of input
1521  if (l.depth < expected_depth) {
1522  if (l.root) {
1523  // this exception distinguishes between the following two cases:
1524  //
1525  // { {{{1, 2, 3}}}, {} }
1526  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1527  // just a List<> which is an apparent mismatch. However, because row 1
1528  // is empty we will allow that to semantically mean
1529  // "a List<List<List<int>>> that's empty at the top level"
1530  //
1531  // { {{{1, 2, 3}}}, {4, 5, 6} }
1532  // In this case, row 1 is a a concrete List<int> with actual values.
1533  // There is no way to rectify the differences so we will treat it as a
1534  // true column mismatch.
1535  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");

1539  }
1540  return *(stubs.back());
1541  }
1542  // the empty hierarchy case
1543  return l.get_view();
1544  });
1545 
1546  return {std::move(cols), std::move(stubs)};
1547  }
