74 __device__ bool is_float(string_view const& d_str)
75 {
76  if (d_str.empty()) return false;
77  // strings allowed by the converter
78  if (d_str.compare("NaN", 3) == 0) return true;
79  if (d_str.compare("Inf", 3) == 0) return true;
80  if (d_str.compare("-Inf", 4) == 0) return true;
81  bool decimal_found = false;
82  bool exponent_found = false;
83  size_type bytes = d_str.size_bytes();
84  const char* data = d_str.data();
85  // sign character allowed at the beginning of the string
86  size_type chidx = (*data == '-' || *data == '+') ? 1 : 0;
87  bool result = chidx < bytes;
88  // check for float chars [0-9] and a single decimal '.'
89  // and scientific notation [eE][+-][0-9]
90  for (; chidx < bytes; ++chidx) {
91  auto chr = data[chidx];
92  if (chr >= '0' && chr <= '9') continue;
93  if (!decimal_found && chr == '.') {
94  decimal_found = true; // no more decimals
95  continue;
96  }
97  if (!exponent_found && (chr == 'e' || chr == 'E')) {
98  if (chidx + 1 < bytes) chr = data[chidx + 1];
99  if (chr == '-' || chr == '+') ++chidx;
100  decimal_found = true; // no decimal allowed in exponent
101  exponent_found = true; // no more exponents
102  continue;
103  }
104  return false;
