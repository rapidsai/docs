312 template <typename B>
313 std::unique_ptr<column> make_fixed_width_column(
314  data_type type,
315  size_type size,
316  B&& null_mask,
317  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
318  rmm::cuda_stream_view stream = rmm::cuda_stream_default,
319  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
320 {
321  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
322  if (is_timestamp(type)) {
323  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
324  } else if (is_duration(type)) {
325  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
326  } else if (is_fixed_point(type)) {
327  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
328  }
329  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
330 }
