38 template <typename T, std::size_t Extent, typename Derived>
39 class span_base {

42  public:

44  using value_type = std::remove_cv<T>;
45  using size_type = std::size_t;
46  using difference_type = std::ptrdiff_t;
47  using pointer = T*;
48  using iterator = T*;
49  using const_pointer = T const*;
50  using reference = T&;
51  using const_reference = T const&;
52 
53  static constexpr std::size_t extent = Extent;

55  constexpr span_base() noexcept : _data(nullptr), _size(0) {}
56  constexpr span_base(pointer data, size_type size) : _data(data), _size(size) {}

58  constexpr span_base(span_base const& other) noexcept = default;
59  constexpr span_base& operator=(span_base const& other) noexcept = default;
60 
61  // not noexcept due to undefined behavior when size = 0
62  constexpr reference front() const { return _data[0]; }
63  // not noexcept due to undefined behavior when size = 0
64  constexpr reference back() const { return _data[_size - 1]; }
65  // not noexcept due to undefined behavior when idx < 0 || idx >= size
66  constexpr reference operator[](size_type idx) const { return _data[idx]; }
67 
68  constexpr iterator begin() const noexcept { return _data; }
69  constexpr iterator end() const noexcept { return _data + _size; }
70  constexpr pointer data() const noexcept { return _data; }
71 
72  constexpr size_type size() const noexcept { return _size; }
73  constexpr size_type size_bytes() const noexcept { return sizeof(T) * _size; }
74  constexpr bool empty() const noexcept { return _size == 0; }

81  constexpr Derived first(size_type count) const noexcept { return Derived(_data, count); }
