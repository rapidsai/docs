908 template <typename ColumnDeviceView, typename ColumnViewIterator>
909 ColumnDeviceView* child_columns_to_device_array(ColumnViewIterator child_begin,
910  ColumnViewIterator child_end,
911  void* h_ptr,
912  void* d_ptr)
913 {
914  ColumnDeviceView* d_children = detail::align_ptr_for_type<ColumnDeviceView>(d_ptr);
915  auto num_children = std::distance(child_begin, child_end);
916  if (num_children > 0) {
917  // The beginning of the memory must be the fixed-sized ColumnDeviceView
918  // struct objects in order for d_children to be used as an array.
919  auto h_column = detail::align_ptr_for_type<ColumnDeviceView>(h_ptr);
920  auto d_column = d_children;
921 
922  // Any child data is assigned past the end of this array: h_end and d_end.
923  auto h_end = reinterpret_cast<int8_t*>(h_column + num_children);
924  auto d_end = reinterpret_cast<int8_t*>(d_column + num_children);
925  std::for_each(child_begin, child_end, [&](auto const& col) {
926  // inplace-new each child into host memory
927  new (h_column) ColumnDeviceView(col, h_end, d_end);
928  h_column++; // advance to next child
929  // update the pointers for holding this child column's child data
930  auto col_child_data_size = ColumnDeviceView::extent(col) - sizeof(ColumnDeviceView);
931  h_end += col_child_data_size;
932  d_end += col_child_data_size;
933  });
934  }
935  return d_children;
936 }
