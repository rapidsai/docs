87 template <typename Rep,
88  Radix Base,
89  typename T,
90  typename simt::std::enable_if_t<(simt::std::is_same<int32_t, T>::value &&
91  is_supported_representation_type<Rep>())>* = nullptr>
92 CUDA_HOST_DEVICE_CALLABLE Rep ipow(T exponent)
93 {
94  if (exponent == 0) return static_cast<Rep>(1);
95  auto extra = static_cast<Rep>(1);
96  auto square = static_cast<Rep>(Base);
97  while (exponent > 1) {
98  if (exponent & 1 /* odd */) {
99  extra *= square;
100  exponent -= 1;
101  }
102  exponent /= 2;
103  square *= square;
104  }
105  return square * extra;
106 }

513  {
514  if (scale == _scale) return *this;

522  explicit operator std::string() const

558 CUDA_HOST_DEVICE_CALLABLE auto addition_overflow(T lhs, T rhs)
559 {

561  : lhs < simt::std::numeric_limits<Rep>::min() - rhs;

587 template <typename Rep, typename T>
588 CUDA_HOST_DEVICE_CALLABLE auto division_overflow(T lhs, T rhs)

601 template <typename Rep, typename T>
602 CUDA_HOST_DEVICE_CALLABLE auto multiplication_overflow(T lhs, T rhs)
603 {
604  auto const min = simt::std::numeric_limits<Rep>::min();
605  auto const max = simt::std::numeric_limits<Rep>::max();
606  if (rhs > 0)
607  return lhs > max / rhs || lhs < min / rhs;
608  else if (rhs < -1)
609  return lhs > min / rhs || lhs < max / rhs;
610  else
611  return rhs == -1 && lhs == min;
612 }
