108 inline __device__ auto null_compare(bool lhs_is_null, bool rhs_is_null, null_order null_precedence)
109 {
110  if (lhs_is_null and rhs_is_null) { // null <? null
111  return weak_ordering::EQUIVALENT;
112  } else if (lhs_is_null) { // null <? x
113  return (null_precedence == null_order::BEFORE) ? weak_ordering::LESS : weak_ordering::GREATER;
114  } else if (rhs_is_null) { // x <? null
115  return (null_precedence == null_order::AFTER) ? weak_ordering::LESS : weak_ordering::GREATER;
116  }
117  return weak_ordering::EQUIVALENT;
118 }

353 class row_lexicographic_comparator {
354  public:
374  row_lexicographic_comparator(Nullate has_nulls,
375  table_device_view lhs,
376  table_device_view rhs,
377  order const* column_order = nullptr,
378  null_order const* null_precedence = nullptr)
379  : _lhs{lhs},
380  _rhs{rhs},
381  _nulls{has_nulls},
382  _column_order{column_order},
383  _null_precedence{null_precedence}
384  {
385  CUDF_EXPECTS(_lhs.num_columns() == _rhs.num_columns(), "Mismatched number of columns.");
386  }
387 
397  __device__ bool operator()(size_type lhs_index, size_type rhs_index) const noexcept
398  {
399  for (size_type i = 0; i < _lhs.num_columns(); ++i) {
400  bool ascending = (_column_order == nullptr) or (_column_order[i] == order::ASCENDING);
401 
402  null_order null_precedence =
403  _null_precedence == nullptr ? null_order::BEFORE : _null_precedence[i];
404 
405  auto comparator =
406  element_relational_comparator{_nulls, _lhs.column(i), _rhs.column(i), null_precedence};
407 
408  weak_ordering state =
409  cudf::type_dispatcher(_lhs.column(i).type(), comparator, lhs_index, rhs_index);
410 
411  if (state == weak_ordering::EQUIVALENT) { continue; }
412 
413  return state == (ascending ? weak_ordering::LESS : weak_ordering::GREATER);
414  }
415  return false;
416  }
417 
418  private:
419  table_device_view _lhs;
420  table_device_view _rhs;
421  Nullate _nulls{};
422  null_order const* _null_precedence{};
423  order const* _column_order{};

489 template <template <typename> class hash_function, typename Nullate>
490 class row_hasher {
491  public:
492  row_hasher() = delete;
493  CUDF_HOST_DEVICE row_hasher(Nullate has_nulls, table_device_view t)
494  : _table{t}, _has_nulls{has_nulls}
495  {
496  }
497  CUDF_HOST_DEVICE row_hasher(Nullate has_nulls, table_device_view t, uint32_t seed)
498  : _table{t}, _seed(seed), _has_nulls{has_nulls}
499  {
500  }
501 
502  __device__ auto operator()(size_type row_index) const
503  {
504  // Hash the first column w/ the seed
505  auto const initial_hash = cudf::detail::hash_combine(
506  hash_value_type{0},
507  type_dispatcher<dispatch_storage_type>(
508  _table.column(0).type(),
509  element_hasher_with_seed<hash_function, Nullate>{_has_nulls, _seed},
510  _table.column(0),
511  row_index));
512 
513  // Hashes an element in a column
514  auto hasher = [=](size_type column_index) {
515  return cudf::type_dispatcher<dispatch_storage_type>(
516  _table.column(column_index).type(),
517  element_hasher<hash_function, Nullate>{_has_nulls},
518  _table.column(column_index),
519  row_index);
520  };
521 
522  // Hash each element and combine all the hash values together
523  return thrust::transform_reduce(
524  thrust::seq,
525  // note that this starts at 1 and not 0 now since we already hashed the first column
526  thrust::make_counting_iterator(1),
527  thrust::make_counting_iterator(_table.num_columns()),
528  hasher,
529  initial_hash,
530  [](hash_value_type lhs, hash_value_type rhs) {
531  return cudf::detail::hash_combine(lhs, rhs);
532  });
533  }
534 
535  private:
536  table_device_view _table;
537  Nullate _has_nulls;
538  uint32_t _seed{DEFAULT_HASH_SEED};
539 };
