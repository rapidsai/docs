316 template <typename T>
317 void throw_or_mimic(const std::string& text)
318 {
319  static_assert(std::is_base_of<std::exception, T>::value,
320  "throw_or_mimic only works on std::exception and "
321  "deriving classes");
322 
323 #ifndef CXXOPTS_NO_EXCEPTIONS
324  // If CXXOPTS_NO_EXCEPTIONS is not defined, just throw
325  throw T{text};
326 #else
327  // Otherwise manually instantiate the exception, print what() to stderr,
328  // and abort
329  T exception{text};
330  std::cerr << exception.what() << std::endl;
331  std::cerr << "Aborting (exceptions disabled)..." << std::endl;
332  std::abort();
333 #endif
334 }

379 template <typename R, typename T>
380 R checked_negate(T&& t, const std::string&, std::true_type)
381 {
382  // if we got to here, then `t` is a positive number that fits into
383  // `R`. So to avoid MSVC C4146, we first cast it to `R`.
384  // See https://github.com/jarro2783/cxxopts/issues/62 for more details.
385  return static_cast<R>(-static_cast<R>(t - 1) - 1);

395 template <typename T>
396 void integer_parser(const std::string& text, T& value)
397 {
398  std::smatch match;
399  std::regex_match(text, match, integer_pattern);
400 
401  if (match.length() == 0) { throw_or_mimic<argument_incorrect_type>(text); }
402 
403  if (match.length(4) > 0) {
404  value = 0;
405  return;
406  }
407 
408  using US = typename std::make_unsigned<T>::type;
409 
410  constexpr bool is_signed = std::numeric_limits<T>::is_signed;
411  const bool negative = match.length(1) > 0;
412  const uint8_t base = match.length(2) > 0 ? 16 : 10;
413 
414  auto value_match = match[3];
415 
416  US result = 0;
417 
418  for (auto iter = value_match.first; iter != value_match.second; ++iter) {
419  US digit = 0;
420 
421  if (*iter >= '0' && *iter <= '9') {
422  digit = static_cast<US>(*iter - '0');
423  } else if (base == 16 && *iter >= 'a' && *iter <= 'f') {
424  digit = static_cast<US>(*iter - 'a' + 10);
425  } else if (base == 16 && *iter >= 'A' && *iter <= 'F') {
426  digit = static_cast<US>(*iter - 'A' + 10);
427  } else {
428  throw_or_mimic<argument_incorrect_type>(text);
429  }
430 
431  const US next = static_cast<US>(result * base + digit);
432  if (result > next) { throw_or_mimic<argument_incorrect_type>(text); }
433 
434  result = next;
435  }
436 
437  detail::check_signed_range<T>(negative, result, text);
438 
439  if (negative) {
440  value = checked_negate<T>(result, text, std::integral_constant<bool, is_signed>());
441  } else {
442  value = static_cast<T>(result);
443  }
444 }

1005 String format_description(const HelpOptionDetails& o, size_t start, size_t width)
1006 {
1007  auto desc = o.desc;
1008 
1009  if (o.has_default && (!o.is_boolean || o.default_value != "false")) {
1010  if (o.default_value != "") {
1011  desc += toLocalString(" (default: " + o.default_value + ")");
1012  } else {
1013  desc += toLocalString(" (default: \"\")");
1014  }
1015  }
1016 
1017  String result;
1018 
1019  auto current = std::begin(desc);
1020  auto startLine = current;
1021  auto lastSpace = current;
1022 
1023  auto size = size_t{};
1024 
1025  while (current != std::end(desc)) {
1026  if (*current == ' ') { lastSpace = current; }
1027 
1028  if (*current == '\n') {
1029  startLine = current + 1;
1030  lastSpace = startLine;
1031  } else if (size > width) {
1032  if (lastSpace == startLine) {
1033  stringAppend(result, startLine, current + 1);
1034  stringAppend(result, "\n");
1035  stringAppend(result, start, ' ');
1036  startLine = current + 1;
1037  lastSpace = startLine;
1038  } else {
1039  stringAppend(result, startLine, lastSpace);
1040  stringAppend(result, "\n");
1041  stringAppend(result, start, ' ');
1042  startLine = lastSpace + 1;
1043  lastSpace = startLine;
1044  }
1045  size = 0;
1046  } else {
1047  ++size;
1048  }
1049 
1050  ++current;
1051  }
1052 
1053  // append whatever is left
1054  stringAppend(result, startLine, current);
1055 
1056  return result;
1057 }

1222 inline void ParseResult::parse(int& argc, char**& argv)
1223 {
1224  int current = 1;
1225 
1226  int nextKeep = 1;
1227 
1228  bool consume_remaining = false;
1229 
1230  while (current != argc) {
1231  if (strcmp(argv[current], "--") == 0) {
1232  consume_remaining = true;
1233  ++current;
1234  break;
1235  }
1236 
1237  std::match_results<const char*> result;
1238  std::regex_match(argv[current], result, option_matcher);
1239 
1240  if (result.empty()) {
1241  // not a flag
1242 
1243  // but if it starts with a `-`, then it's an error
1244  if (argv[current][0] == '-' && argv[current][1] != '\0') {
1245  if (!m_allow_unrecognised) { throw_or_mimic<option_syntax_exception>(argv[current]); }
1246  }
1247 
1248  // if true is returned here then it was consumed, otherwise it is
1249  // ignored
1250  if (consume_positional(argv[current])) {
1251  } else {
1252  argv[nextKeep] = argv[current];
1253  ++nextKeep;
1254  }
1255  // if we return from here then it was parsed successfully, so continue
1256  } else {
1257  // short or long option?
1258  if (result[4].length() != 0) {
1259  const std::string& s = result[4];
1260 
1261  for (std::size_t i = 0; i != s.size(); ++i) {
1262  std::string name(1, s[i]);
1263  auto iter = m_options->find(name);
1264 
1265  if (iter == m_options->end()) {
1266  if (m_allow_unrecognised) {
1267  continue;
1268  } else {
1269  // error
1270  throw_or_mimic<option_not_exists_exception>(name);
1271  }
1272  }
1273 
1274  auto value = iter->second;
1275 
1276  if (i + 1 == s.size()) {
1277  // it must be the last argument
1278  checked_parse_arg(argc, argv, current, value, name);
1279  } else if (value->value().has_implicit()) {
1280  parse_option(value, name, value->value().get_implicit_value());
1281  } else {
1282  // error
1283  throw_or_mimic<option_requires_argument_exception>(name);
1284  }
1285  }
1286  } else if (result[1].length() != 0) {
1287  const std::string& name = result[1];
1288 
1289  auto iter = m_options->find(name);
1290 
1291  if (iter == m_options->end()) {
1292  if (m_allow_unrecognised) {
1293  // keep unrecognised options in argument list, skip to next argument
1294  argv[nextKeep] = argv[current];
1295  ++nextKeep;
1296  ++current;
1297  continue;
1298  } else {
1299  // error
1300  throw_or_mimic<option_not_exists_exception>(name);
1301  }
1302  }
1303 
1304  auto opt = iter->second;
1305 
1306  // equals provided for long option?
1307  if (result[2].length() != 0) {
1308  // parse the option given
1309 
1310  parse_option(opt, name, result[3]);
1311  } else {
1312  // parse the next argument
1313  checked_parse_arg(argc, argv, current, opt, name);
1314  }
1315  }
1316  }
1317 
1318  ++current;
1319  }
1320 
1321  for (auto& opt : *m_options) {
1322  auto& detail = opt.second;
1323  auto& value = detail->value();
1324 
1325  auto& store = m_results[detail];
1326 
1327  if (value.has_default() && !store.count() && !store.has_default()) { parse_default(detail); }
1328  }
1329 
1330  if (consume_remaining) {
1331  while (current < argc) {
1332  if (!consume_positional(argv[current])) { break; }
1333  ++current;
1334  }
1335 
1336  // adjust argv for any that couldn't be swallowed
1337  while (current != argc) {
1338  argv[nextKeep] = argv[current];
1339  ++nextKeep;
1340  ++current;
1341  }
1342  }
1343 
1344  argc = nextKeep;
1345 }

1389 inline String Options::help_one_group(const std::string& g) const
1390 {
1391  typedef std::vector<std::pair<String, String>> OptionHelp;
1392 
1393  auto group = m_help.find(g);
1394  if (group == m_help.end()) { return ""; }
1395 
1396  OptionHelp format;
1397 
1398  size_t longest = 0;
1399 
1400  String result;
1401 
1402  if (!g.empty()) { result += toLocalString(" " + g + " options:\n"); }
1403 
1404  for (const auto& o : group->second.options) {
1405  if (m_positional_set.find(o.l) != m_positional_set.end() && !m_show_positional) { continue; }
1406 
1407  auto s = format_option(o);
1408  longest = (std::max)(longest, stringLength(s));
1409  format.push_back(std::make_pair(s, String()));
1410  }
1411 
1412  longest = (std::min)(longest, static_cast<size_t>(OPTION_LONGEST));
1413 
1414  // widest allowed description
1415  auto allowed = size_t{76} - longest - OPTION_DESC_GAP;
