203 template <typename T, std::enable_if_t<cudf::is_fixed_point<T>()>* = nullptr>
204 std::pair<thrust::host_vector<T>, std::vector<bitmask_type>> to_host(column_view c)
205 {
206  using namespace numeric;
207  using Rep = typename T::rep;
208 
209  auto host_rep_types = thrust::host_vector<Rep>(c.size());
210 
211  CUDF_CUDA_TRY(cudaMemcpy(
212  host_rep_types.data(), c.begin<Rep>(), c.size() * sizeof(Rep), cudaMemcpyDeviceToHost));
213 
214  auto to_fp = [&](Rep val) { return T{scaled_integer<Rep>{val, scale_type{c.type().scale()}}}; };
215  auto begin = thrust::make_transform_iterator(std::cbegin(host_rep_types), to_fp);
216  auto const host_fixed_points = thrust::host_vector<T>(begin, begin + c.size());
217 
218  return {host_fixed_points, bitmask_to_host(c)};
219 }

231 template <>
232 inline std::pair<thrust::host_vector<std::string>, std::vector<bitmask_type>> to_host(column_view c)
233 {
234  auto const scv = strings_column_view(c);
235  auto const h_chars = cudf::detail::make_std_vector_sync<char>(
236  cudf::device_span<char const>(scv.chars().data<char>(), scv.chars().size()),
237  cudf::default_stream_value);
238  auto const h_offsets = cudf::detail::make_std_vector_sync(
239  cudf::device_span<cudf::offset_type const>(
240  scv.offsets().data<cudf::offset_type>() + scv.offset(), scv.size() + 1),
241  cudf::default_stream_value);
242 
243  // build std::string vector from chars and offsets
244  std::vector<std::string> host_data;
245  host_data.reserve(c.size());
246  std::transform(
247  std::begin(h_offsets),
248  std::end(h_offsets) - 1,
249  std::begin(h_offsets) + 1,
250  std::back_inserter(host_data),
251  [&](auto start, auto end) { return std::string(h_chars.data() + start, end - start); });
252 
253  return {host_data, bitmask_to_host(c)};
254 }
