328 template <typename B>
329 std::unique_ptr<column> make_fixed_width_column(
330  data_type type,
331  size_type size,
332  B&& null_mask,
333  size_type null_count = cudf::UNKNOWN_NULL_COUNT,
334  rmm::cuda_stream_view stream = cudf::default_stream_value,
335  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
336 {
337  CUDF_EXPECTS(is_fixed_width(type), "Invalid, non-fixed-width type.");
338  if (is_timestamp(type)) {
339  return make_timestamp_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
340  } else if (is_duration(type)) {
341  return make_duration_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
342  } else if (is_fixed_point(type)) {
343  return make_fixed_point_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
344  }
345  return make_numeric_column(type, size, std::forward<B>(null_mask), null_count, stream, mr);
346 }
