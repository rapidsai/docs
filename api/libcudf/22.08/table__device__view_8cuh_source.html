236 template <typename ColumnDeviceView, typename HostTableView>
237 auto contiguous_copy_column_device_views(HostTableView source_view, rmm::cuda_stream_view stream)
238 {
239  // First calculate the size of memory needed to hold the
240  // table's ColumnDeviceViews. This is done by calling extent()
241  // for each of the table's ColumnViews columns.
242  std::size_t views_size_bytes = std::accumulate(
243  source_view.begin(), source_view.end(), std::size_t{0}, [](std::size_t init, auto col) {
244  return init + ColumnDeviceView::extent(col);
245  });
246  // pad the allocation for aligning the first pointer
247  auto padded_views_size_bytes = views_size_bytes + std::size_t{alignof(ColumnDeviceView) - 1};
248  // A buffer of CPU memory is allocated to hold the ColumnDeviceView
249  // objects. Once filled, the CPU memory is then copied to device memory
250  // and the pointer is set in the d_columns member.
251  std::vector<int8_t> h_buffer(padded_views_size_bytes);
252  // Each ColumnDeviceView instance may have child objects which may
253  // require setting some internal device pointers before being copied
254  // from CPU to device.
255  // Allocate the device memory to be used in the result.
256  // We need this pointer in order to pass it down when creating the
257  // ColumnDeviceViews so the column can set the pointer(s) for any
258  // of its child objects.
259  // align both h_ptr, d_ptr
260  auto descendant_storage = std::make_unique<rmm::device_buffer>(padded_views_size_bytes, stream);
261  void* h_ptr = detail::align_ptr_for_type<ColumnDeviceView>(h_buffer.data());
262  void* d_ptr = detail::align_ptr_for_type<ColumnDeviceView>(descendant_storage->data());
263  auto d_columns = detail::child_columns_to_device_array<ColumnDeviceView>(
264  source_view.begin(), source_view.end(), h_ptr, d_ptr);
265 
266  CUDF_CUDA_TRY(cudaMemcpyAsync(d_ptr, h_ptr, views_size_bytes, cudaMemcpyDefault, stream.value()));
267  stream.synchronize();
268  return std::make_tuple(std::move(descendant_storage), d_columns);
269 }
