40 __device__ inline size_type characters_in_string(const char* str, size_type bytes)
41 {
42  if ((str == 0) || (bytes == 0)) return 0;
43  auto ptr = reinterpret_cast<uint8_t const*>(str);
44  return thrust::count_if(

235 __device__ inline size_type string_view::byte_offset(size_type pos) const
236 {
237  size_type offset = 0;
238  const char* sptr = _data;
239  const char* eptr = sptr + _bytes;
240  if (length() == size_bytes()) return pos;
241  while ((pos > 0) && (sptr < eptr)) {
242  size_type charbytes = strings::detail::bytes_in_utf8_byte(static_cast<uint8_t>(*sptr++));
243  if (charbytes) --pos;
244  offset += charbytes;
245  }
246  return offset;
247 }

254 __device__ inline int string_view::compare(const char* data, size_type bytes) const
255 {
256  size_type const len1 = size_bytes();
257  const unsigned char* ptr1 = reinterpret_cast<const unsigned char*>(this->data());
258  const unsigned char* ptr2 = reinterpret_cast<const unsigned char*>(data);
259  size_type idx = 0;
260  for (; (idx < len1) && (idx < bytes); ++idx) {
261  if (*ptr1 != *ptr2) return static_cast<int32_t>(*ptr1) - static_cast<int32_t>(*ptr2);
262  ++ptr1;
263  ++ptr2;
264  }
265  if (idx < len1) return 1;
266  if (idx < bytes) return -1;
267  return 0;
268 }

309 __device__ inline size_type string_view::find(const char* str,
310  size_type bytes,
311  size_type pos,
312  size_type count) const
313 {
314  const char* sptr = data();
315  if (!str || !bytes) return -1;
316  size_type nchars = length();
317  if (count < 0) count = nchars;
318  size_type end = pos + count;
319  if (end < 0 || end > nchars) end = nchars;
320  size_type spos = byte_offset(pos);
321  size_type epos = byte_offset(end);
322 
323  size_type len2 = bytes;
324  size_type len1 = (epos - spos) - len2 + 1;
325 
326  const char* ptr1 = sptr + spos;
327  const char* ptr2 = str;
328  for (size_type idx = 0; idx < len1; ++idx) {
329  bool match = true;
330  for (size_type jdx = 0; match && (jdx < len2); ++jdx) match = (ptr1[jdx] == ptr2[jdx]);
331  if (match) return character_offset(idx + spos);
332  ptr1++;
333  }
334  return -1;
335 }
