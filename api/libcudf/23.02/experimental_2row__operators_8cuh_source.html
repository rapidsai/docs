216 // The has_nested_columns template parameter of the device_row_comparator is
217 // necessary to help the compiler optimize our code. Without it, the list and
218 // struct view specializations are present in the code paths used for primitive
219 // types, and the compiler fails to inline this nearly as well resulting in a
220 // significant performance drop. As a result, there is some minor tension in
221 // the current design between the presence of this parameter and the way that
222 // the Dremel data is passed around, first as a
223 // std::optional<device_span<dremel_device_view>> in the
224 // preprocessed_table/device_row_comparator (which is always valid when
225 // has_nested_columns and is otherwise invalid) that is then unpacked to a
226 // thrust::optional<dremel_device_view> at the element_comparator level (which
227 // is always valid for a list column and otherwise invalid). We cannot use an
228 // additional template parameter for the element_comparator on a per-column
229 // basis because we cannot conditionally define dremel_device_view member
230 // variables without jumping through extra hoops with inheritance, so the
231 // thrust::optional<dremel_device_view> member must be an optional rather than
232 // a raw dremel_device_view.

372  template <typename Element,
373  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::struct_view>)>
374  __device__ cuda::std::pair<weak_ordering, int> operator()(
375  size_type const lhs_element_index, size_type const rhs_element_index) const noexcept
376  {
377  column_device_view lcol = _lhs;
378  column_device_view rcol = _rhs;
379  int depth = _depth;
380  while (lcol.type().id() == type_id::STRUCT) {
381  bool const lhs_is_null{lcol.is_null(lhs_element_index)};
382  bool const rhs_is_null{rcol.is_null(rhs_element_index)};
383 
384  if (lhs_is_null or rhs_is_null) { // at least one is null
385  weak_ordering state = null_compare(lhs_is_null, rhs_is_null, _null_precedence);
386  return cuda::std::pair(state, depth);
387  }
388 
389  if (lcol.num_child_columns() == 0) {
390  return cuda::std::pair(weak_ordering::EQUIVALENT, depth);
391  }
392 
393  // Non-empty structs have been modified to only have 1 child when using this.
394  lcol = detail::structs_column_device_view(lcol).get_sliced_child(0);
395  rcol = detail::structs_column_device_view(rcol).get_sliced_child(0);
396  ++depth;
397  }
398 
399  return cudf::type_dispatcher<dispatch_void_if_nested>(
400  lcol.type(),
401  element_comparator{_check_nulls, lcol, rcol, _null_precedence, depth, _comparator},
402  lhs_element_index,
403  rhs_element_index);
404  }

406  template <typename Element,
407  CUDF_ENABLE_IF(has_nested_columns and std::is_same_v<Element, cudf::list_view>)>
408  __device__ cuda::std::pair<weak_ordering, int> operator()(size_type lhs_element_index,
409  size_type rhs_element_index)
410  {
411  // only order top-NULLs according to null_order
412  auto const is_l_row_null = _lhs.is_null(lhs_element_index);
413  auto const is_r_row_null = _rhs.is_null(rhs_element_index);
414  if (is_l_row_null || is_r_row_null) {
415  return cuda::std::pair(null_compare(is_l_row_null, is_r_row_null, _null_precedence),
416  _depth);
417  }
418 
419  // These are all the values from the Dremel encoding.
420  auto const l_max_def_level = _l_dremel_device_view->max_def_level;
421  auto const r_max_def_level = _r_dremel_device_view->max_def_level;
422  auto const l_def_levels = _l_dremel_device_view->def_levels;
423  auto const r_def_levels = _r_dremel_device_view->def_levels;
424  auto const l_rep_levels = _l_dremel_device_view->rep_levels;
425  auto const r_rep_levels = _r_dremel_device_view->rep_levels;
426 
427  // Traverse the nested list hierarchy to get a column device view
428  // pointing to the underlying child data.
429  column_device_view lcol = _lhs.slice(lhs_element_index, 1);
430  column_device_view rcol = _rhs.slice(rhs_element_index, 1);
431 
432  while (lcol.type().id() == type_id::LIST) {
433  lcol = detail::lists_column_device_view(lcol).get_sliced_child();
434  rcol = detail::lists_column_device_view(rcol).get_sliced_child();
435  }
436 
437  // These start and end values indicate the start and end points of all
438  // the elements of the lists in the current list element
439  // (`[lhs|rhs]_element_index`) that we are comparing.
440  auto const l_offsets = _l_dremel_device_view->offsets;
441  auto const r_offsets = _r_dremel_device_view->offsets;
442  auto l_start = l_offsets[lhs_element_index];
443  auto l_end = l_offsets[lhs_element_index + 1];
444  auto r_start = r_offsets[rhs_element_index];
445  auto r_end = r_offsets[rhs_element_index + 1];
446 
447  // This comparator will be used to compare leaf (non-nested) data types.
448  auto comparator =
449  element_comparator{_check_nulls, lcol, rcol, _null_precedence, _depth, _comparator};
450 
451  // Loop over each element in the encoding. Note that this includes nulls
452  // and empty lists, so not every index corresponds to an actual element
453  // in the child column. The element_index is used to keep track of the current
454  // child element that we're actually comparing.
455  for (int l_dremel_index = l_start, r_dremel_index = r_start, element_index = 0;
456  l_dremel_index < l_end and r_dremel_index < r_end;
457  ++l_dremel_index, ++r_dremel_index) {
458  auto const l_rep_level = l_rep_levels[l_dremel_index];
459  auto const r_rep_level = r_rep_levels[r_dremel_index];
460 
461  // early exit for smaller sub-list
462  if (l_rep_level != r_rep_level) {
463  // the lower repetition level is a smaller sub-list
464  return l_rep_level < r_rep_level ? cuda::std::pair(weak_ordering::LESS, _depth)
465  : cuda::std::pair(weak_ordering::GREATER, _depth);
466  }
467 
468  // only compare if left and right are at same nesting level
469  auto const l_def_level = l_def_levels[l_dremel_index];
470  auto const r_def_level = r_def_levels[r_dremel_index];
471 
472  // either left or right are empty or NULLs of arbitrary nesting
473  if (l_def_level < l_max_def_level || r_def_level < r_max_def_level) {
474  // in the fully unraveled version of the list column, only the
475  // most nested NULLs and leafs are present
476  // In this rare condition that we get to the most nested NULL, we increment
477  // element_index because either both rows have a deeply nested NULL at the
478  // same position, and we'll "continue" in our iteration, or we will early
479  // exit if only one of the rows has a deeply nested NULL
480  if (lcol.nullable() and l_def_levels[l_dremel_index] == l_max_def_level - 1) {
481  ++element_index;
482  }
483  if (l_def_level == r_def_level) { continue; }
484  // We require [] < [NULL] < [leaf] for nested nulls.
485  // The null_precedence only affects top level nulls.
486  return l_def_level < r_def_level ? cuda::std::pair(weak_ordering::LESS, _depth)
487  : cuda::std::pair(weak_ordering::GREATER, _depth);
488  }
489 
490  // finally, compare leaf to leaf
491  weak_ordering state{weak_ordering::EQUIVALENT};
492  int last_null_depth = _depth;
493  cuda::std::tie(state, last_null_depth) = cudf::type_dispatcher<dispatch_void_if_nested>(
494  lcol.type(), comparator, element_index, element_index);
495  if (state != weak_ordering::EQUIVALENT) { return cuda::std::pair(state, _depth); }
496  ++element_index;
497  }
498 
499  // If we have reached this stage, we know that definition levels,
500  // repetition levels, and actual elements are identical in both list
501  // columns up to the `min(l_end - l_start, r_end - r_start)` element of
502  // the Dremel encoding. However, two lists can only compare equivalent if
503  // they are of the same length. Otherwise, the shorter of the two is less
504  // than the longer. This final check determines the appropriate resulting
505  // ordering by checking how many total elements each list is composed of.
506  return cuda::std::pair(detail::compare_elements(l_end - l_start, r_end - r_start), _depth);
507  }
