73 template <typename T, gdf_dtype type_id>
74 struct wrapper
75 {
76  static constexpr gdf_dtype corresponding_column_type{type_id};
77  using value_type = T;
78  value_type value;
79 
80  template <gdf_dtype dtype = type_id,
81  std::enable_if_t<(dtype != GDF_BOOL8)> * = nullptr>
82  CUDA_HOST_DEVICE_CALLABLE
83  constexpr explicit wrapper(T v) : value{v} {}
84 
85  template <gdf_dtype dtype = type_id,
86  std::enable_if_t<(dtype == GDF_BOOL8)> * = nullptr>
87  CUDA_HOST_DEVICE_CALLABLE constexpr explicit wrapper(T v)
88  : value{static_cast<bool>(v)} {}
89 
90  CUDA_HOST_DEVICE_CALLABLE explicit operator value_type() const {
91  return this->value;
92  }
93 
94  // enable conversion to arithmetic types *only* for the cudf::bool8 wrapper
95  // (defined later in this file as wrapper<gdf_bool8, GDF_BOOL8>)
96  template <gdf_dtype the_type = type_id,
97  typename T_out,
98  typename std::enable_if<(the_type == GDF_BOOL8) &&
99  std::is_arithmetic<T_out>::value,
100  int>::type* = nullptr >
101  CUDA_HOST_DEVICE_CALLABLE
102  explicit operator T_out() const {
103  // Casting a cudf::bool8 to arithmetic type should always be the same as
104  // casting a bool to arithmetic type, and not the same as casting the
105  // underlying type to arithmetic type. Therefore we cast the value to bool
106  // first, then the output type
107  return static_cast<T_out>(static_cast<bool>(this->value));
108  }
109 
110  wrapper(wrapper const& w) = default;
111 
112  wrapper() = default;
113 };
