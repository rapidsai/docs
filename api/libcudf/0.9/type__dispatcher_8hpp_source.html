166 // This pragma disables a compiler warning that complains about the valid usage
167 // of calling a __host__ functor from this function which is __host__ __device__
168 #pragma hd_warning_disable
169 #pragma nv_exec_check_disable
170 template < class functor_t, typename ... Ts>
171 CUDA_HOST_DEVICE_CALLABLE constexpr decltype( auto ) type_dispatcher(gdf_dtype dtype,
172 functor_t f,
173 Ts&&... args) {
174 switch (dtype)
175 {
176 // The .template is known as a "template disambiguator"
177 // See here for more information:
178 // https://stackoverflow.com/questions/3786360/confusing-template-error
179 case GDF_INT8: { return f.template operator()< int8_t >(std::forward<Ts>(args)...); }
180 case GDF_INT16: { return f.template operator()< int16_t >(std::forward<Ts>(args)...); }
181 case GDF_INT32: { return f.template operator()< int32_t >(std::forward<Ts>(args)...); }
182 case GDF_INT64: { return f.template operator()< int64_t >(std::forward<Ts>(args)...); }
183 case GDF_FLOAT32: { return f.template operator()< float >(std::forward<Ts>(args)...); }
184 case GDF_FLOAT64: { return f.template operator()< double >(std::forward<Ts>(args)...); }
185 case GDF_BOOL8: { return f.template operator()< bool8 >(std::forward<Ts>(args)...); }
186 case GDF_DATE32: { return f.template operator()< date32 >(std::forward<Ts>(args)...); }
187 case GDF_DATE64: { return f.template operator()< date64 >(std::forward<Ts>(args)...); }
188 case GDF_TIMESTAMP: { return f.template operator()< timestamp >(std::forward<Ts>(args)...); }
189 case GDF_CATEGORY: { return f.template operator()< category >(std::forward<Ts>(args)...); }
190 case GDF_STRING_CATEGORY:
191 { return f.template operator()< nvstring_category >(std::forward<Ts>(args)...); }
192 default : {
193 #ifdef __CUDA_ARCH__
194
195 // This will cause the calling kernel to crash as well as invalidate
196 // the GPU context
197 release_assert( false && "Invalid gdf_dtype in type_dispatcher" );
198
199 // The following code will never be reached, but the compiler generates a
200 // warning if there isn't a return value.
201
202 // Need to find out what the return type is in order to have a default
203 // return value and solve the compiler warning for lack of a default
204 // return
205 using return_type =
206 decltype(f.template operator ()<int8_t>(std::forward<Ts>(args)...));
207 return return_type();
208 #else
209 // In host-code, the compiler is smart enough to know we don't need a
210 // default return type since we're throwing an exception.
211 throw std::runtime_error( "Invalid gdf_dtype in type_dispatcher" );
212 #endif
213 }
214 }
215 }
