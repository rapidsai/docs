59 gdf_error allocate() {
60 // For strings, just store the ptr + length. Eventually, column's data ptr
61 // is replaced with an NvString instance created from these pairs.
62 const auto num_rows = std::max(col-> size , 1);
63 const auto column_byte_width = (col-> dtype == GDF_STRING)
64 ? sizeof (std::pair<const char *, size_t>)
65 : cudf::byte_width (*col);
66
67 RMM_TRY(RMM_ALLOC(&col-> data , num_rows * column_byte_width, 0));
68 RMM_TRY(RMM_ALLOC(&col-> valid , gdf_valid_allocation_size(num_rows), 0));
69 CUDA_TRY(cudaMemset(col-> valid , 0, gdf_valid_allocation_size(num_rows)));
70
71 return GDF_SUCCESS;
72 }

114 bool insert( const T &data) {
115 if (num_elements < max_elements) {
116 h_data[num_elements] = data;
117 num_elements++;
118 return true ;
119 }
120 return false ;
121 }
122
123 size_t max_size() const noexcept { return max_elements; }
124 size_t size() const noexcept { return num_elements; }
125 size_t memory_size() const noexcept { return sizeof (T) * num_elements; }
126
127 T &operator[]( size_t i) const { return h_data[i]; }
128 T *host_ptr( size_t offset = 0) const { return h_data + offset; }
129 T *device_ptr( size_t offset = 0) const { return d_data + offset; }

131 private :
132 size_t max_elements = 0;
133 size_t num_elements = 0;
134 T *h_data = nullptr ;

145 template < typename T>
146 class device_buffer {
147 T* d_data_ = nullptr ;
148 size_t count_ = 0;
149 cudaStream_t stream_ = 0;
150
151 public :
152 device_buffer () noexcept = default ;
153 device_buffer ( size_t cnt, cudaStream_t stream = 0):
154 stream_(stream){
155 resize(cnt);
156 }
157
158 T* data() const noexcept { return d_data_;}
159 size_t size() const noexcept { return count_;}
160 bool empty() const noexcept { return count_ == 0;}
161
162 void resize( size_t cnt) {
163 if (cnt == count_) {
164 return ;
165 }
166 // new size is zero, free the buffer if not null
167 if (cnt == 0 && d_data_ != nullptr ) {
168 RMM_FREE(d_data_, stream_);
169 d_data_ = nullptr ;
170 count_ = cnt;
171 return ;
172 }
173
174 T* new_ptr = nullptr ;
175 const auto error = RMM_ALLOC(&new_ptr, cnt* sizeof (T), stream_);
176 if (error != RMM_SUCCESS) {
177 cudf::detail::throw_cuda_error(cudaErrorMemoryAllocation, __FILE__, __LINE__);
178 }
179 // Copy to the new buffer, if some memory was already allocated
180 if (count_ != 0) {
181 const size_t copy_bytes = std::min(cnt, count_)* sizeof (T);
182 CUDA_TRY(cudaMemcpyAsync(new_ptr, d_data_, copy_bytes, cudaMemcpyDefault, stream_));
183 RMM_FREE(d_data_, stream_);
184 }
185
186 d_data_ = new_ptr;
187 count_ = cnt;
188 }
189
190 device_buffer ( device_buffer & ) = delete ;
191 device_buffer ( device_buffer && rh) noexcept {
192 d_data_ = rh.d_data_;
193 count_ = rh.count_;
194 stream_ = rh.stream_;
195
196 rh.d_data_ = nullptr ;
197 rh.count_ = 0;
198 }
199
200 device_buffer & operator=( device_buffer & ) = delete ;
201 device_buffer & operator=( device_buffer && rh) noexcept {
202 RMM_FREE(d_data_, stream_);
203
204 d_data_ = rh.d_data_;
205 count_ = rh.count_;
206 stream_ = rh.stream_;
207
208 rh.d_data_ = nullptr ;
209 rh.count_ = 0;
210
211 return * this ;
212 }
213
214 ~ device_buffer () {
215 RMM_FREE(d_data_, stream_);
216 }
217 };
