39 template <typename T, std::size_t Extent, typename Derived>
40 class span_base {
41  static_assert(Extent == dynamic_extent, "Only dynamic extent is supported");
42 
43  public:
44  using element_type = T;
45  using value_type = std::remove_cv<T>;
46  using size_type = std::size_t;
47  using difference_type = std::ptrdiff_t;
48  using pointer = T*;
49  using iterator = T*;
50  using const_pointer = T const*;
51  using reference = T&;
52  using const_reference = T const&;
53 
54  static constexpr std::size_t extent = Extent;
55 
56  constexpr span_base() noexcept : _data(nullptr), _size(0) {}
57  constexpr span_base(pointer data, size_type size) : _data(data), _size(size) {}
58  // constexpr span_base(pointer begin, pointer end) : _data(begin), _size(end - begin) {}
59  constexpr span_base(span_base const& other) noexcept = default;
60  constexpr span_base& operator=(span_base const& other) noexcept = default;
61 
62  // not noexcept due to undefined behavior when size = 0
63  constexpr reference front() const { return _data[0]; }
64  // not noexcept due to undefined behavior when size = 0
65  constexpr reference back() const { return _data[_size - 1]; }
66  // not noexcept due to undefined behavior when idx < 0 || idx >= size
67  constexpr reference operator[](size_type idx) const { return _data[idx]; }
68 
69  constexpr iterator begin() const noexcept { return _data; }
70  constexpr iterator end() const noexcept { return _data + _size; }
71  constexpr pointer data() const noexcept { return _data; }
72 
73  constexpr size_type size() const noexcept { return _size; }
74  constexpr size_type size_bytes() const noexcept { return sizeof(T) * _size; }
75  constexpr bool empty() const noexcept { return _size == 0; }
76 
82  constexpr Derived first(size_type count) const noexcept { return Derived(_data, count); }
83 
89  constexpr Derived last(size_type count) const noexcept
90  {
91  return Derived(_data + _size - count, count);
92  }
93 
94  constexpr Derived subspan(size_type offset, size_type count) const noexcept
95  {
96  return Derived(_data + offset, count);
97  }
98 
99  private:
100  pointer _data;
