92 template <typename From, typename To>
93 struct fixed_width_type_converter {
94  // Are the types same - simply copy elements from [begin, end) to out
95  template <typename FromT = From,
96  typename ToT = To,
97  typename std::enable_if<std::is_same<FromT, ToT>::value, void>::type* = nullptr>
98  constexpr ToT operator()(FromT element) const
99  {
100  return element;
101  }
102 
103  // Are the types convertible or can target be constructed from source?
104  template <typename FromT = From,
105  typename ToT = To,
106  typename std::enable_if<!std::is_same<FromT, ToT>::value &&
107  (cudf::is_convertible<FromT, ToT>::value ||
108  std::is_constructible<ToT, FromT>::value),
109  void>::type* = nullptr>
110  constexpr ToT operator()(FromT element) const
111  {
112  return static_cast<ToT>(element);
113  }
114 
115  // Convert integral values to timestamps
116  template <
117  typename FromT = From,
118  typename ToT = To,
119  typename std::enable_if<std::is_integral<FromT>::value && cudf::is_timestamp_t<ToT>::value,
120  void>::type* = nullptr>
121  constexpr ToT operator()(FromT element) const
122  {
123  return ToT{typename ToT::duration{element}};
124  }
125 };

137 template <typename ElementTo,
138  typename ElementFrom,
139  typename InputIterator,
140  typename std::enable_if_t<not cudf::is_fixed_point<ElementTo>()>* = nullptr>
141 rmm::device_buffer make_elements(InputIterator begin, InputIterator end)
142 {
143  static_assert(cudf::is_fixed_width<ElementTo>(), "Unexpected non-fixed width type.");
144  auto transformer = fixed_width_type_converter<ElementFrom, ElementTo>{};
145  auto transform_begin = thrust::make_transform_iterator(begin, transformer);
146  auto const size = cudf::distance(begin, end);
147  auto const elements = thrust::host_vector<ElementTo>(transform_begin, transform_begin + size);
148  return rmm::device_buffer{elements.data(), size * sizeof(ElementTo)};
149 }

217 template <typename ValidityIterator>
218 std::vector<bitmask_type> make_null_mask_vector(ValidityIterator begin, ValidityIterator end)
219 {
220  auto const size = cudf::distance(begin, end);
221  auto const num_words = cudf::bitmask_allocation_size_bytes(size) / sizeof(bitmask_type);
222 
223  auto null_mask = std::vector<bitmask_type>(num_words, 0);
224  for (auto i = 0; i < size; ++i)
225  if (*(begin + i)) set_bit_unsafe(null_mask.data(), i);
226 
227  return null_mask;

265 template <typename StringsIterator, typename ValidityIterator>
266 auto make_chars_and_offsets(StringsIterator begin, StringsIterator end, ValidityIterator v)
267 {
268  std::vector<char> chars{};
269  std::vector<cudf::size_type> offsets(1, 0);
270  for (auto str = begin; str < end; ++str) {
271  std::string tmp = (*v++) ? std::string(*str) : std::string{};
272  chars.insert(chars.end(), std::cbegin(tmp), std::cend(tmp));
273  offsets.push_back(offsets.back() + tmp.length());
274  }
275  return std::make_pair(std::move(chars), std::move(offsets));
276 };

1488  // nulls are represented as a repeated offset
1489  size_type ret = count;
1490  if (valid) { count += col.size(); }
1491  return ret;
1492  });

1496  cudf::test::fixed_width_column_wrapper<size_type>(offsetv.begin(), offsetv.end()).release();

1501  std::cend(cols),

1604  std::pair<std::vector<column_view>, std::vector<std::unique_ptr<column>>> preprocess_columns(
1605  std::initializer_list<lists_column_wrapper<T, SourceElementT>> const& elements,
1606  column_view& expected_hierarchy,
1607  int expected_depth)
1608  {
1609  std::vector<std::unique_ptr<column>> stubs;
1610  std::vector<column_view> cols;
1611 
1612  // preprocess the incoming lists.
1613  // - unwrap any "root" lists
1614  // - handle incomplete hierarchies
1615  std::transform(elements.begin(),
1616  elements.end(),
1617  std::back_inserter(cols),
1618  [&](lists_column_wrapper const& l) -> column_view {
1619  // depth mismatch. attempt to normalize the short column.
1620  // this function will also catch if this is a legitmately broken
1621  // set of input
1622  if (l.depth < expected_depth) {
1623  if (l.root) {
1624  // this exception distinguishes between the following two cases:
1625  //
1626  // { {{{1, 2, 3}}}, {} }
1627  // In this case, row 0 is a List<List<List<int>>>, whereas row 1 is
1628  // just a List<> which is an apparent mismatch. However, because row 1
1629  // is empty we will allow that to semantically mean
1630  // "a List<List<List<int>>> that's empty at the top level"
1631  //
1632  // { {{{1, 2, 3}}}, {4, 5, 6} }
1633  // In this case, row 1 is a a concrete List<int> with actual values.
1634  // There is no way to rectify the differences so we will treat it as a
1635  // true column mismatch.
1636  CUDF_EXPECTS(l.wrapped->size() == 0, "Mismatch in column types!");
1637  stubs.push_back(empty_like(expected_hierarchy));
1638  } else {
1639  stubs.push_back(normalize_column(l.get_view(), expected_hierarchy));
1640  }
1641  return *(stubs.back());
1642  }
1643  // the empty hierarchy case
1644  return l.get_view();
1645  });
1646 
1647  return {std::move(cols), std::move(stubs)};
1648  }
1649 
1650  column_view get_view() const { return root ? lists_column_view(*wrapped).child() : *wrapped; }
1651 
1652  int depth = 0;
1653  bool root = false;
1654 };
