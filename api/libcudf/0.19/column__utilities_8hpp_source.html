170 template <typename T, typename std::enable_if_t<cudf::is_fixed_point<T>()>* = nullptr>
171 std::pair<thrust::host_vector<T>, std::vector<bitmask_type>> to_host(column_view c)
172 {
173  using namespace numeric;
174  using Rep = typename T::rep;
175 
176  auto host_rep_types = thrust::host_vector<Rep>(c.size());
177 
178  CUDA_TRY(cudaMemcpy(
179  host_rep_types.data(), c.begin<Rep>(), c.size() * sizeof(Rep), cudaMemcpyDeviceToHost));
180 
181  auto to_fp = [&](Rep val) { return T{scaled_integer<Rep>{val, scale_type{c.type().scale()}}}; };
182  auto begin = thrust::make_transform_iterator(std::cbegin(host_rep_types), to_fp);
183  auto const host_fixed_points = thrust::host_vector<T>(begin, begin + c.size());
184 
185  return {host_fixed_points, bitmask_to_host(c)};
186 }

198 template <>
199 inline std::pair<thrust::host_vector<std::string>, std::vector<bitmask_type>> to_host(column_view c)
200 {
201  auto strings_data = cudf::strings::create_offsets(strings_column_view(c));
202  thrust::host_vector<char> h_chars(strings_data.first);
203  thrust::host_vector<size_type> h_offsets(strings_data.second);
204 
205  // build std::string vector from chars and offsets
206  std::vector<std::string> host_data;
207  host_data.reserve(c.size());
208 
209  // When C++17, replace this loop with std::adjacent_difference()
210  for (size_type idx = 0; idx < c.size(); ++idx) {
211  auto offset = h_offsets[idx];
212  auto length = h_offsets[idx + 1] - offset;
213  host_data.push_back(std::string(h_chars.data() + offset, length));
214  }
215 
216  return {host_data, bitmask_to_host(c)};
217 }
