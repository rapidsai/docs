91 __device__ inline size_type string_view::length() const
92 {
93  if (_length == UNKNOWN_STRING_LENGTH)
94  _length = strings::detail::characters_in_string(_data, _bytes);
95  if (_length && (_char_width == UNKNOWN_CHAR_WIDTH)) {
96  uint8_t const* ptr = reinterpret_cast<uint8_t const*>(data());
97  auto const first = strings::detail::bytes_in_utf8_byte(*ptr);
98  // see if they are all the same width
99  _char_width = (thrust::find_if(thrust::seq,
100  ptr,
101  ptr + size_bytes(),
102  [first](auto ch) {
103  auto width = strings::detail::bytes_in_utf8_byte(ch);
104  return (width != 0) && (width != first);
105  })) == (ptr + size_bytes())
106  ? first
107  : VARIABLE_CHAR_WIDTH;
108  }
109  return _length;
110 }

249 __device__ inline size_type string_view::byte_offset(size_type pos) const
250 {
251  size_type offset = 0;
252  const char* sptr = _data;
253  const char* eptr = sptr + _bytes;
254  if (_char_width > 0) return pos * _char_width;
255  while ((pos > 0) && (sptr < eptr)) {
256  size_type charbytes = strings::detail::bytes_in_utf8_byte(static_cast<uint8_t>(*sptr++));
257  if (charbytes) --pos;
258  offset += charbytes;
259  }
260  return offset;
261 }

268 __device__ inline int string_view::compare(const char* data, size_type bytes) const
269 {
270  size_type const len1 = size_bytes();
271  const unsigned char* ptr1 = reinterpret_cast<const unsigned char*>(this->data());
272  const unsigned char* ptr2 = reinterpret_cast<const unsigned char*>(data);
273  size_type idx = 0;
274  for (; (idx < len1) && (idx < bytes); ++idx) {
275  if (*ptr1 != *ptr2) return static_cast<int32_t>(*ptr1) - static_cast<int32_t>(*ptr2);
276  ++ptr1;
277  ++ptr2;
278  }
279  if (idx < len1) return 1;
280  if (idx < bytes) return -1;
281  return 0;
282 }

323 __device__ inline size_type string_view::find(const char* str,
324  size_type bytes,
325  size_type pos,
326  size_type count) const
327 {
328  const char* sptr = data();
329  if (!str || !bytes) return -1;
330  size_type nchars = length();
331  if (count < 0) count = nchars;
332  size_type end = pos + count;
333  if (end < 0 || end > nchars) end = nchars;
334  size_type spos = byte_offset(pos);
335  size_type epos = byte_offset(end);
336 
337  size_type len2 = bytes;
338  size_type len1 = (epos - spos) - len2 + 1;
339 
340  const char* ptr1 = sptr + spos;
341  const char* ptr2 = str;
342  for (size_type idx = 0; idx < len1; ++idx) {
343  bool match = true;
344  for (size_type jdx = 0; match && (jdx < len2); ++jdx) match = (ptr1[jdx] == ptr2[jdx]);
345  if (match) return character_offset(idx + spos);
346  ptr1++;
347  }
348  return -1;
349 }
