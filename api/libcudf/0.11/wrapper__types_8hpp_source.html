74 template <typename T, gdf_dtype type_id>
75 struct wrapper
76 {
77  static constexpr gdf_dtype corresponding_column_type{type_id};
78  using value_type = T;
79  value_type value;
80 
81  template <gdf_dtype dtype = type_id,
82  std::enable_if_t<(dtype != GDF_BOOL8)> * = nullptr>
83  CUDA_HOST_DEVICE_CALLABLE
84  constexpr explicit wrapper(value_type v) : value{v} {}
85 
86  template <typename from_type, gdf_dtype dtype = type_id,
87  std::enable_if_t<(dtype == GDF_BOOL8)> * = nullptr>
88  CUDA_HOST_DEVICE_CALLABLE constexpr explicit wrapper(from_type v)
89  : value{static_cast<bool>(v)} {}
90 
91  template <gdf_dtype dtype = type_id,
92  std::enable_if_t<(dtype != GDF_BOOL8)> * = nullptr>
93  CUDA_HOST_DEVICE_CALLABLE explicit operator value_type() const
94  {
95  return this->value;
96  }
97 
98  template <gdf_dtype dtype = type_id,
99  std::enable_if_t<(dtype == GDF_BOOL8)> * = nullptr>
100  CUDA_HOST_DEVICE_CALLABLE explicit operator value_type() const
101  {
102  return static_cast<value_type>(static_cast<bool>(this->value));
103  }
104 
105  // enable conversion to arithmetic types *only* for the cudf::bool8 wrapper
106  // (defined later in this file as wrapper<int8_t, GDF_BOOL8>)
107  template <typename T_out, gdf_dtype dtype = type_id,
108  typename std::enable_if<(dtype == GDF_BOOL8) &&
109  std::is_arithmetic<T_out>::value,
110  int>::type* = nullptr>
111  CUDA_HOST_DEVICE_CALLABLE explicit operator T_out() const
112  {
113  // Casting a cudf::bool8 to arithmetic type should always be the same as
114  // casting a bool to arithmetic type, and not the same as casting the
115  // underlying type to arithmetic type. Therefore we cast the value to bool
116  // first, then the output type
117  return static_cast<T_out>(static_cast<bool>(this->value));
118  }
119 
120  wrapper(wrapper const& w) = default;
121 
122  wrapper() = default;
123 };
