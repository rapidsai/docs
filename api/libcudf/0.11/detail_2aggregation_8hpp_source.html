142 template < typename F>
143 decltype( auto ) aggregation_dispatcher( aggregation ::Kind k, F f){
144 switch (k){
145 case aggregation::SUM: return f.template operator()<aggregation::SUM>();
146 case aggregation::MIN: return f.template operator()<aggregation::MIN>();
147 case aggregation::MAX: return f.template operator()<aggregation::MAX>();
148 case aggregation::COUNT: return f.template operator()<aggregation::COUNT>();
149 case aggregation::MEAN: return f.template operator()<aggregation::MEAN>();
150 case aggregation::MEDIAN: return f.template operator()<aggregation::MEDIAN>();
151 case aggregation::QUANTILE: return f.template operator()<aggregation::QUANTILE>();
152 }
