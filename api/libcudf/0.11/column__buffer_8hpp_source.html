63 struct column_buffer {
64 using str_pair = thrust::pair<const char*, size_type>;
65
66 column_buffer (
67 data_type type, size_type size, cudaStream_t stream = 0,
68 rmm::mr::device_memory_resource* mr = rmm::mr::get_default_resource()) {
69 if (type. id () == type_id::STRING ) {
70 _strings.resize(size);
71 } else {
72 _data = create_data (type, size, stream, mr);
73 }
74 _null_mask = create_null_mask (size, ALL_NULL , stream, mr);
75 _null_count = 0;
76 }
77
78 auto data() { return _strings.size() ? _strings.data().get() : _data.data(); }
79 auto data_size() {
80 return std::max(_data.size(), _strings.size() * sizeof (str_pair));
81 }
82
83 template < typename T = u int 32_t>
84 auto null_mask() {
85 return static_cast<T*>(_null_mask.data());
86 }
87 auto null_mask_size() { return _null_mask.size(); };
88
89 auto & null_count() { return _null_count; }
90
91 rmm::device_vector<str_pair> _strings;
92 rmm::device_buffer _data{};
93 rmm::device_buffer _null_mask{};
94 size_type _null_count{0};
95 };
