44 template <JoinType join_type,
45 typename output_index_type>
46 gdf_error join_hash( cudf::table const & left_table,
47 cudf::table const & right_table,
48 gdf_column * const output_l,
49 gdf_column * const output_r,
50 bool flip_indices = false )
51 {
52
53 // Hash table is built on the right table.
54 // For inner joins, doesn't matter which table is build/probe, so we want
55 // to build the hash table on the smaller table.
56 if ((join_type == JoinType::INNER_JOIN) &&
57 (right_table. num_rows () > left_table. num_rows ()))
58 {
59 return join_hash<join_type, output_index_type>(right_table,
60 left_table,
61 output_l,
62 output_r,
63 true );
64 }
65
66 return compute_hash_join<join_type, output_index_type>(output_l,
67 output_r,
68 left_table,
69 right_table,
70 flip_indices);
71 }
