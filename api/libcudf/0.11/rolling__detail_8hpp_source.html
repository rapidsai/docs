29 // return true if ColumnType is arithmetic type or
30 // AggOp is min_op/max_op/count_op for wrapper (non-arithmetic) types
31 template < typename ColumnType, class AggOp, bool is_mean>
32 static constexpr bool is_supported()
33 {
34 constexpr bool comparable_countable_op =
35 std::is_same<AggOp, DeviceMin>::value ||
36 std::is_same<AggOp, DeviceMax>::value ||
37 std::is_same<AggOp, DeviceCount>::value;
38
39 constexpr bool timestamp_mean =
40 is_mean &&
41 std::is_same<AggOp, DeviceSum>::value &&
42 cudf::is_timestamp<ColumnType>();
43
44 return !std::is_same<ColumnType, cudf::string_view>::value &&
45 (cudf::is_numeric<ColumnType>() ||
46 comparable_countable_op ||
47 timestamp_mean);
48 }

50 // store functor
51 template < typename T, bool is_mean>
52 struct store_output_functor
53 {
54 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
55 {
56 out = val;
57 }
58 };
59
60 // Specialization for MEAN
61 template < typename _T>
62 struct store_output_functor <_T, true>
63 {
64 // SFINAE for non-bool types
65 template < typename T = _T,
66 std::enable_if_t<!(cudf::is_boolean<T>() || cudf::is_timestamp<T>())>* = nullptr >
67 CUDA_HOST_DEVICE_CALLABLE void operator ()(T &out, T &val, size_type count)
68 {
69 out = val / count;
70 }
71
72 // SFINAE for bool type
73 template <typename T = _T, std::enable_if_t<cudf::is_boolean<T>()>* = nullptr >
74 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
75 {
76 out = static_cast<int32_t>(val) / count;
77 }
78
79 // SFINAE for timestamp types
80 template <typename T = _T, std::enable_if_t<cudf::is_timestamp<T>()>* = nullptr >
81 CUDA_HOST_DEVICE_CALLABLE void operator()(T &out, T &val, size_type count)
82 {
83 out = val.time_since_epoch() / count;
84 }
85 };
