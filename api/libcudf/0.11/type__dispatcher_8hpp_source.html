264 // This pragma disables a compiler warning that complains about the valid usage
265 // of calling a __host__ functor from this function which is __host__ __device__
266 #pragma nv_exec_check_disable
267 template < template <cudf::type_ id > typename IdTypeMap = id_to_type_impl ,
268 typename Functor, typename ... Ts>
269 CUDA_HOST_DEVICE_CALLABLE constexpr decltype( auto ) type_dispatcher (
270 cudf :: data_type dtype, Functor f, Ts&&... args) {
271 switch (dtype.id()) {
272 case BOOL8 :
273 return f.template operator()< typename IdTypeMap<BOOL8>::type>(
274 std::forward<Ts>(args)...);
275 case INT8 :
276 return f.template operator()< typename IdTypeMap<INT8>::type>(
277 std::forward<Ts>(args)...);
278 case INT16 :
279 return f.template operator()< typename IdTypeMap<INT16>::type>(
280 std::forward<Ts>(args)...);
281 case INT32 :
282 return f.template operator()< typename IdTypeMap<INT32>::type>(
283 std::forward<Ts>(args)...);
284 case INT64 :
285 return f.template operator()< typename IdTypeMap<INT64>::type>(
286 std::forward<Ts>(args)...);
287 case FLOAT32 :
288 return f.template operator()< typename IdTypeMap<FLOAT32>::type>(
289 std::forward<Ts>(args)...);
290 case FLOAT64 :
291 return f.template operator()< typename IdTypeMap<FLOAT64>::type>(
292 std::forward<Ts>(args)...);
293 case STRING :
294 return f.template operator()< typename IdTypeMap<STRING>::type>(
295 std::forward<Ts>(args)...);
296 case TIMESTAMP_DAYS :
297 return f.template operator()< typename IdTypeMap<TIMESTAMP_DAYS>::type>(
298 std::forward<Ts>(args)...);
299 case TIMESTAMP_SECONDS :
300 return f.template operator()< typename IdTypeMap<TIMESTAMP_SECONDS>::type>(
301 std::forward<Ts>(args)...);
302 case TIMESTAMP_MILLISECONDS :
303 return f.template operator()< typename IdTypeMap<TIMESTAMP_MILLISECONDS>::type>(
304 std::forward<Ts>(args)...);
305 case TIMESTAMP_MICROSECONDS :
306 return f.template operator()< typename IdTypeMap<TIMESTAMP_MICROSECONDS>::type>(
307 std::forward<Ts>(args)...);
308 case TIMESTAMP_NANOSECONDS :
309 return f.template operator()< typename IdTypeMap<TIMESTAMP_NANOSECONDS>::type>(
310 std::forward<Ts>(args)...);
311 default : {
312 #ifndef __CUDA_ARCH__
313 CUDF_FAIL( "Unsupported type_id." );
314 #else
315 release_assert( false && "Unsuported type_id." );
316
317 // The following code will never be reached, but the compiler generates a
318 // warning if there isn't a return value.
319
320 // Need to find out what the return type is in order to have a default
321 // return value and solve the compiler warning for lack of a default
322 // return
323 using return_type =
324 decltype(f.template operator ()<int8_t>(std::forward<Ts>(args)...));
325 return return_type();
326 #endif
327 }
328 }
329 }
