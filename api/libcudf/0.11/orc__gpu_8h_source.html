30 struct CompressedStreamInfo {
31 CompressedStreamInfo () = default ;
32 explicit constexpr CompressedStreamInfo ( const uint8_t *compressed_data_,
33 size_t compressed_size_)
34 : compressed_data(compressed_data_),
35 compressed_data_size(compressed_size_),
36 uncompressed_data( nullptr ),
37 decctl( nullptr ),
38 decstatus( nullptr ),
39 copyctl( nullptr ),
40 num_compressed_blocks(0),
41 num_uncompressed_blocks(0),
42 max_uncompressed_size(0) {}
43 const uint8_t *compressed_data; // [in] base ptr to compressed stream data
44 uint8_t *uncompressed_data; // [in] base ptr to uncompressed stream data or NULL if not known yet
45 size_t compressed_data_size; // [in] compressed data size for this stream
46 gpu_inflate_input_s *decctl; // [in] base ptr to decompression structure to be filled
47 gpu_inflate_status_s *decstatus; // [in] results of decompression
48 gpu_inflate_input_s *copyctl; // [in] base ptr to copy structure to be filled for uncompressed blocks
49 uint32_t num_compressed_blocks; // [in,out] number of entries in decctl(in), number of compressed blocks(out)
50 uint32_t num_uncompressed_blocks; // [in,out] number of entries in copyctl(in), number of uncompressed blocks(out)
51 uint64_t max_uncompressed_size; // [out] maximum uncompressed data size
52 };

100 uint8_t encoding_kind; // column encoding kind (orc::ColumnEncodingKind)
