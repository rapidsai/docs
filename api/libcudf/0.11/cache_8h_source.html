164 template < typename T, typename FallbackFunc>
165 named_prog<T> getCached(
166 std::string const & name,
167 umap_str_shptr<T>& map,
168 FallbackFunc func) {
169
170 // Find memory cached T object
171 auto it = map.find(name);
172 if ( it != map.end()) {
173 return std::make_pair(name, it->second);
174 }
175 else { // Find file cached T object
176 bool successful_read = false ;
177 std::string serialized;
178 #if defined(JITIFY_USE_CACHE)
179 std::string file_name = getCacheDir() + name;
180 cacheFile file{file_name};
181 serialized = file.read();
182 successful_read = file.is_read_successful();
183 #endif
184 if (not successful_read) {
185 // JIT compile and write to file if possible
186 serialized = func().serialize();
187 #if defined(JITIFY_USE_CACHE)
188 file.write(serialized);
189 #endif
190 }
191 // Add deserialized T to cache and return
192 auto program = std::make_shared<T>(T::deserialize(serialized));
193 map[name] = program;
194 return std::make_pair(name, program);
195 }
196 }
