182 static inline int CountLeadingZeros32(uint32_t value) {
183 #if defined(__clang__) || defined(__GNUC__)
184  if (value == 0) return 32;
185  return static_cast<int>(__builtin_clz(value));
186 #elif defined(_MSC_VER)
187  unsigned long index;
188  return (_BitScanReverse(&index, static_cast<unsigned long>(value)))
189  ? 31 - static_cast<int>(index)
190  : 32;
191 #else
192  int bitpos = 0;
193  while (value != 0) {
194  value >>= 1;
195  ++bitpos;
196  }
197  return 32 - bitpos;
198 #endif
199 }
