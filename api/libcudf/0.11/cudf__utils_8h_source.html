27 inline void set_null_count( gdf_column & column) {
28 if (column.valid == nullptr ) {
29 column.null_count = 0;
30 }
31 else {
32 cudf::size_type valid_count;
33 CUDF_TRY(gdf_count_nonzero_mask(column.valid, column.size, &valid_count));
