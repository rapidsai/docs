171 inline block first_fit(std::set<block>& free_blocks, std::size_t size)
172 {
173  auto const iter = std::find_if(
174  free_blocks.cbegin(), free_blocks.cend(), [size](auto const& b) { return b.fits(size); });
175 
176  if (iter == free_blocks.cend()) {
177  return {};
178  } else {
179  // Remove the block from the free_list.
180  auto const b = *iter;
181  auto const i = free_blocks.erase(iter);
182 
183  if (b.size() > size) {
184  // Split the block and put the remainder back.
185  auto const split = b.split(size);
186  free_blocks.insert(i, split.second);
187  return split.first;
188  } else {
189  return b;
190  }
191  }
192 }

201 inline block coalesce_block(std::set<block>& free_blocks, block const& b)
202 {
203  if (!b.is_valid()) return b;
204 
205  // Find the right place (in ascending address order) to insert the block.
206  auto const next = free_blocks.lower_bound(b);
207  auto const previous = next == free_blocks.cbegin() ? next : std::prev(next);
208 
209  // Coalesce with neighboring blocks.
210  bool const merge_prev = previous->is_contiguous_before(b);
211  bool const merge_next = next != free_blocks.cend() && b.is_contiguous_before(*next);
212 
213  block merged{};
214  if (merge_prev && merge_next) {
215  merged = previous->merge(b).merge(*next);
216  free_blocks.erase(previous);
217  auto const i = free_blocks.erase(next);
218  free_blocks.insert(i, merged);
219  } else if (merge_prev) {
220  merged = previous->merge(b);
221  auto const i = free_blocks.erase(previous);
222  free_blocks.insert(i, merged);
223  } else if (merge_next) {
224  merged = b.merge(*next);
225  auto const i = free_blocks.erase(next);
226  free_blocks.insert(i, merged);
227  } else {
228  free_blocks.emplace(b);
229  merged = b;
230  }
231  return merged;
232 }

269  {

272  "Error, Initial arena size required to be a multiple of 256 bytes");
273  RMM_EXPECTS(maximum_size_ == default_maximum_size || maximum_size_ == align_up(maximum_size_),
274  "Error, Maximum arena size required to be a multiple of 256 bytes");

284  }
