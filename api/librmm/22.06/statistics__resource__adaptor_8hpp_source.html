49  struct counter {
50  int64_t value{0}; // Current value
51  int64_t peak{0}; // Max value of `value`
52  int64_t total{0}; // Sum of all added values
53 
54  counter& operator+=(int64_t val)
55  {
56  value += val;
57  total += val;
58  peak = std::max(value, peak);
59  return *this;
60  }
61 
62  counter& operator-=(int64_t val)
63  {
64  value -= val;
65  return *this;
66  }
67  };

156  void* do_allocate(std::size_t bytes, cuda_stream_view stream) override
157  {
158  void* ptr = upstream_->allocate(bytes, stream);
159 
160  // increment the stats
161  {
162  write_lock_t lock(mtx_);
163 
164  // Increment the allocation_count_ while we have the lock
165  bytes_ += bytes;
166  allocations_ += 1;
167  }

182  {
183  upstream_->deallocate(ptr, bytes, stream);
184 
185  {
186  write_lock_t lock(mtx_);
187 
188  // Decrement the current allocated counts.
189  bytes_ -= bytes;
190  allocations_ -= 1;
191  }
