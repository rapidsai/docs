177  block_type try_to_expand(std::size_t try_size, std::size_t min_size, cuda_stream_view stream)
178  {
179  while (try_size >= min_size) {
180  auto block = block_from_upstream(try_size, stream);
181  if (block.has_value()) {
182  current_pool_size_ += block.value().size();
183  return block.value();
184  }
185  if (try_size == min_size) {
186  break; // only try `size` once
187  }
188  try_size = std::max(min_size, try_size / 2);
189  }
190  RMM_LOG_ERROR("[A][Stream {}][Upstream {}B][FAILURE maximum pool size exceeded]",
191  fmt::ptr(stream.value()),
192  min_size);
193  RMM_FAIL("Maximum pool size exceeded", rmm::out_of_memory);
194  }

208  // NOLINTNEXTLINE(bugprone-easily-swappable-parameters)
209  void initialize_pool(thrust::optional<std::size_t> initial_size,
210  thrust::optional<std::size_t> maximum_size)
211  {
212  auto const try_size = [&]() {
213  if (not initial_size.has_value()) {
214  auto const [free, total] = (get_upstream()->supports_get_mem_info())
215  ? get_upstream()->get_mem_info(cuda_stream_legacy)
216  : rmm::detail::available_device_memory();
217  return rmm::detail::align_up(std::min(free, total / 2),
218  rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
219  }
220  return initial_size.value();
221  }();
222 
223  current_pool_size_ = 0; // try_to_expand will set this if it succeeds
224  maximum_pool_size_ = maximum_size;
225 
226  RMM_EXPECTS(try_size <= maximum_pool_size_.value_or(std::numeric_limits<std::size_t>::max()),
227  "Initial pool size exceeds the maximum pool size!");
228 
229  if (try_size > 0) {
230  auto const block = try_to_expand(try_size, try_size, cuda_stream_legacy);
231  this->insert_block(block, cuda_stream_legacy);
232  }
233  }

244  block_type expand_pool(std::size_t size, free_list& blocks, cuda_stream_view stream)
245  {
246  // Strategy: If maximum_pool_size_ is set, then grow geometrically, e.g. by halfway to the
247  // limit each time. If it is not set, grow exponentially, e.g. by doubling the pool size each
248  // time. Upon failure, attempt to back off exponentially, e.g. by half the attempted size,
249  // until either success or the attempt is less than the requested size.
250  return try_to_expand(size_to_grow(size), size, stream);
251  }

265  [[nodiscard]] std::size_t size_to_grow(std::size_t size) const
266  {
267  if (maximum_pool_size_.has_value()) {
268  auto const unaligned_remaining = maximum_pool_size_.value() - pool_size();
269  using rmm::detail::align_up;
270  auto const remaining = align_up(unaligned_remaining, rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
271  auto const aligned_size = align_up(size, rmm::detail::CUDA_ALLOCATION_ALIGNMENT);
272  return (aligned_size <= remaining) ? std::max(aligned_size, remaining / 2) : 0;
273  }
274  return std::max(size, pool_size());
275  };

285  {

311  {

313 #ifdef RMM_POOL_TRACK_ALLOCATIONS

333  block_type free_block(void* ptr, std::size_t size) noexcept
334  {
335 #ifdef RMM_POOL_TRACK_ALLOCATIONS
336  if (ptr == nullptr) return block_type{};
337  auto const iter = allocated_blocks_.find(static_cast<char*>(ptr));
338  RMM_LOGGING_ASSERT(iter != allocated_blocks_.end());
339 
340  auto block = *iter;
341  RMM_LOGGING_ASSERT(block.size() == rmm::detail::align_up(size, allocation_alignment));
342  allocated_blocks_.erase(iter);
343 
344  return block;
345 #else
346  auto const iter = upstream_blocks_.find(static_cast<char*>(ptr));
347  return block_type{static_cast<char*>(ptr), size, (iter != upstream_blocks_.end())};
348 #endif
349  }
