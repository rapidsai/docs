160  free_list blocks_from_upstream(cuda_stream_view stream)
161  {
162  void* ptr = get_upstream()->allocate(upstream_chunk_size_, stream);
163  block_type block{ptr};
164  upstream_blocks_.push_back(block);
165 
166  auto num_blocks = upstream_chunk_size_ / block_size_;
167 
168  auto block_gen = [ptr, this](int index) {
169  // NOLINTNEXTLINE(cppcoreguidelines-pro-bounds-pointer-arithmetic)
170  return block_type{static_cast<char*>(ptr) + index * block_size_};
171  };
172  auto first =
173  thrust::make_transform_iterator(thrust::make_counting_iterator(std::size_t{0}), block_gen);
174  return free_list(first, first + num_blocks);
175  }
