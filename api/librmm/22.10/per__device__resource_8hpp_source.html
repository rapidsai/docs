164 inline device_memory_resource* set_per_device_resource(cuda_device_id device_id,
165  device_memory_resource* new_mr)
166 {
167  std::lock_guard<std::mutex> lock{detail::map_lock()};
168  auto& map = detail::get_map();
169  auto const old_itr = map.find(device_id.value());
170  // If a resource didn't previously exist for `id`, return pointer to initial_resource
171  auto* old_mr = (old_itr == map.end()) ? detail::initial_resource() : old_itr->second;
172  map[device_id.value()] = (new_mr == nullptr) ? detail::initial_resource() : new_mr;
