174  block_type try_to_expand(std::size_t try_size, std::size_t min_size, cuda_stream_view stream)
175  {
176  while (try_size >= min_size) {
177  auto b = block_from_upstream(try_size, stream);
178  if (b.has_value()) {
179  current_pool_size_ += b.value().size();
180  return b.value();
181  }
182  if (try_size == min_size) break; // only try `size` once
183  try_size = std::max(min_size, try_size / 2);
184  }
185  RMM_LOG_ERROR("[A][Stream {}][Upstream {}B][FAILURE maximum pool size exceeded]",
186  fmt::ptr(stream.value()),
187  min_size);
188  RMM_FAIL("Maximum pool size exceeded", rmm::bad_alloc);
189  }

203  void initialize_pool(thrust::optional<std::size_t> initial_size,
204  thrust::optional<std::size_t> maximum_size)
205  {
206  auto const try_size = [&]() {
207  if (not initial_size.has_value()) {
208  std::size_t free{}, total{};
209  std::tie(free, total) = (get_upstream()->supports_get_mem_info())
210  ? get_upstream()->get_mem_info(cudaStreamLegacy)
211  : detail::available_device_memory();
212  return rmm::detail::align_up(std::min(free, total / 2), allocation_alignment);
213  } else {
214  return initial_size.value();
215  }
216  }();
217 
218  current_pool_size_ = 0; // try_to_expand will set this if it succeeds
219  maximum_pool_size_ = maximum_size;
220 
221  RMM_EXPECTS(try_size <= maximum_pool_size_.value_or(std::numeric_limits<std::size_t>::max()),
222  "Initial pool size exceeds the maximum pool size!");
223 
224  if (try_size > 0) {
225  auto const b = try_to_expand(try_size, try_size, cudaStreamLegacy);
226  this->insert_block(b, cudaStreamLegacy);
227  }
228  }

239  block_type expand_pool(std::size_t size, free_list& blocks, cuda_stream_view stream)
240  {
241  // Strategy: If maximum_pool_size_ is set, then grow geometrically, e.g. by halfway to the
242  // limit each time. If it is not set, grow exponentially, e.g. by doubling the pool size each
243  // time. Upon failure, attempt to back off exponentially, e.g. by half the attempted size,
244  // until either success or the attempt is less than the requested size.
245  return try_to_expand(size_to_grow(size), size, stream);
246  }

260  std::size_t size_to_grow(std::size_t size) const
261  {
262  if (maximum_pool_size_.has_value()) {
263  auto const unaligned_remaining = maximum_pool_size_.value() - pool_size();
264  auto const remaining = rmm::detail::align_up(unaligned_remaining, allocation_alignment);
265  auto const aligned_size = rmm::detail::align_up(size, allocation_alignment);
266  return (aligned_size <= remaining) ? std::max(aligned_size, remaining / 2) : 0;
267  } else
268  return std::max(size, pool_size());
269  };
