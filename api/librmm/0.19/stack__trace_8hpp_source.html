46 class stack_trace {
47  public:
48  stack_trace()
49  {
50 #if defined(RMM_ENABLE_STACK_TRACES)
51  const int MaxStackDepth = 64;
52  void* stack[MaxStackDepth];
53  auto const depth = backtrace(stack, MaxStackDepth);
54  stack_ptrs.insert(stack_ptrs.end(), &stack[0], &stack[depth]);
55 #endif // RMM_ENABLE_STACK_TRACES
56  }
57 
58  friend std::ostream& operator<<(std::ostream& os, const stack_trace& st)
59  {
60 #if defined(RMM_ENABLE_STACK_TRACES)
61  std::unique_ptr<char*, decltype(&::free)> strings(
62  backtrace_symbols(st.stack_ptrs.data(), st.stack_ptrs.size()), &::free);
63  if (strings.get() == nullptr) {
64  os << "But no stack trace could be found!" << std::endl;
65  } else {
67  for (std::size_t i = 0; i < st.stack_ptrs.size(); ++i) {
68  os << "#" << i << " in " << strings.get()[i] << std::endl;
69  }
70  }
71 #else
72  os << "stack traces disabled" << std::endl;
73 #endif // RMM_ENABLE_STACK_TRACES
74  return os;
75  };
