199  device_buffer& operator=(device_buffer const& other)
200  {
201  if (&other != this) {
202  // If the current capacity is large enough and the resources are
203  // compatible, just reuse the existing memory
204  if ((capacity() > other.size()) and _mr->is_equal(*other._mr)) {
205  resize(other.size(), other.stream());
206  copy(other.data(), other.size());
207  } else {
208  // Otherwise, need to deallocate and allocate new memory
209  deallocate();
210  set_stream(other.stream());
211  _mr = other._mr;
212  allocate(other.size());
213  copy(other.data(), other.size());
214  }

289  void resize(std::size_t new_size, cuda_stream_view stream = cuda_stream_view{})
290  {
291  set_stream(stream);
292  // If the requested size is smaller than the current capacity, just update
293  // the size without any allocations
294  if (new_size <= capacity()) {
295  _size = new_size;
296  } else {
297  void* const new_data = _mr->allocate(new_size, this->stream());
298  RMM_CUDA_TRY(
299  cudaMemcpyAsync(new_data, data(), size(), cudaMemcpyDefault, this->stream().value()));
300  deallocate();
301  _data = new_data;
302  _size = new_size;
303  _capacity = new_size;
304  }

320  void shrink_to_fit(cuda_stream_view stream = cuda_stream_view{})
321  {
322  set_stream(stream);
323  if (size() != capacity()) {
324  // Invoke copy ctor on self which only copies `[0, size())` and swap it
325  // with self. The temporary `device_buffer` will hold the old contents
326  // which will then be destroyed
327  auto tmp = device_buffer{*this, stream, _mr};
328  std::swap(tmp, *this);
329  }
