174  void insert(block_type const& b)
175  {
176  if (is_empty()) {
177  free_list::insert(cend(), b);
178  return;
179  }
180 
181  // Find the right place (in ascending ptr order) to insert the block
182  // Can't use binary_search because it's a linked list and will be quadratic
183  auto const next = std::find_if(begin(), end(), [b](block_type const& i) { return b < i; });
184  auto const previous = (next == cbegin()) ? next : std::prev(next);
185 
186  // Coalesce with neighboring blocks or insert the new block if it can't be coalesced
187  bool const merge_prev = previous->is_contiguous_before(b);
188  bool const merge_next = (next != cend()) && b.is_contiguous_before(*next);
189 
190  if (merge_prev && merge_next) {
191  *previous = previous->merge(b).merge(*next);
192  erase(next);
193  } else if (merge_prev) {
194  *previous = previous->merge(b);
195  } else if (merge_next) {
196  *next = b.merge(*next);
197  } else {
198  free_list::insert(next, b); // cannot be coalesced, just insert
199  }
200  }
201 
210  void insert(free_list&& other)
211  {
212  std::for_each(std::make_move_iterator(other.begin()),
213  std::make_move_iterator(other.end()),
214  [this](block_type&& b) { this->insert(std::move(b)); });
215  }
216 
225  block_type get_block(size_t size)
226  {
227  // find best fit block
228  auto const iter =
229  std::min_element(cbegin(), cend(), [size](block_type const& lhs, block_type const& rhs) {
230  return lhs.is_better_fit(size, rhs);
231  });
232 
233  if (iter != cend() && iter->fits(size)) {
234  // Remove the block from the free_list and return it.
235  block_type const found = *iter;
236  erase(iter);
237  return found;
238  }
239 
240  return block_type{}; // not found
241  }
