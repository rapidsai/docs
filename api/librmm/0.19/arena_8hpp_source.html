174 inline block first_fit(std::set<block>& free_blocks, std::size_t size)
175 {
176  auto const iter = std::find_if(
177  free_blocks.cbegin(), free_blocks.cend(), [size](auto const& b) { return b.fits(size); });
178 
179  if (iter == free_blocks.cend()) {
180  return {};
181  } else {
182  // Remove the block from the free_list.
183  auto const b = *iter;
184  auto const i = free_blocks.erase(iter);
185 
186  if (b.size() > size) {
187  // Split the block and put the remainder back.
188  auto const split = b.split(size);
189  free_blocks.insert(i, split.second);
190  return split.first;
191  } else {
192  return b;
193  }
194  }
195 }

204 inline block coalesce_block(std::set<block>& free_blocks, block const& b)
205 {
206  if (!b.is_valid()) return b;
207 
208  // Find the right place (in ascending address order) to insert the block.
209  auto const next = free_blocks.lower_bound(b);
210  auto const previous = next == free_blocks.cbegin() ? next : std::prev(next);
211 
212  // Coalesce with neighboring blocks.
213  bool const merge_prev = previous->is_contiguous_before(b);
214  bool const merge_next = next != free_blocks.cend() && b.is_contiguous_before(*next);
215 
216  block merged{};
217  if (merge_prev && merge_next) {
218  merged = previous->merge(b).merge(*next);
219  free_blocks.erase(previous);
220  auto const i = free_blocks.erase(next);
221  free_blocks.insert(i, merged);
222  } else if (merge_prev) {
223  merged = previous->merge(b);
224  auto const i = free_blocks.erase(previous);
225  free_blocks.insert(i, merged);
226  } else if (merge_next) {
227  merged = b.merge(*next);
228  auto const i = free_blocks.erase(next);
229  free_blocks.insert(i, merged);
230  } else {
231  free_blocks.emplace(b);
232  merged = b;
233  }
234  return merged;
235 }
