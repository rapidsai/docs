162  free_list blocks_from_upstream(cuda_stream_view stream)
163  {
164  void* p = upstream_mr_->allocate(upstream_chunk_size_, stream);
165  block_type b{p};
166  upstream_blocks_.push_back(b);
167 
168  auto num_blocks = upstream_chunk_size_ / block_size_;
169 
170  auto g = [p, this](int i) { return block_type{static_cast<char*>(p) + i * block_size_}; };
171  auto first = thrust::make_transform_iterator(thrust::make_counting_iterator(std::size_t{0}), g);
172  return free_list(first, first + num_blocks);
173  }
