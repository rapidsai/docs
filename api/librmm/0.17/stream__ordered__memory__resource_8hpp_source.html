204  virtual void* do_allocate(std::size_t bytes, cuda_stream_view stream) override
205  {
206  RMM_LOG_TRACE("[A][stream {:p}][{}B]", fmt::ptr(stream.value()), bytes);
207 
208  if (bytes <= 0) return nullptr;
209 
210  lock_guard lock(mtx_);
211 
212  auto stream_event = get_event(stream);
213 
214  bytes = rmm::detail::align_up(bytes, allocation_alignment);
215  RMM_EXPECTS(bytes <= this->underlying().get_maximum_allocation_size(),
216  rmm::bad_alloc,
217  "Maximum allocation size exceeded");
218  auto const b = this->underlying().get_block(bytes, stream_event);
219  auto split = this->underlying().allocate_from_block(b, bytes);
220  if (split.remainder.is_valid()) stream_free_blocks_[stream_event].insert(split.remainder);
221  RMM_LOG_TRACE("[A][stream {:p}][{}B][{:p}]",
222  fmt::ptr(stream_event.stream),
223  bytes,
224  fmt::ptr(split.allocated_pointer));
225 
226  log_summary_trace();
227 
228  return split.allocated_pointer;
229  }

238  virtual void do_deallocate(void* p, std::size_t bytes, cuda_stream_view stream) override
239  {
240  lock_guard lock(mtx_);
241  auto stream_event = get_event(stream);
242  RMM_LOG_TRACE("[D][stream {:p}][{}B][{:p}]", fmt::ptr(stream_event.stream), bytes, p);
243 
244  bytes = rmm::detail::align_up(bytes, allocation_alignment);
245  auto const b = this->underlying().free_block(p, bytes);
246 
247  // TODO: cudaEventRecord has significant overhead on deallocations. For the non-PTDS case
248  // we may be able to delay recording the event in some situations. But using events rather than
249  // streams allows stealing from deleted streams.
250  RMM_ASSERT_CUDA_SUCCESS(cudaEventRecord(stream_event.event, stream.value()));
251 
252  stream_free_blocks_[stream_event].insert(b);
253 
254  log_summary_trace();
255  }

280  stream_event_pair get_event(cuda_stream_view stream)
281  {
282  if (stream.is_per_thread_default()) {
283  // Create a thread-local shared event wrapper. Shared pointers in the thread and in each MR
284  // instance ensures it is destroyed cleaned up only after all are finished with it.
285  thread_local auto event_tls = std::make_shared<event_wrapper>();
286  default_stream_events.insert(event_tls);
287  return stream_event_pair{stream.value(), event_tls.get()->event};
288  }
289  // We use cudaStreamLegacy as the event map key for the default stream for consistency between
290  // PTDS and non-PTDS mode. In PTDS mode, the cudaStreamLegacy map key will only exist if the
291  // user explicitly passes it, so it is used as the default location for the free list
292  // at construction. For consistency, the same key is used for null stream free lists in non-PTDS
293  // mode.
294  auto const stream_to_store = stream.is_default() ? cudaStreamLegacy : stream.value();
295 
296  auto const iter = stream_events_.find(stream_to_store);
297  return (iter != stream_events_.end()) ? iter->second : [&]() {
298  stream_event_pair stream_event{stream_to_store};
299  RMM_ASSERT_CUDA_SUCCESS(
300  cudaEventCreateWithFlags(&stream_event.event, cudaEventDisableTiming));
301  stream_events_[stream_to_store] = stream_event;
302  return stream_event;
303  }();
304  }

313  block_type get_block(size_t size, stream_event_pair stream_event)
314  {
315  // Try to find a satisfactory block in free list for the same stream (no sync required)
316  auto iter = stream_free_blocks_.find(stream_event);
317  if (iter != stream_free_blocks_.end()) {
318  block_type b = iter->second.get_block(size);
319  if (b.is_valid()) { return b; }
320  }
321 
322  free_list& blocks =
323  (iter != stream_free_blocks_.end()) ? iter->second : stream_free_blocks_[stream_event];
324 
325  // Try to find an existing block in another stream
326  {
327  block_type const b = get_block_from_other_stream(size, stream_event, blocks, false);
328  if (b.is_valid()) return b;
329  }
330 
331  // no large enough blocks available on other streams, so sync and merge until we find one
332  {
333  block_type const b = get_block_from_other_stream(size, stream_event, blocks, true);
334  if (b.is_valid()) return b;
335  }
336 
337  log_summary_trace();
338 
339  // no large enough blocks available after merging, so grow the pool
340  return this->underlying().expand_pool(size, blocks, cuda_stream_view{stream_event.stream});
341  }

358  block_type get_block_from_other_stream(size_t size,
359  stream_event_pair stream_event,
360  free_list& blocks,
361  bool merge_first)
362  {
363  for (auto it = stream_free_blocks_.begin(), next_it = it; it != stream_free_blocks_.end();
364  it = next_it) {
365  ++next_it; // Points to element after `it` to allow erasing `it` in the loop body
366  auto other_event = it->first.event;
367  if (other_event != stream_event.event) {
368  auto other_blocks = it->second;
369 
370  block_type const b = [&]() {
371  if (merge_first) {
372  merge_lists(stream_event, blocks, other_event, std::move(other_blocks));
373 
374  RMM_LOG_DEBUG("[A][Stream {:p}][{}B][Merged stream {:p}]",
375  fmt::ptr(stream_event.stream),
376  size,
377  fmt::ptr(it->first.stream));
378 
379  stream_free_blocks_.erase(it);
380 
381  return blocks.get_block(size); // get the best fit block in merged lists
382  } else {
383  return other_blocks.get_block(size); // get the best fit block in other list
384  }
385  }();
386 
387  if (b.is_valid()) {
388  RMM_LOG_DEBUG((merge_first) ? "[A][Stream {:p}][{}B][Found after merging stream {:p}]"
389  : "[A][Stream {:p}][{}B][Taken from stream {:p}]",
390  fmt::ptr(stream_event.stream),
391  size,
392  fmt::ptr(it->first.stream));
393 
394  if (not merge_first) {
395  merge_lists(stream_event, blocks, other_event, std::move(other_blocks));
396  stream_free_blocks_.erase(it);
397  }
398 
399  return b;
400  }
401  }
402  }
403  return block_type{};
404  }

406  void merge_lists(stream_event_pair stream_event,
407  free_list& blocks,
408  cudaEvent_t other_event,

410  {
411  // Since we found a block associated with a different stream, we have to insert a wait
412  // on the stream's associated event into the allocating stream.
413  RMM_CUDA_TRY(cudaStreamWaitEvent(stream_event.stream, other_event, 0));
414 
415  // Merge the two free lists
416  blocks.insert(std::move(other_blocks));
417  }

437  void log_summary_trace()

440  std::size_t num_blocks{0};

444  stream_free_blocks_.cend(),

446  num_blocks += freelist.second.size();

448  max_block = std::max(summary.first, max_block);
