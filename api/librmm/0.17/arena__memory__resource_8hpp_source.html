164  void deallocate_from_other_arena(void* p, std::size_t bytes, cuda_stream_view stream)
165  {
166  stream.synchronize_no_throw();
167 
168  read_lock lock(mtx_);
169 
170  if (use_per_thread_arena(stream)) {
171  auto const id = std::this_thread::get_id();
172  for (auto& kv : thread_arenas_) {
173  // If the arena does not belong to the current thread, try to deallocate from it, and return
174  // if successful.
175  if (kv.first != id && kv.second->deallocate(p, bytes)) return;
176  }
177  } else {
178  for (auto& kv : stream_arenas_) {
179  // If the arena does not belong to the current stream, try to deallocate from it, and return
180  // if successful.
181  if (stream != kv.first && kv.second.deallocate(p, bytes)) return;
182  }
183  }
184 
185  // The thread that originally allocated the block has terminated, deallocate directly in the
186  // global arena.
187  global_arena_.deallocate({p, bytes});
188  }

197  {

199  return get_thread_arena();
200  } else {

202  }

210  arena& get_thread_arena()
211  {
212  auto const id = std::this_thread::get_id();
213  {
214  read_lock lock(mtx_);
215  auto const it = thread_arenas_.find(id);
216  if (it != thread_arenas_.end()) { return *it->second; }
217  }
218  {
219  write_lock lock(mtx_);
220  auto a = std::make_shared<arena>(global_arena_);
221  thread_arenas_.emplace(id, a);
222  thread_local detail::arena::arena_cleaner<Upstream> cleaner{a};
223  return *a;
224  }
225  }

232  arena& get_stream_arena(cuda_stream_view stream)
233  {
234  RMM_LOGGING_ASSERT(!use_per_thread_arena(stream));
235  {
236  read_lock lock(mtx_);
237  auto const it = stream_arenas_.find(stream.value());
238  if (it != stream_arenas_.end()) { return it->second; }
239  }
240  {
241  write_lock lock(mtx_);
242  stream_arenas_.emplace(stream.value(), global_arena_);
243  return stream_arenas_.at(stream.value());
244  }
245  }
