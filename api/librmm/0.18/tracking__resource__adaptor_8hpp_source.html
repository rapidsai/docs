142  void log_outstanding_allocations() const
143  {
144 #if SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_DEBUG
145  read_lock_t lock(mtx);
146  if (not allocations.empty()) {
147  std::ostringstream oss;
148  for (auto const& al : allocations) {
149  oss << al.first << ": " << al.second.allocation_size << " B";
150  if (al.second.strace != nullptr) {
151  oss << " : callstack:" << std::endl << *al.second.strace;
152  }
153  oss << std::endl;
154  }
155  RMM_LOG_DEBUG("Outstanding Allocations: {}", oss.str());
156  }
157 #endif // SPDLOG_ACTIVE_LEVEL <= SPDLOG_LEVEL_DEBUG
158  }

160  private:
174  void* do_allocate(std::size_t bytes, cuda_stream_view stream) override
175  {
176  void* p = upstream_->allocate(bytes, stream);
177 
178  // track it.
179  {
180  write_lock_t lock(mtx_);
181  allocations_.emplace(p, allocation_info{bytes, capture_stacks_});
182  }
183  allocated_bytes_ += bytes;
184 
185  return p;
186  }
187 
197  void do_deallocate(void* p, std::size_t bytes, cuda_stream_view stream) override
198  {
199  upstream_->deallocate(p, bytes, stream);
200  {
201  write_lock_t lock(mtx_);
202  allocations_.erase(p);
203  }
204  allocated_bytes_ -= bytes;
205  }
