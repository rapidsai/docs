137  void add_bin(std::size_t allocation_size, device_memory_resource* bin_resource = nullptr)
138  {
139  allocation_size = rmm::detail::align_up(allocation_size, allocation_alignment);
140 
141  if (nullptr != bin_resource)
142  resource_bins_.insert({allocation_size, bin_resource});
143  else {
144  // If the bin already exists, do nothing.
145  if (resource_bins_.count(allocation_size) == 0) {
146  owned_bin_resources_.push_back(
147  std::make_unique<fixed_size_memory_resource<Upstream>>(upstream_mr_, allocation_size));
148  resource_bins_.insert({allocation_size, owned_bin_resources_.back().get()});
149  }
150  }
