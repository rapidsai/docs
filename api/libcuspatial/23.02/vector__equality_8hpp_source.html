142 template <typename Vector1, typename Vector2>
143 inline void expect_vector_equivalent(Vector1 const& lhs, Vector2 const& rhs)
144 {
145  using T = typename Vector1::value_type;
146  static_assert(std::is_same_v<T, typename Vector2::value_type>, "Value type mismatch.");
147 
148  if constexpr (cuspatial::is_vec_2d<T>()) {
149  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(vec_2d_matcher(), to_host<T>(rhs)));
150  } else if constexpr (std::is_floating_point_v<T>) {
151  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(float_matcher(), to_host<T>(rhs)));
152  } else if constexpr (std::is_integral_v<T>) {
153  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(::testing::Eq(), to_host<T>(rhs)));
154  } else if constexpr (cuspatial::is_optional<T>) {
155  if constexpr (cuspatial::is_vec_2d<typename T::value_type>()) {
156  EXPECT_THAT(to_host<T>(lhs),
157  ::testing::Pointwise(optional_matcher(vec_2d_matcher()), to_host<T>(rhs)));
158  } else if constexpr (std::is_floating_point_v<typename T::value_type>) {
159  EXPECT_THAT(to_host<T>(lhs),
160  ::testing::Pointwise(optional_matcher(float_matcher()), to_host<T>(rhs)));
161  } else if constexpr (std::is_integral_v<typename T::value_type>) {
162  EXPECT_THAT(to_host<T>(lhs),
163  ::testing::Pointwise(optional_matcher(::testing::Eq()), to_host<T>(rhs)));
164  } else {
165  EXPECT_EQ(lhs, rhs);
166  }
167  } else {
168  EXPECT_EQ(lhs, rhs);
169  }
170 }

205 // unpack a `device_vector of structs comprising two `vec_2d`s each into two vectors of `vec_2d`.
206 // Works, e.g., with `cuspatial::box` or `cuspatial::segment`.
207 template <typename PairVector, typename T = typename PairVector::value_type::value_type>
208 std::pair<rmm::device_vector<vec_2d<T>>, rmm::device_vector<vec_2d<T>>> unpack_vec2d_pair_vector(
209  PairVector const& pairs)
210 {
211  using Pair = typename PairVector::value_type;
212 
213  auto first = rmm::device_vector<vec_2d<T>>(pairs.size());
214  auto second = rmm::device_vector<vec_2d<T>>(pairs.size());
215 
216  auto zipped_output = thrust::make_zip_iterator(first.begin(), second.begin());
217 
218  thrust::transform(pairs.begin(), pairs.end(), zipped_output, [] __device__(Pair const& pair) {
219  auto [a, b] = pair;
220  return thrust::make_tuple(a, b);
221  });
222  return {std::move(first), std::move(second)};
223 }
