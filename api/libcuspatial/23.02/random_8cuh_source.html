98 template <typename T>
99 class geometric_distribution : public thrust::random::normal_distribution<integral_to_realType<T>> {
100  using realType = integral_to_realType<T>;
101  using super_t = thrust::random::normal_distribution<realType>;
102  T _lower_bound;
103  T _upper_bound;
104 
105  public:
106  using result_type = T;
107  __host__ __device__ explicit geometric_distribution(T lower_bound, T upper_bound)
108  : super_t(0, std::labs(upper_bound - lower_bound) / 4.0),
109  _lower_bound(lower_bound),
110  _upper_bound(upper_bound)
111  {
112  }
113 
114  template <typename UniformRandomNumberGenerator>
115  __host__ __device__ result_type operator()(UniformRandomNumberGenerator& urng)
116  {
117  return _lower_bound < _upper_bound ? std::abs(super_t::operator()(urng)) + _lower_bound
118  : _lower_bound - std::abs(super_t::operator()(urng));
119  }
120 };
