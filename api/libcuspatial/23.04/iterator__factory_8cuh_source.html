122  typename std::enable_if_t<thrust::is_contiguous_iterator_v<Iter>>> {
123  using element_t = typename std::iterator_traits<Iter>::value_type;
124  using value_type = vec_2d<element_t>;

126  element_t const* ptr;

128  constexpr interleaved_to_vec_2d(Iter it) : ptr{&thrust::raw_reference_cast(*it)}
