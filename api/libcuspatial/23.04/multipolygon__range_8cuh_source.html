179 template <collection_type_id Type,
180  typename T,
181  typename IndexType,
182  typename GeometryColumnView,
183  CUSPATIAL_ENABLE_IF(Type == collection_type_id::SINGLE)>
184 auto make_multipolygon_range(GeometryColumnView const& polygons_column)
185 {
186  CUSPATIAL_EXPECTS(polygons_column.geometry_type() == geometry_type_id::POLYGON,
187  "Must be polygon geometry type.");
188  auto geometry_iter = thrust::make_counting_iterator(0);
189  auto const& part_offsets = polygons_column.offsets();
190  auto const& ring_offsets = polygons_column.child().child(0);
191  auto const& points_xy =
192  polygons_column.child().child(1).child(1); // Ignores x-y offset {0, 2, 4...}
193 
194  auto points_it = make_vec_2d_iterator(points_xy.template begin<T>());
195 
196  return multipolygon_range(geometry_iter,
197  geometry_iter + part_offsets.size(),
198  part_offsets.template begin<IndexType>(),
199  part_offsets.template end<IndexType>(),
200  ring_offsets.template begin<IndexType>(),
201  ring_offsets.template end<IndexType>(),
202  points_it,
203  points_it + points_xy.size() / 2);
204 }
