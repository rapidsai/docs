39 template <typename T>
40 struct multipolygon_generator_parameter {
41  using element_t = T;
42 
43  std::size_t num_multipolygons;
44  std::size_t num_polygons_per_multipolygon;
45  std::size_t num_holes_per_polygon;
46  std::size_t num_edges_per_ring;
47  vec_2d<T> centroid;

50  CUSPATIAL_HOST_DEVICE std::size_t num_polygons()
51  {
52  return num_multipolygons * num_polygons_per_multipolygon;
53  }
54  CUSPATIAL_HOST_DEVICE std::size_t num_rings() { return num_polygons() * num_rings_per_polygon(); }
55  CUSPATIAL_HOST_DEVICE std::size_t num_coords() { return num_rings() * num_vertices_per_ring(); }
56  CUSPATIAL_HOST_DEVICE std::size_t num_vertices_per_ring() { return num_edges_per_ring + 1; }
57  CUSPATIAL_HOST_DEVICE std::size_t num_rings_per_polygon() { return num_holes_per_polygon + 1; }
58  CUSPATIAL_HOST_DEVICE T hole_radius() { return radius / (num_holes_per_polygon + 1); }
59 };

74 template <typename T>
75 vec_2d<T> __device__ generate_ring_coordinate(std::size_t point_local_idx,
76  std::size_t num_edges,
77  vec_2d<T> centroid,
78  T radius)
79 {
80  // Overrides last coordinate to make sure ring is closed.
81  if (point_local_idx == num_edges) return vec_2d<T>{centroid.x + radius, centroid.y};
82 
83  T angle = (2.0 * M_PI * point_local_idx) / num_edges;
84 
85  return vec_2d<T>{centroid.x + radius * cos(angle), centroid.y + radius * sin(angle)};
86 }

100 template <typename T>
101 vec_2d<T> __device__ polygon_centroid_displacement(vec_2d<T> centroid,
102  std::size_t part_local_idx,

141 template <typename T>
142 vec_2d<T> __device__
143 ring_centroid_displacement(vec_2d<T> centroid, std::size_t ring_local_idx, T radius, T hole_radius)
144 {
145  // This is a shell
146  if (ring_local_idx == 0) { return centroid; }
147 
148  // This is a hole
149  ring_local_idx -= 1; // offset hole indices to be 0-based
150  T max_hole_displacement = radius - hole_radius;
151  T displacement_x = -max_hole_displacement + ring_local_idx * hole_radius * 2;
152  T displacement_y = 0.0;
153  return centroid + vec_2d<T>{displacement_x, displacement_y};
154 }

167 template <typename T, typename MultipolygonRange>
168 void __global__ generate_multipolygon_array_coordinates(MultipolygonRange multipolygons,
169  multipolygon_generator_parameter<T> params)
170 {
171  for (auto idx = threadIdx.x + blockIdx.x * blockDim.x; idx < multipolygons.num_points();
172  idx += gridDim.x * blockDim.x) {
173  auto ring_idx = multipolygons.ring_idx_from_point_idx(idx);
174  auto part_idx = multipolygons.part_idx_from_ring_idx(ring_idx);
175  auto geometry_idx = multipolygons.geometry_idx_from_part_idx(part_idx);
176 
177  auto point_local_idx = idx - params.num_vertices_per_ring() * ring_idx;
178  auto ring_local_idx = ring_idx - params.num_rings_per_polygon() * part_idx;
179  auto part_local_idx = part_idx - params.num_polygons_per_multipolygon * geometry_idx;
180 
181  auto centroid = ring_centroid_displacement(
182  polygon_centroid_displacement(params.centroid, part_local_idx, params.radius),
183  ring_local_idx,
184  params.radius,
185  params.hole_radius());
186 
187  if (ring_local_idx == 0) // Generate coordinate for shell
188  multipolygons.point_begin()[idx] = generate_ring_coordinate(
189  point_local_idx, params.num_edges_per_ring, centroid, params.radius);
190  else // Generate coordinate for holes
191  multipolygons.point_begin()[idx] = generate_ring_coordinate(
192  point_local_idx, params.num_edges_per_ring, centroid, params.hole_radius());
193  }
194 }

204 template <typename T>
205 auto generate_multipolygon_array(multipolygon_generator_parameter<T> params,
206  rmm::cuda_stream_view stream)
207 {
208  rmm::device_uvector<std::size_t> geometry_offsets(params.num_multipolygons + 1, stream);
209  rmm::device_uvector<std::size_t> part_offsets(params.num_polygons() + 1, stream);
210  rmm::device_uvector<std::size_t> ring_offsets(params.num_rings() + 1, stream);
211  rmm::device_uvector<vec_2d<T>> coordinates(params.num_coords(), stream);
212 
213  thrust::sequence(rmm::exec_policy(stream),
214  ring_offsets.begin(),
215  ring_offsets.end(),
216  std::size_t{0},
217  params.num_vertices_per_ring());
218 
219  thrust::sequence(rmm::exec_policy(stream),
220  part_offsets.begin(),
221  part_offsets.end(),
222  std::size_t{0},
223  params.num_rings_per_polygon());
224 
225  thrust::sequence(rmm::exec_policy(stream),
226  geometry_offsets.begin(),
227  geometry_offsets.end(),
228  std::size_t{0},
229  params.num_polygons_per_multipolygon);
230 
231  auto multipolygons = multipolygon_range(geometry_offsets.begin(),
232  geometry_offsets.end(),
233  part_offsets.begin(),
234  part_offsets.end(),
235  ring_offsets.begin(),
236  ring_offsets.end(),
237  coordinates.begin(),
238  coordinates.end());
239 
240  auto [tpb, nblocks] = grid_1d(multipolygons.num_points());
241 
242  generate_multipolygon_array_coordinates<T><<<nblocks, tpb, 0, stream>>>(multipolygons, params);
243 
244  CUSPATIAL_CHECK_CUDA(stream.value());
245 
246  return make_multipolygon_array<std::size_t, vec_2d<T>>(std::move(geometry_offsets),
247  std::move(part_offsets),
248  std::move(ring_offsets),
249  std::move(coordinates));
250 }
