294 template <collection_type_id Type,
295  typename T,
296  typename IndexType,
297  typename GeometryColumnView,
298  CUSPATIAL_ENABLE_IF(Type == collection_type_id::SINGLE)>
299 auto make_multilinestring_range(GeometryColumnView const& linestrings_column)
300 {
301  CUSPATIAL_EXPECTS(linestrings_column.geometry_type() == geometry_type_id::LINESTRING,
302  "Must be Linestring geometry type.");
303  auto geometry_iter = thrust::make_counting_iterator(0);
304  auto const& part_offsets = linestrings_column.offsets();
305  auto const& points_xy = linestrings_column.child().child(1); // Ignores x-y offset {0, 2, 4...}
306 
307  auto points_it = make_vec_2d_iterator(points_xy.template begin<T>());
308 
309  return multilinestring_range(geometry_iter,
310  geometry_iter + part_offsets.size(),
311  part_offsets.template begin<IndexType>(),
312  part_offsets.template end<IndexType>(),
313  points_it,
314  points_it + points_xy.size() / 2);
315 }
