242 auto make_multilinestring_range(GeometryColumnView const& linestrings_column)

244  CUSPATIAL_EXPECTS(linestrings_column.geometry_type() == geometry_type_id::LINESTRING,
245  "Must be Linestring geometry type.");
246  auto geometry_iter = thrust::make_counting_iterator(0);

252  return multilinestring_range(geometry_iter,
