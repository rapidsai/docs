224 template <typename T>
225 auto make_multipoints_array(std::initializer_list<std::initializer_list<vec_2d<T>>> inl)
226 {
227  std::vector<std::size_t> offsets{0};
228  std::transform(inl.begin(), inl.end(), std::back_inserter(offsets), [](auto multipoint) {
229  return multipoint.size();
230  });
231  std::inclusive_scan(offsets.begin(), offsets.end(), offsets.begin());
232 
233  std::vector<vec_2d<T>> coordinates = std::accumulate(
234  inl.begin(), inl.end(), std::vector<vec_2d<T>>{}, [](std::vector<vec_2d<T>>& init, auto cur) {
235  init.insert(init.end(), cur.begin(), cur.end());
236  return init;
237  });
238 
239  return multipoint_array{rmm::device_vector<std::size_t>(offsets),
240  rmm::device_vector<vec_2d<T>>(coordinates)};
241 }
