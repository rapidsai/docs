159 template <typename T>
160 auto make_multipoints_array(std::initializer_list<std::initializer_list<vec_2d<T>>> inl)
161 {
162  std::vector<std::size_t> offsets{0};
163  std::transform(inl.begin(), inl.end(), std::back_inserter(offsets), [](auto multipoint) {
164  return multipoint.size();
165  });
166  std::inclusive_scan(offsets.begin(), offsets.end(), offsets.begin());
167 
168  std::vector<vec_2d<T>> coordinates = std::accumulate(
169  inl.begin(), inl.end(), std::vector<vec_2d<T>>{}, [](std::vector<vec_2d<T>>& init, auto cur) {
170  init.insert(init.end(), cur.begin(), cur.end());
171  return init;
172  });
173 
174  return multipoint_array{rmm::device_vector<std::size_t>(offsets),
175  rmm::device_vector<vec_2d<T>>(coordinates)};
176 }
