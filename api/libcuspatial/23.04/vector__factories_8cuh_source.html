310 template <typename T>
311 auto make_multipoints_array(std::initializer_list<std::initializer_list<vec_2d<T>>> inl)
312 {
313  std::vector<std::size_t> offsets{0};
314  std::transform(inl.begin(), inl.end(), std::back_inserter(offsets), [](auto multipoint) {
315  return multipoint.size();
316  });
317  std::inclusive_scan(offsets.begin(), offsets.end(), offsets.begin());
318 
319  std::vector<vec_2d<T>> coordinates = std::accumulate(
320  inl.begin(), inl.end(), std::vector<vec_2d<T>>{}, [](std::vector<vec_2d<T>>& init, auto cur) {
321  init.insert(init.end(), cur.begin(), cur.end());
322  return init;
323  });
324 
325  return multipoint_array{rmm::device_vector<std::size_t>(offsets),
326  rmm::device_vector<vec_2d<T>>(coordinates)};
327 }
