141 template <typename Vector1, typename Vector2>
142 inline void expect_vector_equivalent(Vector1 const& lhs, Vector2 const& rhs)
143 {
144  using T = typename Vector1::value_type;
145  static_assert(std::is_same_v<T, typename Vector2::value_type>, "Value type mismatch.");
146 
147  if constexpr (cuspatial::is_vec_2d<T>()) {
148  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(vec_2d_matcher(), to_host<T>(rhs)));
149  } else if constexpr (std::is_floating_point_v<T>) {
150  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(float_matcher(), to_host<T>(rhs)));
151  } else if constexpr (std::is_integral_v<T>) {
152  EXPECT_THAT(to_host<T>(lhs), ::testing::Pointwise(::testing::Eq(), to_host<T>(rhs)));
153  } else if constexpr (cuspatial::is_optional<T>) {
154  if constexpr (cuspatial::is_vec_2d<typename T::value_type>()) {
155  EXPECT_THAT(to_host<T>(lhs),
156  ::testing::Pointwise(optional_matcher(vec_2d_matcher()), to_host<T>(rhs)));
157  } else if constexpr (std::is_floating_point_v<typename T::value_type>) {
158  EXPECT_THAT(to_host<T>(lhs),
159  ::testing::Pointwise(optional_matcher(float_matcher()), to_host<T>(rhs)));
160  } else if constexpr (std::is_integral_v<typename T::value_type>) {
161  EXPECT_THAT(to_host<T>(lhs),
162  ::testing::Pointwise(optional_matcher(::testing::Eq()), to_host<T>(rhs)));
163  } else {
164  EXPECT_EQ(lhs, rhs);
165  }
166  } else {
167  EXPECT_EQ(lhs, rhs);
168  }
169 }
