143 template <typename T>
144 auto make_multipoints_array(std::initializer_list<std::initializer_list<vec_2d<T>>> inl)
145 {
146  std::vector<std::size_t> offsets{0};
147  std::transform(inl.begin(), inl.end(), std::back_inserter(offsets), [](auto multipoint) {
148  return multipoint.size();
149  });
150  std::inclusive_scan(offsets.begin(), offsets.end(), offsets.begin());
151 
152  std::vector<vec_2d<T>> coordinates = std::accumulate(
153  inl.begin(), inl.end(), std::vector<vec_2d<T>>{}, [](std::vector<vec_2d<T>>& init, auto cur) {
154  init.insert(init.end(), cur.begin(), cur.end());
155  return init;
156  });
157 
158  return multipoint_array{rmm::device_vector<std::size_t>(offsets),
159  rmm::device_vector<vec_2d<T>>(coordinates)};
160 }
