34 template <typename vertex_t, typename edge_t, typename weight_t>
35 void exact_fa2(GraphCOOView<vertex_t, edge_t, weight_t> &graph,
36  float *pos,
37  const int max_iter = 500,
38  float *x_start = nullptr,
39  float *y_start = nullptr,
40  bool outbound_attraction_distribution = true,
41  bool lin_log_mode = false,
42  bool prevent_overlapping = false,
43  const float edge_weight_influence = 1.0,
44  const float jitter_tolerance = 1.0,
45  const float scaling_ratio = 2.0,
46  bool strong_gravity_mode = false,
47  const float gravity = 1.0,
48  bool verbose = false,
49  internals::GraphBasedDimRedCallback *callback = nullptr)
50 {
51  const edge_t e = graph.number_of_edges;
52  const vertex_t n = graph.number_of_vertices;
53 
54  float *d_repel{nullptr};
55  float *d_attract{nullptr};
56  float *d_old_forces{nullptr};
57  edge_t *d_mass{nullptr};
58  float *d_swinging{nullptr};
59  float *d_traction{nullptr};
60 
61  rmm::device_vector<float> repel(n * 2, 0);
62  rmm::device_vector<float> attract(n * 2, 0);
63  rmm::device_vector<float> old_forces(n * 2, 0);
64  // FA2 requires degree + 1.
65  rmm::device_vector<edge_t> mass(n, 1);
66  rmm::device_vector<float> swinging(n, 0);
67  rmm::device_vector<float> traction(n, 0);
68 
69  d_repel = repel.data().get();
70  d_attract = attract.data().get();
71  d_old_forces = old_forces.data().get();
72  d_mass = mass.data().get();
73  d_swinging = swinging.data().get();
74  d_traction = traction.data().get();
75 
76  int random_state = 0;
77  random_vector(pos, n * 2, random_state);
78 
79  if (x_start && y_start) {
80  copy(n, x_start, pos);
81  copy(n, y_start, pos + n);
82  }
83 
84  // Sort COO for coalesced memory access.
85  cudaStream_t stream = {nullptr};
86  sort(graph, stream);
87  CHECK_CUDA(stream);
88  // FIXME: this function should work on "stream"
89  graph.degree(d_mass, cugraph::DegreeDirection::OUT);
90  CHECK_CUDA(stream);
91 
92  const vertex_t *row = graph.src_indices;
93  const vertex_t *col = graph.dst_indices;
94  const weight_t *v = graph.edge_data;
95 
96  float speed = 1.f;
97  float speed_efficiency = 1.f;
98  float outbound_att_compensation = 1.f;
99  float jt = 0.f;
100 
101  if (outbound_attraction_distribution) {
102  int sum = thrust::reduce(rmm::exec_policy(nullptr)->on(nullptr), mass.begin(), mass.end());
103  outbound_att_compensation = sum / (float)n;
104  }
105 
106  if (callback) {
107  callback->setup<float>(n, 2);
108  callback->on_preprocess_end(pos);
109  }
110 
111  for (int iter = 0; iter < max_iter; ++iter) {
112  // Reset force arrays
113  fill(n * 2, d_repel, 0.f);
114  fill(n * 2, d_attract, 0.f);
115  fill(n, d_swinging, 0.f);
116  fill(n, d_traction, 0.f);
117 
118  // Exact repulsion
119  apply_repulsion<vertex_t>(pos, pos + n, d_repel, d_repel + n, d_mass, scaling_ratio, n);
120 
121  apply_gravity<vertex_t>(pos,
122  pos + n,
123  d_attract,
124  d_attract + n,
125  d_mass,
126  gravity,
127  strong_gravity_mode,
128  scaling_ratio,
129  n);
130 
131  apply_attraction<vertex_t, edge_t, weight_t>(row,
132  col,
133  v,
134  e,
135  pos,
136  pos + n,
137  d_attract,
138  d_attract + n,
139  d_mass,
140  outbound_attraction_distribution,
141  lin_log_mode,
142  edge_weight_influence,
143  outbound_att_compensation);
144 
145  compute_local_speed(d_repel,
146  d_repel + n,
147  d_attract,
148  d_attract + n,
149  d_old_forces,
150  d_old_forces + n,
151  d_mass,
152  d_swinging,
153  d_traction,
154  n);
155 
156  // Compute global swinging and traction values.
157  const float s =
158  thrust::reduce(rmm::exec_policy(nullptr)->on(nullptr), swinging.begin(), swinging.end());
159  const float t =
160  thrust::reduce(rmm::exec_policy(nullptr)->on(nullptr), traction.begin(), traction.end());
161 
162  adapt_speed<vertex_t>(jitter_tolerance, &jt, &speed, &speed_efficiency, s, t, n);
163 
164  apply_forces<vertex_t>(pos,
165  pos + n,
166  d_repel,
167  d_repel + n,
168  d_attract,
169  d_attract + n,
170  d_old_forces,
171  d_old_forces + n,
172  d_swinging,
173  speed,
174  n);
175 
176  if (callback) callback->on_epoch_end(pos);
177 
178  if (verbose) {
179  printf("iteration %i, speed: %f, speed_efficiency: %f, ", iter + 1, speed, speed_efficiency);
180  printf("jt: %f, ", jt);
181  printf("swinging: %f, traction: %f\n", s, t);
182  }
183  }
184 
185  if (callback) callback->on_train_end(pos);
186 }

188 } // namespace detail
