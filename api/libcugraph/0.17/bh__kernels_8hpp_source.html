70 __global__ __launch_bounds__(THREADS1, FACTOR1) void BoundingBoxKernel(int *restrict startd,
71  int *restrict childd,
72  int *restrict massd,
73  float *restrict posxd,
74  float *restrict posyd,
75  float *restrict maxxd,
76  float *restrict maxyd,
77  float *restrict minxd,
78  float *restrict minyd,
79  const int FOUR_NNODES,
80  const int NNODES,
81  const int N,
82  unsigned *restrict limiter,
83  float *restrict radiusd)
84 {
85  float val, minx, maxx, miny, maxy;
86  __shared__ float sminx[THREADS1], smaxx[THREADS1], sminy[THREADS1], smaxy[THREADS1];
87 
88  // initialize with valid data (in case #bodies < #threads)
89  minx = maxx = posxd[0];
90  miny = maxy = posyd[0];
91 
92  // scan all bodies
93  const int i = threadIdx.x;
94  const int inc = THREADS1 * gridDim.x;
95  for (int j = i + blockIdx.x * THREADS1; j < N; j += inc) {
96  val = posxd[j];
97  if (val < minx)
98  minx = val;
99  else if (val > maxx)
100  maxx = val;
101 
102  val = posyd[j];
103  if (val < miny)
104  miny = val;
105  else if (val > maxy)
106  maxy = val;
107  }
108 
109  // reduction in shared memory
110  sminx[i] = minx;
111  smaxx[i] = maxx;
112  sminy[i] = miny;
113  smaxy[i] = maxy;
114 
115  for (int j = THREADS1 / 2; j > i; j /= 2) {
116  __syncthreads();
117  const int k = i + j;
118  sminx[i] = minx = fminf(minx, sminx[k]);
119  smaxx[i] = maxx = fmaxf(maxx, smaxx[k]);
120  sminy[i] = miny = fminf(miny, sminy[k]);
121  smaxy[i] = maxy = fmaxf(maxy, smaxy[k]);
122  }
123 
124  if (i == 0) {
125  // write block result to global memory
126  const int k = blockIdx.x;
127  minxd[k] = minx;
128  maxxd[k] = maxx;
129  minyd[k] = miny;
130  maxyd[k] = maxy;
131  __threadfence();
132 
133  const int inc = gridDim.x - 1;
134  if (inc != atomicInc(limiter, inc)) return;
135 
136  // I'm the last block, so combine all block results
137  for (int j = 0; j <= inc; j++) {
138  minx = fminf(minx, minxd[j]);
139  maxx = fmaxf(maxx, maxxd[j]);
140  miny = fminf(miny, minyd[j]);
141  maxy = fmaxf(maxy, maxyd[j]);
142  }
143 
144  // compute 'radius'
145  atomicExch(radiusd, fmaxf(maxx - minx, maxy - miny) * 0.5f + 1e-5f);
146 
147  massd[NNODES] = -1;
148  startd[NNODES] = 0;
149  posxd[NNODES] = (minx + maxx) * 0.5f;
150  posyd[NNODES] = (miny + maxy) * 0.5f;
151 
152 #pragma unroll
153  for (int a = 0; a < 4; a++) childd[FOUR_NNODES + a] = -1;
154  }
155 }

176 __global__ __launch_bounds__(THREADS2,
177  FACTOR2) void TreeBuildingKernel(int *restrict childd,
178  const float *restrict posxd,
179  const float *restrict posyd,
180  const int NNODES,
181  const int N,
182  int *restrict maxdepthd,
183  int *restrict bottomd,
184  const float *restrict radiusd)
185 {
186  int j, depth;
187  float x, y, r;
188  float px, py;
189  int ch, n, locked, patch;
190 
191  // cache root data
192  const float radius = radiusd[0];
193  const float rootx = posxd[NNODES];
194  const float rooty = posyd[NNODES];
195 
196  int localmaxdepth = 1;
197  int skip = 1;
198  const int inc = blockDim.x * gridDim.x;
199  int i = threadIdx.x + blockIdx.x * blockDim.x;
200 
201  // iterate over all bodies assigned to thread
202  while (i < N) {
203  if (skip != 0) {
204  // new body, so start traversing at root
205  skip = 0;
206  n = NNODES;
207  depth = 1;
208  r = radius * 0.5f;
209 
210  x = rootx + ((rootx < (px = posxd[i])) ? (j = 1, r) : (j = 0, -r));
211 
212  y = rooty + ((rooty < (py = posyd[i])) ? (j |= 2, r) : (-r));
213  }
214 
215  // follow path to leaf cell
216  while ((ch = childd[n * 4 + j]) >= N) {
217  n = ch;
218  depth++;
219  r *= 0.5f;
220 
221  // determine which child to follow
222  x += ((x < px) ? (j = 1, r) : (j = 0, -r));
223 
224  y += ((y < py) ? (j |= 2, r) : (-r));
225  }
226 
227  if (ch != -2) {
228  // skip if child pointer is locked and try again later
229  locked = n * 4 + j;
230 
231  if (ch == -1) {
232  if (atomicCAS(&childd[locked], -1, i) == -1) {
233  if (depth > localmaxdepth) localmaxdepth = depth;
234 
235  i += inc; // move on to next body
236  skip = 1;
237  }
238  } else {
239  if (ch == atomicCAS(&childd[locked], ch, -2)) {
240  // try to lock
241  patch = -1;
242 
243  while (ch >= 0) {
244  depth++;
245 
246  // Add new cell
247  const int cell = atomicSub(bottomd, 1) - 1;
248  if (cell <= N) {
249  // out of cell memory
250  atomicExch(bottomd, N);
251  }
252 
253  if (patch != -1) childd[n * 4 + j] = cell;
254 
255  if (cell > patch) patch = cell;
256 
257  j = (x < posxd[ch]) ? 1 : 0;
258  if (y < posyd[ch]) j |= 2;
259 
260  childd[cell * 4 + j] = ch;
261  n = cell;
262  r *= 0.5f;
263 
264  x += ((x < px) ? (j = 1, r) : (j = 0, -r));
265 
266  y += ((y < py) ? (j |= 2, r) : (-r));
267 
268  ch = childd[n * 4 + j];
269 
270  if (r <= 1e-10) break;
271  }
272 
273  // Add new body
274  childd[n * 4 + j] = i;
275 
276  if (depth > localmaxdepth) localmaxdepth = depth;
277 
278  i += inc; // move on to next body
279  skip = 2;
280  }
281  }
282  }
283  __threadfence();
284 
285  if (skip == 2) childd[locked] = patch;
286  }
287 
288  // record maximum tree depth
289  if (localmaxdepth > 32) localmaxdepth = 32;
290 
291  atomicMax(maxdepthd, localmaxdepth);
292 }

318 __global__ __launch_bounds__(THREADS3,
319  FACTOR3) void SummarizationKernel(int *restrict countd,
320  const int *restrict childd,
321  volatile int *restrict massd,
322  float *restrict posxd,
323  float *restrict posyd,
324  const int NNODES,
325  const int N,
326  const int *restrict bottomd)
327 {
328  bool flag = 0;
329  float cm, px, py;
330  __shared__ int child[THREADS3 * 4];
331  __shared__ int mass[THREADS3 * 4];
332 
333  const int bottom = bottomd[0];
334  const int inc = blockDim.x * gridDim.x;
335  int k = (bottom & -32) + threadIdx.x + blockIdx.x * blockDim.x;
336  if (k < bottom) k += inc;
337 
338  const int restart = k;
339 
340  for (int j = 0; j < 5; j++) // wait-free pre-passes
341  {
342  // iterate over all cells assigned to thread
343  while (k <= NNODES) {
344  if (massd[k] < 0) {
345  for (int i = 0; i < 4; i++) {
346  const int ch = childd[k * 4 + i];
347  child[i * THREADS3 + threadIdx.x] = ch;
348 
349  if ((ch >= N) and ((mass[i * THREADS3 + threadIdx.x] = massd[ch]) < 0))
350  goto CONTINUE_LOOP;
351  }
352 
353  // all children are ready
354  cm = 0.0f;
355  px = 0.0f;
356  py = 0.0f;
357  int cnt = 0;
358 
359 #pragma unroll
360  for (int i = 0; i < 4; i++) {
361  const int ch = child[i * THREADS3 + threadIdx.x];
362  if (ch >= 0) {
363  const float m = (ch >= N) ? (cnt += countd[ch], mass[i * THREADS3 + threadIdx.x])
364  : (cnt++, massd[ch]);
365  // add child's contribution
366  cm += m;
367  px += posxd[ch] * m;
368  py += posyd[ch] * m;
369  }
370  }
371 
372  countd[k] = cnt;
373  const float m = 1.0f / cm;
374  posxd[k] = px * m;
375  posyd[k] = py * m;
376  __threadfence(); // make sure data are visible before setting mass
377  massd[k] = cm;
378  }
379 
380  CONTINUE_LOOP:
381  k += inc; // move on to next cell
382  }
383  k = restart;
384  }
385 
386  int j = 0;
387  // iterate over all cells assigned to thread
388  while (k <= NNODES) {
389  if (massd[k] >= 0) {
390  k += inc;
391  goto SKIP_LOOP;
392  }
393 
394  if (j == 0) {
395  j = 4;
396  for (int i = 0; i < 4; i++) {
397  const int ch = childd[k * 4 + i];
398 
399  child[i * THREADS3 + threadIdx.x] = ch;
400  if ((ch < N) or ((mass[i * THREADS3 + threadIdx.x] = massd[ch]) >= 0)) j--;
401  }
402 
403  } else {
404  j = 4;
405  for (int i = 0; i < 4; i++) {
406  const int ch = child[i * THREADS3 + threadIdx.x];
407  if ((ch < N) or (mass[i * THREADS3 + threadIdx.x] >= 0) or
408  ((mass[i * THREADS3 + threadIdx.x] = massd[ch]) >= 0))
409  j--;
410  }
411  }
412 
413  if (j == 0) {
414  // all children are ready
415  cm = 0.0f;
416  px = 0.0f;
417  py = 0.0f;
418  int cnt = 0;
419 
420 #pragma unroll
421  for (int i = 0; i < 4; i++) {
422  const int ch = child[i * THREADS3 + threadIdx.x];
423  if (ch >= 0) {
424  const float m =
425  (ch >= N) ? (cnt += countd[ch], mass[i * THREADS3 + threadIdx.x]) : (cnt++, massd[ch]);
426  // add child's contribution
427  cm += m;
428  px += posxd[ch] * m;
429  py += posyd[ch] * m;
430  }
431  }
432 
433  countd[k] = cnt;
434  const float m = 1.0f / cm;
435  posxd[k] = px * m;
436  posyd[k] = py * m;
437  flag = 1;
438  }
439 
440  // All children mass are computed we can update the current one
441  SKIP_LOOP:
442  __threadfence();
443  if (flag != 0) {
444  massd[k] = cm;
445  k += inc;
446  flag = 0;
447  }
448  }
449 }

454 __global__ __launch_bounds__(THREADS4, FACTOR4) void SortKernel(int *restrict sortd,
455  const int *restrict countd,
456  volatile int *restrict startd,
457  int *restrict childd,
458  const int NNODES,
459  const int N,
460  const int *restrict bottomd)
461 {
462  const int bottom = bottomd[0];
463  const int dec = blockDim.x * gridDim.x;
464  int k = NNODES + 1 - dec + threadIdx.x + blockIdx.x * blockDim.x;
465  int start;
466  int limiter = 0;
467 
468  // iterate over all cells assigned to thread
469  while (k >= bottom) {
470  // To control possible infinite loops
471  if (++limiter > NNODES) break;
472 
473  // Not a child so skip
474  if ((start = startd[k]) < 0) continue;
475 
476  int j = 0;
477  for (int i = 0; i < 4; i++) {
478  const int ch = childd[k * 4 + i];
479  if (ch >= 0) {
480  if (i != j) {
481  // move children to front (needed later for speed)
482  childd[k * 4 + i] = -1;
483  childd[k * 4 + j] = ch;
484  }
485  if (ch >= N) {
486  // child is a cell
487  startd[ch] = start;
488  start += countd[ch]; // add #bodies in subtree
489  } else if (start <= NNODES and start >= 0) {
490  // child is a body
491  sortd[start++] = ch;
492  }
493  j++;
494  }
495  }
496  k -= dec; // move on to next cell
497  }
498 }

503 __global__ __launch_bounds__(
504  THREADS5, FACTOR5) void RepulsionKernel(/* int *restrict errd, */
505  const float scaling_ratio,
506  const float theta,
507  const float epssqd, // correction for zero distance
508  const int *restrict sortd,
509  const int *restrict childd,
510  const int *restrict massd,
511  const float *restrict posxd,
512  const float *restrict posyd,
513  float *restrict velxd,
514  float *restrict velyd,
515  const float theta_squared,
516  const int NNODES,
517  const int FOUR_NNODES,
518  const int N,
519  const float *restrict radiusd_squared,
520  const int *restrict maxdepthd)
521 {
522  __shared__ int pos[THREADS5], node[THREADS5];
523  __shared__ float dq[THREADS5];
524 
525  if (threadIdx.x == 0) {
526  const int max_depth = maxdepthd[0];
527  dq[0] = __fdividef(radiusd_squared[0], theta_squared);
528 
529  for (int i = 1; i < max_depth; i++) {
530  dq[i] = dq[i - 1] * 0.25f;
531  dq[i - 1] += epssqd;
532  }
533  dq[max_depth - 1] += epssqd;
534  }
535 
536  __syncthreads();
537  // figure out first thread in each warp (lane 0)
538  // const int base = threadIdx.x / 32;
539  // const int sbase = base * 32;
540  const int sbase = (threadIdx.x / 32) * 32;
541  const bool SBASE_EQ_THREAD = (sbase == threadIdx.x);
542 
543  const int diff = threadIdx.x - sbase;
544  // make multiple copies to avoid index calculations later
545  // Always true
546  dq[diff + sbase] = dq[diff];
547 
548  __threadfence_block();
549 
550  // iterate over all bodies assigned to thread
551  const int MAX_SIZE = FOUR_NNODES + 4;
552 
553  for (int k = threadIdx.x + blockIdx.x * blockDim.x; k < N; k += blockDim.x * gridDim.x) {
554  const int i = sortd[k]; // get permuted/sorted index
555  // cache position info
556  if (i < 0 or i >= MAX_SIZE) continue;
557 
558  const float px = posxd[i];
559  const float py = posyd[i];
560 
561  float vx = 0.0f;
562  float vy = 0.0f;
563 
564  // initialize iteration stack, i.e., push root node onto stack
565  int depth = sbase;
566 
567  if (SBASE_EQ_THREAD == true) {
568  pos[sbase] = 0;
569  node[sbase] = FOUR_NNODES;
570  }
571 
572  do {
573  // stack is not empty
574  int pd = pos[depth];
575  int nd = node[depth];
576 
577  while (pd < 4 && depth < THREADS5) {
578  const int index = nd + pd++;
579  if (index < 0 or index >= MAX_SIZE) break;
580 
581  const int n = childd[index]; // load child pointer
582 
583  // Non child
584  if (n < 0 or n > NNODES) break;
585 
586  const float dx = px - posxd[n];
587  const float dy = py - posyd[n];
588  const float dxy1 = dx * dx + dy * dy + epssqd;
589 
590  if ((n < N) or __all_sync(__activemask(), dxy1 >= dq[depth])) {
591  const float tdist_2 = __fdividef(scaling_ratio * massd[i] * massd[n], dxy1);
592  vx += dx * tdist_2;
593  vy += dy * tdist_2;
594  } else {
595  // push cell onto stack
596  if (SBASE_EQ_THREAD == true) {
597  pos[depth] = pd;
598  node[depth] = nd;
599  }
600  depth++;
601  pd = 0;
602  nd = n * 4;
603  }
604  }
605  } while (--depth >= sbase); // done with this level
606 
607  // update velocity
608  velxd[i] += vx;
609  velyd[i] += vy;
610  }
611 }

613 __global__ __launch_bounds__(THREADS6,
614  FACTOR6) void apply_forces_bh(float *restrict Y_x,
615  float *restrict Y_y,
616  const float *restrict attract_x,
617  const float *restrict attract_y,
618  const float *restrict repel_x,
619  const float *restrict repel_y,
620  float *restrict old_dx,
621  float *restrict old_dy,
622  const float *restrict swinging,
623  const float speed,
624  const int n)
625 {
626  // For evrery vertex
627  for (int i = threadIdx.x + blockIdx.x * blockDim.x; i < n; i += gridDim.x * blockDim.x) {
628  // Store displacement needed for next iteration.
629  const float dx = (repel_x[i] + attract_x[i]);
630  const float dy = (repel_y[i] + attract_y[i]);
631  old_dx[i] = dx;
632  old_dy[i] = dy;
633 
634  // Update positions
635  float factor = speed / (1.0 + sqrt(speed * swinging[i]));
636  Y_x[i] += dx * factor;
637  Y_y[i] += dy * factor;
638  }
639 }
