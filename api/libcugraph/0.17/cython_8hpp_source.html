51 // "container" for a graph type instance which insulates the owner from the
52 // specifics of the actual graph type. This is intended to be used in Cython
53 // code that only needs to pass a graph object to another wrapped C++ API. This
54 // greatly simplifies the Cython code since the Cython definition only needs to
55 // define the container and not the various individual graph types in Cython.
56 struct graph_container_t {
57  // FIXME: This union is in place only to support legacy calls, remove when
58  // migration to graph_t types is complete, or when legacy graph objects are
59  // constructed in the call_<<algo> wrappers instead of the
60  // populate_graph_container_legacy() function.
61  union graphPtrUnion {
62  ~graphPtrUnion() {}

64  void* null;
65  std::unique_ptr<GraphCSRView<int32_t, int32_t, float>> GraphCSRViewFloatPtr;
66  std::unique_ptr<GraphCSRView<int32_t, int32_t, double>> GraphCSRViewDoublePtr;
67  std::unique_ptr<GraphCSCView<int32_t, int32_t, float>> GraphCSCViewFloatPtr;

76  // The expected usage of a graph_container_t is for it to be created as part
77  // of a cython wrapper simply for passing a templated instantiation of a
78  // particular graph class from one call to another, and not to exist outside

81  // to an instance are not supported and these methods are deleted.
