35 template <typename vertex_t, typename edge_t, typename weight_t>
36 void barnes_hut(GraphCOOView<vertex_t, edge_t, weight_t> &graph,
37  float *pos,
38  const int max_iter = 500,
39  float *x_start = nullptr,
40  float *y_start = nullptr,
41  bool outbound_attraction_distribution = true,
42  bool lin_log_mode = false,
43  bool prevent_overlapping = false,
44  const float edge_weight_influence = 1.0,
45  const float jitter_tolerance = 1.0,
46  const float theta = 0.5,
47  const float scaling_ratio = 2.0,
48  bool strong_gravity_mode = false,
49  const float gravity = 1.0,
50  bool verbose = false,
51  internals::GraphBasedDimRedCallback *callback = nullptr)
52 {
53  cudaStream_t stream = {nullptr};
54  const edge_t e = graph.number_of_edges;
55  const vertex_t n = graph.number_of_vertices;
56 
57  const int blocks = getMultiProcessorCount();
58  // A tiny jitter to promote numerical stability/
59  const float epssq = 0.0025;
60  // We use the same array for nodes and cells.
61  int nnodes = n * 2;
62  if (nnodes < 1024 * blocks) nnodes = 1024 * blocks;
63  while ((nnodes & (32 - 1)) != 0) nnodes++;
64  nnodes--;
65 
66  // Allocate more space
67  //---------------------------------------------------
68  rmm::device_vector<unsigned> d_limiter(1);
69  rmm::device_vector<int> d_maxdepthd(1);
70  rmm::device_vector<int> d_bottomd(1);
71  rmm::device_vector<float> d_radiusd(1);
72 
73  unsigned *limiter = d_limiter.data().get();
74  int *maxdepthd = d_maxdepthd.data().get();
75  int *bottomd = d_bottomd.data().get();
76  float *radiusd = d_radiusd.data().get();
77 
78  InitializationKernel<<<1, 1, 0, stream>>>(limiter, maxdepthd, radiusd);
79  CHECK_CUDA(stream);
80 
81  const int FOUR_NNODES = 4 * nnodes;
82  const int FOUR_N = 4 * n;
83  const float theta_squared = theta * theta;
84  const int NNODES = nnodes;
85 
86  rmm::device_vector<int> d_startl(nnodes + 1, 0);
87  rmm::device_vector<int> d_childl((nnodes + 1) * 4, 0);
88  // FA2 requires degree + 1
89  rmm::device_vector<int> d_massl(nnodes + 1, 1.f);
90 
91  rmm::device_vector<float> d_maxxl(blocks * FACTOR1, 0);
92  rmm::device_vector<float> d_maxyl(blocks * FACTOR1, 0);
93  rmm::device_vector<float> d_minxl(blocks * FACTOR1, 0);
94  rmm::device_vector<float> d_minyl(blocks * FACTOR1, 0);
95 
96  // Actual mallocs
97  int *startl = d_startl.data().get();
98  int *childl = d_childl.data().get();
99  int *massl = d_massl.data().get();
100 
101  float *maxxl = d_maxxl.data().get();
102  float *maxyl = d_maxyl.data().get();
103  float *minxl = d_minxl.data().get();
104  float *minyl = d_minyl.data().get();
105 
106  // SummarizationKernel
107  rmm::device_vector<int> d_countl(nnodes + 1, 0);
108  int *countl = d_countl.data().get();
109 
110  // SortKernel
111  rmm::device_vector<int> d_sortl(nnodes + 1, 0);
112  int *sortl = d_sortl.data().get();
113 
114  // RepulsionKernel
115  rmm::device_vector<float> d_rep_forces((nnodes + 1) * 2, 0);
116  float *rep_forces = d_rep_forces.data().get();
117 
118  rmm::device_vector<float> d_radius_squared(1, 0);
119  float *radiusd_squared = d_radius_squared.data().get();
120 
121  rmm::device_vector<float> d_nodes_pos((nnodes + 1) * 2, 0);
122  float *nodes_pos = d_nodes_pos.data().get();
123 
124  // Initialize positions with random values
125  int random_state = 0;
126 
127  // Copy start x and y positions.
128  if (x_start && y_start) {
129  copy(n, x_start, nodes_pos);
130  copy(n, y_start, nodes_pos + nnodes + 1);
131  } else {
132  random_vector(nodes_pos, (nnodes + 1) * 2, random_state, stream);
133  }
134 
135  // Allocate arrays for force computation
136  float *attract{nullptr};
137  float *old_forces{nullptr};
138  float *swinging{nullptr};
139  float *traction{nullptr};
140 
141  rmm::device_vector<float> d_attract(n * 2, 0);
142  rmm::device_vector<float> d_old_forces(n * 2, 0);
143  rmm::device_vector<float> d_swinging(n, 0);
144  rmm::device_vector<float> d_traction(n, 0);
145 
146  attract = d_attract.data().get();
147  old_forces = d_old_forces.data().get();
148  swinging = d_swinging.data().get();
149  traction = d_traction.data().get();
150 
151  // Sort COO for coalesced memory access.
152  sort(graph, stream);
153  CHECK_CUDA(stream);
154 
155  graph.degree(massl, cugraph::DegreeDirection::OUT);
156  CHECK_CUDA(stream);
157 
158  const vertex_t *row = graph.src_indices;
159  const vertex_t *col = graph.dst_indices;
160  const weight_t *v = graph.edge_data;
161 
162  // Scalars used to adapt global speed.
163  float speed = 1.f;
164  float speed_efficiency = 1.f;
165  float outbound_att_compensation = 1.f;
166  float jt = 0.f;
167 
168  // If outboundAttractionDistribution active, compensate.
169  if (outbound_attraction_distribution) {
170  int sum =
171  thrust::reduce(rmm::exec_policy(stream)->on(stream), d_massl.begin(), d_massl.begin() + n);
172  outbound_att_compensation = sum / (float)n;
173  }
174 
175  //
176  // Set cache levels for faster algorithm execution
177  //---------------------------------------------------
178  cudaFuncSetCacheConfig(BoundingBoxKernel, cudaFuncCachePreferShared);
179  cudaFuncSetCacheConfig(TreeBuildingKernel, cudaFuncCachePreferL1);
180  cudaFuncSetCacheConfig(ClearKernel1, cudaFuncCachePreferL1);
181  cudaFuncSetCacheConfig(ClearKernel2, cudaFuncCachePreferL1);
182  cudaFuncSetCacheConfig(SummarizationKernel, cudaFuncCachePreferShared);
183  cudaFuncSetCacheConfig(SortKernel, cudaFuncCachePreferL1);
184  cudaFuncSetCacheConfig(RepulsionKernel, cudaFuncCachePreferL1);
185  cudaFuncSetCacheConfig(apply_forces_bh, cudaFuncCachePreferL1);
186 
187  if (callback) {
188  callback->setup<float>(nnodes + 1, 2);
189  callback->on_preprocess_end(nodes_pos);
190  }
191 
192  for (int iter = 0; iter < max_iter; ++iter) {
193  // Reset force values
194  fill((nnodes + 1) * 2, rep_forces, 0.f);
195  fill(n * 2, attract, 0.f);
196  fill(n, swinging, 0.f);
197  fill(n, traction, 0.f);
198 
199  ResetKernel<<<1, 1, 0, stream>>>(radiusd_squared, bottomd, NNODES, radiusd);
200  CHECK_CUDA(stream);
201 
202  // Compute bounding box arround all bodies
203  BoundingBoxKernel<<<blocks * FACTOR1, THREADS1, 0, stream>>>(startl,
204  childl,
205  massl,
206  nodes_pos,
207  nodes_pos + nnodes + 1,
208  maxxl,
209  maxyl,
210  minxl,
211  minyl,
212  FOUR_NNODES,
213  NNODES,
214  n,
215  limiter,
216  radiusd);
217  CHECK_CUDA(stream);
218 
219  ClearKernel1<<<blocks, 1024, 0, stream>>>(childl, FOUR_NNODES, FOUR_N);
220  CHECK_CUDA(stream);
221 
222  // Build quadtree
223  TreeBuildingKernel<<<blocks * FACTOR2, THREADS2, 0, stream>>>(
224  childl, nodes_pos, nodes_pos + nnodes + 1, NNODES, n, maxdepthd, bottomd, radiusd);
225  CHECK_CUDA(stream);
226 
227  ClearKernel2<<<blocks, 1024, 0, stream>>>(startl, massl, NNODES, bottomd);
228  CHECK_CUDA(stream);
229 
230  // Summarizes mass and position for each cell, bottom up approach
231  SummarizationKernel<<<blocks * FACTOR3, THREADS3, 0, stream>>>(
232  countl, childl, massl, nodes_pos, nodes_pos + nnodes + 1, NNODES, n, bottomd);
233  CHECK_CUDA(stream);
234 
235  // Group closed bodies together, used to speed up Repulsion kernel
236  SortKernel<<<blocks * FACTOR4, THREADS4, 0, stream>>>(
237  sortl, countl, startl, childl, NNODES, n, bottomd);
238  CHECK_CUDA(stream);
239 
240  // Force computation O(n . log(n))
241  RepulsionKernel<<<blocks * FACTOR5, THREADS5, 0, stream>>>(scaling_ratio,
242  theta,
243  epssq,
244  sortl,
245  childl,
246  massl,
247  nodes_pos,
248  nodes_pos + nnodes + 1,
249  rep_forces,
250  rep_forces + nnodes + 1,
251  theta_squared,
252  NNODES,
253  FOUR_NNODES,
254  n,
255  radiusd_squared,
256  maxdepthd);
257  CHECK_CUDA(stream);
258 
259  apply_gravity<vertex_t>(nodes_pos,
260  nodes_pos + nnodes + 1,
261  attract,
262  attract + n,
263  massl,
264  gravity,
265  strong_gravity_mode,
266  scaling_ratio,
267  n,
268  stream);
269 
270  apply_attraction<vertex_t, edge_t, weight_t>(row,
271  col,
272  v,
273  e,
274  nodes_pos,
275  nodes_pos + nnodes + 1,
276  attract,
277  attract + n,
278  massl,
279  outbound_attraction_distribution,
280  lin_log_mode,
281  edge_weight_influence,
282  outbound_att_compensation,
283  stream);
284 
285  compute_local_speed(rep_forces,
286  rep_forces + nnodes + 1,
287  attract,
288  attract + n,
289  old_forces,
290  old_forces + n,
291  massl,
292  swinging,
293  traction,
294  n,
295  stream);
296 
297  // Compute global swinging and traction values
298  const float s =
299  thrust::reduce(rmm::exec_policy(stream)->on(stream), d_swinging.begin(), d_swinging.end());
300 
301  const float t =
302  thrust::reduce(rmm::exec_policy(stream)->on(stream), d_traction.begin(), d_traction.end());
303 
304  // Compute global speed based on gloab and local swinging and traction.
305  adapt_speed<vertex_t>(jitter_tolerance, &jt, &speed, &speed_efficiency, s, t, n);
306 
307  // Update positions
308  apply_forces_bh<<<blocks * FACTOR6, THREADS6, 0, stream>>>(nodes_pos,
309  nodes_pos + nnodes + 1,
310  attract,
311  attract + n,
312  rep_forces,
313  rep_forces + nnodes + 1,
314  old_forces,
315  old_forces + n,
316  swinging,
317  speed,
318  n);
319 
320  if (callback) callback->on_epoch_end(nodes_pos);
321 
322  if (verbose) {
323  printf("iteration %i, speed: %f, speed_efficiency: %f, ", iter + 1, speed, speed_efficiency);
324  printf("jt: %f, ", jt);
325  printf("swinging: %f, traction: %f\n", s, t);
326  }
327  }
328 
329  // Copy nodes positions into final output pos
330  copy(n, nodes_pos, pos);
331  copy(n, nodes_pos + nnodes + 1, pos + n);
332 
333  if (callback) callback->on_train_end(nodes_pos);
334 }
