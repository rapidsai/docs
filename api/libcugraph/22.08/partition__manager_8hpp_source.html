57 // FIXME: This class is a misnomer since the python layer is currently
58 // responsible for creating and managing partitioning. Consider renaming it or
59 // refactoring it away.
60 //
61 // class responsible for creating 2D partition sub-comms:
62 // this is instantiated by each worker (processing element, PE)
63 // for the row/column it belongs to;
64 //
65 // naming policy defaults to simplified naming:
66 // one key per row subcomms, one per column subcomms;
67 //
68 template <typename name_policy_t = key_naming_t>
69 class subcomm_factory_t {
70  public:
71  subcomm_factory_t(raft::handle_t& handle, int row_size) : handle_(handle), row_size_(row_size)
72  {
73  init_row_col_comms();
74  }
75  virtual ~subcomm_factory_t(void) {}
76 
77  pair_comms_t const& row_col_comms(void) const { return row_col_subcomms_; }
78 
79  protected:
80  virtual void init_row_col_comms(void)
81  {
82  name_policy_t key;
83  raft::comms::comms_t const& communicator = handle_.get_comms();
84 
85  int const rank = communicator.get_rank();
86  int row_index = rank / row_size_;
87  int col_index = rank % row_size_;
88 
89  auto row_comm =
90  std::make_shared<raft::comms::comms_t>(communicator.comm_split(row_index, col_index));
91  handle_.set_subcomm(key.row_name(), row_comm);
92 
93  auto col_comm =
94  std::make_shared<raft::comms::comms_t>(communicator.comm_split(col_index, row_index));
95  handle_.set_subcomm(key.col_name(), col_comm);
96 
97  row_col_subcomms_.first = row_comm;
98  row_col_subcomms_.second = col_comm;
99  }
100 
101  private:
102  raft::handle_t& handle_;
103  int row_size_;
104  pair_comms_t row_col_subcomms_;
105 };
