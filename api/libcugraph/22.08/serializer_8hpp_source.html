145  template <typename graph_t>
146  static std::pair<size_t, size_t> get_device_graph_sz_bytes(
147  graph_meta_t<graph_t> const& graph_meta)
148  {
149  using vertex_t = typename graph_t::vertex_type;
150  using edge_t = typename graph_t::edge_type;
151  using weight_t = typename graph_t::weight_type;
152 
153  if constexpr (!graph_t::is_multi_gpu) {
154  size_t num_vertices = graph_meta.num_vertices_;
155  size_t num_edges = graph_meta.num_edges_;
156 
157  size_t weight_storage_sz = graph_meta.is_weighted_ ? num_edges * sizeof(weight_t) : 0;
158 
159  size_t device_ser_sz =
160  (num_vertices + 1) * sizeof(edge_t) + num_edges * sizeof(vertex_t) + weight_storage_sz;
161 
162  size_t host_ser_sz = graph_meta.get_device_sz_bytes();
163 
164  return std::make_pair(
165  device_ser_sz,
166  host_ser_sz); // FIXME: remove when host_bcast() becomes available for host vectors
167 
168  } else {
169  CUGRAPH_FAIL("Unsupported graph type for un/serialization.");
170 
171  return std::pair<size_t, size_t>{};
172  }
173  }
