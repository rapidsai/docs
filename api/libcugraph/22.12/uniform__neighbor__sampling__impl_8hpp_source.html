37 uniform_nbr_sample_impl(raft::handle_t const& handle,
38  graph_view_t const& graph_view,
39  rmm::device_uvector<typename graph_view_t::vertex_type>& d_in,
40  raft::host_span<const int> h_fan_out,
41  bool with_replacement,
42  uint64_t seed)
43 {
44  using vertex_t = typename graph_view_t::vertex_type;
45  using edge_t = typename graph_view_t::edge_type;
46  using weight_t = typename graph_view_t::weight_type;
47 
48 #ifdef NO_CUGRAPH_OPS
49  CUGRAPH_FAIL(
50  "uniform_nbr_sampl_impl not supported in this configuration, built with NO_CUGRAPH_OPS");
51 #else
52 
53  CUGRAPH_EXPECTS(h_fan_out.size() > 0,
54  "Invalid input argument: number of levels must be non-zero.");
55 
56  rmm::device_uvector<vertex_t> d_result_src(0, handle.get_stream());
57  rmm::device_uvector<vertex_t> d_result_dst(0, handle.get_stream());
58  auto d_result_indices =
59  thrust::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
60 
61  size_t level{0};
62  size_t comm_size{1};
63 
64  if constexpr (graph_view_t::is_multi_gpu) {
65  seed += handle.get_comms().get_rank();
66  comm_size = handle.get_comms().get_size();
67  }
68 
69  for (auto&& k_level : h_fan_out) {
70  // prep step for extracting out-degs(sources):
71  if constexpr (graph_view_t::is_multi_gpu) {
72  d_in = shuffle_int_vertices_by_gpu_id(
73  handle, std::move(d_in), graph_view.vertex_partition_range_lasts());
74  }
75 
76  rmm::device_uvector<vertex_t> d_out_src(0, handle.get_stream());
77  rmm::device_uvector<vertex_t> d_out_dst(0, handle.get_stream());
78  auto d_out_indices = std::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
79 
80  if (k_level > 0) {
81  raft::random::RngState rng_state(seed);
82  seed += d_in.size() * k_level * comm_size;
83 
84  std::tie(d_out_src, d_out_dst, d_out_indices) = sample_edges(
85  handle, graph_view, rng_state, d_in, static_cast<size_t>(k_level), with_replacement);
86  } else {
87  std::tie(d_out_src, d_out_dst, d_out_indices) =
88  gather_one_hop_edgelist(handle, graph_view, d_in);
89  }
90 
91  // resize accumulators:
92  auto old_sz = d_result_dst.size();
93  auto add_sz = d_out_dst.size();
94  auto new_sz = old_sz + add_sz;
95 
96  d_result_src.resize(new_sz, handle.get_stream());
97  d_result_dst.resize(new_sz, handle.get_stream());
98  d_result_indices->resize(new_sz, handle.get_stream());
99 
100  raft::copy(
101  d_result_src.begin() + old_sz, d_out_src.begin(), d_out_src.size(), handle.get_stream());
102  raft::copy(
103  d_result_dst.begin() + old_sz, d_out_dst.begin(), d_out_dst.size(), handle.get_stream());
104  raft::copy(d_result_indices->begin() + old_sz,
105  d_out_indices->begin(),
106  d_out_indices->size(),
107  handle.get_stream());
108 
109  d_in = std::move(d_out_dst);
110 
111  ++level;
112  }
113 
114  return count_and_remove_duplicates<vertex_t, edge_t, weight_t>(
115  handle, std::move(d_result_src), std::move(d_result_dst), std::move(*d_result_indices));
116 #endif
117 }
