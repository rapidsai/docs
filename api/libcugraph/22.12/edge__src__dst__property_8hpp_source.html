152 template <typename vertex_t, typename T>
153 class edge_major_property_t {
154  public:
155  using buffer_type = decltype(allocate_dataframe_buffer<T>(size_t{0}, rmm::cuda_stream_view{}));
156 
157  edge_major_property_t(raft::handle_t const& handle) {}
158 
159  edge_major_property_t(raft::handle_t const& handle,
160  std::vector<vertex_t> const& edge_partition_major_range_sizes,
161  std::vector<vertex_t> const& edge_partition_major_range_firsts)
162  : edge_partition_major_range_firsts_(edge_partition_major_range_firsts)
163  {
164  buffers_.reserve(edge_partition_major_range_firsts_.size());
165  for (size_t i = 0; i < edge_partition_major_range_firsts_.size(); ++i) {
166  buffers_.push_back(
167  allocate_dataframe_buffer<T>(edge_partition_major_range_sizes[i], handle.get_stream()));
168  }
169  }
170 
171  edge_major_property_t(
172  raft::handle_t const& handle,
173  raft::host_span<raft::device_span<vertex_t const> const> edge_partition_keys,
174  raft::host_span<raft::device_span<vertex_t const> const> edge_partition_key_chunk_start_offsets,
175  size_t key_chunk_size,
176  std::vector<vertex_t> const& edge_partition_major_range_firsts)
177  : edge_partition_keys_(edge_partition_keys),
178  edge_partition_key_chunk_start_offsets_(edge_partition_key_chunk_start_offsets),
179  key_chunk_size_(key_chunk_size),
180  edge_partition_major_range_firsts_(edge_partition_major_range_firsts)
181  {
182  buffers_.reserve(edge_partition_major_range_firsts_.size());
183  for (size_t i = 0; i < edge_partition_major_range_firsts_.size(); ++i) {
184  buffers_.push_back(
185  allocate_dataframe_buffer<T>(edge_partition_keys[i].size(), handle.get_stream()));
186  }
187  }
188 
189  void clear(raft::handle_t const& handle)
190  {
191  edge_partition_keys_ = std::nullopt;
192  edge_partition_key_chunk_start_offsets_ = std::nullopt;
193  key_chunk_size_ = std::nullopt;
194 
195  buffers_.clear();
196  buffers_.shrink_to_fit();
197  edge_partition_major_range_firsts_.clear();
198  edge_partition_major_range_firsts_.shrink_to_fit();
199  }
200 
201  auto view() const
202  {
203  using const_value_iterator = decltype(get_dataframe_buffer_cbegin(buffers_[0]));
204 
205  std::vector<const_value_iterator> edge_partition_value_firsts(buffers_.size());
206  for (size_t i = 0; i < edge_partition_value_firsts.size(); ++i) {
207  edge_partition_value_firsts[i] = get_dataframe_buffer_cbegin(buffers_[i]);
208  }
209 
210  if (edge_partition_keys_) {
211  return edge_major_property_view_t<vertex_t, const_value_iterator>(
212  *edge_partition_keys_,
213  *edge_partition_key_chunk_start_offsets_,
214  *key_chunk_size_,
215  edge_partition_value_firsts,
216  edge_partition_major_range_firsts_);
217  } else {
218  return edge_major_property_view_t<vertex_t, const_value_iterator>(
219  edge_partition_value_firsts, edge_partition_major_range_firsts_);
220  }
221  }
222 
223  auto mutable_view()
224  {
225  using value_iterator = decltype(get_dataframe_buffer_begin(buffers_[0]));
226 
227  std::vector<value_iterator> edge_partition_value_firsts(buffers_.size());
228  for (size_t i = 0; i < edge_partition_value_firsts.size(); ++i) {
229  edge_partition_value_firsts[i] = get_dataframe_buffer_begin(buffers_[i]);
230  }
231 
232  if (edge_partition_keys_) {
233  return edge_major_property_view_t<vertex_t, value_iterator>(
234  *edge_partition_keys_,
235  *edge_partition_key_chunk_start_offsets_,
236  *key_chunk_size_,
237  edge_partition_value_firsts,
238  edge_partition_major_range_firsts_);
239  } else {
240  return edge_major_property_view_t<vertex_t, value_iterator>(
241  edge_partition_value_firsts, edge_partition_major_range_firsts_);
242  }
243  }
244 
245  private:
246  std::optional<raft::host_span<raft::device_span<vertex_t const> const>> edge_partition_keys_{
247  std::nullopt};
248  std::optional<raft::host_span<raft::device_span<vertex_t const> const>>
249  edge_partition_key_chunk_start_offsets_{std::nullopt};
250  std::optional<size_t> key_chunk_size_{std::nullopt};
251 
252  std::vector<buffer_type> buffers_{};
253  std::vector<vertex_t> edge_partition_major_range_firsts_{};
254 };

272 
273  edge_minor_property_t(raft::handle_t const& handle,
274  raft::device_span<vertex_t const> keys,
275  raft::device_span<vertex_t const> key_chunk_start_offsets,
276  size_t key_chunk_size,
277  vertex_t minor_range_first)
278  : keys_(keys),
279  key_chunk_start_offsets_(key_chunk_start_offsets),
280  key_chunk_size_(key_chunk_size),

282  minor_range_first_(minor_range_first)
283  {

285 
286  void clear(raft::handle_t const& handle)
287  {
288  keys_ = std::nullopt;
289  key_chunk_start_offsets_ = std::nullopt;
290  key_chunk_size_ = std::nullopt;
291 
292  resize_dataframe_buffer(buffer_, size_t{0}, handle.get_stream());
293  shrink_to_fit_dataframe_buffer(buffer_, handle.get_stream());
294  minor_range_first_ = vertex_t{0};
295  }
296 
297  auto view() const
298  {
299  auto value_first = get_dataframe_buffer_cbegin(buffer_);
300  if (keys_) {
301  return edge_minor_property_view_t<vertex_t, decltype(value_first)>(
302  *keys_, *key_chunk_start_offsets_, *key_chunk_size_, value_first, minor_range_first_);
303  } else {
304  return edge_minor_property_view_t<vertex_t, decltype(value_first)>(value_first,
305  minor_range_first_);
306  }
307  }
308 
309  auto mutable_view()
310  {
311  auto value_first = get_dataframe_buffer_begin(buffer_);
312  if (keys_) {
313  return edge_minor_property_view_t<vertex_t, decltype(value_first)>(
314  *keys_, *key_chunk_start_offsets_, *key_chunk_size_, value_first, minor_range_first_);
315  } else {
316  return edge_minor_property_view_t<vertex_t, decltype(value_first)>(value_first,
317  minor_range_first_);
318  }
319  }
320 
321  private:
322  std::optional<raft::device_span<vertex_t const>> keys_{std::nullopt};
323  std::optional<raft::device_span<vertex_t const>> key_chunk_start_offsets_{std::nullopt};
324  std::optional<size_t> key_chunk_size_{std::nullopt};
325 
326  decltype(allocate_dataframe_buffer<T>(size_t{0}, rmm::cuda_stream_view{})) buffer_;
327  vertex_t minor_range_first_{};
328 };

361 class edge_src_property_t {
362  public:
363  using value_type = T;
364  static_assert(is_arithmetic_or_thrust_tuple_of_arithmetic<T>::value);
365 
366  edge_src_property_t(raft::handle_t const& handle) : property_(handle) {}
367 
368  edge_src_property_t(raft::handle_t const& handle, GraphViewType const& graph_view)
369  : property_(handle)
370  {
371  using vertex_t = typename GraphViewType::vertex_type;
372 
373  auto key_chunk_size = graph_view.local_sorted_unique_edge_src_chunk_size();
374  if (key_chunk_size) {
375  if constexpr (GraphViewType::is_multi_gpu) {
376  if constexpr (GraphViewType::is_storage_transposed) {
377  property_ = detail::edge_minor_property_t<vertex_t, T>(
378  handle,
379  *(graph_view.local_sorted_unique_edge_srcs()),
380  *(graph_view.local_sorted_unique_edge_src_chunk_start_offsets()),
381  *key_chunk_size,
382  graph_view.local_edge_partition_src_range_first());
383  } else {
384  edge_partition_keys_ = std::vector<raft::device_span<vertex_t const>>(
385  graph_view.number_of_local_edge_partitions());
386  edge_partition_key_chunk_start_offsets_ = std::vector<raft::device_span<vertex_t const>>(
387  graph_view.number_of_local_edge_partitions());
388  std::vector<vertex_t> major_range_firsts(graph_view.number_of_local_edge_partitions());
389  for (size_t i = 0; i < graph_view.number_of_local_edge_partitions(); ++i) {
390  (*edge_partition_keys_)[i] = *(graph_view.local_sorted_unique_edge_srcs(i));
391  (*edge_partition_key_chunk_start_offsets_)[i] =
392  *(graph_view.local_sorted_unique_edge_src_chunk_start_offsets(i));
393  major_range_firsts[i] = graph_view.local_edge_partition_src_range_first(i);
394  }
395  property_ = detail::edge_major_property_t<vertex_t, T>(
396  handle,
397  raft::host_span<raft::device_span<vertex_t const>>((*edge_partition_keys_).data(),
398  (*edge_partition_keys_).size()),
399  raft::host_span<raft::device_span<vertex_t const>>(
400  (*edge_partition_key_chunk_start_offsets_).data(),
401  (*edge_partition_key_chunk_start_offsets_).size()),
402  *key_chunk_size,
403  std::move(major_range_firsts));
404  }
405  } else {
406  assert(false);
407  }
408  } else {
409  if constexpr (GraphViewType::is_storage_transposed) {
410  property_ = detail::edge_minor_property_t<vertex_t, T>(
411  handle,
412  graph_view.local_edge_partition_src_range_size(),
413  graph_view.local_edge_partition_src_range_first());
414  } else {
415  std::vector<vertex_t> major_range_sizes(graph_view.number_of_local_edge_partitions(),
416  vertex_t{0});
417  std::vector<vertex_t> major_range_firsts(graph_view.number_of_local_edge_partitions());
418  for (size_t i = 0; i < graph_view.number_of_local_edge_partitions(); ++i) {
419  major_range_sizes[i] = graph_view.local_edge_partition_src_range_size(i);
420  major_range_firsts[i] = graph_view.local_edge_partition_src_range_first(i);
421  }
422  property_ = detail::edge_major_property_t<vertex_t, T>(
423  handle, std::move(major_range_sizes), std::move(major_range_firsts));
424  }
425  }
426  }
427 
428  void clear(raft::handle_t const& handle)
429  {
430  property_.clear(handle);
431 
432  edge_partition_keys_ = std::nullopt;
