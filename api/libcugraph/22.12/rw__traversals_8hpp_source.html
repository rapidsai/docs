131 // Uniform RW selector logic:
132 //
133 template <typename vertex_t, typename edge_t, typename weight_t, typename real_t>
134 struct uniform_selector_t {
135  struct sampler_t {
136  sampler_t(edge_t const* ro,
137  vertex_t const* ci,
138  weight_t const* w,
139  edge_t const* ptr_d_cache_out_degs)
140  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_cache_out_degs_(ptr_d_cache_out_degs)
141  {
142  }
143 
144  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
145  vertex_t src_v,
146  real_t rnd_val,
147  vertex_t = 0 /* not used*/,
148  edge_t = 0 /* not used*/,
149  bool = false /* not used*/) const
150  {
151  auto crt_out_deg = ptr_d_cache_out_degs_[src_v];
152  if (crt_out_deg == 0) return thrust::nullopt; // src_v is a sink
153 
154  vertex_t v_indx =
155  static_cast<vertex_t>(rnd_val >= 1.0 ? crt_out_deg - 1 : rnd_val * crt_out_deg);
156  auto col_indx = v_indx >= crt_out_deg ? crt_out_deg - 1 : v_indx;
157  auto start_row = row_offsets_[src_v];
158 
159  auto weight_value =
160  (values_ == nullptr ? weight_t{1}
161  : values_[start_row + col_indx]); // account for un-weighted graphs
162  return thrust::optional{thrust::make_tuple(col_indices_[start_row + col_indx], weight_value)};
163  }
164 
165  private:
166  edge_t const* row_offsets_;
167  vertex_t const* col_indices_;
168  weight_t const* values_;
169 
170  edge_t const* ptr_d_cache_out_degs_;
171  };

199 // Biased RW selection logic:
200 //
201 template <typename vertex_t, typename edge_t, typename weight_t, typename real_t>
202 struct biased_selector_t {
203  struct sampler_t {
204  sampler_t(edge_t const* ro,
205  vertex_t const* ci,
206  weight_t const* w,
207  weight_t const* ptr_d_sum_weights)
208  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_sum_weights_(ptr_d_sum_weights)
209  {
210  }
211 
212  // pre-condition:
213  //
214  // Sum(weights(neighborhood(src_v))) are pre-computed and
215  // stored in ptr_d_sum_weights_ (too expensive to check, here);
216  //
217  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
218  vertex_t src_v,
219  real_t rnd_val,
220  vertex_t = 0 /* not used*/,
221  edge_t = 0 /* not used*/,
222  bool = false /* not used*/) const
223  {
224  weight_t run_sum_w{0};
225  auto rnd_sum_weights = rnd_val * ptr_d_sum_weights_[src_v];
226 
227  auto col_indx_begin = row_offsets_[src_v];
228  auto col_indx_end = row_offsets_[src_v + 1];
229  if (col_indx_begin == col_indx_end) return thrust::nullopt; // src_v is a sink
230 
231  auto col_indx = col_indx_begin;
232  auto prev_col_indx = col_indx;
233 
234  for (; col_indx < col_indx_end; ++col_indx) {
235  if (rnd_sum_weights < run_sum_w) break;
236 
237  run_sum_w += values_[col_indx];
238  prev_col_indx = col_indx;
239  }
240  return thrust::optional{
241  thrust::make_tuple(col_indices_[prev_col_indx], values_[prev_col_indx])};
242  }
243 
244  private:
245  edge_t const* row_offsets_;
246  vertex_t const* col_indices_;
247  weight_t const* values_;
248 
249  weight_t const* ptr_d_sum_weights_;
250  };

272 // node2vec RW selection logic:
273 // uses biased selector on scaled weights,
274 // to be computed (and possibly cached) according to
275 // `node2vec` logic (see `get_alpha()`);
276 // works on unweighted graphs (for which unscalled weights are 1.0);
277 //
278 // TODO: need to decide logic on very 1st step of traversal
279 // (which has no `prev_v` vertex);
280 //
281 template <typename vertex_t, typename edge_t, typename weight_t, typename real_t>
282 struct node2vec_selector_t {
283  struct sampler_t {
284  sampler_t(edge_t const* ro,
285  vertex_t const* ci,
286  weight_t const* w,
287  weight_t p,
288  weight_t q,
289  vertex_t max_degree,
290  edge_t num_paths,
291  weight_t* ptr_alpha)
292  : row_offsets_(ro),
293  col_indices_(ci),
294  values_(w),
295  p_(p),
296  q_(q),
297  coalesced_alpha_{
298  (max_degree > 0) && (num_paths > 0) && (ptr_alpha != nullptr)
299  ? thrust::optional<thrust::tuple<vertex_t, edge_t, weight_t*>>{thrust::make_tuple(
300  max_degree, num_paths, ptr_alpha)}
301  : thrust::nullopt}
302  {
303  }
304 
305  // node2vec alpha scalling logic:
306  // pre-condition: assume column_indices[] is seg-sorted;
307  // (each row has column_indices[] sorted)
308  //
309  __device__ weight_t get_alpha(vertex_t prev_v, vertex_t src_v, vertex_t next_v) const
310  {
311  if (next_v == prev_v) {
312  return 1.0 / p_;
313  } else {
314  // binary-search `next_v` in the adj(prev_v)
315  //
316  auto prev_indx_begin = row_offsets_[prev_v];
317  auto prev_indx_end = row_offsets_[prev_v + 1];
318 
319  auto found_next_from_prev = thrust::binary_search(
320  thrust::seq, col_indices_ + prev_indx_begin, col_indices_ + prev_indx_end, next_v);
321 
322  if (found_next_from_prev) {
323  return 1;
324  } else {
325  return 1.0 / q_;
326  }
327  }
328  }
329 
330  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
331  vertex_t src_v, real_t rnd_val, vertex_t prev_v, edge_t path_index, bool start_path) const
332  {
333  auto const offset_indx_begin = row_offsets_[src_v];
334  auto const offset_indx_end = row_offsets_[src_v + 1];
335 
336  weight_t sum_scaled_weights{0};
337  auto offset_indx = offset_indx_begin;
338 
339  if (offset_indx_begin == offset_indx_end) return thrust::nullopt; // src_v is a sink
340 
341  // for 1st vertex in path just use biased random selection:
342  //
343  if (start_path) { // `src_v` is starting vertex in path
344  for (; offset_indx < offset_indx_end; ++offset_indx) {
345  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
346 
347  sum_scaled_weights += crt_weight;
348  }
349 
350  weight_t run_sum_w{0};
351  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
352  offset_indx = offset_indx_begin;
353  auto prev_offset_indx = offset_indx;
354 
355  // biased sampling selection loop:
356  // (Note: re-compute `scaled_weight`, since no cache is available);
357  //
358  for (; offset_indx < offset_indx_end; ++offset_indx) {
359  if (rnd_sum_weights < run_sum_w) break;
360 
361  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
362  run_sum_w += crt_weight;
363  prev_offset_indx = offset_indx;
364  }
365  return thrust::optional{
366  thrust::make_tuple(col_indices_[prev_offset_indx],
367  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
368  }
369 
370  // cached solution, for increased performance, but memory expensive:
371  //
372  if (coalesced_alpha_.has_value()) {
373  auto&& tpl = *coalesced_alpha_;
374 
375  auto max_out_deg = thrust::get<0>(tpl);
376  auto num_paths = thrust::get<1>(tpl);
377  weight_t* ptr_d_scaled_weights = thrust::get<2>(tpl);
378 
379  // sum-scaled-weights reduction loop:
380  //
381  auto const start_alpha_offset = max_out_deg * path_index;
382  for (vertex_t nghbr_indx = 0; offset_indx < offset_indx_end; ++offset_indx, ++nghbr_indx) {
383  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
384  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
385  auto scaled_weight = crt_weight * crt_alpha;
386 
387  // caching is available, hence cache the alpha's for next step
388  // (the actual sampling step);
389  //
390  ptr_d_scaled_weights[start_alpha_offset + nghbr_indx] = scaled_weight;
391 
392  sum_scaled_weights += scaled_weight;
393  }
394 
395  weight_t run_sum_w{0};
396  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
397  offset_indx = offset_indx_begin;
398  auto prev_offset_indx = offset_indx;
399 
400  // biased sampling selection loop:
401  //
402  for (vertex_t nghbr_indx = 0; offset_indx < offset_indx_end; ++offset_indx, ++nghbr_indx) {
403  if (rnd_sum_weights < run_sum_w) break;
404 
405  run_sum_w += ptr_d_scaled_weights[start_alpha_offset + nghbr_indx];
406  prev_offset_indx = offset_indx;
407  }
408  return thrust::optional{
409  thrust::make_tuple(col_indices_[prev_offset_indx],
410  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
411 
412  } else { // uncached solution, with much lower memory footprint but not as efficient
413 
414  for (; offset_indx < offset_indx_end; ++offset_indx) {
415  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
416 
417  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
418 
419  auto scaled_weight = crt_weight * crt_alpha;
420  sum_scaled_weights += scaled_weight;
421  }
422 
423  weight_t run_sum_w{0};
424  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
425  offset_indx = offset_indx_begin;
426  auto prev_offset_indx = offset_indx;
427 
428  // biased sampling selection loop:
429  // (Note: re-compute `scaled_weight`, since no cache is available);
430  //
431  for (; offset_indx < offset_indx_end; ++offset_indx) {
432  if (rnd_sum_weights < run_sum_w) break;
433 
434  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
435  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
436  auto scaled_weight = crt_weight * crt_alpha;
437 
438  run_sum_w += scaled_weight;
439  prev_offset_indx = offset_indx;
440  }
441  return thrust::optional{
442  thrust::make_tuple(col_indices_[prev_offset_indx],
443  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
444  }
445  }
446 
447  decltype(auto) get_alpha_buffer(void) const { return coalesced_alpha_; }
448 
449  private:
450  edge_t const* row_offsets_;
451  vertex_t const* col_indices_;
452  weight_t const* values_;
453 
454  weight_t const p_;
455  weight_t const q_;
456 
457  // alpha scaling coalesced buffer (per path):
458  // (use as cache since the per-path alpha-buffer
459  // is used twice for each node transition:
460  // (1) for computing sum_scaled weights;
461  // (2) for using scaled_weights for the biased next vertex selection)
462  // this is information related to a scratchpad buffer, used as cache, hence mutable;
463  // (necessary, because get_strategy() is const)
464  //
465  mutable thrust::optional<thrust::tuple<vertex_t, edge_t, weight_t*>>
466  coalesced_alpha_; // tuple<max_vertex_degree,
467  // num_paths, alpha_buffer[max_vertex_degree*num_paths]>
468  };

513 // vertical traversal proxy:
514 // a device vector of next vertices is generated for each path;
515 // when a vertex is a sink the corresponding path doesn't advance anymore;
516 //
517 // smaller memory footprint;
518 //
519 struct vertical_traversal_t {
520  vertical_traversal_t(size_t num_paths, size_t max_depth)
521  : num_paths_(num_paths), max_depth_(max_depth)
522  {
523  }
524 
525  template <typename vertex_t,
526  typename edge_t,
527  typename weight_t,
528  typename random_walker_t,
529  typename selector_t,
530  typename index_t,
531  typename real_t,
532  typename seed_t>
533  void operator()(
534  graph_view_t<vertex_t, edge_t, false, false> const& graph_view, // graph being traversed
535  std::optional<edge_property_view_t<edge_t, weight_t const*>> edge_weight_view,
536  random_walker_t const& rand_walker, // random walker object for which traversal is driven
537  selector_t const& selector, // sampling type (uniform, biased, etc.)
538  seed_t seed0, // initial seed value
539  device_vec_t<vertex_t>& d_coalesced_v, // crt coalesced vertex set
540  device_vec_t<weight_t>& d_coalesced_w, // crt coalesced weight set
541  device_vec_t<index_t>& d_paths_sz, // crt paths sizes
542  device_vec_t<edge_t>& d_crt_out_degs, // crt out-degs for current set of vertices
543  device_vec_t<real_t>& d_random, // crt set of random real values
544  device_vec_t<vertex_t>& d_col_indx) // crt col col indices to be used for retrieving next step
545  const
546  {
547  auto const& handle = rand_walker.get_handle();
548 
549  // start from 1, as 0-th was initialized above:
550  //
551  for (decltype(max_depth_) step_indx = 1; step_indx < max_depth_; ++step_indx) {
552  // take one-step in-sync for each path in parallel:
553  //
554  rand_walker.step(graph_view,
555  edge_weight_view,
556  selector,
557  seed0 + static_cast<seed_t>(step_indx),
558  d_coalesced_v,
559  d_coalesced_w,
560  d_paths_sz,
561  d_crt_out_degs,
562  d_random,
563  d_col_indx);
564 
565  // early exit: all paths have reached sinks:
566  //
567  if (rand_walker.all_paths_stopped(d_crt_out_degs)) break;
568  }
569  }
570 
571  size_t get_random_buff_sz(void) const { return num_paths_; }
572  size_t get_tmp_buff_sz(void) const { return num_paths_; }
573 
574  private:
575  size_t num_paths_;
576  size_t max_depth_;
577 };
