42  typename weight_t,
43  bool store_transposed,
44  bool multi_gpu>
45 cugraph_error_code_t transpose_storage(raft::handle_t const& handle,
46  cugraph_graph_t* graph,
47  cugraph_error_t* error)
48 {
49  if (store_transposed == graph->store_transposed_) {
50  auto p_graph =
51  reinterpret_cast<cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>*>(
52  graph->graph_);
53 
54  auto number_map = reinterpret_cast<rmm::device_uvector<vertex_t>*>(graph->number_map_);
55 
56  auto graph_transposed =
57  new cugraph::graph_t<vertex_t, edge_t, weight_t, !store_transposed, multi_gpu>(handle);
58 
59  std::optional<rmm::device_uvector<vertex_t>> new_number_map;
60 
61  std::tie(*graph_transposed, new_number_map) =
62  p_graph->transpose_storage(handle, std::move(*number_map), true);
63 
64  *number_map = std::move(new_number_map.value());
65 
66  delete p_graph;
67 
68  graph->graph_ = graph_transposed;
69  graph->store_transposed_ = !store_transposed;
70 
71  return CUGRAPH_SUCCESS;
72  } else {
73  error->error_message_ = "transpose failed, value of transpose does not match graph";
74  return CUGRAPH_INVALID_INPUT;
75  }
