27 class HighResTimer {

32  void start(std::string label)
33  {
34  auto it = timers.find(label);
35  if (it == timers.end()) {
36  it = timers.insert(std::make_pair(label, std::make_pair<int, int64_t>(int{0}, int64_t{0})))
37  .first;
38  }
39 
40  timespec start_time;
41  clock_gettime(CLOCK_REALTIME, &start_time);
42  it->second.second -= start_time.tv_sec * 1000000000 + start_time.tv_nsec;
43 
44  open_label = label;
45  }
46 
47  void stop()
48  {
49  timespec stop_time;
50  clock_gettime(CLOCK_REALTIME, &stop_time);
51 
52  auto it = timers.find(open_label);
53  it->second.first++;
54  it->second.second += stop_time.tv_sec * 1000000000 + stop_time.tv_nsec;
55  }
56 
57  double get_average_runtime(std::string const& label)
58  {
59  auto it = timers.find(label);
60  if (it != timers.end()) {
61  return (static_cast<double>(it->second.second) / (1000000.0 * it->second.first));
62  } else {
63  std::stringstream ss;
64  ss << "ERROR: timing label: " << label << "not found.";
65 
66  throw std::runtime_error(ss.str());
67  }
68  }
