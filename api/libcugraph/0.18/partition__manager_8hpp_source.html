57 // FIXME: This class is a misnomer since the python layer is currently
58 // responsible for creating and managing partitioning. Consider renaming it or
59 // refactoring it away.
60 //
61 // class responsible for creating 2D partition sub-comms:
62 // this is instantiated by each worker (processing element, PE)
63 // for the row/column it belongs to;
64 //
65 // naming policy defaults to simplified naming:
66 // one key per row subcomms, one per column subcomms;
67 //
68 template <typename name_policy_t = key_naming_t, typename size_type = int>
69 class subcomm_factory_t {
70  public:
71  subcomm_factory_t(raft::handle_t& handle, size_type row_size)
72  : handle_(handle), row_size_(row_size)
73  {
74  init_row_col_comms();
75  }
76  virtual ~subcomm_factory_t(void) {}
77 
78  pair_comms_t const& row_col_comms(void) const { return row_col_subcomms_; }
79 
80  protected:
81  virtual void init_row_col_comms(void)
82  {
83  name_policy_t key;
84  raft::comms::comms_t const& communicator = handle_.get_comms();
85 
86  int const rank = communicator.get_rank();
87  int row_index = rank / row_size_;
88  int col_index = rank % row_size_;
89 
90  auto row_comm =
91  std::make_shared<raft::comms::comms_t>(communicator.comm_split(row_index, col_index));
92  handle_.set_subcomm(key.row_name(), row_comm);
93 
94  auto col_comm =
95  std::make_shared<raft::comms::comms_t>(communicator.comm_split(col_index, row_index));
96  handle_.set_subcomm(key.col_name(), col_comm);
97 
98  row_col_subcomms_.first = row_comm;
99  row_col_subcomms_.second = col_comm;
100  }
101 
102  private:
103  raft::handle_t& handle_;
104  size_type row_size_;
105  pair_comms_t row_col_subcomms_;
106 };
