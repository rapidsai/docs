46 __device__ void random_init(float const *posx,
47  float const *posy,
48  int const *vtx_ptr,
49  int *path,
50  float *px,
51  float *py,
52  int const nstart,
53  int const nodes)
54 {
55  // Fill values
56  for (int i = threadIdx.x; i <= nodes; i += blockDim.x) {
57  px[i] = posx[i];
58  py[i] = posy[i];
59  path[i] = vtx_ptr[i];
60  }
61 
62  __syncthreads();
63 
64  if (threadIdx.x == 0) { /* serial permutation as starting point */
65  // swap to start at nstart node
66  raft::swapVals(px[0], px[nstart]);
67  raft::swapVals(py[0], py[nstart]);
68  raft::swapVals(path[0], path[nstart]);
69 
70  curandState rndstate;
71  curand_init(blockIdx.x, 0, 0, &rndstate);
72  for (int i = 1; i < nodes; i++) {
73  int j = curand(&rndstate) % (nodes - 1 - i) + i;
74  if (i == j) continue;
75  raft::swapVals(px[i], px[j]);
76  raft::swapVals(py[i], py[j]);
77  raft::swapVals(path[i], path[j]);
78  }
79  px[nodes] = px[0]; /* close the loop now, avoid special cases later */
80  py[nodes] = py[0];
81  path[nodes] = path[0];
82  }
83 }

86 __device__ void knn_init(float const *posx,
87  float const *posy,
88  int const *vtx_ptr,
89  int64_t const *neighbors,
90  int *buf,
91  int *path,
92  float *px,
93  float *py,
94  int const nstart,
95  int const nodes,
96  int const K)
97 {
98  for (int i = threadIdx.x; i < nodes; i += blockDim.x) buf[i] = 0;
99 
100  __syncthreads();
101 
102  if (threadIdx.x == 0) {
103  curandState rndstate;
104  curand_init(blockIdx.x, 0, 0, &rndstate);
105  int progress = 0;
106  int initlen = 0;
107 
108  px[0] = posx[nstart];
109  py[0] = posy[nstart];
110  path[0] = vtx_ptr[nstart];
111  int head = nstart;
112  int v = 0;
113  buf[head] = 1;
114  while (progress < nodes - 1) { // beam search as starting point
115  for (int i = 1; i <= progress; i++) buf[i] = 0;
116  progress = 0; // reset current location in path and visited array
117  initlen = 0;
118  int randjumps = 0;
119  while (progress < nodes - 1) {
120  int nj = curand(&rndstate) % K;
121  int linked = 0;
122  for (int nh = 0; nh < K; ++nh) {
123  // offset (idx / K) + 1 filters the points as their own nearest neighbors.
124  int offset = (K * head + nj) / K + 1;
125  v = neighbors[K * head + nj + offset];
126  if (v < nodes && buf[v] == 0) {
127  head = v;
128  progress += 1;
129  buf[head] = 1;
130  linked = 1;
131  break;
132  }
133  nj = (nj + 1) % K;
134  }
135  if (linked == 0) {
136  if (randjumps > nodes - 1)
137  break; // give up on this traversal, we failed to find a next link
138  randjumps += 1;
139  int nr = (head + 1) % nodes; // jump to next node
140  while (buf[nr] == 1) { nr = (nr + 1) % nodes; }
141  head = nr;
142  progress += 1;
143  buf[head] = 1;
144  }
145  // copy from input into beam-search order, update len
146  px[progress] = posx[head];
147  py[progress] = posy[head];
148  path[progress] = vtx_ptr[head];
149  initlen += __float2int_rn(euclidean_dist(px, py, progress, progress - 1));
150  }
151  }
152  px[nodes] = px[nstart];
153  py[nodes] = py[nstart];
154  path[nodes] = path[nstart];
155  initlen += __float2int_rn(euclidean_dist(px, py, nodes, nstart));
156  }
157 }

159 __device__ void two_opt_search(
160  int *buf, float *px, float *py, int *shbuf, int *minchange, int *mini, int *minj, int const nodes)
161 {
162  __shared__ float shmem_x[tilesize];
163  __shared__ float shmem_y[tilesize];
164 
165  for (int ii = 0; ii < nodes - 2; ii += blockDim.x) {
166  int i = ii + threadIdx.x;
167  float pxi0, pyi0, pxi1, pyi1, pxj1, pyj1;
168  if (i < nodes - 2) {
169  minchange[0] -= buf[i];
170  pxi0 = px[i];
171  pyi0 = py[i];
172  pxi1 = px[i + 1];
173  pyi1 = py[i + 1];
174  pxj1 = px[nodes];
175  pyj1 = py[nodes];
176  }
177  for (int jj = nodes - 1; jj >= ii + 2; jj -= tilesize) {
178  int bound = jj - tilesize + 1;
179  for (int k = threadIdx.x; k < tilesize; k += blockDim.x) {
180  if (k + bound >= ii + 2) {
181  shmem_x[k] = px[k + bound];
182  shmem_y[k] = py[k + bound];
183  shbuf[k] = buf[k + bound];
184  }
185  }
186  __syncthreads();
187 
188  int lower = bound;
189  if (lower < (i + 2)) lower = i + 2;
190  for (int j = jj; j >= lower; j--) {
191  int jm = j - bound;
192  float pxj0 = shmem_x[jm];
193  float pyj0 = shmem_y[jm];
194  int delta =
195  shbuf[jm] +
196  __float2int_rn(sqrtf((pxi0 - pxj0) * (pxi0 - pxj0) + (pyi0 - pyj0) * (pyi0 - pyj0))) +
197  __float2int_rn(sqrtf((pxi1 - pxj1) * (pxi1 - pxj1) + (pyi1 - pyj1) * (pyi1 - pyj1)));
198  pxj1 = pxj0;
199  pyj1 = pyj0;
200 
201  if (delta < minchange[0]) {
202  minchange[0] = delta;
203  mini[0] = i;
204  minj[0] = j;
205  }
206  }
207  __syncthreads();
208  }
209 
210  if (i < nodes - 2) { minchange[0] += buf[i]; }
211  }
212 }

215 __device__ void hill_climbing(
216  float *px, float *py, int *buf, int *path, int *shbuf, int const nodes, int *climbs)
217 {
218  __shared__ int best_change[kswaps];
219  __shared__ int best_i[kswaps];
220  __shared__ int best_j[kswaps];
221 
222  int minchange;
223  int mini;
224  int minj;
225  int kswaps_active = kswaps;
226  int myswaps = 0;
227 
228  // Hill climbing, iteratively improve from the starting guess
229  do {
230  if (threadIdx.x == 0) {
231  for (int k = 0; k < kswaps; k++) {
232  best_change[k] = 0;
233  best_i[k] = 0;
234  best_j[k] = 0;
235  }
236  }
237  __syncthreads();
238  for (int i = threadIdx.x; i < nodes; i += blockDim.x) {
239  buf[i] = -__float2int_rn(euclidean_dist(px, py, i, i + 1));
240  }
241  __syncthreads();
242 
243  // Reset
244  minchange = 0;
245  mini = 0;
246  minj = 0;
247 
248  // Find best indices
249  two_opt_search(buf, px, py, shbuf, &minchange, &mini, &minj, nodes);
250  __syncthreads();
251 
252  // Stats only
253  if (threadIdx.x == 0) atomicAdd(climbs, 1);
254 
255  shbuf[threadIdx.x] = minchange;
256 
257  int j = blockDim.x; // warp reduction to find best thread results
258  do {
259  int k = (j + 1) / 2;
260  if ((threadIdx.x + k) < j) {
261  shbuf[threadIdx.x] = min(shbuf[threadIdx.x + k], shbuf[threadIdx.x]);
262  }
263  j = k;
264  __syncthreads();
265  } while (j > 1); // thread winner for this k is in shbuf[0]
266 
267  if (threadIdx.x == 0) {
268  best_change[0] = shbuf[0]; // sort best result in shared
269  }
270  __syncthreads();
271 
272  if (minchange == shbuf[0]) { // My thread is as good as the winner
273  shbuf[1] = threadIdx.x; // store thread ID in shbuf[1]
274  }
275  __syncthreads();
276 
277  if (threadIdx.x == shbuf[1]) { // move from thread local to shared
278  best_i[0] = mini; // shared best indices for compatibility checks
279  best_j[0] = minj;
280  }
281  __syncthreads();
282 
283  // look for more compatible swaps
284  for (int kmin = 1; kmin < kswaps_active; kmin++) {
285  // disallow swaps that conflict with ones already picked
286  for (int kchk = kmin - 1; kchk >= 0; --kchk) {
287  if ((mini < (best_j[kchk] + 1)) && (minj > (best_i[kchk] - 1))) {
288  minchange = shbuf[threadIdx.x] = 0;
289  }
290  __syncthreads();
291  }
292  shbuf[threadIdx.x] = minchange;
293 
294  j = blockDim.x;
295  do {
296  int k = (j + 1) / 2;
297  if ((threadIdx.x + k) < j) {
298  shbuf[threadIdx.x] = min(shbuf[threadIdx.x + k], shbuf[threadIdx.x]);
299  }
300  j = k;
301  __syncthreads();
302  } while (j > 1); // thread winner for this k is in shbuf[0]
303 
304  if (threadIdx.x == 0) {
305  best_change[kmin] = shbuf[0]; // store best result in shared
306  }
307  __syncthreads();
308 
309  if (minchange == shbuf[0]) { // My thread is as good as the winner
310  shbuf[1] = threadIdx.x; // store thread ID in shbuf[1]
311  __threadfence_block();
312  }
313  __syncthreads();
314 
315  if (threadIdx.x == shbuf[1]) { // move from thread local to shared
316  best_i[kmin] = mini; // store swap targets
317  best_j[kmin] = minj;
318  __threadfence_block();
319  }
320  __syncthreads();
321  // look for the best compatible move
322  } // end loop over kmin
323  minchange = best_change[0];
324  myswaps += 1;
325  for (int kmin = 0; kmin < kswaps_active; kmin++) {
326  int sum = best_i[kmin] + best_j[kmin] + 1; // = mini + minj +1
327  // this is a reversal of all nodes included in the range [ i+1, j ]
328  for (int i = threadIdx.x; (i + i) < sum; i += blockDim.x) {
329  if (best_i[kmin] < i) {
330  int j = sum - i;
331  raft::swapVals(px[i], px[j]);
332  raft::swapVals(py[i], py[j]);
333  raft::swapVals(path[i], path[j]);
334  }
335  }
336  __syncthreads();
337  }
338  } while (minchange < 0 && myswaps < 2 * nodes);
339 }

341 __device__ void get_optimal_tour(
342  int *mylock, int *best_tour, float *px, float *py, int *path, int *shbuf, int const nodes)
343 {
344  // Now find actual length of the last tour, result of the climb
345  int term = 0;
346  for (int i = threadIdx.x; i < nodes; i += blockDim.x) {
347  term += __float2int_rn(euclidean_dist(px, py, i, i + 1));
348  }
349  shbuf[threadIdx.x] = term;
350  __syncthreads();
351 
352  int j = blockDim.x; // block level reduction
353  do {
354  int k = (j + 1) / 2;
355  if ((threadIdx.x + k) < j) { shbuf[threadIdx.x] += shbuf[threadIdx.x + k]; }
356  j = k; // divide active warp size in half
357  __syncthreads();
358  } while (j > 1);
359  term = shbuf[0];
360 
361  if (threadIdx.x == 0) {
362  atomicMin(best_tour, term);
363  while (atomicExch(mylock, 1) != 0)
364  ; // acquire
365  if (best_tour[0] == term) {
366  best_soln = px;
367  best_route = path;
368  }
369  *mylock = 0; // release
370  __threadfence();
371  }
372 }

374 __global__ __launch_bounds__(2048, 2) void search_solution(int *mylock,
375  int *best_tour,
376  int const *vtx_ptr,
377  bool beam_search,
378  int const K,
379  int nodes,
380  int64_t const *neighbors,
381  float const *posx,
382  float const *posy,
383  int *work,
384  int const nstart,
385  float *times,
386  int *climbs,
387  int threads)
388 {
389  int *buf = &work[blockIdx.x * ((4 * nodes + 3 + 31) / 32 * 32)];
390  float *px = (float *)(&buf[nodes]);
391  float *py = &px[nodes + 1];
392  int *path = (int *)(&py[nodes + 1]);
393  __shared__ int shbuf[tilesize];
394  clock_t start;
395 
396  start = clock64();
397  if (!beam_search)
398  random_init(posx, posy, vtx_ptr, path, px, py, nstart, nodes);
399  else
400  knn_init(posx, posy, vtx_ptr, neighbors, buf, path, px, py, nstart, nodes, K);
401  __syncthreads();
402  times[threadIdx.x] = clock64() - start;
403 
404  start = clock64();
405  hill_climbing(px, py, buf, path, shbuf, nodes, climbs);
406  __syncthreads();
407  times[threads + threadIdx.x + 1] = clock64() - start;
408 
409  start = clock64();
410  get_optimal_tour(mylock, best_tour, px, py, path, shbuf, nodes);
411  times[2 * threads + threadIdx.x + 1] = clock64() - start;
412 }
