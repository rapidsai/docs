40 template <typename graph_t>
41 graph_t graph_broadcast(raft::handle_t const& handle, graph_t* graph_ptr)
42 {
43  using namespace cugraph::serializer;
44 
45  using vertex_t = typename graph_t::vertex_type;
46  using edge_t = typename graph_t::edge_type;
47  using weight_t = typename graph_t::weight_type;
48 
49  if constexpr (!graph_t::is_multi_gpu) {
50  if (handle.get_comms().get_rank() == 0) {
51  CUGRAPH_EXPECTS(graph_ptr != nullptr, "Cannot serialize nullptr graph pointer.");
52 
53  auto pair = serializer_t::get_device_graph_sz_bytes(*graph_ptr);
54  thrust::tuple<size_t, size_t> dev_sz_host_sz_bytes =
55  thrust::make_tuple(pair.first, pair.second);
56 
57  auto total_graph_dev_sz = pair.first + pair.second;
58 
59  serializer_t ser(handle, total_graph_dev_sz);
60  serializer_t::graph_meta_t<graph_t> graph_meta{};
61  ser.serialize(*graph_ptr, graph_meta);
62 
63  int root{0};
64  host_scalar_bcast(handle.get_comms(), dev_sz_host_sz_bytes, root, handle.get_stream());
65  device_bcast(handle.get_comms(),
66  ser.get_storage(),
67  ser.get_storage(),
68  total_graph_dev_sz,
69  root,
70  handle.get_stream());
71 
72  return std::move(*graph_ptr);
73  } else {
74  thrust::tuple<size_t, size_t> dev_sz_host_sz_bytes(0, 0);
75 
76  int root{0};
77  dev_sz_host_sz_bytes =
78  host_scalar_bcast(handle.get_comms(), dev_sz_host_sz_bytes, root, handle.get_stream());
79  //
80  auto total_graph_dev_sz =
81  thrust::get<0>(dev_sz_host_sz_bytes) + thrust::get<1>(dev_sz_host_sz_bytes);
82 
83  CUGRAPH_EXPECTS(total_graph_dev_sz > 0, "Graph size comm failure.");
84 
85  rmm::device_uvector<serializer_t::byte_t> data_buffer(total_graph_dev_sz,
86  handle.get_stream());
87 
88  device_bcast(handle.get_comms(),
89  data_buffer.data(),
90  data_buffer.data(),
91  total_graph_dev_sz,
92  root,
93  handle.get_stream());
94 
95  serializer_t ser(handle, data_buffer.data());
96  auto graph = ser.unserialize<graph_t>(thrust::get<0>(dev_sz_host_sz_bytes),
97  thrust::get<1>(dev_sz_host_sz_bytes));
98 
99  return graph;
100  }
101  } else {
102  CUGRAPH_FAIL("Unsupported graph type for broadcasting.");
103 
104  return graph_t{handle};
105  }
106 }
