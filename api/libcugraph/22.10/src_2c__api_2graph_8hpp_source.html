49 cugraph_error_code_t transpose_storage(raft::handle_t const& handle,
50  cugraph_graph_t* graph,
51  cugraph_error_t* error)
52 {
53  if (store_transposed == graph->store_transposed_) {
54  auto p_graph =
55  reinterpret_cast<cugraph::graph_t<vertex_t, edge_t, weight_t, store_transposed, multi_gpu>*>(
56  graph->graph_);
57 
58  auto number_map = reinterpret_cast<rmm::device_uvector<vertex_t>*>(graph->number_map_);
59 
60  auto graph_transposed =
61  new cugraph::graph_t<vertex_t, edge_t, weight_t, !store_transposed, multi_gpu>(handle);
62 
63  std::optional<rmm::device_uvector<vertex_t>> new_number_map;
64 
65  std::tie(*graph_transposed, new_number_map) =
66  p_graph->transpose_storage(handle, std::move(*number_map), true);
67 
68  *number_map = std::move(new_number_map.value());
69 
70  delete p_graph;
71 
72  graph->graph_ = graph_transposed;
73  graph->store_transposed_ = !store_transposed;
74 
75  return CUGRAPH_SUCCESS;
76  } else {
77  error->error_message_ = "transpose failed, value of transpose does not match graph";
