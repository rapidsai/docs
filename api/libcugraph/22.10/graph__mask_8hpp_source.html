35 template <typename mask_t>
36 __host__ __device__ constexpr int log_bits()
37 {
38  switch (std::numeric_limits<mask_t>::digits) {
39  case 8: return 3;
40  case 16: return 4;

42  case 64: return 6;
43  default: return log2(std::numeric_limits<mask_t>::digits);
44  }

64 template <typename mask_type>
65 __device__ __host__ inline void _set_bit(mask_type* arr, mask_type h)
66 {
67  mask_type bit = bit_mod<mask_type>(h);
68  mask_type idx = h >> log_bits<mask_type>();
69  atomicOr(arr + idx, 1 << bit);
70 }

75 template <typename mask_type>
76 __device__ __host__ inline void _unset_bit(mask_type* arr, mask_type h)
77 {
78  mask_type bit = bit_mod<mask_type>(h);
79  mask_type idx = h >> log_bits<mask_type>();
80  atomicAnd(arr + idx, ~(1 << bit));
81 }

87 template <typename mask_type>
88 __device__ __host__ inline bool _is_set(mask_type* arr, mask_type h)
89 {

91  mask_type idx = h >> log_bits<mask_type>();
