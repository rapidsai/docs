82  partition_t(std::vector<vertex_t> const& vertex_partition_range_offsets,
83  int row_comm_size,
84  int col_comm_size,
85  int row_comm_rank,
86  int col_comm_rank)
87  : vertex_partition_range_offsets_(vertex_partition_range_offsets),
88  comm_rank_(col_comm_rank * row_comm_size + row_comm_rank),
89  row_comm_size_(row_comm_size),
90  col_comm_size_(col_comm_size),
91  row_comm_rank_(row_comm_rank),
92  col_comm_rank_(col_comm_rank)
93  {
94  CUGRAPH_EXPECTS(vertex_partition_range_offsets.size() ==
95  static_cast<size_t>(row_comm_size * col_comm_size + 1),
96  "Invalid API parameter: erroneous vertex_partition_range_offsets.size().");
97 
98  CUGRAPH_EXPECTS(std::is_sorted(vertex_partition_range_offsets_.begin(),
99  vertex_partition_range_offsets_.end()),
100  "Invalid API parameter: partition.vertex_partition_range_offsets values should "
101  "be non-descending.");
102  CUGRAPH_EXPECTS(
103  vertex_partition_range_offsets_[0] == vertex_t{0},
104  "Invalid API parameter: partition.vertex_partition_range_offsets[0] should be 0.");
105 
106  vertex_t start_offset{0};
107  edge_partition_major_value_start_offsets_.assign(number_of_local_edge_partitions(), 0);
108  for (size_t i = 0; i < edge_partition_major_value_start_offsets_.size(); ++i) {
109  edge_partition_major_value_start_offsets_[i] = start_offset;
110  start_offset +=
111  local_edge_partition_major_range_last(i) - local_edge_partition_major_range_first(i);
112  }
113  }

617  edge_partition_view_t<vertex_t, edge_t, weight_t, true> local_edge_partition_view(
618  size_t partition_idx) const
619  {
620  vertex_t major_range_first{};
621  vertex_t major_range_last{};
622  vertex_t minor_range_first{};
623  vertex_t minor_range_last{};
624  vertex_t major_value_range_start_offset{};
625  if constexpr (store_transposed) {
626  major_range_first = this->local_edge_partition_dst_range_first(partition_idx);
627  major_range_last = this->local_edge_partition_dst_range_last(partition_idx);
628  minor_range_first = this->local_edge_partition_src_range_first();
629  minor_range_last = this->local_edge_partition_src_range_last();
630  major_value_range_start_offset =
