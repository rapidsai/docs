85  partition_t(std::vector<vertex_t> const& vertex_partition_range_offsets,
86  int row_comm_size,
87  int col_comm_size,
88  int row_comm_rank,
89  int col_comm_rank)
90  : vertex_partition_range_offsets_(vertex_partition_range_offsets),
91  comm_rank_(col_comm_rank * row_comm_size + row_comm_rank),
92  row_comm_size_(row_comm_size),
93  col_comm_size_(col_comm_size),
94  row_comm_rank_(row_comm_rank),
95  col_comm_rank_(col_comm_rank)
96  {
97  CUGRAPH_EXPECTS(vertex_partition_range_offsets.size() ==
98  static_cast<size_t>(row_comm_size * col_comm_size + 1),
99  "Invalid API parameter: erroneous vertex_partition_range_offsets.size().");
100 
101  CUGRAPH_EXPECTS(std::is_sorted(vertex_partition_range_offsets_.begin(),
102  vertex_partition_range_offsets_.end()),
103  "Invalid API parameter: partition.vertex_partition_range_offsets values should "
104  "be non-descending.");
105  CUGRAPH_EXPECTS(
106  vertex_partition_range_offsets_[0] == vertex_t{0},
107  "Invalid API parameter: partition.vertex_partition_range_offsets[0] should be 0.");
108 
109  vertex_t start_offset{0};
110  edge_partition_major_value_start_offsets_.assign(number_of_local_edge_partitions(), 0);
111  for (size_t i = 0; i < edge_partition_major_value_start_offsets_.size(); ++i) {
112  edge_partition_major_value_start_offsets_[i] = start_offset;
113  start_offset +=
114  local_edge_partition_major_range_last(i) - local_edge_partition_major_range_first(i);
115  }

614  edge_partition_view_t<vertex_t, edge_t, weight_t, true> local_edge_partition_view(
615  size_t partition_idx) const
616  {
617  vertex_t major_range_first{};
618  vertex_t major_range_last{};
619  vertex_t minor_range_first{};
620  vertex_t minor_range_last{};
621  vertex_t major_value_range_start_offset{};
622  if constexpr (store_transposed) {
623  major_range_first = this->local_edge_partition_dst_range_first(partition_idx);
624  major_range_last = this->local_edge_partition_dst_range_last(partition_idx);
625  minor_range_first = this->local_edge_partition_src_range_first();
626  minor_range_last = this->local_edge_partition_src_range_last();
627  major_value_range_start_offset =
628  this->local_edge_partition_dst_value_start_offset(partition_idx);
