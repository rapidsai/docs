214 template <typename BufferType, typename Op>
215 void transform(const BufferType& input, BufferType& output, Op&& op)
216 {
217  if constexpr (is_std_tuple_of_arithmetic_vectors<BufferType>::value) {
218  size_t constexpr tuple_size = std::tuple_size<BufferType>::value;
219  detail::transform_tuple_impl(std::make_index_sequence<tuple_size>(), input, output, op);
220  } else if constexpr (is_arithmetic_vector<BufferType, rmm::device_uvector>::value) {
221  std::invoke(op, input, output);
222  }
