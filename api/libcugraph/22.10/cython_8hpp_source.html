54 // "container" for a graph type instance which insulates the owner from the
55 // specifics of the actual graph type. This is intended to be used in Cython
56 // code that only needs to pass a graph object to another wrapped C++ API. This
57 // greatly simplifies the Cython code since the Cython definition only needs to
58 // define the container and not the various individual graph types in Cython.
59 struct graph_container_t {
60  // FIXME: This union is in place only to support legacy calls, remove when
61  // migration to graph_t types is complete, or when legacy graph objects are
62  // constructed in the call_<<algo> wrappers instead of the
63  // populate_graph_container_legacy() function.
64  union graphPtrUnion {
65  ~graphPtrUnion() {}
66 
67  void* null;
68  std::unique_ptr<legacy::GraphCSRView<int32_t, int32_t, float>> GraphCSRViewFloatPtr;
69  std::unique_ptr<legacy::GraphCSRView<int32_t, int32_t, double>> GraphCSRViewDoublePtr;
70  std::unique_ptr<legacy::GraphCOOView<int32_t, int32_t, float>> GraphCOOViewFloatPtr;
71  std::unique_ptr<legacy::GraphCOOView<int32_t, int32_t, double>> GraphCOOViewDoublePtr;

78  // of a cython wrapper simply for passing a templated instantiation of a
79  // particular graph class from one call to another, and not to exist outside

81  // scope once the wrapper function returns). Therefore, copys and assignments
82  // to an instance are not supported and these methods are deleted.
