405  : number_of_vertices_p(graph.number_of_vertices),

419  vertex_t *dst_indices(void) { return static_cast<vertex_t *>(dst_indices_p.data()); }

421 
422  GraphCOOContents<vertex_t, edge_t, weight_t> release() noexcept

424  vertex_t number_of_vertices = number_of_vertices_p;

426  number_of_vertices_p = 0;
427  number_of_edges_p = 0;
428  return GraphCOOContents<vertex_t, edge_t, weight_t>{
429  number_of_vertices,
430  number_of_edges,
431  std::make_unique<rmm::device_buffer>(std::move(src_indices_p)),
432  std::make_unique<rmm::device_buffer>(std::move(dst_indices_p)),
433  std::make_unique<rmm::device_buffer>(std::move(edge_data_p))};
434  }
