128 // Uniform RW selector logic:
129 //
130 template <typename graph_type, typename real_t>
131 struct uniform_selector_t {
132  using vertex_t = typename graph_type::vertex_type;
133  using edge_t = typename graph_type::edge_type;
134  using weight_t = typename graph_type::weight_type;
135 
136  struct sampler_t {
137  sampler_t(edge_t const* ro,
138  vertex_t const* ci,
139  weight_t const* w,
140  edge_t const* ptr_d_cache_out_degs)
141  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_cache_out_degs_(ptr_d_cache_out_degs)
142  {
143  }
144 
145  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
146  vertex_t src_v,
147  real_t rnd_val,
148  vertex_t = 0 /* not used*/,
149  edge_t = 0 /* not used*/,
150  bool = false /* not used*/) const
151  {
152  auto crt_out_deg = ptr_d_cache_out_degs_[src_v];
153  if (crt_out_deg == 0) return thrust::nullopt; // src_v is a sink
154 
155  vertex_t v_indx =
156  static_cast<vertex_t>(rnd_val >= 1.0 ? crt_out_deg - 1 : rnd_val * crt_out_deg);
157  auto col_indx = v_indx >= crt_out_deg ? crt_out_deg - 1 : v_indx;
158  auto start_row = row_offsets_[src_v];
159 
160  auto weight_value =
161  (values_ == nullptr ? weight_t{1}
162  : values_[start_row + col_indx]); // account for un-weighted graphs
163  return thrust::optional{thrust::make_tuple(col_indices_[start_row + col_indx], weight_value)};
164  }
165 
166  private:
167  edge_t const* row_offsets_;
168  vertex_t const* col_indices_;
169  weight_t const* values_;
170 
171  edge_t const* ptr_d_cache_out_degs_;
172  };
173 
174  using sampler_type = sampler_t;
175 
176  uniform_selector_t(raft::handle_t const& handle, graph_type const& graph, real_t tag)
177  : d_cache_out_degs_(graph.compute_out_degrees(handle)),
178  sampler_{graph.local_edge_partition_view().offsets(),
179  graph.local_edge_partition_view().indices(),
180  graph.local_edge_partition_view().weights()
181  ? *(graph.local_edge_partition_view().weights())
182  : static_cast<weight_t*>(nullptr),
183  d_cache_out_degs_.data()}
184  {
185  }
186 
187  device_vec_t<edge_t> const& get_cached_out_degs(void) const { return d_cache_out_degs_; }
188 
189  sampler_t const& get_strategy(void) const { return sampler_; }
190 
191  private:
192  device_vec_t<edge_t> d_cache_out_degs_; // selector-specific: selector must own this resource
193  sampler_t sampler_; // which is why the sampling must be separated into a separate object:
194  // it must be captured by device a calling lambda, which is not possible for selector object,
195  // because it ows a device_vec;
196 };

202 struct visitor_aggregate_weights_t : visitors::visitor_t {
203  visitor_aggregate_weights_t(
204  raft::handle_t const& handle,
205  size_t num_vertices,
206  binary_op_t aggregate_op = thrust::plus<weight_t>{},
207  weight_t initial_value = 0) // different aggregation ops require different initial values;
208  : handle_(handle),

215  void visit_graph(graph_envelope_t::base_graph_t const& graph_v) override
216  {
217  auto const& graph_view =
218  static_cast<graph_view_t<vertex_t, edge_t, weight_t, false, false> const&>(graph_v);
219 
220  auto opt_weights = graph_view.local_edge_partition_view().weights();
221  CUGRAPH_EXPECTS(opt_weights.has_value(), "Cannot aggregate weights of un-weighted graph.");
222 
223  size_t num_vertices = d_aggregate_weights_.size();
224 
225  edge_t const* offsets = graph_view.local_edge_partition_view().offsets();
226 
227  weight_t const* values = *opt_weights;
228 
229  // Determine temporary device storage requirements:
230  //
231  void* ptr_d_temp_storage{nullptr};
232  size_t temp_storage_bytes = 0;
233  cub::DeviceSegmentedReduce::Reduce(ptr_d_temp_storage,
234  temp_storage_bytes,
235  values,
236  d_aggregate_weights_.data(),
237  num_vertices,
238  offsets,
239  offsets + 1,
240  aggregator_,
241  initial_value_,
242  handle_.get_stream());
243  // Allocate temporary storage
244  //
245  rmm::device_uvector<std::byte> d_temp_storage(temp_storage_bytes, handle_.get_stream());
246  ptr_d_temp_storage = d_temp_storage.data();
247 
248  // Run reduction:
249  //
250  cub::DeviceSegmentedReduce::Reduce(ptr_d_temp_storage,
251  temp_storage_bytes,
252  values,
253  d_aggregate_weights_.data(),
254  num_vertices,
255  offsets,
256  offsets + 1,
257  aggregator_,
258  initial_value_,
259  handle_.get_stream());
260  }

286 // Biased RW selection logic:
287 //
288 template <typename graph_type, typename real_t>
289 struct biased_selector_t {
290  using vertex_t = typename graph_type::vertex_type;
291  using edge_t = typename graph_type::edge_type;
292  using weight_t = typename graph_type::weight_type;
293 
294  struct sampler_t {
295  sampler_t(edge_t const* ro,
296  vertex_t const* ci,
297  weight_t const* w,
298  weight_t const* ptr_d_sum_weights)
299  : row_offsets_(ro), col_indices_(ci), values_(w), ptr_d_sum_weights_(ptr_d_sum_weights)
300  {
301  }
302 
303  // pre-condition:
304  //
305  // Sum(weights(neighborhood(src_v))) are pre-computed and
306  // stored in ptr_d_sum_weights_ (too expensive to check, here);
307  //
308  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
309  vertex_t src_v,
310  real_t rnd_val,
311  vertex_t = 0 /* not used*/,
312  edge_t = 0 /* not used*/,
313  bool = false /* not used*/) const
314  {
315  weight_t run_sum_w{0};
316  auto rnd_sum_weights = rnd_val * ptr_d_sum_weights_[src_v];
317 
318  auto col_indx_begin = row_offsets_[src_v];
319  auto col_indx_end = row_offsets_[src_v + 1];
320  if (col_indx_begin == col_indx_end) return thrust::nullopt; // src_v is a sink
321 
322  auto col_indx = col_indx_begin;
323  auto prev_col_indx = col_indx;
324 
325  for (; col_indx < col_indx_end; ++col_indx) {
326  if (rnd_sum_weights < run_sum_w) break;
327 
328  run_sum_w += values_[col_indx];
329  prev_col_indx = col_indx;
330  }
331  return thrust::optional{
332  thrust::make_tuple(col_indices_[prev_col_indx], values_[prev_col_indx])};
333  }
334 
335  private:
336  edge_t const* row_offsets_;
337  vertex_t const* col_indices_;
338  weight_t const* values_;
339 
340  weight_t const* ptr_d_sum_weights_;
341  };
342 
343  using sampler_type = sampler_t;
344 
345  biased_selector_t(raft::handle_t const& handle, graph_type const& graph, real_t tag)
346  : sum_calculator_(handle, graph.number_of_vertices()),
347  sampler_{graph.local_edge_partition_view().offsets(),
348  graph.local_edge_partition_view().indices(),
349  graph.local_edge_partition_view().weights()
350  ? *(graph.local_edge_partition_view().weights())
351  : static_cast<weight_t*>(nullptr),
352  sum_calculator_.get_aggregated_weights().data()}
353  {
354  graph.apply(sum_calculator_);
355  }
356 
357  sampler_t const& get_strategy(void) const { return sampler_; }
358 
359  decltype(auto) get_sum_weights(void) const { return sum_calculator_.get_aggregated_weights(); }
360 
361  private:
362  visitor_aggregate_weights_t<vertex_t, edge_t, weight_t> sum_calculator_;
363  sampler_t sampler_;
364 };

366 // node2vec RW selection logic:
367 // uses biased selector on scaled weights,
368 // to be computed (and possibly cached) according to
369 // `node2vec` logic (see `get_alpha()`);
370 // works on unweighted graphs (for which unscalled weights are 1.0);
371 //
372 // TODO: need to decide logic on very 1st step of traversal
373 // (which has no `prev_v` vertex);
374 //
375 template <typename graph_type, typename real_t>
376 struct node2vec_selector_t {
377  using vertex_t = typename graph_type::vertex_type;
378  using edge_t = typename graph_type::edge_type;
379  using weight_t = typename graph_type::weight_type;
380 
381  struct sampler_t {
382  sampler_t(edge_t const* ro,
383  vertex_t const* ci,
384  weight_t const* w,
385  weight_t p,
386  weight_t q,
387  vertex_t max_degree,
388  edge_t num_paths,
389  weight_t* ptr_alpha)
390  : row_offsets_(ro),
391  col_indices_(ci),
392  values_(w),
393  p_(p),
394  q_(q),
395  coalesced_alpha_{
396  (max_degree > 0) && (num_paths > 0) && (ptr_alpha != nullptr)
397  ? thrust::optional<thrust::tuple<vertex_t, edge_t, weight_t*>>{thrust::make_tuple(
398  max_degree, num_paths, ptr_alpha)}
399  : thrust::nullopt}
400  {
401  }
402 
403  // node2vec alpha scalling logic:
404  // pre-condition: assume column_indices[] is seg-sorted;
405  // (each row has column_indices[] sorted)
406  //
407  __device__ weight_t get_alpha(vertex_t prev_v, vertex_t src_v, vertex_t next_v) const
408  {
409  if (next_v == prev_v) {
410  return 1.0 / p_;
411  } else {
412  // binary-search `next_v` in the adj(prev_v)
413  //
414  auto prev_indx_begin = row_offsets_[prev_v];
415  auto prev_indx_end = row_offsets_[prev_v + 1];
416 
417  auto found_next_from_prev = thrust::binary_search(
418  thrust::seq, col_indices_ + prev_indx_begin, col_indices_ + prev_indx_end, next_v);
419 
420  if (found_next_from_prev) {
421  return 1;
422  } else {
423  return 1.0 / q_;
424  }
425  }
426  }
427 
428  __device__ thrust::optional<thrust::tuple<vertex_t, weight_t>> operator()(
429  vertex_t src_v, real_t rnd_val, vertex_t prev_v, edge_t path_index, bool start_path) const
430  {
431  auto const offset_indx_begin = row_offsets_[src_v];
432  auto const offset_indx_end = row_offsets_[src_v + 1];
433 
434  weight_t sum_scaled_weights{0};
435  auto offset_indx = offset_indx_begin;
436 
437  if (offset_indx_begin == offset_indx_end) return thrust::nullopt; // src_v is a sink
438 
439  // for 1st vertex in path just use biased random selection:
440  //
441  if (start_path) { // `src_v` is starting vertex in path
442  for (; offset_indx < offset_indx_end; ++offset_indx) {
443  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
444 
445  sum_scaled_weights += crt_weight;
446  }
447 
448  weight_t run_sum_w{0};
449  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
450  offset_indx = offset_indx_begin;
451  auto prev_offset_indx = offset_indx;
452 
453  // biased sampling selection loop:
454  // (Note: re-compute `scaled_weight`, since no cache is available);
455  //
456  for (; offset_indx < offset_indx_end; ++offset_indx) {
457  if (rnd_sum_weights < run_sum_w) break;
458 
459  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
460  run_sum_w += crt_weight;
461  prev_offset_indx = offset_indx;
462  }
463  return thrust::optional{
464  thrust::make_tuple(col_indices_[prev_offset_indx],
465  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
466  }
467 
468  // cached solution, for increased performance, but memory expensive:
469  //
470  if (coalesced_alpha_.has_value()) {
471  auto&& tpl = *coalesced_alpha_;
472 
473  auto max_out_deg = thrust::get<0>(tpl);
474  auto num_paths = thrust::get<1>(tpl);
475  weight_t* ptr_d_scaled_weights = thrust::get<2>(tpl);
476 
477  // sum-scaled-weights reduction loop:
478  //
479  auto const start_alpha_offset = max_out_deg * path_index;
480  for (vertex_t nghbr_indx = 0; offset_indx < offset_indx_end; ++offset_indx, ++nghbr_indx) {
481  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
482  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
483  auto scaled_weight = crt_weight * crt_alpha;
484 
485  // caching is available, hence cache the alpha's for next step
486  // (the actual sampling step);
487  //
488  ptr_d_scaled_weights[start_alpha_offset + nghbr_indx] = scaled_weight;
489 
490  sum_scaled_weights += scaled_weight;
491  }
492 
493  weight_t run_sum_w{0};
494  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
495  offset_indx = offset_indx_begin;
496  auto prev_offset_indx = offset_indx;
497 
498  // biased sampling selection loop:
499  //
500  for (vertex_t nghbr_indx = 0; offset_indx < offset_indx_end; ++offset_indx, ++nghbr_indx) {
501  if (rnd_sum_weights < run_sum_w) break;
502 
503  run_sum_w += ptr_d_scaled_weights[start_alpha_offset + nghbr_indx];
504  prev_offset_indx = offset_indx;
505  }
506  return thrust::optional{
507  thrust::make_tuple(col_indices_[prev_offset_indx],
508  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
509 
510  } else { // uncached solution, with much lower memory footprint but not as efficient
511 
512  for (; offset_indx < offset_indx_end; ++offset_indx) {
513  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
514 
515  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
516 
517  auto scaled_weight = crt_weight * crt_alpha;
518  sum_scaled_weights += scaled_weight;
519  }
520 
521  weight_t run_sum_w{0};
522  auto rnd_sum_weights = rnd_val * sum_scaled_weights;
523  offset_indx = offset_indx_begin;
524  auto prev_offset_indx = offset_indx;
525 
526  // biased sampling selection loop:
527  // (Note: re-compute `scaled_weight`, since no cache is available);
528  //
529  for (; offset_indx < offset_indx_end; ++offset_indx) {
530  if (rnd_sum_weights < run_sum_w) break;
531 
532  auto crt_alpha = get_alpha(prev_v, src_v, col_indices_[offset_indx]);
533  weight_t crt_weight = (values_ == nullptr ? weight_t{1} : values_[offset_indx]);
534  auto scaled_weight = crt_weight * crt_alpha;
535 
536  run_sum_w += scaled_weight;
537  prev_offset_indx = offset_indx;
538  }
539  return thrust::optional{
540  thrust::make_tuple(col_indices_[prev_offset_indx],
541  values_ == nullptr ? weight_t{1} : values_[prev_offset_indx])};
542  }
543  }
544 
545  decltype(auto) get_alpha_buffer(void) const { return coalesced_alpha_; }
546 
547  private:
548  edge_t const* row_offsets_;
549  vertex_t const* col_indices_;
550  weight_t const* values_;
551 
552  weight_t const p_;
553  weight_t const q_;
554 
555  // alpha scaling coalesced buffer (per path):
556  // (use as cache since the per-path alpha-buffer
557  // is used twice for each node transition:
558  // (1) for computing sum_scaled weights;
559  // (2) for using scaled_weights for the biased next vertex selection)
560  // this is information related to a scratchpad buffer, used as cache, hence mutable;
561  // (necessary, because get_strategy() is const)
562  //
563  mutable thrust::optional<thrust::tuple<vertex_t, edge_t, weight_t*>>
564  coalesced_alpha_; // tuple<max_vertex_degree,
565  // num_paths, alpha_buffer[max_vertex_degree*num_paths]>
566  };
567 
568  using sampler_type = sampler_t;
569 
570  node2vec_selector_t(raft::handle_t const& handle,
571  graph_type const& graph,
572  real_t tag,
573  weight_t p,
574  weight_t q,
575  edge_t num_paths = 0)
576  : max_out_degree_(num_paths > 0 ? graph.compute_max_out_degree(handle) : 0),
577  d_coalesced_alpha_{max_out_degree_ * num_paths, handle.get_stream()},
578  sampler_{graph.local_edge_partition_view().offsets(),
579  graph.local_edge_partition_view().indices(),
580  graph.local_edge_partition_view().weights()
581  ? *(graph.local_edge_partition_view().weights())
582  : static_cast<weight_t*>(nullptr),

584  q,

586  num_paths,
587  raw_ptr(d_coalesced_alpha_)}
588  {
589  }
590 
591  sampler_t const& get_strategy(void) const { return sampler_; }
592 
593  device_vec_t<weight_t> const& get_alpha_cache(void) const { return d_coalesced_alpha_; }
594 
595  private:
596  size_t max_out_degree_{0};
597 
598  // alpha scaling coalesced buffer (per path):
599  // (use as cache since the per-path alpha-buffer
600  // is used twice for each node transition:
601  // (1) for computing sum_scaled weights;

603  //

605  sampler_t sampler_;

745  // (d_random, d_coalesced_w)

748
