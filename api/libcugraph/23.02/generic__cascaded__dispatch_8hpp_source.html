36 // multi_gpu bool dispatcher:
37 // resolves bool `multi_gpu`
38 // and using template arguments vertex_t, edge_t, weight_t, store_transpose
39 // Calls functor
40 //
41 template <typename vertex_t,
42  typename edge_t,
43  typename weight_t,
44  typename edge_type_t,
45  bool store_transposed,
46  typename functor_t>
47 constexpr decltype(auto) multi_gpu_dispatcher(bool multi_gpu, functor_t& functor)
48 {
49  if (multi_gpu) {
50  return functor
51  .template operator()<vertex_t, edge_t, weight_t, edge_type_t, store_transposed, true>();
52  } else {
53  return functor
54  .template operator()<vertex_t, edge_t, weight_t, edge_type_t, store_transposed, false>();
55  }
56 }

58 // transpose bool dispatcher:
59 // resolves bool `store_transpose`
60 // and using template arguments vertex_t, edge_t, weight_t
61 // cascades into next level
62 // multi_gpu_dispatcher()
63 //
64 template <typename vertex_t,
65  typename edge_t,
66  typename weight_t,
67  typename edge_type_t,
68  typename functor_t>
69 constexpr decltype(auto) transpose_dispatcher(bool store_transposed,
70  bool multi_gpu,
71  functor_t& functor)
72 {
73  if (store_transposed) {
74  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, edge_type_t, true>(multi_gpu, functor);
75  } else {
76  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, edge_type_t, false>(multi_gpu, functor);
77  }
78 }

80 // edge_type_type type dispatcher:
81 // resolves weight_t from weight_type enum
82 // and using template arguments vertex_t, edge_t
83 // cascades into next level
84 // transpose_dispatcher()
85 //
86 template <typename vertex_t, typename edge_t, typename weight_t, typename functor_t>
87 constexpr decltype(auto) edge_type_type_dispatcher(cugraph::visitors::DTypes edge_type_type,
88  bool store_transposed,
89  bool multi_gpu,
90  functor_t& functor)
91 {
92  switch (edge_type_type) {
93  case cugraph::visitors::DTypes::INT32: {
94  using edge_type_t =
95  typename cugraph::visitors::DMapType<cugraph::visitors::DTypes::INT32>::type;
96  return transpose_dispatcher<vertex_t, edge_t, weight_t, edge_type_t>(
97  store_transposed, multi_gpu, functor);
98  }
99  case cugraph::visitors::DTypes::INT64: {
100  throw std::runtime_error(
101  "ERROR: Data type INT64 not allowed for edge type (valid types: INT32).");
102  break;
103  }
104  case cugraph::visitors::DTypes::FLOAT32: {
105  throw std::runtime_error(
106  "ERROR: Data type FLOAT32 not allowed for edge type (valid types: INT32).");
107  break;
108  }
109  case cugraph::visitors::DTypes::FLOAT64: {
110  throw std::runtime_error(
111  "ERROR: Data type FLOAT64 not allowed for edge type (valid types: INT32).");
112  break;
113  }
114 
115  default: {
116  std::stringstream ss;
117  ss << "ERROR: Unknown type enum:" << static_cast<int>(edge_type_type);
118  throw std::runtime_error(ss.str());
119  }
120  }
121 }
