125 template <typename T>
126 std::enable_if_t<std::is_arithmetic<T>::value, T> host_scalar_allreduce(
127  raft::comms::comms_t const& comm, T input, raft::comms::op_t op, cudaStream_t stream)
128 {
129  rmm::device_uvector<T> d_input(1, stream);
130  raft::update_device(d_input.data(), &input, 1, stream);
131  comm.allreduce(d_input.data(), d_input.data(), 1, op, stream);
132  T h_input{};
133  raft::update_host(&h_input, d_input.data(), 1, stream);
134  auto status = comm.sync_stream(stream);
135  CUGRAPH_EXPECTS(status == raft::comms::status_t::SUCCESS, "sync_stream() failure.");
136  return h_input;

253 template <typename T>
254 std::enable_if_t<std::is_arithmetic<T>::value, std::vector<T>> host_scalar_allgather(
255  raft::comms::comms_t const& comm, T input, cudaStream_t stream)
256 {
257  std::vector<size_t> rx_counts(comm.get_size(), size_t{1});
258  std::vector<size_t> displacements(rx_counts.size(), size_t{0});
259  std::iota(displacements.begin(), displacements.end(), size_t{0});
260  rmm::device_uvector<T> d_outputs(rx_counts.size(), stream);
261  raft::update_device(d_outputs.data() + comm.get_rank(), &input, 1, stream);
262  // FIXME: better use allgather
263  comm.allgatherv(d_outputs.data() + comm.get_rank(),
264  d_outputs.data(),
265  rx_counts.data(),
266  displacements.data(),
267  stream);
268  std::vector<T> h_outputs(rx_counts.size());
269  raft::update_host(h_outputs.data(), d_outputs.data(), rx_counts.size(), stream);
270  auto status = comm.sync_stream(stream);
271  CUGRAPH_EXPECTS(status == raft::comms::status_t::SUCCESS, "sync_stream() failure.");
272  return h_outputs;
