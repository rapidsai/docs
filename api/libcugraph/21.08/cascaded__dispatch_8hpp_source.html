57 // final step of cascading: calls factory on erased pack:
58 //
59 template <typename vertex_t,
60  typename edge_t,
61  typename weight_t,
62  bool tr,
63  bool mg,
64  std::enable_if_t<is_candidate<vertex_t, edge_t, weight_t>::value, void*> = nullptr>
65 constexpr pair_uniques_t graph_dispatcher(GTypes graph_type, erased_pack_t& ep)
66 {
67  switch (graph_type) {
68  case GTypes::GRAPH_T: {
69  using graph_t = typename GMapType<vertex_t, edge_t, weight_t, tr, mg, GTypes::GRAPH_T>::type;
70  graph_factory_t<graph_t> factory;
71 
72  pair_uniques_t p_uniques =
73  std::make_pair(factory.make_graph(ep),
74  std::make_unique<dependent_factory_t<vertex_t, edge_t, weight_t, tr, mg>>());
75 
76  return p_uniques;
77  } break;
78 
79  default: {
80  std::stringstream ss;
81  ss << "ERROR: Unknown type enum:" << static_cast<int>(graph_type);
82  throw std::runtime_error(ss.str());
83  }
84  }
85 }

87 // multi_gpu bool dispatcher:
88 // resolves bool `multi_gpu`
89 // and using template arguments vertex_t, edge_t, weight_t, store_transpose
90 // cascades into next level
91 // graph_dispatcher()
92 //
93 template <typename vertex_t, typename edge_t, typename weight_t, bool store_transposed>
94 constexpr decltype(auto) multi_gpu_dispatcher(bool multi_gpu, GTypes graph_type, erased_pack_t& ep)
95 {
96  switch (multi_gpu) {
97  case true: {
98  return graph_dispatcher<vertex_t, edge_t, weight_t, store_transposed, true>(graph_type, ep);
99  } break;
100  case false: {
101  return graph_dispatcher<vertex_t, edge_t, weight_t, store_transposed, false>(graph_type, ep);
102  }
103  }
104 }

106 // transpose bool dispatcher:
107 // resolves bool `store_transpose`
108 // and using template arguments vertex_t, edge_t, weight_t
109 // cascades into next level
110 // multi_gpu_dispatcher()
111 //
112 template <typename vertex_t, typename edge_t, typename weight_t>
113 constexpr decltype(auto) transp_dispatcher(bool store_transposed,
114  bool multi_gpu,
115  GTypes graph_type,
116  erased_pack_t& ep)
117 {
118  switch (store_transposed) {
119  case true: {
120  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, true>(multi_gpu, graph_type, ep);
121  } break;
122  case false: {
123  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, false>(multi_gpu, graph_type, ep);
124  }
125  }
126 }

128 // weight type dispatcher:
129 // resolves weigth_t from weight_type enum
130 // and using template arguments vertex_t, edge_t
131 // cascades into next level
132 // transp_dispatcher()
133 //
134 template <typename vertex_t, typename edge_t>
135 constexpr decltype(auto) weight_dispatcher(
136  DTypes weight_type, bool store_transposed, bool multi_gpu, GTypes graph_type, erased_pack_t& ep)
137 {
138  switch (weight_type) {
139  case DTypes::INT32: {
140  using weight_t = typename DMapType<DTypes::INT32>::type;
141  return transp_dispatcher<vertex_t, edge_t, weight_t>(
142  store_transposed, multi_gpu, graph_type, ep);
143  } break;
144  case DTypes::INT64: {
145  using weight_t = typename DMapType<DTypes::INT64>::type;
146  return transp_dispatcher<vertex_t, edge_t, weight_t>(
147  store_transposed, multi_gpu, graph_type, ep);
148  } break;
149  case DTypes::FLOAT32: {
150  using weight_t = typename DMapType<DTypes::FLOAT32>::type;
151  return transp_dispatcher<vertex_t, edge_t, weight_t>(
152  store_transposed, multi_gpu, graph_type, ep);
153  } break;
154  case DTypes::FLOAT64: {
155  using weight_t = typename DMapType<DTypes::FLOAT64>::type;
156  return transp_dispatcher<vertex_t, edge_t, weight_t>(
157  store_transposed, multi_gpu, graph_type, ep);
158  } break;
159  default: {
160  std::stringstream ss;
161  ss << "ERROR: Unknown type enum:" << static_cast<int>(weight_type);
162  throw std::runtime_error(ss.str());
163  }
164  }
165 }
