34 template <typename vertex_t, typename edge_t, typename weight_t>
35 void exact_fa2(GraphCOOView<vertex_t, edge_t, weight_t> &graph,
36  float *pos,
37  const int max_iter = 500,
38  float *x_start = nullptr,
39  float *y_start = nullptr,
40  bool outbound_attraction_distribution = true,
41  bool lin_log_mode = false,
42  bool prevent_overlapping = false,
43  const float edge_weight_influence = 1.0,
44  const float jitter_tolerance = 1.0,
45  const float scaling_ratio = 2.0,
46  bool strong_gravity_mode = false,
47  const float gravity = 1.0,
48  bool verbose = false,
49  internals::GraphBasedDimRedCallback *callback = nullptr)
50 {
51  cudaStream_t stream = {nullptr};
52  const edge_t e = graph.number_of_edges;
53  const vertex_t n = graph.number_of_vertices;
54 
55  float *d_repel{nullptr};
56  float *d_attract{nullptr};
57  float *d_old_forces{nullptr};
58  int *d_mass{nullptr};
59  float *d_swinging{nullptr};
60  float *d_traction{nullptr};
61 
62  rmm::device_vector<float> repel(n * 2, 0);
63  rmm::device_vector<float> attract(n * 2, 0);
64  rmm::device_vector<float> old_forces(n * 2, 0);
65  // FA2 requires degree + 1.
66  rmm::device_vector<int> mass(n, 1);
67  rmm::device_vector<float> swinging(n, 0);
68  rmm::device_vector<float> traction(n, 0);
69 
70  d_repel = repel.data().get();
71  d_attract = attract.data().get();
72  d_old_forces = old_forces.data().get();
73  d_mass = mass.data().get();
74  d_swinging = swinging.data().get();
75  d_traction = traction.data().get();
76 
77  int random_state = 0;
78  random_vector(pos, n * 2, random_state, stream);
79 
80  if (x_start && y_start) {
81  copy(n, x_start, pos);
82  copy(n, y_start, pos + n);
83  }
84 
85  // Sort COO for coalesced memory access.
86  sort(graph, stream);
87  CHECK_CUDA(stream);
88 
89  graph.degree(d_mass, cugraph::DegreeDirection::OUT);
90  CHECK_CUDA(stream);
91 
92  const vertex_t *row = graph.src_indices;
93  const vertex_t *col = graph.dst_indices;
94  const weight_t *v = graph.edge_data;
95 
96  float speed = 1.f;
97  float speed_efficiency = 1.f;
98  float outbound_att_compensation = 1.f;
99  float jt = 0.f;
100 
101  if (outbound_attraction_distribution) {
102  int sum = thrust::reduce(rmm::exec_policy(stream)->on(stream), mass.begin(), mass.end());
103  outbound_att_compensation = sum / (float)n;
104  }
105 
106  if (callback) {
107  callback->setup<float>(n, 2);
108  callback->on_preprocess_end(pos);
109  }
110 
111  for (int iter = 0; iter < max_iter; ++iter) {
112  // Reset force arrays
113  fill(n * 2, d_repel, 0.f);
114  fill(n * 2, d_attract, 0.f);
115  fill(n, d_swinging, 0.f);
116  fill(n, d_traction, 0.f);
117 
118  // Exact repulsion
119  apply_repulsion<vertex_t>(pos, pos + n, d_repel, d_repel + n, d_mass, scaling_ratio, n, stream);
120 
121  apply_gravity<vertex_t>(pos,
122  pos + n,
123  d_attract,
124  d_attract + n,
125  d_mass,
126  gravity,
127  strong_gravity_mode,
128  scaling_ratio,
129  n,
130  stream);
131 
132  apply_attraction<vertex_t, edge_t, weight_t>(row,
133  col,
134  v,
135  e,
136  pos,
137  pos + n,
138  d_attract,
139  d_attract + n,
140  d_mass,
141  outbound_attraction_distribution,
142  lin_log_mode,
143  edge_weight_influence,
144  outbound_att_compensation,
145  stream);
146 
147  compute_local_speed(d_repel,
148  d_repel + n,
149  d_attract,
150  d_attract + n,
151  d_old_forces,
152  d_old_forces + n,
153  d_mass,
154  d_swinging,
155  d_traction,
156  n,
157  stream);
158 
159  // Compute global swinging and traction values.
160  const float s =
161  thrust::reduce(rmm::exec_policy(stream)->on(stream), swinging.begin(), swinging.end());
162  const float t =
163  thrust::reduce(rmm::exec_policy(stream)->on(stream), traction.begin(), traction.end());
164 
165  adapt_speed<vertex_t>(jitter_tolerance, &jt, &speed, &speed_efficiency, s, t, n);
166 
167  apply_forces<vertex_t>(pos,
168  pos + n,
169  d_repel,
170  d_repel + n,
171  d_attract,
172  d_attract + n,
173  d_old_forces,
174  d_old_forces + n,
175  d_swinging,
176  speed,
177  n,
178  stream);
179 
180  if (callback) callback->on_epoch_end(pos);
181 
182  if (verbose) {
183  printf("iteration %i, speed: %f, speed_efficiency: %f, ", iter + 1, speed, speed_efficiency);
184  printf("jt: %f, ", jt);
185  printf("swinging: %f, traction: %f\n", s, t);
186  }
187  }
188 
189  if (callback) callback->on_train_end(pos);
190 }
191 
192 } // namespace detail
