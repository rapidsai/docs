24 template <typename vertex_t>
25 __global__ void repulsion_kernel(const float *restrict x_pos,
26  const float *restrict y_pos,
27  float *restrict repel_x,
28  float *restrict repel_y,
29  const int *restrict mass,
30  const float scaling_ratio,
31  const vertex_t n)
32 {
33  int j = (blockIdx.x * blockDim.x) + threadIdx.x; // for every item in row
34  int i = (blockIdx.y * blockDim.y) + threadIdx.y; // for every row
35  for (; i < n; i += gridDim.y * blockDim.y) {
36  for (; j < n; j += gridDim.x * blockDim.x) {
37  if (j >= i) return;
38  float x_dist = x_pos[i] - x_pos[j];
39  float y_dist = y_pos[i] - y_pos[j];
40  float distance = x_dist * x_dist + y_dist * y_dist;
41  distance += FLT_EPSILON;
42  float factor = scaling_ratio * mass[i] * mass[j] / distance;
43  // Add forces
44  atomicAdd(&repel_x[i], x_dist * factor);
45  atomicAdd(&repel_x[j], -x_dist * factor);
46  atomicAdd(&repel_y[i], y_dist * factor);
47  atomicAdd(&repel_y[j], -y_dist * factor);
48  }
49  }
50 }

52 template <typename vertex_t, int TPB_X = 32, int TPB_Y = 32>
53 void apply_repulsion(const float *restrict x_pos,
54  const float *restrict y_pos,
55  float *restrict repel_x,
56  float *restrict repel_y,
57  const int *restrict mass,
58  const float scaling_ratio,
59  const vertex_t n,
60  cudaStream_t stream)
61 {
62  dim3 nthreads(TPB_X, TPB_Y);
63  dim3 nblocks(min((n + nthreads.x - 1) / nthreads.x, CUDA_MAX_BLOCKS_2D),
64  min((n + nthreads.y - 1) / nthreads.y, CUDA_MAX_BLOCKS_2D));
65 
66  repulsion_kernel<vertex_t>
67  <<<nblocks, nthreads, 0, stream>>>(x_pos, y_pos, repel_x, repel_y, mass, scaling_ratio, n);
68  CHECK_CUDA(stream);
69 }
