409  : number_of_vertices_p(graph.number_of_vertices),

422  vertex_t *src_indices(void) { return static_cast<vertex_t *>(src_indices_p.data()); }
423  vertex_t *dst_indices(void) { return static_cast<vertex_t *>(dst_indices_p.data()); }
424  weight_t *edge_data(void) { return static_cast<weight_t *>(edge_data_p.data()); }
425 
426  GraphCOOContents<vertex_t, edge_t, weight_t> release() noexcept
427  {
428  vertex_t number_of_vertices = number_of_vertices_p;
429  edge_t number_of_edges = number_of_edges_p;
430  number_of_vertices_p = 0;
431  number_of_edges_p = 0;
432  return GraphCOOContents<vertex_t, edge_t, weight_t>{
433  number_of_vertices,
434  number_of_edges,
435  std::make_unique<rmm::device_buffer>(std::move(src_indices_p)),
436  std::make_unique<rmm::device_buffer>(std::move(dst_indices_p)),
437  std::make_unique<rmm::device_buffer>(std::move(edge_data_p))};
438  }
439
