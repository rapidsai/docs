60  // Courtesy: https://www.gnu.org/software/libc/manual/html_node/Backtraces.html
61  void collectCallStack() throw()
62  {
63 #ifdef __GNUC__
64  const int MaxStackDepth = 64;
65  void* stack[MaxStackDepth];
66  auto depth = backtrace(stack, MaxStackDepth);
67  std::ostringstream oss;
68  oss << std::endl << "Obtained " << depth << " stack frames" << std::endl;
69  std::unique_ptr<char*, decltype(&::free)> strings(backtrace_symbols(stack, depth), &::free);
70  if (strings.get() == nullptr) {
71  oss << "But no stack trace could be found!" << std::endl;
72  msg += oss.str();
73  return;
74  }
76  for (int i = 0; i < depth; ++i) {
77  oss << "#" << i << " in " << strings.get()[i] << std::endl;
78  }
79  msg += oss.str();
80 #endif // __GNUC__
81  }
