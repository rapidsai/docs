32 template <typename vertex_t,
33  typename edge_t,
34  typename weight_t,
35  bool store_transposed,
36  bool multi_gpu>
37 std::tuple<rmm::device_uvector<vertex_t>,
38  rmm::device_uvector<vertex_t>,
39  rmm::device_uvector<weight_t>,

41 uniform_nbr_sample_impl(
42  raft::handle_t const& handle,
43  graph_view_t<vertex_t, edge_t, store_transposed, multi_gpu> const& graph_view,
44  std::optional<edge_property_view_t<edge_t, weight_t const*>> edge_weight_view,
45  rmm::device_uvector<vertex_t>& d_in,
46  raft::host_span<const int> h_fan_out,
47  bool with_replacement,
48  uint64_t seed)
49 {
50 #ifdef NO_CUGRAPH_OPS
51  CUGRAPH_FAIL(
52  "uniform_nbr_sampl_impl not supported in this configuration, built with NO_CUGRAPH_OPS");
53 #else
54 
55  CUGRAPH_EXPECTS(h_fan_out.size() > 0,
56  "Invalid input argument: number of levels must be non-zero.");
57 
58  rmm::device_uvector<vertex_t> d_result_src(0, handle.get_stream());
59  rmm::device_uvector<vertex_t> d_result_dst(0, handle.get_stream());
60  auto d_result_weights =
61  thrust::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
62 
63  int32_t hop{0};
64  size_t comm_size{1};
65 
66  if constexpr (multi_gpu) {
67  seed += handle.get_comms().get_rank();
68  comm_size = handle.get_comms().get_size();
69  }
70 
71  for (auto&& k_level : h_fan_out) {
72  // prep step for extracting out-degs(sources):
73  if constexpr (multi_gpu) {
74  d_in = shuffle_int_vertices_to_local_gpu_by_vertex_partitioning(
75  handle, std::move(d_in), graph_view.vertex_partition_range_lasts());
76  }
77 
78  rmm::device_uvector<vertex_t> d_out_src(0, handle.get_stream());
79  rmm::device_uvector<vertex_t> d_out_dst(0, handle.get_stream());
80  auto d_out_weights = std::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()));
81 
82  if (k_level > 0) {
83  raft::random::RngState rng_state(seed);
84  seed += d_in.size() * k_level * comm_size;
85 
86  std::tie(d_out_src, d_out_dst, d_out_weights) = sample_edges(handle,
87  graph_view,
88  edge_weight_view,
89  rng_state,
90  d_in,
91  static_cast<size_t>(k_level),
92  with_replacement);
93  } else {
94  std::tie(d_out_src, d_out_dst, d_out_weights) =
95  gather_one_hop_edgelist(handle, graph_view, edge_weight_view, d_in);
96  }
97 
98  // resize accumulators:
99  auto old_sz = d_result_dst.size();
100  auto add_sz = d_out_dst.size();
101  auto new_sz = old_sz + add_sz;
102 
103  d_result_src.resize(new_sz, handle.get_stream());
104  d_result_dst.resize(new_sz, handle.get_stream());
105  d_result_weights->resize(new_sz, handle.get_stream());
106 
107  raft::copy(
108  d_result_src.begin() + old_sz, d_out_src.begin(), d_out_src.size(), handle.get_stream());
109  raft::copy(
110  d_result_dst.begin() + old_sz, d_out_dst.begin(), d_out_dst.size(), handle.get_stream());
111  raft::copy(d_result_weights->begin() + old_sz,
112  d_out_weights->begin(),
113  d_out_weights->size(),
114  handle.get_stream());
115 
116  d_in = std::move(d_out_dst);
117 
118  ++hop;
119  }
120 
121  return count_and_remove_duplicates<vertex_t, edge_t, weight_t>(
122  handle, std::move(d_result_src), std::move(d_result_dst), std::move(*d_result_weights));
123 #endif
124 }

137  rmm::device_uvector<int32_t>,
138  std::optional<rmm::device_uvector<int32_t>>>
139 uniform_neighbor_sample_impl(
140  raft::handle_t const& handle,
141  graph_view_t<vertex_t, edge_t, store_transposed, multi_gpu> const& graph_view,
142  std::optional<edge_property_view_t<edge_t, weight_t const*>> edge_weight_view,
143  std::optional<
144  edge_property_view_t<edge_t,
145  thrust::zip_iterator<thrust::tuple<edge_t const*, edge_type_t const*>>>>
146  edge_id_type_view,
147  rmm::device_uvector<vertex_t>&& d_in,
148  std::optional<rmm::device_uvector<int32_t>>&& d_labels,
149  raft::host_span<int32_t const> h_fan_out,
150  bool with_replacement,
151  raft::random::RngState& rng_state)
152 {
153 #ifdef NO_CUGRAPH_OPS
154  CUGRAPH_FAIL(
155  "uniform_neighbor_sampl_impl not supported in this configuration, built with NO_CUGRAPH_OPS");
156 #else
157 
158  CUGRAPH_EXPECTS(h_fan_out.size() > 0,
159  "Invalid input argument: number of levels must be non-zero.");
160 
161  rmm::device_uvector<vertex_t> d_result_src(0, handle.get_stream());
162  rmm::device_uvector<vertex_t> d_result_dst(0, handle.get_stream());
163  auto d_result_edge_id =
164  edge_id_type_view ? std::make_optional(rmm::device_uvector<edge_t>(0, handle.get_stream()))
165  : std::nullopt;
166  auto d_result_weight =
167  edge_weight_view ? std::make_optional(rmm::device_uvector<weight_t>(0, handle.get_stream()))
168  : std::nullopt;
169  auto d_result_edge_type =
170  edge_id_type_view ? std::make_optional(rmm::device_uvector<edge_type_t>(0, handle.get_stream()))
171  : std::nullopt;
172  rmm::device_uvector<int32_t> d_result_hop(0, handle.get_stream());
173  auto d_result_label = d_labels
174  ? std::make_optional(rmm::device_uvector<int32_t>(0, handle.get_stream()))
175  : std::nullopt;
176 
177  int32_t hop{0};
178  size_t comm_size{1};
179 
180  if constexpr (multi_gpu) { comm_size = handle.get_comms().get_size(); }
181 
182  for (auto&& k_level : h_fan_out) {
183  // prep step for extracting out-degs(sources):
184  if constexpr (multi_gpu) {
185  if (d_labels) {
186  std::tie(d_in, *d_labels) =
187  shuffle_int_vertex_value_pairs_to_local_gpu_by_vertex_partitioning(
188  handle,
189  std::move(d_in),
190  std::move(*d_labels),
191  graph_view.vertex_partition_range_lasts());
192  } else {
193  d_in = shuffle_int_vertices_to_local_gpu_by_vertex_partitioning(
194  handle, std::move(d_in), graph_view.vertex_partition_range_lasts());
195  }
196  }
197 
198  rmm::device_uvector<vertex_t> d_out_src(0, handle.get_stream());
199  rmm::device_uvector<vertex_t> d_out_dst(0, handle.get_stream());
200  std::optional<rmm::device_uvector<edge_t>> d_out_edge_id{std::nullopt};
201  std::optional<rmm::device_uvector<weight_t>> d_out_weight{std::nullopt};
202  std::optional<rmm::device_uvector<edge_type_t>> d_out_edge_type{std::nullopt};
203  std::optional<rmm::device_uvector<int32_t>> d_out_label{std::nullopt};
204 
205  if (k_level > 0) {
206  std::tie(d_out_src, d_out_dst, d_out_weight, d_out_edge_id, d_out_edge_type, d_out_label) =
207  sample_edges(handle,
208  graph_view,
209  edge_weight_view,
210  edge_id_type_view,
211  rng_state,
212  d_in,
213  d_labels,
214  static_cast<size_t>(k_level),
215  with_replacement);
216  } else {
217  std::tie(d_out_src, d_out_dst, d_out_weight, d_out_edge_id, d_out_edge_type, d_out_label) =
218  gather_one_hop_edgelist(
219  handle, graph_view, edge_weight_view, edge_id_type_view, d_in, d_labels);
220  }
221 
222  // FIXME: potential optimization, continual resizing/copying
223  // of these device vectors is inefficient. If we're going to
224  // do higher hop sampling it might be more efficient to create
225  // a separate vector for each hop and concatenate them all
226  // at the end rather than recopying each time.
227  // resize accumulators:
228  auto old_sz = d_result_dst.size();
229  auto add_sz = d_out_dst.size();
230  auto new_sz = old_sz + add_sz;
231 
232  d_result_src.resize(new_sz, handle.get_stream());
233  d_result_dst.resize(new_sz, handle.get_stream());
234  d_result_hop.resize(new_sz, handle.get_stream());
235  if (d_result_weight) d_result_weight->resize(new_sz, handle.get_stream());
236  if (d_result_edge_id) d_result_edge_id->resize(new_sz, handle.get_stream());
237  if (d_result_edge_type) d_result_edge_type->resize(new_sz, handle.get_stream());
238  if (d_result_label) d_result_label->resize(new_sz, handle.get_stream());
239 
240  raft::copy(
241  d_result_src.begin() + old_sz, d_out_src.begin(), d_out_src.size(), handle.get_stream());
242  raft::copy(
243  d_result_dst.begin() + old_sz, d_out_dst.begin(), d_out_dst.size(), handle.get_stream());
244  if (d_out_edge_id)
245  raft::copy(d_result_edge_id->begin() + old_sz,
246  d_out_edge_id->begin(),
247  d_out_edge_id->size(),
248  handle.get_stream());
249 
250  detail::scalar_fill(handle, d_result_hop.data() + old_sz, add_sz, hop);
251 
252  if (d_result_weight)
253  raft::copy(d_result_weight->begin() + old_sz,
254  d_out_weight->begin(),
255  d_out_weight->size(),
256  handle.get_stream());
257  if (d_result_edge_type)
258  raft::copy(d_result_edge_type->begin() + old_sz,
259  d_out_edge_type->begin(),
260  d_out_edge_type->size(),
261  handle.get_stream());
262  if (d_result_label) {
263  raft::copy(d_result_label->begin() + old_sz,
264  d_out_label->begin(),
265  d_out_label->size(),
266  handle.get_stream());
267  }
268 
269  d_in = std::move(d_out_dst);
270  d_labels = std::move(d_out_label);
271 
272  ++hop;
273  }
274 
275  return std::make_tuple(std::move(d_result_src),
276  std::move(d_result_dst),
277  std::move(d_result_weight),
278  std::move(d_result_edge_id),
279  std::move(d_result_edge_type),
280  std::move(d_result_hop),
281  std::move(d_result_label));
282 #endif
283 }
