47 uniform_neighbor_sample_impl(
48  raft::handle_t const& handle,
49  graph_view_t<vertex_t, edge_t, store_transposed, multi_gpu> const& graph_view,
50  std::optional<edge_property_view_t<edge_t, weight_t const*>> edge_weight_view,
51  std::optional<edge_property_view_t<edge_t, edge_t const*>> edge_id_view,
52  std::optional<edge_property_view_t<edge_t, edge_type_t const*>> edge_type_view,
53  raft::device_span<vertex_t const> starting_vertices,
54  std::optional<raft::device_span<label_t const>> starting_vertex_labels,
55  std::optional<std::tuple<raft::device_span<label_t const>, raft::device_span<int32_t const>>>
56  label_to_output_comm_rank,
57  raft::host_span<int32_t const> fan_out,
58  bool return_hops,
59  bool with_replacement,
60  raft::random::RngState& rng_state,

63 #ifdef NO_CUGRAPH_OPS
64  CUGRAPH_FAIL(
65  "uniform_neighbor_sample_impl not supported in this configuration, built with NO_CUGRAPH_OPS");
66 #else
67  CUGRAPH_EXPECTS(fan_out.size() > 0, "Invalid input argument: number of levels must be non-zero.");
68  CUGRAPH_EXPECTS(
69  fan_out.size() <= static_cast<size_t>(std::numeric_limits<int32_t>::max()),
70  "Invalid input argument: number of levels should not overflow int32_t"); // as we use int32_t
71  // to store hops
72 
73  if constexpr (!multi_gpu) {
74  CUGRAPH_EXPECTS(!label_to_output_comm_rank,
75  "cannot specify output GPU mapping in SG implementation");
76  }
77 
78  CUGRAPH_EXPECTS(
79  !label_to_output_comm_rank || starting_vertex_labels,
80  "cannot specify output GPU mapping without also specifying starting_vertex_labels");
81 
82  if (do_expensive_check) {
83  if (label_to_output_comm_rank) {
84  CUGRAPH_EXPECTS(cugraph::detail::is_sorted(handle, std::get<0>(*label_to_output_comm_rank)),
85  "Labels in label_to_output_comm_rank must be sorted");
86  }
87  }
88 
89  std::vector<rmm::device_uvector<vertex_t>> level_result_src_vectors{};
90  std::vector<rmm::device_uvector<vertex_t>> level_result_dst_vectors{};
91  auto level_result_weight_vectors =
92  edge_weight_view ? std::make_optional(std::vector<rmm::device_uvector<weight_t>>{})
93  : std::nullopt;
94  auto level_result_edge_id_vectors =
95  edge_id_view ? std::make_optional(std::vector<rmm::device_uvector<edge_t>>{}) : std::nullopt;
96  auto level_result_edge_type_vectors =
97  edge_type_view ? std::make_optional(std::vector<rmm::device_uvector<edge_type_t>>{})
98  : std::nullopt;
99  auto level_result_label_vectors =
100  starting_vertex_labels ? std::make_optional(std::vector<rmm::device_uvector<label_t>>{})
101  : std::nullopt;
102 
103  level_result_src_vectors.reserve(fan_out.size());
104  level_result_dst_vectors.reserve(fan_out.size());
105  if (level_result_weight_vectors) { (*level_result_weight_vectors).reserve(fan_out.size()); }
106  if (level_result_edge_id_vectors) { (*level_result_edge_id_vectors).reserve(fan_out.size()); }
107  if (level_result_edge_type_vectors) { (*level_result_edge_type_vectors).reserve(fan_out.size()); }
108  if (level_result_label_vectors) { (*level_result_label_vectors).reserve(fan_out.size()); }
109 
110  rmm::device_uvector<vertex_t> frontier_vertices(0, handle.get_stream());
111  auto frontier_vertex_labels =
112  starting_vertex_labels
113  ? std::make_optional(rmm::device_uvector<label_t>{0, handle.get_stream()})
114  : std::nullopt;
115 
116  std::vector<size_t> level_sizes{};
117  int32_t hop{0};
118  for (auto&& k_level : fan_out) {
119  rmm::device_uvector<vertex_t> srcs(0, handle.get_stream());
120  rmm::device_uvector<vertex_t> dsts(0, handle.get_stream());
121  std::optional<rmm::device_uvector<weight_t>> weights{std::nullopt};
122  std::optional<rmm::device_uvector<edge_t>> edge_ids{std::nullopt};
123  std::optional<rmm::device_uvector<edge_type_t>> edge_types{std::nullopt};
124  std::optional<rmm::device_uvector<int32_t>> labels{std::nullopt};
125 
126  if (k_level > 0) {
127  std::tie(srcs, dsts, weights, edge_ids, edge_types, labels) =
128  sample_edges(handle,
129  graph_view,
130  edge_weight_view,
131  edge_id_view,
132  edge_type_view,
133  rng_state,
134  starting_vertices,
135  starting_vertex_labels,
136  static_cast<size_t>(k_level),
137  with_replacement);
138  } else {
139  std::tie(srcs, dsts, weights, edge_ids, edge_types, labels) =
140  gather_one_hop_edgelist(handle,
141  graph_view,
142  edge_weight_view,
143  edge_id_view,
144  edge_type_view,
145  starting_vertices,
146  starting_vertex_labels);
147  }
148 
149  level_sizes.push_back(srcs.size());
150 
151  level_result_src_vectors.push_back(std::move(srcs));
152  level_result_dst_vectors.push_back(std::move(dsts));
153  if (weights) { (*level_result_weight_vectors).push_back(std::move(*weights)); }
154  if (edge_ids) { (*level_result_edge_id_vectors).push_back(std::move(*edge_ids)); }
155  if (edge_types) { (*level_result_edge_type_vectors).push_back(std::move(*edge_types)); }
156  if (labels) { (*level_result_label_vectors).push_back(std::move(*labels)); }
157 
158  ++hop;
159  if (hop < fan_out.size()) {
160  if constexpr (multi_gpu) {
161  size_t frontier_size = level_result_dst_vectors.back().size();
162  frontier_vertices.resize(frontier_size, handle.get_stream());
163 
164  raft::copy(frontier_vertices.begin(),
165  level_result_dst_vectors.back().data(),
166  level_result_dst_vectors.back().size(),
167  handle.get_stream());
168 
169  if (starting_vertex_labels) {
170  frontier_vertex_labels->resize(frontier_size, handle.get_stream());
171  raft::copy(frontier_vertex_labels->begin(),
172  level_result_label_vectors->back().data(),
173  frontier_size,
174  handle.get_stream());
175 
176  std::tie(frontier_vertices, *frontier_vertex_labels) =
177  shuffle_int_vertex_value_pairs_to_local_gpu_by_vertex_partitioning(
178  handle,
179  std::move(frontier_vertices),
180  std::move(*frontier_vertex_labels),
181  graph_view.vertex_partition_range_lasts());
182 
183  starting_vertices =
184  raft::device_span<vertex_t const>(frontier_vertices.data(), frontier_vertices.size());
185 
186  starting_vertex_labels = raft::device_span<label_t const>(frontier_vertex_labels->data(),
187  frontier_vertex_labels->size());
188  } else {
189  frontier_vertices = shuffle_int_vertices_to_local_gpu_by_vertex_partitioning(
190  handle, std::move(frontier_vertices), graph_view.vertex_partition_range_lasts());
191 
192  starting_vertices =
193  raft::device_span<vertex_t const>(frontier_vertices.data(), frontier_vertices.size());
194  }
195  } else {
196  starting_vertices = raft::device_span<vertex_t const>(
197  level_result_dst_vectors.back().data(), level_result_dst_vectors.back().size());
198  if (starting_vertex_labels) {
199  starting_vertex_labels = raft::device_span<label_t const>(
200  level_result_label_vectors->back().data(), level_result_label_vectors->back().size());
201  }
202  }
203  }
204  }
205 
206  auto result_size = std::reduce(level_sizes.begin(), level_sizes.end());
207  size_t output_offset{};
208 
209  rmm::device_uvector<vertex_t> result_srcs(result_size, handle.get_stream());
210  output_offset = 0;
211  for (size_t i = 0; i < level_result_src_vectors.size(); ++i) {
212  raft::copy(result_srcs.begin() + output_offset,
213  level_result_src_vectors[i].begin(),
214  level_sizes[i],
215  handle.get_stream());
216  output_offset += level_sizes[i];
217  }
218  level_result_src_vectors.clear();
219  level_result_src_vectors.shrink_to_fit();
220 
221  rmm::device_uvector<vertex_t> result_dsts(result_size, handle.get_stream());
222  output_offset = 0;
223  for (size_t i = 0; i < level_result_dst_vectors.size(); ++i) {
224  raft::copy(result_dsts.begin() + output_offset,
225  level_result_dst_vectors[i].begin(),
226  level_sizes[i],
227  handle.get_stream());
228  output_offset += level_sizes[i];
229  }
230  level_result_dst_vectors.clear();
231  level_result_dst_vectors.shrink_to_fit();
232 
233  auto result_weights =
234  level_result_weight_vectors
235  ? std::make_optional(rmm::device_uvector<weight_t>(result_size, handle.get_stream()))
236  : std::nullopt;
237  if (result_weights) {
238  output_offset = 0;
239  for (size_t i = 0; i < (*level_result_weight_vectors).size(); ++i) {
240  raft::copy((*result_weights).begin() + output_offset,
241  (*level_result_weight_vectors)[i].begin(),
242  level_sizes[i],
243  handle.get_stream());
244  output_offset += level_sizes[i];
245  }
246  level_result_weight_vectors = std::nullopt;
247  }
248 
249  auto result_edge_ids =
250  level_result_edge_id_vectors
251  ? std::make_optional(rmm::device_uvector<edge_t>(result_size, handle.get_stream()))
252  : std::nullopt;
253  if (result_edge_ids) {
254  output_offset = 0;
255  for (size_t i = 0; i < (*level_result_edge_id_vectors).size(); ++i) {
256  raft::copy((*result_edge_ids).begin() + output_offset,
257  (*level_result_edge_id_vectors)[i].begin(),
258  level_sizes[i],
259  handle.get_stream());
260  output_offset += level_sizes[i];
261  }
262  level_result_edge_id_vectors = std::nullopt;
263  }
264 
265  auto result_edge_types =
266  level_result_edge_type_vectors
267  ? std::make_optional(rmm::device_uvector<edge_type_t>(result_size, handle.get_stream()))
268  : std::nullopt;
269  if (result_edge_types) {
270  output_offset = 0;
271  for (size_t i = 0; i < (*level_result_edge_type_vectors).size(); ++i) {
272  raft::copy((*result_edge_types).begin() + output_offset,
273  (*level_result_edge_type_vectors)[i].begin(),
274  level_sizes[i],
275  handle.get_stream());
276  output_offset += level_sizes[i];
277  }
278  level_result_edge_type_vectors = std::nullopt;
279  }
280 
281  std::optional<rmm::device_uvector<int32_t>> result_hops{std::nullopt};
282 
283  if (return_hops) {
284  result_hops = rmm::device_uvector<int32_t>(result_size, handle.get_stream());
285  output_offset = 0;
286  for (size_t i = 0; i < fan_out.size(); ++i) {
287  scalar_fill(
288  handle, result_hops->data() + output_offset, level_sizes[i], static_cast<int32_t>(i));
289  output_offset += level_sizes[i];
290  }
291  }
292 
293  auto result_labels =
294  level_result_label_vectors
295  ? std::make_optional(rmm::device_uvector<label_t>(result_size, handle.get_stream()))
296  : std::nullopt;
297  if (result_labels) {
298  output_offset = 0;
299  for (size_t i = 0; i < (*level_result_label_vectors).size(); ++i) {
300  raft::copy((*result_labels).begin() + output_offset,
301  (*level_result_label_vectors)[i].begin(),
302  level_sizes[i],
303  handle.get_stream());
304  output_offset += level_sizes[i];
305  }
306  level_result_label_vectors = std::nullopt;
307  }
308 
309  return detail::shuffle_and_organize_output(handle,
310  std::move(result_srcs),
311  std::move(result_dsts),
312  std::move(result_weights),
313  std::move(result_edge_ids),
314  std::move(result_edge_types),
315  std::move(result_hops),
316  std::move(result_labels),
317  label_to_output_comm_rank);
318 #endif
