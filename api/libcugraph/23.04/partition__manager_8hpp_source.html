29 class partition_manager {
30  public:
31  // we 2D partition both a graph adjacency matrix and the GPUs. The graph adjacency matrix is 2D
32  // partitioned along the major axis and the minor axis. The GPUs are 2D partitioned to
33  // gpu_col_comm_size * gpu_row_comm_size where gpu_col_comm_size is the size of the column
34  // direction communicator (GPUs in the same column in the GPU 2D partitioning belongs to the same
35  // column sub-communicator) and row_comm_size is the size of the row direction communicator (GPUs
36  // in the same row belongs to the same row sub-communicator). GPUs in the same GPU row
37  // communicator have consecutive process IDs (and may be physically closer in hierarchical
38  // interconnects). Graph algorithms require communications due to the graph adjacency matrix
39  // partitioning along the major axis (major sub-communicator is responsible for this) and along
40  // the minor axis (minor sub-communicator is responsible for this). This variable controls whether
41  // to map the major sub-communicator to the GPU row communicator or the GPU column communicator.
42  static constexpr bool map_major_comm_to_gpu_row_comm = true;
43 
44 #ifdef __CUDACC__
45  __host__ __device__
46 #endif
47  static int
48  compute_global_comm_rank_from_vertex_partition_id(int major_comm_size,
49  int minor_comm_size,
50  int vertex_partition_id)
51  {
52  return map_major_comm_to_gpu_row_comm
53  ? vertex_partition_id
54  : (vertex_partition_id % major_comm_size) * minor_comm_size +
55  (vertex_partition_id / major_comm_size);
56  }
57 
58 #ifdef __CUDACC__
59  __host__ __device__
60 #endif
61  static int
62  compute_global_comm_rank_from_graph_subcomm_ranks(int major_comm_size,
63  int minor_comm_size,
64  int major_comm_rank,
65  int minor_comm_rank)
66  {
67  return map_major_comm_to_gpu_row_comm ? (minor_comm_rank * major_comm_size + major_comm_rank)
68  : (major_comm_rank * minor_comm_size + minor_comm_rank);
69  }
70 
71 #ifdef __CUDACC__
72  __host__ __device__
73 #endif
74  static int
75  compute_vertex_partition_id_from_graph_subcomm_ranks(int major_comm_size,
76  int minor_comm_size,
77  int major_comm_rank,
78  int minor_comm_rank)
79  {
80  return map_major_comm_to_gpu_row_comm
81  ? compute_global_comm_rank_from_graph_subcomm_ranks(
82  major_comm_size, minor_comm_size, major_comm_rank, minor_comm_rank)
83  : minor_comm_rank * major_comm_size + major_comm_rank;
84  }
85 
86  static std::string major_comm_name()
87  {
88  return std::string(map_major_comm_to_gpu_row_comm ? "gpu_row_comm" : "gpu_col_comm");
89  }
90 
91  static std::string minor_comm_name()
92  {
93  return std::string(map_major_comm_to_gpu_row_comm ? "gpu_col_comm" : "gpu_row_comm");
94  }
95 
96  static void init_subcomm(raft::handle_t& handle, int gpu_row_comm_size)
97  {
98  auto& comm = handle.get_comms();
99 
100  auto rank = comm.get_rank();
101  int row_idx = rank / gpu_row_comm_size;
102  int col_idx = rank % gpu_row_comm_size;
103 
104  handle.set_subcomm("gpu_row_comm",
105  std::make_shared<raft::comms::comms_t>(comm.comm_split(row_idx, col_idx)));
106  handle.set_subcomm("gpu_col_comm",
107  std::make_shared<raft::comms::comms_t>(comm.comm_split(col_idx, row_idx)));
108  };
109 };
