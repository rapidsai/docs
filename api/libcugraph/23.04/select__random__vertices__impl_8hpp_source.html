38 template <typename vertex_t, typename edge_t, bool store_transposed, bool multi_gpu>
39 rmm::device_uvector<vertex_t> select_random_vertices(
40  raft::handle_t const& handle,
41  graph_view_t<vertex_t, edge_t, store_transposed, multi_gpu> const& graph_view,
42  raft::random::RngState& rng_state,
43  size_t select_count,
44  bool with_replacement,
45  bool sort_vertices)
46 {
47  CUGRAPH_EXPECTS(
48  with_replacement || select_count <= static_cast<size_t>(graph_view.number_of_vertices()),
49  "Invalid input arguments: select_count should not exceed the number of vertices if "
50  "with_replacement == false.");
51 
52  rmm::device_uvector<vertex_t> mg_sample_buffer(0, handle.get_stream());
53 
54  size_t this_gpu_select_count{0};
55  if constexpr (multi_gpu) {
56  auto const comm_rank = handle.get_comms().get_rank();
57  auto const comm_size = handle.get_comms().get_size();
58 
59  this_gpu_select_count =
60  select_count / static_cast<size_t>(comm_size) +
61  (static_cast<size_t>(comm_rank) < static_cast<size_t>(select_count % comm_size) ? size_t{1}
62  : size_t{0});
63  } else {
64  this_gpu_select_count = select_count;
65  }
66 
67  if (with_replacement) {
68  // FIXME: need to double check uniform_random_fill generates random numbers in [0, V) (not [0,
69  // V])
70  mg_sample_buffer.resize(this_gpu_select_count, handle.get_stream());
71  cugraph::detail::uniform_random_fill(handle.get_stream(),
72  mg_sample_buffer.data(),
73  mg_sample_buffer.size(),
74  vertex_t{0},
75  graph_view.number_of_vertices(),
76  rng_state);
77  } else {
78  auto local_vertex_partition_range_first = graph_view.local_vertex_partition_range_first();
79  auto local_vertex_partition_range_last = graph_view.local_vertex_partition_range_last();
80 
81  mg_sample_buffer = rmm::device_uvector<vertex_t>(
82  local_vertex_partition_range_last - local_vertex_partition_range_first, handle.get_stream());
83  thrust::sequence(handle.get_thrust_policy(),
84  mg_sample_buffer.begin(),
85  mg_sample_buffer.end(),
86  local_vertex_partition_range_first);
87 
88  { // random shuffle (use this instead of thrust::shuffle to use raft::random::RngState)
89  rmm::device_uvector<float> random_numbers(mg_sample_buffer.size(), handle.get_stream());
90  cugraph::detail::uniform_random_fill(handle.get_stream(),
91  random_numbers.data(),
92  random_numbers.size(),
93  float{0.0},
94  float{1.0},
95  rng_state);
96  thrust::sort_by_key(handle.get_thrust_policy(),
97  random_numbers.begin(),
98  random_numbers.end(),
99  mg_sample_buffer.begin());
100  }
101 
102  if constexpr (multi_gpu) {
103  auto const comm_rank = handle.get_comms().get_rank();
104  auto const comm_size = handle.get_comms().get_size();
105 
106  std::vector<size_t> tx_value_counts(comm_size);
107  for (int i = 0; i < comm_size; ++i) {
108  tx_value_counts[i] =
109  mg_sample_buffer.size() / comm_size +
110  (static_cast<size_t>(i) < static_cast<size_t>(mg_sample_buffer.size() % comm_size) ? 1
111  : 0);
112  }
113  std::tie(mg_sample_buffer, std::ignore) = cugraph::shuffle_values(
114  handle.get_comms(), mg_sample_buffer.begin(), tx_value_counts, handle.get_stream());
115 
116  { // random shuffle (use this instead of thrust::shuffle to use raft::random::RngState)
117  rmm::device_uvector<float> random_numbers(mg_sample_buffer.size(), handle.get_stream());
118  cugraph::detail::uniform_random_fill(handle.get_stream(),
119  random_numbers.data(),
120  random_numbers.size(),
121  float{0.0},
122  float{1.0},
123  rng_state);
124  thrust::sort_by_key(handle.get_thrust_policy(),
125  random_numbers.begin(),
126  random_numbers.end(),
127  mg_sample_buffer.begin());
128  }
129 
130  auto buffer_sizes = cugraph::host_scalar_allgather(
131  handle.get_comms(), mg_sample_buffer.size(), handle.get_stream());
132  auto min_buffer_size = *std::min_element(buffer_sizes.begin(), buffer_sizes.end());
133  if (min_buffer_size <= select_count / comm_size) {
134  auto new_sizes = std::vector<size_t>(comm_size, min_buffer_size);
135  auto num_deficits = select_count - min_buffer_size * comm_size;
136  for (int i = 0; i < comm_size; ++i) {
137  auto delta = std::min(num_deficits, buffer_sizes[i] - min_buffer_size);
138  new_sizes[i] += delta;
139  num_deficits -= delta;
140  }
141  this_gpu_select_count = new_sizes[comm_rank];
142  }
143  }
144 
145  mg_sample_buffer.resize(this_gpu_select_count, handle.get_stream());
146  mg_sample_buffer.shrink_to_fit(handle.get_stream());
147  }
148 
149  if constexpr (multi_gpu) {
150  mg_sample_buffer = cugraph::detail::shuffle_int_vertices_to_local_gpu_by_vertex_partitioning(
151  handle, std::move(mg_sample_buffer), graph_view.vertex_partition_range_lasts());
152  }
153 
154  if (sort_vertices) {
155  thrust::sort(handle.get_thrust_policy(), mg_sample_buffer.begin(), mg_sample_buffer.end());
156  }
157 
158  return mg_sample_buffer;
159 }
