54 // multi_gpu bool dispatcher:
55 // resolves bool `multi_gpu`
56 // and using template arguments vertex_t, edge_t, weight_t, store_transpose
57 // Calls functor
58 //
59 template <typename vertex_t,
60  typename edge_t,
61  typename weight_t,
62  typename edge_type_t,
63  bool store_transposed,
64  typename functor_t>
65 constexpr decltype(auto) multi_gpu_dispatcher(bool multi_gpu, functor_t& functor)
66 {
67  if (multi_gpu) {
68  return functor
69  .template operator()<vertex_t, edge_t, weight_t, edge_type_t, store_transposed, true>();
70  } else {
71  return functor
72  .template operator()<vertex_t, edge_t, weight_t, edge_type_t, store_transposed, false>();
73  }
74 }

76 // transpose bool dispatcher:
77 // resolves bool `store_transpose`
78 // and using template arguments vertex_t, edge_t, weight_t
79 // cascades into next level
80 // multi_gpu_dispatcher()
81 //
82 template <typename vertex_t,
83  typename edge_t,
84  typename weight_t,
85  typename edge_type_t,
86  typename functor_t>
87 constexpr decltype(auto) transpose_dispatcher(bool store_transposed,
88  bool multi_gpu,
89  functor_t& functor)
90 {
91  if (store_transposed) {
92  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, edge_type_t, true>(multi_gpu, functor);
93  } else {
94  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, edge_type_t, false>(multi_gpu, functor);
95  }
96 }

98 // edge_type_type type dispatcher:
99 // resolves weight_t from weight_type enum
100 // and using template arguments vertex_t, edge_t
101 // cascades into next level
102 // transpose_dispatcher()
103 //
104 template <typename vertex_t, typename edge_t, typename weight_t, typename functor_t>
105 constexpr decltype(auto) edge_type_type_dispatcher(cugraph_data_type_id_t edge_type_type,
106  bool store_transposed,
107  bool multi_gpu,
108  functor_t& functor)
109 {
110  switch (edge_type_type) {
111  case cugraph_data_type_id_t::INT32: {
112  using edge_type_t = int32_t;
113  return transpose_dispatcher<vertex_t, edge_t, weight_t, edge_type_t>(
114  store_transposed, multi_gpu, functor);
115  }
116  case cugraph_data_type_id_t::INT64: {
117  throw std::runtime_error(
118  "ERROR: Data type INT64 not allowed for edge type (valid types: INT32).");
119  break;
120  }
121  case cugraph_data_type_id_t::FLOAT32: {
122  throw std::runtime_error(
123  "ERROR: Data type FLOAT32 not allowed for edge type (valid types: INT32).");
124  break;
125  }
126  case cugraph_data_type_id_t::FLOAT64: {
127  throw std::runtime_error(
128  "ERROR: Data type FLOAT64 not allowed for edge type (valid types: INT32).");
129  break;
130  }
131 
132  default: {
133  std::stringstream ss;
134  ss << "ERROR: Unknown type enum:" << static_cast<int>(edge_type_type);
135  throw std::runtime_error(ss.str());
136  }
137  }
138 }

182 // edge type dispatcher:
183 // resolves edge_t from edge_type enum
184 // and using template argument vertex_t

188 template <typename vertex_t, typename functor_t>
189 constexpr decltype(auto) edge_dispatcher(cugraph_data_type_id_t edge_type,
190  cugraph_data_type_id_t weight_type,
191  cugraph_data_type_id_t edge_type_type,
192  bool store_transposed,

195 {
