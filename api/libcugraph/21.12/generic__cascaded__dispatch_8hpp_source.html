36 // multi_gpu bool dispatcher:
37 // resolves bool `multi_gpu`
38 // and using template arguments vertex_t, edge_t, weight_t, store_transpose
39 // Calls functor
40 //
41 template <typename vertex_t,
42  typename edge_t,
43  typename weight_t,
44  bool store_transposed,
45  typename functor_t>
46 constexpr decltype(auto) multi_gpu_dispatcher(bool multi_gpu, functor_t& functor)
47 {
48  if (multi_gpu) {
49  return functor.template operator()<vertex_t, edge_t, weight_t, store_transposed, true>();
50  } else {
51  return functor.template operator()<vertex_t, edge_t, weight_t, store_transposed, false>();
52  }
53 }

55 // transpose bool dispatcher:
56 // resolves bool `store_transpose`
57 // and using template arguments vertex_t, edge_t, weight_t
58 // cascades into next level
59 // multi_gpu_dispatcher()
60 //
61 template <typename vertex_t, typename edge_t, typename weight_t, typename functor_t>
62 constexpr decltype(auto) transpose_dispatcher(bool store_transposed,
63  bool multi_gpu,
64  functor_t& functor)
65 {
66  if (store_transposed) {
67  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, true>(multi_gpu, functor);
68  } else {
69  return multi_gpu_dispatcher<vertex_t, edge_t, weight_t, false>(multi_gpu, functor);
70  }
71 }

73 // weight type dispatcher:
74 // resolves weigth_t from weight_type enum
75 // and using template arguments vertex_t, edge_t
76 // cascades into next level
77 // transpose_dispatcher()
78 //
79 template <typename vertex_t, typename edge_t, typename functor_t>
80 constexpr decltype(auto) weight_dispatcher(cugraph::visitors::DTypes weight_type,
81  bool store_transposed,
82  bool multi_gpu,
83  functor_t& functor)
84 {
85  switch (weight_type) {
86  case cugraph::visitors::DTypes::INT32: {
87  using weight_t = typename cugraph::visitors::DMapType<cugraph::visitors::DTypes::INT32>::type;
88  return transpose_dispatcher<vertex_t, edge_t, weight_t>(store_transposed, multi_gpu, functor);
89  } break;
90  case cugraph::visitors::DTypes::INT64: {
91  using weight_t = typename cugraph::visitors::DMapType<cugraph::visitors::DTypes::INT64>::type;
92  return transpose_dispatcher<vertex_t, edge_t, weight_t>(store_transposed, multi_gpu, functor);
93  } break;
94  case cugraph::visitors::DTypes::FLOAT32: {
95  using weight_t =
96  typename cugraph::visitors::DMapType<cugraph::visitors::DTypes::FLOAT32>::type;
97  return transpose_dispatcher<vertex_t, edge_t, weight_t>(store_transposed, multi_gpu, functor);
98  } break;
99  case cugraph::visitors::DTypes::FLOAT64: {
100  using weight_t =
101  typename cugraph::visitors::DMapType<cugraph::visitors::DTypes::FLOAT64>::type;
102  return transpose_dispatcher<vertex_t, edge_t, weight_t>(store_transposed, multi_gpu, functor);
103  } break;
104  default: {
105  std::stringstream ss;
106  ss << "ERROR: Unknown type enum:" << static_cast<int>(weight_type);
107  throw std::runtime_error(ss.str());
108  }
109  }
110 }
