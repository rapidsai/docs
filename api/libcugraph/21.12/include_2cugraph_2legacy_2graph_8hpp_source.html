388  public:

400  bool has_data = false,
401  cudaStream_t stream = nullptr,
402  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
403  : number_of_vertices_p(number_of_vertices),
404  number_of_edges_p(number_of_edges),
405  src_indices_p(sizeof(vertex_t) * number_of_edges, stream, mr),
406  dst_indices_p(sizeof(vertex_t) * number_of_edges, stream, mr),

411  GraphCOO(GraphCOOView<vertex_t, edge_t, weight_t> const& graph,
412  cudaStream_t stream = nullptr,
413  rmm::mr::device_memory_resource* mr = rmm::mr::get_current_device_resource())
414  : number_of_vertices_p(graph.number_of_vertices),
415  number_of_edges_p(graph.number_of_edges),
416  src_indices_p(graph.src_indices, graph.number_of_edges * sizeof(vertex_t), stream, mr),
417  dst_indices_p(graph.dst_indices, graph.number_of_edges * sizeof(vertex_t), stream, mr)
418  {
419  if (graph.has_data()) {
420  edge_data_p =
421  rmm::device_buffer{graph.edge_data, graph.number_of_edges * sizeof(weight_t), stream, mr};
422  }
423  }
424  GraphCOO(GraphCOOContents<vertex_t, edge_t, weight_t>&& contents)
425  : number_of_vertices_p(contents.number_of_vertices),
426  number_of_edges_p(contents.number_of_edges),
427  src_indices_p(std::move(*(contents.src_indices.release()))),
428  dst_indices_p(std::move(*(contents.dst_indices.release()))),
429  edge_data_p(std::move(*(contents.edge_data.release())))
430  {
431  }
432 
433  vertex_t number_of_vertices(void) { return number_of_vertices_p; }
434  edge_t number_of_edges(void) { return number_of_edges_p; }
435  vertex_t* src_indices(void) { return static_cast<vertex_t*>(src_indices_p.data()); }
436  vertex_t* dst_indices(void) { return static_cast<vertex_t*>(dst_indices_p.data()); }
437  weight_t* edge_data(void) { return static_cast<weight_t*>(edge_data_p.data()); }
438 
439  GraphCOOContents<vertex_t, edge_t, weight_t> release() noexcept
440  {
441  vertex_t number_of_vertices = number_of_vertices_p;
442  edge_t number_of_edges = number_of_edges_p;
443  number_of_vertices_p = 0;
444  number_of_edges_p = 0;
445  return GraphCOOContents<vertex_t, edge_t, weight_t>{
446  number_of_vertices,
447  number_of_edges,
448  std::make_unique<rmm::device_buffer>(std::move(src_indices_p)),
449  std::make_unique<rmm::device_buffer>(std::move(dst_indices_p)),
450  std::make_unique<rmm::device_buffer>(std::move(edge_data_p))};
451  }
452 
453  GraphCOOView<vertex_t, edge_t, weight_t> view(void) noexcept
454  {
455  return GraphCOOView<vertex_t, edge_t, weight_t>(
