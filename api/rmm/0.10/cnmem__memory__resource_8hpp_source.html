67 cnmem_memory_resource (std::size_t initial_pool_size, memory_kind mem_kind)
68 {
69 cnmemDevice_t dev;
70
71 // TODO Update exception
72 if (cudaSuccess != cudaGetDevice(&(dev. device ))) {
73 throw std::runtime_error{ "Failed to get CUDA device" };
74 }
75 // Note: cnmem defaults to half GPU memory
76 dev. size = initial_pool_size;
77 dev. numStreams = 1;
78 cudaStream_t streams[1];
79 streams[0] = 0;
80 dev. streams = streams;
81 dev. streamSizes = 0;
82
83 unsigned long flags = (mem_kind == memory_kind::MANAGED ) ?
84 CNMEM_FLAGS_MANAGED : 0;
85
86 // TODO Update exception
87 auto status = cnmemInit(1, &dev, flags);
88 if (CNMEM_STATUS_SUCCESS != status) {
89 std::string msg = cnmemGetErrorString(status);
90 throw std::runtime_error{ "Failed to initialize cnmem: " + msg};
91 }
92 }

106 void * do_allocate(std::size_t bytes, cudaStream_t stream) override {
107 register_stream(stream);
108 void * p{ nullptr };
109 auto status = cnmemMalloc(&p, bytes, stream);

135 void register_stream(cudaStream_t stream) {
136 // Don't register null stream with CNMEM
137 if (stream != 0) {
138 // TODO Probably don't want to have to take a lock for every memory
139 // allocation
140 std::lock_guard<std::mutex> lock(streams_mutex);
141 auto result = registered_streams.insert(stream);
142
143 if (result.second == true ) {
144 auto status = cnmemRegisterStream(stream);
145 if (CNMEM_STATUS_SUCCESS != status) {
146 std::string msg = cnmemGetErrorString(status);
147 throw std::runtime_error{ "Failed to register stream with cnmem: " +
148 msg};
149 }
150 }
151 }
