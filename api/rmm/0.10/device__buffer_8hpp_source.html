283 void resize(std::size_t new_size) {
284 // If the requested size is smaller, just update the size without any
285 // allocations
286 if (new_size <= _size) {
287 _size = new_size;
288 } else {
289 void * const new_data = _mr->allocate(new_size, _stream);
290 if (_size > 0) {
291 auto status =
292 cudaMemcpyAsync(new_data, _data, _size, cudaMemcpyDefault, _stream);
293
294 if (cudaSuccess != status) {
295 throw std::runtime_error{ "Device memory copy failed." };
296 }
297 }
298 _mr->deallocate(_data, _size, _stream);
299 _data = new_data;
300 _size = new_size;
301 _capacity = new_size;
302 }
303 }

318 void shrink_to_fit() {
319 if (size() != capacity()) {
320 void * const new_data = _mr->allocate(size(), stream());
321 if (size() > 0) {
322 auto status = cudaMemcpyAsync(new_data, _data, size(),
323 cudaMemcpyDefault, stream());
324 if (cudaSuccess != status) {
325 throw std::runtime_error{ "Device memory copy failed." };
326 }
327 }
328 _mr->deallocate(_data, size(), stream());
329 _data = new_data;
330 _capacity = size();
331 }
