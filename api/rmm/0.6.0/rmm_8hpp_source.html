119 inline rmmError_t alloc(T** ptr, size_t size, cudaStream_t stream, const char* file,
120  unsigned int line) {
121  rmm::LogIt log(rmm::Logger::Alloc, 0, size, stream, file, line);
122 
123  if (!ptr && !size) {
124  return RMM_SUCCESS;
125  } else if( !size ) {
126  ptr[0] = nullptr;
127  return RMM_SUCCESS;
128  }
129 
130  if (!ptr) return RMM_ERROR_INVALID_ARGUMENT;
131 
132  if (rmm::Manager::usePoolAllocator()) {
133  RMM_CHECK(rmm::Manager::getInstance().registerStream(stream));
134  RMM_CHECK_CNMEM(cnmemMalloc(reinterpret_cast<void**>(ptr), size, stream));
135  } else if (rmm::Manager::useManagedMemory()) {
136  RMM_CHECK_CUDA(cudaMallocManaged(reinterpret_cast<void**>(ptr), size));
137  } else
138  RMM_CHECK_CUDA(cudaMalloc(reinterpret_cast<void**>(ptr), size));
139 
140  log.setPointer(*ptr);
141  return RMM_SUCCESS;
142 }

161 inline rmmError_t realloc(T** ptr, size_t new_size, cudaStream_t stream,
162  const char* file, unsigned int line) {
163  rmm::LogIt log(rmm::Logger::Realloc, ptr, new_size, stream, file, line);
164 
165  if (!ptr && !new_size) {
166  return RMM_SUCCESS;
167  }
168 
169  if (!ptr) return RMM_ERROR_INVALID_ARGUMENT;
170 
171  if (rmm::Manager::usePoolAllocator()) {
172  RMM_CHECK(rmm::Manager::getInstance().registerStream(stream));
173  RMM_CHECK_CNMEM(cnmemFree(*reinterpret_cast<void**>(ptr), stream));
174  RMM_CHECK_CNMEM(
175  cnmemMalloc(reinterpret_cast<void**>(ptr), new_size, stream));
176  } else {
177  RMM_CHECK_CUDA(cudaFree(*ptr));
178  if (!new_size)
179  ptr[0] = nullptr;
180  else if (rmm::Manager::useManagedMemory())
181  RMM_CHECK_CUDA(cudaMallocManaged(reinterpret_cast<void**>(ptr), new_size));
182  else
183  RMM_CHECK_CUDA(cudaMalloc(reinterpret_cast<void**>(ptr), new_size));
184  }
185 
186  log.setPointer(*ptr);
187  return RMM_SUCCESS;
188 }
