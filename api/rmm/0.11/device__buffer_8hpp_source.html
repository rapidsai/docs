113 device_buffer( void const * source_data, std::size_t size,
114 cudaStream_t stream = 0,
115 mr::device_memory_resource* mr = mr::get_default_resource())
116 : _size{size}, _capacity{size}, _stream{stream}, _mr{mr} {

198 device_buffer& operator=(device_buffer const & other) {

286 void resize(std::size_t new_size, cudaStream_t stream = 0) {
287 set_stream(stream);
288 // If the requested size is smaller, just update the size without any
289 // allocations
290 if (new_size <= _size) {
291 _size = new_size;
292 } else {
293 void * const new_data = _mr->allocate(new_size, this->stream());
294 if (_size > 0) {
295 auto status =
296 cudaMemcpyAsync(new_data, _data, _size, cudaMemcpyDefault,
297 this->stream());
298
299 if (cudaSuccess != status) {
300 throw std::runtime_error{ "Device memory copy failed." };
301 }
302 }
303 _mr->deallocate(_data, _size, this->stream());
304 _data = new_data;
305 _size = new_size;
306 _capacity = new_size;
307 }
308 }

323 void shrink_to_fit(cudaStream_t stream = 0) {
324 set_stream(stream);
325 if (size() != capacity()) {
326 void * const new_data = _mr->allocate(size(), this->stream());
327 if (size() > 0) {
328 auto status = cudaMemcpyAsync(new_data, _data, size(),
329 cudaMemcpyDefault, this->stream());
330 if (cudaSuccess != status) {
331 throw std::runtime_error{ "Device memory copy failed." };
332 }
333 }
334 _mr->deallocate(_data, size(), this->stream());
335 _data = new_data;
336 _capacity = size();
337 }
