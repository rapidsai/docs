71 cnmem_memory_resource (std::size_t initial_pool_size,
72 std::vector<int> const & devices,
73 memory_kind pool_type) {
74 std::vector<cnmemDevice_t> cnmem_devices;
75
76 // If no devices were specified, use the current one
77 if (devices.empty()) {
78 int current_device{};
79 auto status = cudaGetDevice(&current_device);
80 if (status != cudaSuccess){
81 throw std::runtime_error{ "Failed to get current device." };
82 }
83 cnmemDevice_t dev{};
84 dev. device = current_device;
85 dev.size = initial_pool_size;
86 cnmem_devices.push_back(dev);
87 } else {
88 for ( auto const & d : devices) {
89 cnmemDevice_t dev{};
90 dev. device = d;
91 dev.size = initial_pool_size;
92 cnmem_devices.push_back(dev);
93 }
94 }
95
96 unsigned long flags =
97 (pool_type == memory_kind::MANAGED ) ? CNMEM_FLAGS_MANAGED : 0;
98 // TODO Update exception
99 auto status = cnmemInit(cnmem_devices.size(), cnmem_devices.data(), flags);
100 if (CNMEM_STATUS_SUCCESS != status) {
101 std::string msg = cnmemGetErrorString(status);
102 throw std::runtime_error{ "Failed to initialize cnmem: " + msg};
103 }
104 }
105
106 private :
118 void * do_allocate(std::size_t bytes, cudaStream_t stream) override {
119 register_stream(stream);
120 void * p{ nullptr };
121 auto status = cnmemMalloc(&p, bytes, stream);
122 if (CNMEM_STATUS_SUCCESS != status) {
123 #ifndef NDEBUG
124 std::cerr << "cnmemMalloc failed: " << cnmemGetErrorString(status)
125 << "\n" ;
126 #endif
127 throw std::bad_alloc{};
128 }
129 return p;
130 }

139 void do_deallocate( void * p, std::size_t, cudaStream_t stream) override {
140 auto status = cnmemFree(p, stream);
141 if (CNMEM_STATUS_SUCCESS != status) {
142 #ifndef NDEBUG
143 std::cerr << "cnmemFree failed: " << cnmemGetErrorString(status) << "\n" ;
144 #endif
145 }
146 }

148 void register_stream(cudaStream_t stream) {
149 // Don't register null stream with CNMEM
150 if (stream != 0) {
151 // TODO Probably don't want to have to take a lock for every memory
152 // allocation
153 std::lock_guard<std::mutex> lock(streams_mutex);
154 auto result = registered_streams.insert(stream);
155
156 if (result.second == true ) {
157 auto status = cnmemRegisterStream(stream);
158 if (CNMEM_STATUS_SUCCESS != status) {
159 std::string msg = cnmemGetErrorString(status);
160 throw std::runtime_error{ "Failed to register stream with cnmem: " +
161 msg};
162 }
163 }
164 }
165 }

175 std::pair<size_t, size_t> do_get_mem_info(cudaStream_t stream) const {
176 std::size_t free_size;
177 std::size_t total_size;
178 auto status = cnmemMemGetInfo(&free_size, &total_size, stream);
179 if (CNMEM_STATUS_SUCCESS != status) {
180 std::string msg = cnmemGetErrorString(status);
181 throw std::runtime_error{ "cnmemMemGetInfo failed: " + msg};
182 }
183 return std::make_pair(free_size, total_size);
184 }
