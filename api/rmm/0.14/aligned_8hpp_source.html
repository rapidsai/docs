73 template <typename Alloc>
74 void *aligned_allocate(std::size_t bytes, std::size_t alignment, Alloc alloc)
75 {
76  assert(is_pow2(alignment));
77 
78  // allocate memory for bytes, plus potential alignment correction,
79  // plus store of the correction offset
80  std::size_t padded_allocation_size{bytes + alignment + sizeof(std::ptrdiff_t)};
81 
82  char *const original = static_cast<char *>(alloc(padded_allocation_size));
83 
84  // account for storage of offset immediately prior to the aligned pointer
85  void *aligned{original + sizeof(std::ptrdiff_t)};
86 
87  // std::align modifies `aligned` to point to the first aligned location
88  std::align(alignment, bytes, aligned, padded_allocation_size);
89 
90  // Compute the offset between the original and aligned pointers
91  std::ptrdiff_t offset = static_cast<char *>(aligned) - original;
92 
93  // Store the offset immediately before the aligned pointer
94  *(static_cast<std::ptrdiff_t *>(aligned) - 1) = offset;
95 
96  return aligned;
97 }
