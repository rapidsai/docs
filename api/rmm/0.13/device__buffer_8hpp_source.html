195 device_buffer& operator=(device_buffer const & other) {
196 if (&other != this ) {
197 // If the current capacity is large enough and the resources are
198 // compatible, just reuse the existing memory
199 if ((capacity() > other.size()) and _mr->is_equal(*other._mr)) {
200 set_stream(other.stream());
201 resize(other.size());
202 copy(other.data(), other.size());
203 } else {
204 // Otherwise, need to deallocate and allocate new memory
205 deallocate();
206 set_stream(other.stream());
207 _mr = other._mr;
208 allocate(other.size());
209 copy(other.data(), other.size());
210 }
211 }
212 return * this ;
213 }
214
227 device_buffer& operator=(device_buffer&& other) noexcept {
228 if (&other != this ) {
229 deallocate();
230
231 _data = other._data;
232 _size = other._size;
233 _capacity = other._capacity;
234 set_stream(other.stream());
235 _mr = other._mr;
236
237 other._data = nullptr ;
238 other._size = 0;
239 other._capacity = 0;
240 other.set_stream(0);
241 }
242 return * this ;
243 }

283 void resize(std::size_t new_size, cudaStream_t stream = 0) {
284 set_stream(stream);
285 // If the requested size is smaller than the current capacity, just update
286 // the size without any allocations
287 if (new_size <= capacity()) {
288 _size = new_size;
289 } else {
290 void * const new_data = _mr->allocate(new_size, this->stream());
291 RMM_CUDA_TRY(cudaMemcpyAsync(new_data, data(), size(), cudaMemcpyDefault,
292 this->stream()));
293 deallocate();
294 _data = new_data;
295 _size = new_size;
296 _capacity = new_size;
297 }
298 }

313 void shrink_to_fit(cudaStream_t stream = 0) {
314 set_stream(stream);
315 if (size() != capacity()) {
316 // Invoke copy ctor on self which only copies `[0, size())` and swap it
317 // with self. The temporary `device_buffer` will hold the old contents
318 // which will then be destroyed
319 auto tmp = device_buffer{* this , stream};
320 std::swap(tmp, * this );
321 }
322 }

397 void allocate(std::size_t bytes) {
