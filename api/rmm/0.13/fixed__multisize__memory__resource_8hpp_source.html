47 // Integer pow function
48 std::size_t ipow(std::size_t base, std::size_t exp) {
49  std::size_t ret = 1;
50  while (exp > 0) {
51  if (exp & 1) ret *= base; // multiply the result by the current base
52  base *= base; // square the base
53  exp = exp >> 1; // divide the exponent in half
54  }
55  return ret;
56 }

112  fixed_size_mr_.emplace_back(new fixed_size_memory_resource<Upstream>(

114  }

171  device_memory_resource* get_resource(std::size_t bytes) {
172  assert(bytes <= get_max_size());
173 
174  auto exponentiate = [this](std::size_t const& k) { return ipow(size_base_, k); };
175  auto min_exp = thrust::make_transform_iterator(
176  thrust::make_counting_iterator(min_size_exponent_), exponentiate);
177 
178  auto iter = std::upper_bound(min_exp, min_exp + (max_size_exponent_ - min_size_exponent_),
179  bytes, [](std::size_t const& a, std::size_t const& b) { return a <= b; });
180 
181  return fixed_size_mr_[std::distance(min_exp, iter)].get();
182  }
