132  void insert(block const& b) {
133  if (is_empty()) {
134  insert(blocks.end(), b);
135  return;
136  }
137 
138  // Find the right place (in ascending ptr order) to insert the block
139  // Can't use binary_search because it's a linked list and will be quadratic
140  auto const next = std::find_if(blocks.begin(), blocks.end(),
141  [b](block const& i) { return i.ptr > b.ptr; });
142  auto const previous = (next == blocks.begin()) ? next : std::prev(next);
143 
144  // Coalesce with neighboring blocks or insert the new block if it can't be coalesced
145  bool const merge_prev = previous->is_contiguous_before(b);
146  bool const merge_next = (next != blocks.end()) && b.is_contiguous_before(*next);
147 
148  if (merge_prev && merge_next) {
149  *previous = detail::merge_blocks(*previous, b);
150  *previous = detail::merge_blocks(*previous, *next);
151  erase(next);
152  } else if (merge_prev) {
153  *previous = detail::merge_blocks(*previous, b);
154  } else if (merge_next) {
155  *next = detail::merge_blocks(b, *next);
156  } else {
157  insert(next, b); // cannot be coalesced, just insert
158  }
159  }
