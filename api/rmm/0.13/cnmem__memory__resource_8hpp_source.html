102 cnmem_memory_resource (std::size_t initial_pool_size,
103 std::vector<int> const & devices,
104 memory_kind pool_type) {
105 std::vector<cnmemDevice_t> cnmem_devices;
106
107 // If no devices were specified, use the current one
108 if (devices.empty()) {
109 int current_device{};
110 RMM_CUDA_TRY(cudaGetDevice(&current_device));
111 cnmemDevice_t dev{};
112 dev. device = current_device;
113 dev.size = initial_pool_size;
114 cnmem_devices.push_back(dev);
115 } else {
116 for ( auto const & d : devices) {
117 cnmemDevice_t dev{};
118 dev. device = d;
119 dev.size = initial_pool_size;
120 cnmem_devices.push_back(dev);
121 }
122 }
123
124 unsigned long flags =
125 (pool_type == memory_kind::MANAGED ) ? CNMEM_FLAGS_MANAGED : 0;
126
127 CNMEM_TRY(cnmemInit(cnmem_devices.size(), cnmem_devices.data(), flags));
128 }

142 void * do_allocate(std::size_t bytes, cudaStream_t stream) override {

175 void register_stream(cudaStream_t stream) {
176 // Don't register null stream with CNMEM
177 if (stream != 0) {
178 // TODO Probably don't want to have to take a lock for every memory
179 // allocation
180 std::lock_guard<std::mutex> lock(streams_mutex);
181 auto result = registered_streams.insert(stream);
182 if (result.second == true ) {
183 CNMEM_TRY(cnmemRegisterStream(stream));
184 }
185 }

196 std::pair<size_t, size_t> do_get_mem_info(cudaStream_t stream) const {
197 std::size_t free_size;
