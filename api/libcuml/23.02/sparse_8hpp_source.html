69  const T alpha,

72  const bool transB,
73  const T beta,

75  cudaStream_t stream) const override
76  {
77  const SimpleSparseMat<T>& B = *this;
78  int kA = A.n;
79  int kB = B.m;
80 
81  if (transA) {
82  ASSERT(A.n == C.m, "GEMM invalid dims: m");
83  kA = A.m;
84  } else {
85  ASSERT(A.m == C.m, "GEMM invalid dims: m");
86  }
87 
88  if (transB) {
89  ASSERT(B.m == C.n, "GEMM invalid dims: n");
90  kB = B.n;
91  } else {
92  ASSERT(B.n == C.n, "GEMM invalid dims: n");
93  }
94  ASSERT(kA == kB, "GEMM invalid dims: k");
95 
96  // matrix C must change the order and be transposed, because we need
97  // to swap arguments A and B in cusparseSpMM.
98  cusparseDnMatDescr_t descrC;
99  auto order = C.ord == COL_MAJOR ? CUSPARSE_ORDER_ROW : CUSPARSE_ORDER_COL;
100  RAFT_CUSPARSE_TRY(raft::sparse::detail::cusparsecreatednmat(
101  &descrC, C.n, C.m, order == CUSPARSE_ORDER_COL ? C.n : C.m, C.data, order));
102 
103  /*
104  The matrix A must have the same order as the matrix C in the input
105  of function cusparseSpMM (i.e. swapped order w.r.t. original C).
106  To account this requirement, I may need to flip transA (whether to transpose A).
107 
108  C C' rowsC' colsC' ldC' A A' rowsA' colsA' ldA' flipTransA
109  c r n m m c r n m m x
110  c r n m m r r m n n o
111  r c n m n c c m n m o
112  r c n m n r c n m n x
113 
114  where:
115  c/r - column/row major order
116  A,C - input to gemmb
117  A', C' - input to cusparseSpMM
118  ldX' - leading dimension - m or n, depending on order and transX
119  */
120  cusparseDnMatDescr_t descrA;
121  RAFT_CUSPARSE_TRY(raft::sparse::detail::cusparsecreatednmat(&descrA,
122  C.ord == A.ord ? A.n : A.m,
123  C.ord == A.ord ? A.m : A.n,
124  A.ord == COL_MAJOR ? A.m : A.n,
125  A.data,
126  order));
127  auto opA =
128  transA ^ (C.ord == A.ord) ? CUSPARSE_OPERATION_NON_TRANSPOSE : CUSPARSE_OPERATION_TRANSPOSE;
129 
130  cusparseSpMatDescr_t descrB;
131  RAFT_CUSPARSE_TRY(raft::sparse::detail::cusparsecreatecsr(
132  &descrB, B.m, B.n, B.nnz, B.row_ids, B.cols, B.values));
133  auto opB = transB ? CUSPARSE_OPERATION_NON_TRANSPOSE : CUSPARSE_OPERATION_TRANSPOSE;
134 
135  auto alg = order == CUSPARSE_ORDER_COL ? CUSPARSE_SPMM_CSR_ALG1 : CUSPARSE_SPMM_CSR_ALG2;
136 
137  size_t bufferSize;
138  RAFT_CUSPARSE_TRY(raft::sparse::detail::cusparsespmm_bufferSize(handle.get_cusparse_handle(),
139  opB,
140  opA,
141  &alpha,
142  descrB,
143  descrA,
144  &beta,
145  descrC,
146  alg,
147  &bufferSize,
148  stream));
149 
150  raft::interruptible::synchronize(stream);
151  rmm::device_uvector<T> tmp(bufferSize, stream);
152 
153  RAFT_CUSPARSE_TRY(raft::sparse::detail::cusparsespmm(handle.get_cusparse_handle(),
154  opB,
155  opA,
156  &alpha,
157  descrB,
158  descrA,
159  &beta,
160  descrC,
161  alg,
162  tmp.data(),
163  stream));
164 
165  RAFT_CUSPARSE_TRY(cusparseDestroyDnMat(descrA));
166  RAFT_CUSPARSE_TRY(cusparseDestroySpMat(descrB));
167  RAFT_CUSPARSE_TRY(cusparseDestroyDnMat(descrC));
168  }

192  raft::update_host(&cols[0], mat.cols, mat.nnz, rmm::cuda_stream_default);
193  raft::update_host(&row_ids[0], mat.row_ids, mat.m + 1, rmm::cuda_stream_default);
194  raft::interruptible::synchronize(rmm::cuda_stream_view());
195 
196  int i, row_end = 0;
197  for (int row = 0; row < mat.m; row++) {
198  i = row_end;
199  row_end = row_ids[row + 1];
200  for (int col = 0; col < mat.n; col++) {
201  if (i >= row_end || col < cols[i]) {
202  os << "0";
203  } else {
204  os << values[i];
205  i++;
206  }
207  if (col < mat.n - 1) os << ",";
208  }
209 
210  os << std::endl;
211  }
