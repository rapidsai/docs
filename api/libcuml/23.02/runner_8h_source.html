51 template <typename value_idx, typename value_t>
52 struct FixConnectivitiesRedOp {
53  value_idx* colors;
54  value_t* core_dists;
55  value_idx m;
56 
57  FixConnectivitiesRedOp(value_idx* colors_, value_t* core_dists_, value_idx m_)
58  : colors(colors_), core_dists(core_dists_), m(m_){};
59 
60  typedef typename raft::KeyValuePair<value_idx, value_t> KVP;
61  DI void operator()(value_idx rit, KVP* out, const KVP& other)
62  {
63  if (rit < m && other.value < std::numeric_limits<value_t>::max() &&
64  colors[rit] != colors[other.key]) {
65  value_t core_dist_rit = core_dists[rit];
66  value_t core_dist_other = max(core_dist_rit, max(core_dists[other.key], other.value));
67 
68  value_t core_dist_out;
69  if (out->key > -1) {
70  core_dist_out = max(core_dist_rit, max(core_dists[out->key], out->value));
71  } else {
72  core_dist_out = out->value;
73  }
74 
75  bool smaller = core_dist_other < core_dist_out;
76  out->key = smaller ? other.key : out->key;
77  out->value = smaller ? core_dist_other : core_dist_out;
78  }
79  }
80 
81  DI KVP operator()(value_idx rit, const KVP& a, const KVP& b)
82  {
83  if (rit < m && a.key > -1 && colors[rit] != colors[a.key]) {
84  value_t core_dist_rit = core_dists[rit];
85  value_t core_dist_a = max(core_dist_rit, max(core_dists[a.key], a.value));
86 
87  value_t core_dist_b;
88  if (b.key > -1) {
89  core_dist_b = max(core_dist_rit, max(core_dists[b.key], b.value));
90  } else {
91  core_dist_b = b.value;
92  }
93 
94  return core_dist_a < core_dist_b ? KVP(a.key, core_dist_a) : KVP(b.key, core_dist_b);
95  }
96 
97  return b;
98  }
99 
100  DI void init(value_t* out, value_t maxVal) { *out = maxVal; }
101  DI void init(KVP* out, value_t maxVal)
102  {
103  out->key = -1;
104  out->value = maxVal;
105  }
106 };
