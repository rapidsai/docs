461 inline auto get_forest_variant_index(
462  bool use_double_thresholds,
463  index_type max_node_offset,
464  index_type num_features,
465  index_type num_categorical_nodes = index_type{},
466  index_type max_num_categories = index_type{},
467  index_type num_vector_leaves = index_type{},
468  tree_layout layout = preferred_tree_layout
469 ) {
470  using small_index_t = typename detail::specialization_types<preferred_tree_layout, false, false>::index_type;
471  auto max_local_categories = index_type(sizeof(small_index_t) * 8);
472  // If the index required for pointing to categorical storage bins or vector
473  // leaf output exceeds what we can store in a uint32_t, uint64_t will be used
474  //
475  // TODO(wphicks): We are overestimating categorical storage required here
476  auto double_indexes_required = (
477  max_num_categories > max_local_categories
478  && (
479  (
480  raft_proto::ceildiv(max_num_categories, max_local_categories) + 1
481  * num_categorical_nodes
482  ) > std::numeric_limits<small_index_t>::max()
483  )
484  ) || num_vector_leaves > std::numeric_limits<small_index_t>::max();
485 
486  auto double_precision = use_double_thresholds || double_indexes_required;
487 
488  using small_metadata_t = typename detail::specialization_types<preferred_tree_layout, false, false>::metadata_type;
489  using small_offset_t = typename detail::specialization_types<preferred_tree_layout, false, false>::offset_type;
490 
491  auto large_trees = (
492  num_features > (
493  std::numeric_limits<small_metadata_t>::max() >> reserved_node_metadata_bits
494  ) || max_node_offset > std::numeric_limits<small_offset_t>::max()
495  );
496 
497  auto layout_value = static_cast<std::underlying_type_t<tree_layout>>(layout);
498 
499  return (
500  (index_type{layout_value} << index_type{2})
501  + (index_type{double_precision} << index_type{1})
502  + index_type{large_trees}
503  );
504 }
