41 /*
42  * Perform postprocessing on raw forest output
43  *
44  * @param val Pointer to the raw forest output
45  * @param output_count The number of output values per row
46  * @param out Pointer to the output buffer
47  * @param stride Number of elements between the first element that must be
48  * summed for a particular output element and the next. This is typically
49  * equal to the number of "groves" of trees over which the computation
50  * was divided.
51  * @param average_factor The factor by which to divide during the
52  * normalization step of postprocessing
53  * @param bias The bias factor to subtract off during the
54  * normalization step of postprocessing
55  * @param constant If the postprocessing operation requires a constant,
56  * it can be passed here.
57  */
58  template<
59  row_op row_wise_v,
60  element_op elem_wise_v,
61  typename io_t
62  >
63  HOST DEVICE void postprocess(
64  io_t* val,
65  index_type output_count,
66  io_t* out,
67  index_type stride=index_type{1},
68  io_t average_factor=io_t{1},
69  io_t bias=io_t{0},
70  io_t constant=io_t{1}
71  ) {
72 #pragma GCC diagnostic push
73 #pragma GCC diagnostic ignored "-Wunused-but-set-variable"
74  auto max_index = index_type{};
75  auto max_value = std::numeric_limits<io_t>::lowest();
76 #pragma GCC diagnostic pop
77  for (auto output_index=index_type{}; output_index < output_count; ++output_index) {
78  auto workspace_index = output_index * stride;
79  val[workspace_index] = val[workspace_index] / average_factor + bias;
80  if constexpr (elem_wise_v == element_op::signed_square) {
81  val[workspace_index] = copysign(val[workspace_index] * val[workspace_index], val[workspace_index]);
82  } else if constexpr (elem_wise_v == element_op::hinge) {
83  val[workspace_index] = io_t(val[workspace_index] > io_t{});
84  } else if constexpr (elem_wise_v == element_op::sigmoid) {
85  val[workspace_index] = io_t{1} / (io_t{1} + exp(-constant * val[workspace_index]));
86  } else if constexpr (elem_wise_v == element_op::exponential) {
87  val[workspace_index] = exp(val[workspace_index] / constant);
88  } else if constexpr (elem_wise_v == element_op::logarithm_one_plus_exp) {
89  val[workspace_index] = log1p(exp(val[workspace_index] / constant));
90  }
91  if constexpr (row_wise_v == row_op::softmax || row_wise_v == row_op::max_index) {
92  auto is_new_max = val[workspace_index] > max_value;
93  max_index = is_new_max * output_index + (!is_new_max) * max_index;
94  max_value = is_new_max * val[workspace_index] + (!is_new_max) * max_value;
95  }
96  }
97 
98  if constexpr (row_wise_v == row_op::max_index) {
99  *out = max_index;
100  } else {
101 #pragma GCC diagnostic push
102 #pragma GCC diagnostic ignored "-Wunused-but-set-variable"
103  auto softmax_normalization = io_t{};
104 #pragma GCC diagnostic pop
105  if constexpr (row_wise_v == row_op::softmax) {
106  for (auto workspace_index=index_type{}; workspace_index < output_count * stride; workspace_index += stride) {
107  val[workspace_index] = exp(val[workspace_index] - max_value);
108  softmax_normalization += val[workspace_index];
109  }
110  }
111 
112  for (auto output_index=index_type{}; output_index < output_count; ++output_index) {
113  auto workspace_index = output_index * stride;
114  if constexpr (row_wise_v == row_op::softmax) {
115  out[output_index] = val[workspace_index] / softmax_normalization;
116  } else {
117  out[output_index] = val[workspace_index];
118  }
119  }
120  }
121  }
