60  /* Add a root node, indicating the beginning of a new tree */
61  void start_new_tree() {
62  if (root_node_indexes_.empty()) {
63  root_node_indexes_.emplace_back();
64  } else {
65  max_tree_size_ = std::max(cur_tree_size_, max_tree_size_);
66  if (alignment_ != index_type{}) {
67  if (cur_tree_size_ % alignment_ != index_type{}) {
68  auto padding = (alignment_ - cur_tree_size_ % alignment_);
69  for (auto i = index_type{}; i < padding; ++i) {
70  add_node(typename node_type::threshold_type{});
71  }
72  }
73  }
74  root_node_indexes_.push_back(
75  root_node_indexes_.back() + cur_tree_size_
76  );
77  cur_tree_size_ = index_type{};
78  }

81  /* Add a node with a categorical split */
82  template<typename iter_t>
83  void add_categorical_node(
84  iter_t vec_begin,
85  iter_t vec_end,
86  bool default_to_distant_child=false,
87  typename node_type::metadata_storage_type feature = typename node_type::metadata_storage_type{},
88  typename node_type::offset_type offset = typename node_type::offset_type{}
89  ) {
90  auto constexpr const bin_width = index_type(sizeof(typename node_type::index_type) * 8);
91  auto node_value = typename node_type::index_type{};
92  auto set_storage = &node_value;
93  auto max_node_categories = *std::max_element(vec_begin, vec_end) + 1;
94  if (max_num_categories_ > bin_width) {
95  // TODO(wphicks): Check for overflow here
96  node_value = categorical_storage_.size();
97  auto bins_required = raft_proto::ceildiv(max_node_categories, bin_width);
98  categorical_storage_.push_back(max_node_categories);
99  categorical_storage_.resize(categorical_storage_.size() + bins_required);
100  set_storage = &(categorical_storage_[node_value + 1]);
101  }
102  auto set = bitset{set_storage, max_node_categories};
103  std::for_each(
104  vec_begin,
105  vec_end,
106  [&set](auto&& cat_index) {
107  set.set(cat_index);
108  }
109  );
110 
111  add_node(
112  node_value,
113  false,
114  default_to_distant_child,
115  true,
116  feature,
117  offset,
118  false
119  );
120  }
