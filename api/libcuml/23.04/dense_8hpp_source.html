63  // Implemented GEMM as a static method here to improve readability
64  inline static void gemm(const raft::handle_t& handle,
65  const T alpha,
66  const SimpleDenseMat<T>& A,
67  const bool transA,
68  const SimpleDenseMat<T>& B,
69  const bool transB,
70  const T beta,
71  SimpleDenseMat<T>& C,
72  cudaStream_t stream)
73  {
74  int kA = A.n;
75  int kB = B.m;
76 
77  if (transA) {
78  ASSERT(A.n == C.m, "GEMM invalid dims: m");
79  kA = A.m;
80  } else {
81  ASSERT(A.m == C.m, "GEMM invalid dims: m");
82  }
83 
84  if (transB) {
85  ASSERT(B.m == C.n, "GEMM invalid dims: n");
86  kB = B.n;
87  } else {
88  ASSERT(B.n == C.n, "GEMM invalid dims: n");
89  }
90  ASSERT(kA == kB, "GEMM invalid dims: k");
91 
92  if (A.ord == COL_MAJOR && B.ord == COL_MAJOR && C.ord == COL_MAJOR) {
93  // #TODO: Call from public API when ready
94  raft::linalg::detail::cublasgemm(handle.get_cublas_handle(), // handle
95  transA ? CUBLAS_OP_T : CUBLAS_OP_N, // transA
96  transB ? CUBLAS_OP_T : CUBLAS_OP_N, // transB
97  C.m,
98  C.n,
99  kA, // dimensions m,n,k
100  &alpha,
101  A.data,
102  A.m, // lda
103  B.data,
104  B.m, // ldb
105  &beta,
106  C.data,
107  C.m, // ldc,
108  stream);
109  return;
110  }
111  if (A.ord == ROW_MAJOR) {
112  const SimpleDenseMat<T> Acm(A.data, A.n, A.m, COL_MAJOR);
113  gemm(handle, alpha, Acm, !transA, B, transB, beta, C, stream);
114  return;
115  }
116  if (B.ord == ROW_MAJOR) {
117  const SimpleDenseMat<T> Bcm(B.data, B.n, B.m, COL_MAJOR);
118  gemm(handle, alpha, A, transA, Bcm, !transB, beta, C, stream);
119  return;
120  }
121  if (C.ord == ROW_MAJOR) {
122  SimpleDenseMat<T> Ccm(C.data, C.n, C.m, COL_MAJOR);
123  gemm(handle, alpha, B, !transB, A, !transA, beta, Ccm, stream);
124  return;
125  }
126  }

287 {
288  auto f = [] __device__(const T x, const T y) { return x * y; };
289  raft::linalg::mapThenSumReduce(tmp_dev, u.len, f, stream, u.data, v.data);
290  T tmp_host;
291  raft::update_host(&tmp_host, tmp_dev, 1, stream);
292 
293  raft::interruptible::synchronize(stream);
294  return tmp_host;
