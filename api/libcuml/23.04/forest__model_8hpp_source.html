172  template <typename io_t>
173  void predict(
174  raft_proto::handle_t const& handle,
175  raft_proto::buffer<io_t>& output,
176  raft_proto::buffer<io_t> const& input,
177  infer_kind predict_type=infer_kind::default_kind,
178  std::optional<index_type> specified_chunk_size=std::nullopt
179  ) {
180  std::visit([this, predict_type, &handle, &output, &input, &specified_chunk_size](auto&& concrete_forest) {
181  using model_io_t = typename std::remove_reference_t<decltype(concrete_forest)>::io_type;
182  if constexpr(std::is_same_v<model_io_t, io_t>) {
183  if (output.memory_type() == memory_type() && input.memory_type() == memory_type()) {
184  concrete_forest.predict(
185  output,
186  input,
187  handle.get_next_usable_stream(),
188  predict_type,
189  specified_chunk_size
190  );
191  } else {
192  auto constexpr static const MIN_CHUNKS_PER_PARTITION = std::size_t{64};
193  auto constexpr static const MAX_CHUNK_SIZE = std::size_t{64};
194 
195  auto row_count = input.size() / num_features();
196  auto partition_size = std::max(
197  raft_proto::ceildiv(row_count, handle.get_usable_stream_count()),
198  specified_chunk_size.value_or(MAX_CHUNK_SIZE) * MIN_CHUNKS_PER_PARTITION
199  );
200  auto partition_count = raft_proto::ceildiv(row_count, partition_size);
201  for (auto i = std::size_t{}; i < partition_count; ++i) {
202  auto stream = handle.get_next_usable_stream();
203  auto rows_in_this_partition = std::min(partition_size, row_count - i * partition_size);
204  auto partition_in = raft_proto::buffer<io_t>{};
205  if (input.memory_type() != memory_type()) {
206  partition_in = raft_proto::buffer<io_t>{
207  rows_in_this_partition * num_features(),
208  memory_type()
209  };
210  raft_proto::copy<raft_proto::DEBUG_ENABLED>(
211  partition_in,
212  input,
213  0,
214  i * partition_size * num_features(),
215  partition_in.size(),
216  stream
217  );
218  } else {
219  partition_in = raft_proto::buffer<io_t>{
220  input.data() + i * partition_size * num_features(),
221  rows_in_this_partition * num_features(),
222  memory_type()
223  };
224  }
225  auto partition_out = raft_proto::buffer<io_t>{};
226  if (output.memory_type() != memory_type()) {
227  partition_out = raft_proto::buffer<io_t>{
228  rows_in_this_partition * num_outputs(),
229  memory_type()
230  };
231  } else {
232  partition_out = raft_proto::buffer<io_t>{
233  output.data() + i * partition_size * num_outputs(),
234  rows_in_this_partition * num_outputs(),
235  memory_type()
236  };
237  }
238  concrete_forest.predict(
239  partition_out,
240  partition_in,
241  stream,
242  predict_type,
243  specified_chunk_size
244  );
245  if (output.memory_type() != memory_type()) {
246  raft_proto::copy<raft_proto::DEBUG_ENABLED>(
247  output,
248  partition_out,
249  i * partition_size * num_outputs(),
250  0,
251  partition_out.size(),
252  stream
253  );
254  }
255  }
256  }
257  } else {
258  throw type_error("Input type does not match model_type");
259  }
260  }, decision_forest_);
261  }
