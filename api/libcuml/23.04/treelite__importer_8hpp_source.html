40 template <tree_layout layout, typename T>
41 struct traversal_container {
42  using backing_container_t = std::conditional_t<
43  layout == tree_layout::depth_first,
44  std::stack<T>,
45  std::queue<T>
46  >;
47  void add(T const& val) {
48  data_.push(val);
49  }
50  void add(T const& hot, T const& distant) {
51  if constexpr (layout == tree_layout::depth_first) {
52  data_.push(distant);
53  data_.push(hot);
54  } else {
55  data_.push(hot);
56  data_.push(distant);
57  }
58  }
59  auto next() {
60  if constexpr (std::is_same_v<backing_container_t, std::stack<T>>) {
61  auto result = data_.top();
62  data_.pop();
63  return result;
64  } else {
65  auto result = data_.front();
66  data_.pop();
67  return result;
68  }
69  }
70  auto peek() {
71  if constexpr (std::is_same_v<backing_container_t, std::stack<T>>) {
72  return data_.top();
73  } else {
74  return data_.front();
75  }
76  }
77  [[nodiscard]] auto empty() {
78  return data_.empty();
79  }
80  auto size() {
81  return data_.size();
82  }
83  private:
84  backing_container_t data_;
85 };

134  auto default_distant() {
135  auto result = false;
136  auto default_child = tree.DefaultChild(node_id);
137  if (is_categorical()) {
138  if (tree.CategoriesListRightChild(node_id)) {
139  result = (default_child == tree.RightChild(node_id));
140  } else {
141  result = (default_child == tree.LeftChild(node_id));
142  }
143  } else {
144  auto tl_operator = tree.ComparisonOp(node_id);
145  if (tl_operator == treelite::Operator::kLT || tl_operator == treelite::Operator::kLE) {
146  result = (default_child == tree.LeftChild(node_id));
147  } else {
148  result = (default_child == tree.RightChild(node_id));
149  }
150  }
151  return result;
152  }

156  }
157 
158  auto categories() {

160  if (is_categorical()) {

162  }

172  template<typename tl_threshold_t, typename tl_output_t, typename lambda_t>
173  void node_for_each(treelite::Tree<tl_threshold_t, tl_output_t> const& tl_tree, lambda_t&& lambda) {
174  using node_index_t = decltype(tl_tree.LeftChild(0));
175  auto to_be_visited = detail::traversal_container<layout, node_index_t>{};
176  to_be_visited.add(node_index_t{});
177 
178  auto parent_indices = detail::traversal_container<layout, index_type>{};
179  auto cur_index = index_type{};
180  parent_indices.add(cur_index);
181 
182  while (!to_be_visited.empty()) {
183  auto node_id = to_be_visited.next();
184  auto remaining_size = to_be_visited.size();
185 
186  auto tl_node = treelite_node<tl_threshold_t, tl_output_t>{
187  tl_tree, node_id, parent_indices.next(), cur_index
188  };
189  lambda(tl_node);
190 
191  if (!tl_tree.IsLeaf(node_id)) {
192  auto tl_left_id = tl_tree.LeftChild(node_id);
193  auto tl_right_id = tl_tree.RightChild(node_id);
194  auto tl_operator = tl_tree.ComparisonOp(node_id);
195  if (!tl_node.is_categorical()) {
196  if (tl_operator == treelite::Operator::kLT || tl_operator == treelite::Operator::kLE) {
197  to_be_visited.add(tl_right_id, tl_left_id);
198  } else if (tl_operator == treelite::Operator::kGT || tl_operator == treelite::Operator::kGE) {
199  to_be_visited.add(tl_left_id, tl_right_id);
200  } else {
201  throw model_import_error("Unrecognized Treelite operator");
202  }
203  } else {
204  if (tl_tree.CategoriesListRightChild(node_id)) {
205  to_be_visited.add(tl_left_id, tl_right_id);
206  } else {
207  to_be_visited.add(tl_right_id, tl_left_id);
208  }
209  }
210  parent_indices.add(cur_index, cur_index);
211  }
212  ++cur_index;
213  }
214  }

247  template<typename tl_threshold_t, typename tl_output_t>
248  auto get_offsets(treelite::Tree<tl_threshold_t, tl_output_t> const& tl_tree) {
249  auto result = std::vector<index_type>(tl_tree.num_nodes);
250  auto nodes = get_nodes(tl_tree);
251  for (auto i = index_type{}; i < nodes.size(); ++i) {
252  // Current index should always be greater than or equal to parent index.
253  // Later children will overwrite values set by earlier children, ensuring
254  // that most distant offset is used.
255  result[nodes[i].parent_index] = index_type{i - nodes[i].parent_index};
256  }
257 
258  return result;
259  }

391  auto get_postproc_params(treelite::Model const& tl_model) {
392  auto result = detail::postproc_params_t{};
393  tl_model.Dispatch([&result](auto&& concrete_tl_model) {
394  auto tl_pred_transform = std::string{concrete_tl_model.param.pred_transform};
395  if (
396  tl_pred_transform == std::string{"identity"} ||
397  tl_pred_transform == std::string{"identity_multiclass"}) {
398  result.element = element_op::disable;
399  result.row = row_op::disable;
400  } else if (tl_pred_transform == std::string{"signed_square"}) {
401  result.element = element_op::signed_square;
402  } else if (tl_pred_transform == std::string{"hinge"}) {
403  result.element = element_op::hinge;
404  } else if (tl_pred_transform == std::string{"sigmoid"}) {
405  result.constant = concrete_tl_model.param.sigmoid_alpha;
406  result.element = element_op::sigmoid;
407  } else if (tl_pred_transform == std::string{"exponential"}) {
408  result.element = element_op::exponential;
409  } else if (tl_pred_transform == std::string{"exponential_standard_ratio"}) {
410  result.constant = -concrete_tl_model.param.ratio_c / std::log(2);
411  result.element = element_op::exponential;
412  } else if (tl_pred_transform == std::string{"logarithm_one_plus_exp"}) {
413  result.element = element_op::logarithm_one_plus_exp;
414  } else if (tl_pred_transform == std::string{"max_index"}) {
415  result.row = row_op::max_index;
416  } else if (tl_pred_transform == std::string{"softmax"}) {
417  result.row = row_op::softmax;
418  } else if (tl_pred_transform == std::string{"multiclass_ova"}) {
419  result.constant = concrete_tl_model.param.sigmoid_alpha;
420  result.element = element_op::sigmoid;
421  } else {
422  throw model_import_error{"Unrecognized Treelite pred_transform string"};
423  }
424  });
425  return result;

483  template<index_type variant_index>
484  auto import_to_specific_variant(
485  index_type target_variant_index,
486  treelite::Model const& tl_model,
487  index_type num_class,
488  index_type num_feature,
489  index_type max_num_categories,
490  std::vector<std::vector<index_type>> const& offsets,
491  index_type align_bytes = index_type{},
492  raft_proto::device_type mem_type=raft_proto::device_type::cpu,
493  int device=0,
494  raft_proto::cuda_stream stream=raft_proto::cuda_stream{}
495  ) {
496  auto result = decision_forest_variant{};
497  if constexpr (variant_index != std::variant_size_v<decision_forest_variant>) {
498  if (variant_index == target_variant_index) {
499  using forest_model_t = std::variant_alternative_t<variant_index, decision_forest_variant>;
500  auto builder = detail::decision_forest_builder<forest_model_t>(max_num_categories, align_bytes);
501  auto tree_count = num_trees(tl_model);
502  auto tree_index = index_type{};
503  tree_for_each(tl_model, [this, &builder, &tree_index, &offsets](auto&& tree) {
504  builder.start_new_tree();
505  auto node_index = index_type{};
506  node_for_each(tree, [&builder, &tree_index, &node_index, &offsets](auto&& node) {
507  if (node.is_leaf()) {
508  auto output = node.get_output();
509  builder.set_output_size(output.size());
510  if (output.size() > index_type{1}) {
511  builder.add_leaf_vector_node(
512  std::begin(output),
513  std::end(output)
514  );
515  } else {
516  builder.add_node(
517  typename forest_model_t::io_type(output[0]),
518  true
519  );
520  }
521  } else {
522  if (node.is_categorical()) {
523  auto categories = node.get_categories();
524  builder.add_categorical_node(
525  std::begin(categories),
526  std::end(categories),
527  node.default_distant(),
528  node.get_feature(),
529  offsets[tree_index][node_index]
530  );
531  } else {
532  builder.add_node(
533  typename forest_model_t::threshold_type(node.threshold()),
534  false,
535  node.default_distant(),
536  false,
537  node.get_feature(),
538  offsets[tree_index][node_index],
539  node.is_inclusive()
540  );
541  }
542  }
543  ++node_index;
544  });
545  ++tree_index;
546  });
547 
548  builder.set_average_factor(get_average_factor(tl_model));
549  builder.set_bias(get_bias(tl_model));
550  auto postproc_params = get_postproc_params(tl_model);
551  builder.set_element_postproc(postproc_params.element);
552  builder.set_row_postproc(postproc_params.row);
553  builder.set_postproc_constant(postproc_params.constant);
554 
555  result.template emplace<variant_index>(builder.get_decision_forest(num_feature, num_class, mem_type, device, stream));
556  } else {
557  result = import_to_specific_variant<variant_index + 1>(
558  target_variant_index,
559  tl_model,
560  num_class,
561  num_feature,
562  max_num_categories,
563  offsets,
564  align_bytes,
565  mem_type,
566  device,
567  stream
568  );
569  }
570  }
571  return result;
572  }

596  auto import(

598  index_type align_bytes = index_type{},

600  raft_proto::device_type dev_type=raft_proto::device_type::cpu,
601  int device=0,
602  raft_proto::cuda_stream stream=raft_proto::cuda_stream{}
603  ) {

609  auto use_double_thresholds = use_double_precision.value_or(uses_double_thresholds(tl_model));

623  std::end(offsets),
624  std::back_inserter(tree_sizes),

628  );

638  );
