29 /*
30  * Return the byte size to which a node with the given types should be aligned
31  */
32 template<typename threshold_t, typename index_t, typename metadata_storage_t, typename offset_t>
33 auto constexpr get_node_alignment() {
34  auto total = index_type(
35  std::max(sizeof(threshold_t), sizeof(index_t))
36  + sizeof(metadata_storage_t)
37  + sizeof(offset_t)
38  );
39  auto result = index_type{8};
40  if (total > result) {
41  result = index_type{16};
42  }
43  if (total > result) {
44  result = index_type{32};
45  }
46  if (total > result) {
47  result = index_type{64};
48  }
49  if (total > result) {
50  result = index_type{128};
51  }
52  if (total > result) {
53  result = index_type{256};
54  }
55  if (total > result) {
56  result = total;
57  }
58  return result;
59 }

63 /* @brief A single node in a forest model
64  *
65  * Note that this implementation includes NO error checking for poorly-chosen
66  * template types. If the types are not large enough to hold the required data,
67  * an incorrect node will be silently constructed. Error checking occurs
68  * instead at the time of construction of the entire forest.
69  *
70  * @tparam layout_v The layout for nodes within the forest
71  *
72  * @tparam threshold_t The type used as a threshold for evaluating a non-leaf
73  * node or (when possible) the output of a leaf node. For non-categorical
74  * nodes, if an input value is less than this threshold, the node evaluates to
75  * true. For leaf nodes, output values will only be stored as this type if it
76  * matches the leaf output type expected by the forest. Typically, this type is
77  * either float or double.
78  *
79  * @tparam index_t The type used as an index to the output data for leaf nodes,
80  * or to the categorical set for a categorical non-leaf node. This type should
81  * be large enough to index the entire array of output data or categorical sets
82  * stored in the forest. Typically, this type is either uint32_t or uint64_t.
83  * Smaller types offer no benefit, since this value is stored in a union with
84  * threshold_t, which is at least 32 bits.
85  *
86  * @tparam metadata_storage_t An unsigned integral type used for a bit-wise
87  * representation of metadata about this node. The first three bits encode
88  * whether or not this is a leaf node, whether or not we should default to the
89  * more distant child in case of missing values, and whether or not this node
90  * is categorical. The remaining bits are used to encode the feature index for
91  * this node. Thus, uint8_t may be used for 2**(8 - 3) = 32 or fewer features,
92  * uint16_t for 2**(16 - 3) = 8192 or fewer, and uint32_t for 536870912 or
93  * fewer features.
94  *
95  * @tparam offset_t An integral type used to indicate the offset from
96  * this node to its most distant child. This type must be large enough to store
97  * the largest such offset in the forest model.
98  */
99 template <tree_layout layout_v, typename threshold_t, typename index_t, typename metadata_storage_t, typename offset_t>
100 struct alignas(
102  detail::get_node_alignment<threshold_t, index_t, metadata_storage_t, offset_t>()
104 ) node {
105  // @brief An alias for layout_v
106  auto constexpr static const layout = layout_v;
107  // @brief An alias for threshold_t
108  using threshold_type = threshold_t;
109  // @brief An alias for index_t
110  using index_type = index_t;
111  /* @brief A union to hold either a threshold value or index
112  *
113  * All nodes will need EITHER a threshold value, an output value, OR an index
114  * to data elsewhere that wil be used either for evaluating the node (e.g. an
115  * index to a categorical set) or creating an output (e.g. an index to vector
116  * leaf output). This union allows us to store either of these values without
117  * using additional space for the unused value.
118  */
119  union value_type {
120  threshold_t value;
121  index_t index;
122  };
124  using metadata_storage_type = metadata_storage_t;
126  using offset_type = offset_t;
127

165  /* The index of the feature for this node */
166  HOST DEVICE auto constexpr feature_index() const {
167  return aligned_data.inner_data.metadata & FEATURE_MASK;
168  }
169  /* Whether or not this node is a leaf node */
170  HOST DEVICE auto constexpr is_leaf() const {
171  return !bool(aligned_data.inner_data.distant_offset);
172  }
173  /* Whether or not to default to distant child in case of missing values */
174  HOST DEVICE auto constexpr default_distant() const {
175  return bool(aligned_data.inner_data.metadata & DEFAULT_DISTANT_MASK);
176  }
177  /* Whether or not this node is a categorical node */
178  HOST DEVICE auto constexpr is_categorical() const {
179  return bool(aligned_data.inner_data.metadata & CATEGORICAL_MASK);
180  }
181  /* The offset to the child of this node if it evaluates to given condition */
182  HOST DEVICE auto constexpr child_offset(bool condition) const {
183  if constexpr (layout == tree_layout::depth_first) {
184  return offset_type{1} + condition * (aligned_data.inner_data.distant_offset - offset_type{1});
185  } else if constexpr (layout == tree_layout::breadth_first) {
186  return condition * offset_type{1} + (aligned_data.inner_data.distant_offset - offset_type{1});
187  } else {

189  }

213  private:
214  /* Define all bit masks required to extract information from the stored
215  * metadata. The first bit tells us whether or not this is a leaf node, the
216  * second tells us whether or not we should default to the distant child in
217  * the case of a missing value, and the third tells us whether or not this is
218  * a categorical node. The remaining bits indicate the index of the feature
219  * for this node */
220  auto constexpr static const LEAF_BIT = metadata_storage_type(
221  index_type(sizeof(metadata_storage_type) * 8 - 1)
222  );
223  auto constexpr static const LEAF_MASK = metadata_storage_type(1 << LEAF_BIT);
224  auto constexpr static const DEFAULT_DISTANT_BIT = metadata_storage_type(LEAF_BIT - 1);
225  auto constexpr static const DEFAULT_DISTANT_MASK = metadata_storage_type(1 << DEFAULT_DISTANT_BIT);
226  auto constexpr static const CATEGORICAL_BIT = metadata_storage_type(DEFAULT_DISTANT_BIT - 1);
227  auto constexpr static const CATEGORICAL_MASK = metadata_storage_type(1 << CATEGORICAL_BIT);
228  auto constexpr static const FEATURE_MASK = metadata_storage_type(
229  ~(LEAF_MASK | DEFAULT_DISTANT_MASK | CATEGORICAL_MASK)
230  );

241  (default_to_distant_child << DEFAULT_DISTANT_BIT) +
242  (is_categorical_node << CATEGORICAL_BIT) +
243  (feature & FEATURE_MASK)
244  );
245  }
