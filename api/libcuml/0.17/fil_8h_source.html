204 struct forest_params_t {

213  // leaf_algo determines what the leaves store (predict)
214  leaf_algo_t leaf_algo;
215  // algo is the inference algorithm;
216  // sparse forests do not distinguish between NAIVE and TREE_REORG
217  algo_t algo;
218  // output is the desired output type
219  output_t output;
220  // threshold is used to for classification if leaf_algo == FLOAT_UNARY_BINARY && (output & OUTPUT_CLASS) != 0 && !predict_proba,
221  // and is ignored otherwise
222  float threshold;
223  // global_bias is added to the sum of tree predictions
224  // (after averaging, if it is used, but before any further transformations)
225  float global_bias;
226  // only used for CATEGORICAL_LEAF inference. since we're storing the
227  // labels in leaves instead of the whole vector, this keeps track
228  // of the number of classes
229  int num_classes;
230  // blocks_per_sm, if nonzero, works as a limit to improve cache hit rate for larger forests
231  // suggested values (if nonzero) are from 2 to 7
232  // if zero, launches ceildiv(num_rows, NITEMS) blocks
233  int blocks_per_sm;
234 };
