45 template <typename value_idx, typename value_t>
46 struct FixConnectivitiesRedOp {
47  value_idx* colors;
48  value_t* core_dists;
49  value_idx m;
50 
51  FixConnectivitiesRedOp(value_idx* colors_, value_t* core_dists_, value_idx m_)
52  : colors(colors_), core_dists(core_dists_), m(m_){};
53 
54  typedef typename cub::KeyValuePair<value_idx, value_t> KVP;
55  DI void operator()(value_idx rit, KVP* out, const KVP& other)
56  {
57  if (rit < m && other.value < std::numeric_limits<value_t>::max() &&
58  colors[rit] != colors[other.key]) {
59  value_t core_dist_rit = core_dists[rit];
60  value_t core_dist_other = max(core_dist_rit, max(core_dists[other.key], other.value));
61 
62  value_t core_dist_out;
63  if (out->key > -1) {
64  core_dist_out = max(core_dist_rit, max(core_dists[out->key], out->value));
65  } else {
66  core_dist_out = out->value;
67  }
68 
69  bool smaller = core_dist_other < core_dist_out;
70  out->key = smaller ? other.key : out->key;
71  out->value = smaller ? core_dist_other : core_dist_out;
72  }
73  }
74 
75  DI KVP operator()(value_idx rit, const KVP& a, const KVP& b)
76  {
77  if (rit < m && a.key > -1 && colors[rit] != colors[a.key]) {
78  value_t core_dist_rit = core_dists[rit];
79  value_t core_dist_a = max(core_dist_rit, max(core_dists[a.key], a.value));
80 
81  value_t core_dist_b;
82  if (b.key > -1) {
83  core_dist_b = max(core_dist_rit, max(core_dists[b.key], b.value));
84  } else {
85  core_dist_b = b.value;
86  }
87 
88  return core_dist_a < core_dist_b ? KVP(a.key, core_dist_a) : KVP(b.key, core_dist_b);
89  }
90 
91  return b;
92  }
93 
94  DI void init(value_t* out, value_t maxVal) { *out = maxVal; }
95  DI void init(KVP* out, value_t maxVal)
96  {
97  out->key = -1;
98  out->value = maxVal;
99  }
100 };
