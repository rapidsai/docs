33 struct ARIMAOrder {
34  int p; // Basic order
35  int d;
36  int q;
37  int P; // Seasonal order
38  int D;
39  int Q;
40  int s; // Seasonal period
41  int k; // Fit intercept?
42  int n_exog; // Number of exogenous regressors
43 
44  inline int n_diff() const { return d + s * D; }
45  inline int n_phi() const { return p + s * P; }
46  inline int n_theta() const { return q + s * Q; }
47  inline int r() const { return std::max(n_phi(), n_theta() + 1); }
48  inline int rd() const { return n_diff() + r(); }
49  inline int complexity() const { return p + P + q + Q + k + n_exog + 1; }
50  inline bool need_diff() const { return static_cast<bool>(d + D); }
51 };

122  void pack(const ARIMAOrder& order, int batch_size, DataT* param_vec, cudaStream_t stream) const
123  {
124  int N = order.complexity();
125  auto counting = thrust::make_counting_iterator(0);
126  // The device lambda can't capture structure members...
127  const DataT *_mu = mu, *_beta = beta, *_ar = ar, *_ma = ma, *_sar = sar, *_sma = sma,
128  *_sigma2 = sigma2;
129  thrust::for_each(
130  thrust::cuda::par.on(stream), counting, counting + batch_size, [=] __device__(int bid) {
131  DataT* param = param_vec + bid * N;
132  if (order.k) {
133  *param = _mu[bid];
134  param++;
135  }
136  for (int i = 0; i < order.n_exog; i++) {
137  param[i] = _beta[order.n_exog * bid + i];
138  }
139  param += order.n_exog;
140  for (int ip = 0; ip < order.p; ip++) {
141  param[ip] = _ar[order.p * bid + ip];
142  }
143  param += order.p;
144  for (int iq = 0; iq < order.q; iq++) {
145  param[iq] = _ma[order.q * bid + iq];
146  }
147  param += order.q;
148  for (int iP = 0; iP < order.P; iP++) {
149  param[iP] = _sar[order.P * bid + iP];
150  }
151  param += order.P;
152  for (int iQ = 0; iQ < order.Q; iQ++) {
153  param[iQ] = _sma[order.Q * bid + iQ];
154  }
155  param += order.Q;
156  *param = _sigma2[bid];
157  });
158  }

169  void unpack(const ARIMAOrder& order, int batch_size, const DataT* param_vec, cudaStream_t stream)
170  {
171  int N = order.complexity();
172  auto counting = thrust::make_counting_iterator(0);
173  // The device lambda can't capture structure members...
174  DataT *_mu = mu, *_beta = beta, *_ar = ar, *_ma = ma, *_sar = sar, *_sma = sma,
175  *_sigma2 = sigma2;
176  thrust::for_each(
177  thrust::cuda::par.on(stream), counting, counting + batch_size, [=] __device__(int bid) {
178  const DataT* param = param_vec + bid * N;
179  if (order.k) {
180  _mu[bid] = *param;
181  param++;
182  }
183  for (int i = 0; i < order.n_exog; i++) {
184  _beta[order.n_exog * bid + i] = param[i];
185  }
186  param += order.n_exog;
187  for (int ip = 0; ip < order.p; ip++) {
188  _ar[order.p * bid + ip] = param[ip];
189  }
190  param += order.p;
191  for (int iq = 0; iq < order.q; iq++) {
192  _ma[order.q * bid + iq] = param[iq];
193  }
194  param += order.q;
195  for (int iP = 0; iP < order.P; iP++) {
196  _sar[order.P * bid + iP] = param[iP];
197  }
198  param += order.P;
199  for (int iQ = 0; iQ < order.Q; iQ++) {
200  _sma[order.Q * bid + iQ] = param[iQ];
201  }
202  param += order.Q;
