63  // Implemented GEMM as a static method here to improve readability
64  inline static void gemm(const raft::handle_t& handle,
65  const T alpha,
66  const SimpleDenseMat<T>& A,
67  const bool transA,
68  const SimpleDenseMat<T>& B,
69  const bool transB,
70  const T beta,
71  SimpleDenseMat<T>& C,
72  cudaStream_t stream)
73  {
74  int kA = A.n;
75  int kB = B.m;
76 
77  if (transA) {
78  ASSERT(A.n == C.m, "GEMM invalid dims: m");
79  kA = A.m;
80  } else {
81  ASSERT(A.m == C.m, "GEMM invalid dims: m");
82  }
83 
84  if (transB) {
85  ASSERT(B.m == C.n, "GEMM invalid dims: n");
86  kB = B.n;
87  } else {
88  ASSERT(B.n == C.n, "GEMM invalid dims: n");
89  }
90  ASSERT(kA == kB, "GEMM invalid dims: k");
91 
92  if (A.ord == COL_MAJOR && B.ord == COL_MAJOR && C.ord == COL_MAJOR) {
93  raft::linalg::cublasgemm(handle.get_cublas_handle(), // handle
94  transA ? CUBLAS_OP_T : CUBLAS_OP_N, // transA
95  transB ? CUBLAS_OP_T : CUBLAS_OP_N, // transB
96  C.m,
97  C.n,
98  kA, // dimensions m,n,k
99  &alpha,
100  A.data,
101  A.m, // lda
102  B.data,
103  B.m, // ldb
104  &beta,
105  C.data,
106  C.m, // ldc,
107  stream);
108  return;
109  }
110  if (A.ord == ROW_MAJOR) {
111  const SimpleDenseMat<T> Acm(A.data, A.n, A.m, COL_MAJOR);
112  gemm(handle, alpha, Acm, !transA, B, transB, beta, C, stream);
113  return;
114  }
115  if (B.ord == ROW_MAJOR) {
116  const SimpleDenseMat<T> Bcm(B.data, B.n, B.m, COL_MAJOR);
117  gemm(handle, alpha, A, transA, Bcm, !transB, beta, C, stream);
118  return;
119  }
120  if (C.ord == ROW_MAJOR) {
121  SimpleDenseMat<T> Ccm(C.data, C.n, C.m, COL_MAJOR);
122  gemm(handle, alpha, B, !transB, A, !transA, beta, Ccm, stream);
123  return;
124  }
125  }

286 {
287  auto f = [] __device__(const T x, const T y) { return x * y; };
288  raft::linalg::mapThenSumReduce(tmp_dev, u.len, f, stream, u.data, v.data);
289  T tmp_host;
290  raft::update_host(&tmp_host, tmp_dev, 1, stream);
291  cudaStreamSynchronize(stream);
292  return tmp_host;
293 }
