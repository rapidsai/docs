66  inline void gemmb(const raft::handle_t &handle, const T alpha,
67  const SimpleDenseMat<T> &A, const bool transA,
68  const bool transB, const T beta, SimpleDenseMat<T> &C,
69  cudaStream_t stream) const override {
70  const SimpleSparseMat<T> &B = *this;
71  int kA = A.n;
72  int kB = B.m;
73 
74  if (transA) {
75  ASSERT(A.n == C.m, "GEMM invalid dims: m");
76  kA = A.m;
77  } else {
78  ASSERT(A.m == C.m, "GEMM invalid dims: m");
79  }
80 
81  if (transB) {
82  ASSERT(B.m == C.n, "GEMM invalid dims: n");
83  kB = B.n;
84  } else {
85  ASSERT(B.n == C.n, "GEMM invalid dims: n");
86  }
87  ASSERT(kA == kB, "GEMM invalid dims: k");
88 
89  // matrix C must change the order and be transposed, because we need
90  // to swap arguments A and B in cusparseSpMM.
91  cusparseDnMatDescr_t descrC;
92  auto order = C.ord == COL_MAJOR ? CUSPARSE_ORDER_ROW : CUSPARSE_ORDER_COL;
93  CUSPARSE_CHECK(raft::sparse::cusparsecreatednmat(
94  &descrC, C.n, C.m, order == CUSPARSE_ORDER_COL ? C.n : C.m, C.data,
95  order));
96 
97  /*
98  The matrix A must have the same order as the matrix C in the input
99  of function cusparseSpMM (i.e. swapped order w.r.t. original C).
100  To account this requirement, I may need to flip transA (whether to transpose A).
101 
102  C C' rowsC' colsC' ldC' A A' rowsA' colsA' ldA' flipTransA
103  c r n m m c r n m m x
104  c r n m m r r m n n o
105  r c n m n c c m n m o
106  r c n m n r c n m n x
107 
108  where:
109  c/r - column/row major order
110  A,C - input to gemmb
111  A', C' - input to cusparseSpMM
112  ldX' - leading dimension - m or n, depending on order and transX
113  */
114  cusparseDnMatDescr_t descrA;
115  CUSPARSE_CHECK(raft::sparse::cusparsecreatednmat(
116  &descrA, C.ord == A.ord ? A.n : A.m, C.ord == A.ord ? A.m : A.n,
117  A.ord == COL_MAJOR ? A.m : A.n, A.data, order));
118  auto opA = transA ^ (C.ord == A.ord) ? CUSPARSE_OPERATION_NON_TRANSPOSE
119  : CUSPARSE_OPERATION_TRANSPOSE;
120 
121  cusparseSpMatDescr_t descrB;
122  CUSPARSE_CHECK(raft::sparse::cusparsecreatecsr(
123  &descrB, B.m, B.n, B.nnz, B.row_ids, B.cols, B.values));
124  auto opB =
125  transB ? CUSPARSE_OPERATION_NON_TRANSPOSE : CUSPARSE_OPERATION_TRANSPOSE;
126 
127  auto alg = order == CUSPARSE_ORDER_COL ? CUSPARSE_SPMM_CSR_ALG1
128  : CUSPARSE_SPMM_CSR_ALG2;
129 
130  size_t bufferSize;
131  CUSPARSE_CHECK(raft::sparse::cusparsespmm_bufferSize(
132  handle.get_cusparse_handle(), opB, opA, &alpha, descrB, descrA, &beta,
133  descrC, alg, &bufferSize, stream));
134 
135  CUDA_CHECK(cudaStreamSynchronize(stream));
136  rmm::device_uvector<T> tmp(bufferSize, stream);
137 
138  CUSPARSE_CHECK(raft::sparse::cusparsespmm(
139  handle.get_cusparse_handle(), opB, opA, &alpha, descrB, descrA, &beta,
140  descrC, alg, tmp.data(), stream));
141 
142  CUSPARSE_CHECK(cusparseDestroyDnMat(descrA));
143  CUSPARSE_CHECK(cusparseDestroySpMat(descrB));
144  CUSPARSE_CHECK(cusparseDestroyDnMat(descrC));
145  }

150  int row_ids_nnz;
151  raft::update_host(&row_ids_nnz, &mat.row_ids[mat.m], 1, stream);
