100 template <typename value_idx, typename value_t>
101 Common::CondensedHierarchy<value_idx, value_t> make_cluster_tree(
102  const raft::handle_t& handle, Common::CondensedHierarchy<value_idx, value_t>& condensed_tree)
103 {
104  auto stream = handle.get_stream();
105  auto thrust_policy = handle.get_thrust_policy();
106  auto parents = condensed_tree.get_parents();
107  auto children = condensed_tree.get_children();
108  auto lambdas = condensed_tree.get_lambdas();
109  auto sizes = condensed_tree.get_sizes();
110 
111  value_idx cluster_tree_edges = thrust::transform_reduce(
112  thrust_policy,
113  sizes,
114  sizes + condensed_tree.get_n_edges(),
115  [=] __device__(value_idx a) { return a > 1; },
116  0,
117  thrust::plus<value_idx>());
118 
119  // remove leaves from condensed tree
120  rmm::device_uvector<value_idx> cluster_parents(cluster_tree_edges, stream);
121  rmm::device_uvector<value_idx> cluster_children(cluster_tree_edges, stream);
122  rmm::device_uvector<value_t> cluster_lambdas(cluster_tree_edges, stream);
123  rmm::device_uvector<value_idx> cluster_sizes(cluster_tree_edges, stream);
124 
125  auto in = thrust::make_zip_iterator(thrust::make_tuple(parents, children, lambdas, sizes));
126 
127  auto out = thrust::make_zip_iterator(thrust::make_tuple(
128  cluster_parents.data(), cluster_children.data(), cluster_lambdas.data(), cluster_sizes.data()));
129 
130  thrust::copy_if(thrust_policy,
131  in,
132  in + (condensed_tree.get_n_edges()),
133  sizes,
134  out,
135  [=] __device__(value_idx a) { return a > 1; });
136 
137  auto n_leaves = condensed_tree.get_n_leaves();
138  thrust::transform(thrust_policy,
139  cluster_parents.begin(),
140  cluster_parents.end(),
141  cluster_parents.begin(),
142  [n_leaves] __device__(value_idx a) { return a - n_leaves; });
143  thrust::transform(thrust_policy,
144  cluster_children.begin(),
145  cluster_children.end(),
146  cluster_children.begin(),
147  [n_leaves] __device__(value_idx a) { return a - n_leaves; });
148 
149  return Common::CondensedHierarchy<value_idx, value_t>(handle,
150  condensed_tree.get_n_leaves(),
151  cluster_tree_edges,
152  condensed_tree.get_n_clusters(),
153  std::move(cluster_parents),
154  std::move(cluster_children),
155  std::move(cluster_lambdas),
156  std::move(cluster_sizes));
157 }
