131  void pack(const ARIMAOrder& order, int batch_size, DataT* param_vec,
132  cudaStream_t stream) const {
133  int N = order.complexity();
134  auto counting = thrust::make_counting_iterator(0);
135  // The device lambda can't capture structure members...
136  const DataT *_mu = mu, *_ar = ar, *_ma = ma, *_sar = sar, *_sma = sma,
137  *_sigma2 = sigma2;
138  thrust::for_each(thrust::cuda::par.on(stream), counting,
139  counting + batch_size, [=] __device__(int bid) {
140  DataT* param = param_vec + bid * N;
141  if (order.k) {
142  *param = _mu[bid];
143  param++;
144  }
145  for (int ip = 0; ip < order.p; ip++) {
146  param[ip] = _ar[order.p * bid + ip];
147  }
148  param += order.p;
149  for (int iq = 0; iq < order.q; iq++) {
150  param[iq] = _ma[order.q * bid + iq];
151  }
