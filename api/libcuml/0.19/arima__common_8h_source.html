132  void pack(const ARIMAOrder& order, int batch_size, DataT* param_vec,
133  cudaStream_t stream) const {
134  int N = order.complexity();
135  auto counting = thrust::make_counting_iterator(0);
136  // The device lambda can't capture structure members...
137  const DataT *_mu = mu, *_ar = ar, *_ma = ma, *_sar = sar, *_sma = sma,
138  *_sigma2 = sigma2;
139  thrust::for_each(thrust::cuda::par.on(stream), counting,
140  counting + batch_size, [=] __device__(int bid) {
141  DataT* param = param_vec + bid * N;
142  if (order.k) {
143  *param = _mu[bid];
144  param++;
145  }
146  for (int ip = 0; ip < order.p; ip++) {
147  param[ip] = _ar[order.p * bid + ip];
148  }
149  param += order.p;
150  for (int iq = 0; iq < order.q; iq++) {
151  param[iq] = _ma[order.q * bid + iq];
